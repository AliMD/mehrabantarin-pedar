ﻧﺎﺑﺮﺩﻩ ﺭﻧﺞ ﮔﻨﺞ ﻣﻴﺴﺮ ﻧﻤﯽ ﺷﻮﺩ!

ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﮔﻨﺞ ﭘﺮﻧﻌﻤﺘﯽ ﺍﺳـﺖ ﻛـﻪ ﺧﺪﺍﻭﻧﺪ ﺁﻥ ﺭﺍ ﺩﺭ ﻣﻴـﺎﻥ ﻣﺎ ﻧﻬﺎﺩﻩ ﺍﺳـﺖ. ﺑﻪ ﻣﺎ ﺗﻮﺻﻴﻪ ﺷـﺪﻩ ﻛﻪ ﺑﺮﺍی ﺍﺻـﻼﺡ ﺗﻤﺎﻡ ﺍﻣﻮﺭ ﺯﻧﺪﮔﻴﻤﺎﻥ ﺑﻪ ﺍﻳﺸـﺎﻥ ﻣﺮﺍﺟﻌـﻪ ﻛﻨﻴﻢ ﻭ ﺍﺯ ﺁﻥ ﺣﻀﺮﺕ ﻳـﺎﺭی ﺑﺠﻮﺋﻴﻢ. ﺍﻳﻦ ﻣﻌﻨﺎی «ﺗﻤﺴـﮏ ﺑﻪ ﺍﻣﺎﻡ» ﺍﺳـﺖ. ﺍﮔﺮ ﻣﺎ ﺧﺎﻟﺼﺎﻧﻪ ﺍﺯ ﺍﻭ ﺍﺳـﺘﻤﺪﺍﺩ ﻛﻨﻴـﻢ، ﻳﻘﻴﻨـﺎ ﺍﻭ ﻳﺎﺭﻳﻤﺎﻥ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﺍﻭ ﺧﻠﻴﻔﻪ ﺧﺪﺍ ﺩﺭ ﺯﻣﻴﻦ ﺍﺳـﺖ. ﺍﻭ ﺩﺳـﺖ ﺑﺨﺸﻨﺪﻩ ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﻣﻴﺎﻥ ﻣﺎﺳﺖ. ﺍﻭ ﻋﺼﺎﺭﻩ ﺑﻬﺸﺖ ﺍﺳﺖ ﻛـﻪ ﺧـﺪﺍ ﺩﺭ ﻫﻤﻴـﻦ ﺩﻧﻴﺎ ﺑﺮﺍی ﻣﺎ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺍﺳـﺖ. ﺍﻣﺎ ﺁﻳﺎ ﺩﺳـﺘﻴﺎﺑﯽ ﺑﻪ ﺍﻳﻦ ﮔﻨﺞ، ﺑﺪﻭﻥ ﺭﻧﺞ ﺍﻣﻜﺎﻥ ﭘﺬﻳﺮ ﺍﺳـﺖ؟ ﻭ ﺁﻳﺎ ﺑﻬﺮﻩ ﻣﻨﺪ ﺷـﺪﻥ ﺍﺯ ﺍﻳﻦ ﺑﻬﺸﺖ ﺑﯽ ﻫﻴﭻ ﺩﻏﺪﻏﻪ ﻣﻴﺴﺮ ﺍﺳﺖ؟

ﺧﺪﺍﻭﻧـﺪ ﻣﯽ ﻓﺮﻣﺎﻳـﺪ: «ﺁﻳـﺎ ﮔﻤـﺎﻥ ﻛﺮﺩﻩ ﺍﻳـﺪ ﻛـﻪ ﺩﺍﺧﻞ ﺑﻬﺸـﺖ ﻣﯽ ﺷـﻮﻳﺪ ﺩﺭ ﺣﺎﻟـﯽ ﻛـﻪ ﻫﻨـﻮﺯ ﺁﺯﻣﻮﻥ ﻫﺎﻳﯽ ﻛﻪ ﭘﻴﺸـﻴﻨﻴﺎﻥ ﺷـﻤﺎ ﺭﺍ ﺩﺭ ﺑﺮﮔﺮﻓـﺖ، ﺑﺮﺍی ﺷـﻤﺎ ﺭﻭی ﻧﺪﺍﺩﻩ ﺍﺳـﺖ؟ ﺁﻧﻬﺎ ﺩﭼـﺎﺭ ﮔﺮﻓﺘﺎﺭی ﻫﺎ ﻭ ﺳﺨﺘﯽ ﻫﺎی ﺑﻴﭽﺎﺭﻩ ﻛﻨﻨﺪﻩ ﺷﺪﻧﺪ ﻭ ﺑﻪ ﻟﻐﺰﺵ ﻭ ﺗﺰﻟﺰﻝ ﺍﻓﺘﺎﺩﻧﺪ.»(10) ﻧﻴـﺰ ﻣﯽ ﻓﺮﻣﺎﻳـﺪ: «ﺁﻳﺎ ﻣـﺮﺩﻡ ﮔﻤﺎﻥ ﻛﺮﺩﻩ ﺍﻧـﺪ ﺑﻪ ﻫﻤﻴﻦ ﺣـﺪ ﻛﻪ ﺑﻪ ﺯﺑـﺎﻥ ﮔﻔﺘﻨﺪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻳـﻢ ﻛﺎﻓﯽ ﺍﺳـﺖ ﻭ ﺩﭼﺎﺭ ﺍﻣﺘﺤـﺎﻥ ﻧﺨﻮﺍﻫﻨﺪ ﺷﺪ؟»(11)

ﻧـﻪ ! ﭼﻨﻴﻦ ﻧﻴﺴـﺖ. ﺧﺪﺍﻭﻧﺪ ﺑﻬﺸـﺖ ﺁﺧﺮﺕ ﺭﺍ ﺭﺍﻳﮕﺎﻥ ﺑﻪ ﻛﺴـﯽ ﻧﻤﯽ ﺩﻫـﺪ ﻭ ﺑـﻪ ﻫﻤﻴﻦ ﺗﺮﺗﻴﺐ ﮔﻨـﺞ ﭘﺮﻧﻌﻤﺘﯽ ﻫﻤﭽﻮﻥ ﺍﻣـﺎﻡ ﻣﻬﺪی ﺭﺍ ﻧﻴـﺰ ﺍﺭﺯﺍﻥ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻛﺴـﯽ ﻧﻤﯽ ﮔـﺬﺍﺭﺩ، ﺑﻠﻜﻪ ﺑﺮﺍی ﺑﺮﺧـﻮﺭﺩﺍﺭی ﺍﺯ ﻣﺤﻀﺮ ﭘﺮﻓﺎﻳﺪﻩ ﺍﻭ ﺍﻧﺴﺎﻥ ﻫﺎ ﺭﺍ ﻣﯽ ﺁﺯﻣﺎﻳﺪ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻣﺎﻡ ﻛﻪ ﺟﺎﻥ ﻫﺎی ﻣﺎ ﻓﺪﺍﻳﺶ ﺑﺎﺩ ﺩﺭﺟﺎﺕ ﻭ ﺳﻄﻮﺡ ﻣﺘﻔـﺎﻭﺕ ﺩﺍﺭﺩ، ﻭ ﺧﺪﺍﻭﻧـﺪ ﺩﺭ ﻫﺮ ﻣﺮﺣﻠﻪ ﻭ ﺩﺭ ﻫﺮ ﺩﺭﺟﻪ ﺑﻪ ﺗﻨﺎﺳـﺐ، ﻣﺎ ﺭﺍ ﻣﯽ ﺁﺯﻣﺎﻳﺪ.

ﺍﮔـﺮ ﭼﻨﻴـﻦ ﺑﻮﺩ ﻛﻪ ﻫﺮ ﻛﺴـﯽ ﻛﻪ ﺑﻪ ﺧﺎﻧـﻪ ﭘﻴﺎﻣﺒﺮﺍﻥ ﻭ ﺟﺎﻧﺸـﻴﻨﺎﻥ ﺍﻳﺸـﺎﻥ ﻣﺮﺍﺟﻌﻪ ﻣﯽ ﻛـﺮﺩ، ﺑﻼﻓﺎﺻﻠﻪ ﻫﺮﭼـﻪ ﺭﺍ ﻣﯽ ﺧﻮﺍﺳـﺖ (ﺍﻋﻢ ﺍﺯ ﺣﻮﺍﻳـﺞ ﻣﺎﺩ ی ﻭ ﻣﻌﻨﻮی ) ﺩﺭﻳﺎﻓﺖ ﻣﯽ ﻧﻤـﻮﺩ، ﺁﻳﺎ ﺩﻳﮕﺮ ﻳﮏ ﻛﺎﻓﺮ ﺑﺎﻗﯽ ﻣﯽ ﻣﺎﻧﺪ؟ ﻣﺴﻠﻤﺎ ﺳﻮﺩﺟﻮﻳﺎﻥ ﺑﯽ ﺍﻳﻤﺎﻥ ﻧﻴﺰ ﺑﺮﺍی ﺟﻠﺐ ﻣﻨﺎﻓﻊ ﺷﺨﺼﯽ ﺩﺭ ﺁﺳـﺘﺎﻧﻪ ﺍﻳﻦ ﺑﺰﺭﮔﺎﻥ ﺭﺣﻞ ﺍﻗﺎﻣﺖ ﻣﯽ ﺍﻓﻜﻨﺪﻧﺪ ﻭ ﺣﺘﯽ ﺑﺮﺍی ﻛﺴـﺐ ﻣﻨﺎﻓـﻊ ﺑﻴﺸـﺘﺮ ﺍﺯ ﺑﺮﺧـﯽ ﻣﺆﻣﻨـﺎﻥ ﺣﻘﻴﻘﯽ ﻧﻴﺰ ﭘﻴﺸـﯽ ﻣﯽ ﮔﺮﻓﺘﻨـﺪ. ﺍﮔﺮ ﭼﻨﻴـﻦ ﺑﻮﺩ ﺁﻳـﺎ ﺍﻳﻤﺎﻥ ﭼﻨﻴﻦ ﻣﺪﻋﻴﺎﻧﯽ ﺍﺭﺯﺷـﯽ ﻫﻢ ﺩﺍﺷـﺖ؟ ﻭ ﺁﻳﺎ ﺁﻧﻬﺎ ﻣﺴـﺘﺤﻖ ﺛﻮﺍﺑﯽ ﻫﻢ ﺑﻮﺩﻧﺪ؟ ﺩﺭ ﺻﻮﺭﺕ ﺍﻳﻦ ﻓﺮﺽ ﺁﻳﺎ ﻣﺆﻣﻨﺎﻥ ﺣﻘﻴﻘﯽ ﺍﺯ ﻣﺪﻋﻴـﺎﻥ ﺩﺭﻭﻏﻴـﻦ ﺍﻳﻤـﺎﻥ ﺑﺎﺯﺷـﻨﺎﺧﺘﻪ ﻣﯽ ﺷـﺪﻧﺪ؟ ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﭼﻪ ﺑﺴـﺎ ﺩﻧﻴﺎﭘﺮﺳـﺘﺎﻧﯽ ﻛﻪ ﺩﺭ ﺑﺎﻃﻦ ﻛﻴﻨـﻪ ﺍﻭﻟﻴﺎی ﺧﺪﺍ ﺭﺍ ﺩﺍﺷـﺘﻨﺪ ﻭ ﺑﻪ ﺍﻳﺸـﺎﻥ ﺣﺴـﺪ ﻣﯽ ﺑﺮﺩﻧﺪ، ﺩﺭ ﻣﺮﺍﺟﻌﻪ ﺑﻪ ﭘﻴﺎﻣﺒـﺮﺍﻥ ﻭ ﺍﻣﺎﻣﺎﻥ ﺍﺯ ﺩﻳﮕﺮﺍﻥ ﭘﻴﺸـﯽ ﻣﯽ ﮔﺮﻓﺘﻨـﺪ ﻭ ﺑﺎ ﺩﻭﺭﻭﻳﯽ ﺧـﻮﺩ ﺭﺍ ﺑﻪ ﺁﻧﺎﻥ ﻧﺰﺩﻳـﮏ ﻭ ﻧﺰﺩﻳﮏﺗﺮ ﻣﯽ ﺳﺎﺧﺘﻨﺪ، ﺗﺎ ﺍﺯ ﮔﻨﺠﻴﻨﻪ ﻛﺮﺍﻣﺎﺕ ﺁﻧﻬﺎ ﺑﻴﺸﺘﺮ ﺑﻬﺮﻩ ﺑﺒﺮﻧﺪ.(12)

ﺍﻣﺎ ﺍﺯ ﺁﻧﺠﺎ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺍﺭﺍﺩﻩ ﻛﺮﺩﻩ ﺗﺎ ﺳـﺮﺍﻧﺠﺎﻡ ﻣﺆﻣﻦ ﺭﺍﺳﺘﻴﻦ ﺍﺯ ﻣﺪﻋـﯽ ﺩﺭﻭﻏﻴﻦ ﺗﻔﻜﻴﮏ ﮔﺮﺩﺩ، ﺩﻧﻴﺎ ﺭﺍ ﺍﺯ ﺁﺯﻣﻮﻥ ﻫﺎی ﮔﻮﻧﺎﮔﻮﻥ ﻟﺒﺮﻳﺰ ﺳـﺎﺧﺘﻪ ﻭ ﺩﺭ ﺑﻮﺗﻪ ﭘﺮﺣﺮﺍﺭﺕ ﺍﻳﻦ ﺁﺯﻣﺎﻳﺶﻫﺎ ﻋﻴﺎﺭ ﺍﻳﻤﺎﻥ ﻣﺆﻣﻨﺎﻥ ﺭﺍ ﺑﻪ ﻣﺤﮏ ﺻﺒﺮ ﻭ ﭼﮕﻮﻧﮕﯽ ﺭﻭﻳﺎﺭﻭﻳﯽ ﺑﺎ ﺣﻮﺍﺩﺙ ﺭﻭﺯﮔﺎﺭ ﻣﯽ ﺳﻨﺠﺪ.

ﺭﺍﺳﺘﯽ ﻛﻪ ﻋﺠﺐ ﺗﻘﺪﻳﺮ ﺷﮕﻔﺖ ﻭ ﺧﻴﺮﻩ ﻛﻨﻨﺪﻩ ﺍی ﺩﺍﺭﺩ، ﺧﺪﺍ!

ﺍﺯ ﻳﮏﺳـﻮ ﺑﺎ ﻗﺮﺍﺭﺩﺍﺩﻥ ﮔﻨﺠﯽ ﭼﻮﻥ ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﺩﺭ ﻣﻴﺎﻥ ﻣﺎ ﻣﺮﺩﻡ ﺭﺍﻩ ﻣﺎ ﺭﺍ ﺑﻪ ﺳـﻮی ﺑﻬﺮﻩ ﻣﻨـﺪی ﺍﺯ ﺍﻳﻦ ﮔﻨﺞ ﭘﺮﻧﻌﻤﺖ ﺑﺎﺯ ﻛﺮﺩﻩ ﺍﺳـﺖ ﻭ ﺍﺯ ﺳـﻮی ﺩﻳﮕـﺮ ﺑﺎ ﻗـﺮﺍﺭ ﺩﺍﺩﻥ ﺳـﻨﮓﻫﺎ ﻭ ﺭﻧﺞﻫﺎﻳـﯽ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ، ﺩﺳﺖ ﻧﺎﺍﻫﻼﻥ ﺭﺍ ﺍﺯ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻳﻦ ﮔﻨﺞ ﺑﯽ ﻧﻈﻴﺮ ﻛﻮﺗﺎﻩ ﻣﯽ ﻛﻨﺪ ﻭ ﺻﺒﻮﺭی ﻣﺸﺘﺎﻗﺎﻥ ﻭﺍﻗﻌﯽ ﺭﺍ ﻣﯽ ﺁﺯﻣﺎﻳﺪ.

ﺁﺭی ، ﻫﻤـﺎﻥ ﮔﻮﻧﻪ ﻛﻪ ﮔﻨﺞ ﻫﺴـﺖ ﻭ ﺣﻘﻴﻘﺖ ﺩﺍﺭﺩ، ﺭﻧﺞ ﻫﻢ ﺩﺭ ﺭﺍﻩ ﺭﺳﻴﺪﻥ ﺑﻪ ﮔﻨﺞ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﻫﺪﻑ ﺍﻳﻦ ﮔﻔﺘﺎﺭ ﺗﺮﺳﺎﻧﺪﻥ ﻭ ﻧﺎﺍﻣﻴﺪ ﻛﺮﺩﻥ ﻛﺴﺎﻧﯽ ﻧﻴﺴﺖ ﻛﻪ ﺗﺎﺯﻩ ﺩﺭﻳﺎﻓﺘﻪ ﺍﻧﺪ ﭼﻨﻴﻦ ﮔﻨﺠﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻠﻜﻪ ﻫﺪﻑ، ﻧﻴﺮﻭ ﺩﺍﺩﻥ ﻭ ﺗﺸﻮﻳﻖ ﭼﻨﻴﻦ ﺍﻓﺮﺍﺩی ﺍﺳﺖ ﺑﻪ ﭘﺎﻳﺪﺍﺭی ﺩﺭ ﺭﺍﻩ ﺭﺳﻴﺪﻥ ﺑﻪ ﺁﻥ ﮔﻨﺞ ﻭ ﺑﻪ ﺍﻳﻨﻜﻪ ﺑﺎ ﻣﻮﺍﺟﻪ ﺷـﺪﻥ ﺑﺎ ﻣﻮﺍﻧﻊ ﺍﺯ ﺟﺴﺘﺠﻮی ﺁﻥ ﺧﺴﺘﻪ ﻭ ﻧﺎﺍﻣﻴﺪ ﻧﺸﻮﻧﺪ. ﻫﺪﻑ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺍﺭﺯﺵ ﺍﻳﻤﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﺪﺍﻧﻴﻢ ﻭ ﺁﮔﺎﻩ ﺑﺎﺷـﻴﻢ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺍﻳﻤـﺎﻥ ﻣـﺎ ﺭﺍ ﺑﺎ ﺻﺒـﺮ ﻣﯽ ﺁﺯﻣﺎﻳـﺪ. ﺑﺮﺍی ﻓﻬـﻢ ﺍﻳﻦ ﻣﻄﻠﺐ «ﺩﻋـﺎ» ﻣﺜﺎﻝ ﺧﻮﺑﯽ ﺍﺳﺖ.

ﻣﺎ ﺷـﮏ ﻧﺪﺍﺭﻳﻢ ﻛﻪ ﺧﺪﺍ ﺑﺮ ﺣﻞ ﻣﺸـﻜﻼﺕ ﻣﺎ ﺗﻮﺍﻧﺎﺳﺖ ﻭ ﺷﮏ ﻧﺪﺍﺭﻳـﻢ ﻛﻪ ﺍﻭ ﺑﻪ ﻣﺎ ﮔﻔﺘﻪ ﻛﻪ ﺩﻋﺎ ﻛﻨﻴﺪ ﻭ ﺍﺯ ﻣﻦ ﺑﺨﻮﺍﻫﻴﺪ ﺗﺎ ﻣﺸـﻜﻼﺕ ﺷـﻤﺎ ﺭﺍ ﺣـﻞ ﻛﻨﻢ. ﺷـﮏ ﻧﺪﺍﺭﻳﻢ ﻛـﻪ ﺍﻭ ﻓﺮﻣﻮﺩﻩ ﻣﻦ ﺑﻪ ﺩﻋﺎﻫﺎی ﺷـﻤﺎ ﭘﺎﺳـﺦ ﺧﻮﺍﻫـﻢ ﺩﺍﺩ. ﺍﻣﺎ ﺁﻳﺎ ﻫﺮ ﺑﺎﺭ ﻛﻪ ﭼﻴﺰی ﺍﺯ ﺧﺪﺍ ﺧﻮﺍﺳـﺘﻪ ﺍﻳﻢ، ﺩﺭ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﺧﻮﺍﺳـﺘﻪ ﻣﺎ ﺑﻪ ﻣﺎ ﺩﺍﺩﻩ ﺷـﺪﻩ ﺍﺳﺖ؟ ﺗﺠﺮﺑﻪ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻛﻪ ﺧﻴﻠﯽ ﻭﻗﺖﻫﺎ ﭘﺎﺳـﺦ ﺧﺪﺍﻭﻧﺪ ﻓﻮﺭی ﺑﻪ ﻣﺎ ﻧﻤﯽ ﺭﺳـﺪ. ﻳﻜﯽ ﺍﺯ ﺩﻻﻳﻞ ﺍﻳـﻦ ﺍﻣﺮ ﺍﻳﻦ ﺍﺳـﺖ ﻛـﻪ ﺍﻳﻤﺎﻥ ﻣﺎ ﺑﻪ ﺷـﻨﻮﺍ ﺑﻮﺩﻥ ﺧﺪﺍﻭﻧﺪ، ﺑـﻪ ﺍﻳﻨﻜﻪ ﺍﻭ ﺧﻠﻒ ﻭﻋﺪﻩ ﻧﻤﯽ ﻛﻨﺪ ﻭ ﺍﻳﻨﻜﻪ ﺍﻭ ﺑﺮ ﺣﻞ ﻣﺸـﻜﻞ ﻣﺎ ﺗﻮﺍﻧﺎﺳـﺖ ﻭ ﺍﻳﻨﻜﻪ ﺍﻭ ﺑﺮﺁﻭﺭﻧﺪﻩ ﻧﻴﺎﺯﻫﺎﺳﺖ، ﺁﺯﻣﻮﺩﻩ ﺷﻮﺩ.

ﺧﺪﺍﻭﻧـﺪ ﮔﻨـﺞ ﺍﺟﺎﺑﺖ ﺭﺍ ﺩﺭ ﭘـﺲ ﺑﺮﺧﯽ ﺍﺯ ﺩﻋﺎﻫـﺎی ﻣﺎ ﻣﺨﻔﯽ ﻛـﺮﺩﻩ ﺍﺳـﺖ ﺗـﺎ ﺍﺯ ﺍﻳﻦ ﺭﻫﮕـﺬﺭ، ﭘﻴﻮﻧﺪ ﻣـﺎ ﺭﺍ ﺑـﺎ ﺧﻮﺩ ﺗﺪﺍﻭﻡ ﺑﺨﺸـﺪ ﻭ ﺷـﺪﺕ ﺍﻳﻤـﺎﻥ ﻣـﺎ ﺭﺍ ﺑﻪ ﺳـﻨﺠﺶ ﺑﮕﺬﺍﺭﺩ. ﺍﮔـﺮ ﺑﺎ ﺍﻭﻟﻴـﻦ ﺩﻋﺎ، ﮔﻨﺞ ﺍﺟﺎﺑـﺖ ﺑـﻪ ﻣﺎ ﺩﺍﺩﻩ ﺷـﻮﺩ، ﺩﻳﮕﺮ ﺩﻋﺎ ﻛـﺮﺩﻥ ﺍﺭﺯﺵ ﺧـﻮﺩ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳـﮏ ﻋﺒﺎﺩﺕ ﺍﺯ ﺩﺳـﺖ ﻣﯽ ﺩﻫﺪ. ﺑـﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺧﺪﺍﻭﻧـﺪ، ﺍﺟﺎﺑﺖ ﺭﺍ ﺩﺭ ﻣﻴﺎﻥ ﺩﻋﺎﻫﺎی ﻣﺎ ﭘﻨﻬﺎﻥ ﻛﺮﺩﻩ ﺍﺳـﺖ. ﻣﻌﻨﯽ ﺍﻳﻦ ﻛﻼﻡ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﻣﺜﻼ ﻫﻔﺘﺎﺩ ﺑﺎﺭ ﺍﺯ ﺍﻭ ﻃﻠﺐ ﻣﻐﻔﺮﺕ ﻣﯽ ﻛﻨﻴﻢ، ﺍﻣﺎ ﻣﻐﻔﺮﺕ ﺍﻟﻬﯽ ﺗﻨﻬﺎ ﺩﺭ ﺍﺛـﺮ ﻳﻜﯽ ﺍﺯ ﺍﻳﻦ ﻫﻔﺘﺎﺩ ﺑﺎﺭ ﻛﻪ ﻣﻌﻠﻮﻡ ﻧﻴﺴـﺖ ﻛﺪﺍﻡ ﻳﮏ ﺍﺯ ﺁﻧﻬﺎﺳـﺖ ﺑﻪ ﻣﺎ ﻋﻄﺎ ﻣﯽ ﺷﻮﺩ. ﻟﺬﺍ ﺩﺭ ﺩﻳﻦ ﺗﻮﺻﻴﻪ ﺷﺪﻩ ﻛﻪ ﺑﻪ ﺩﻋﺎ ﺩﺭ ﻫﻴﭻ ﻣﻮﻗﻌﻴﺘﯽ ﻛﻢ ﺗﻮﺟﻬـﯽ ﻧﻜﻨﻴﻢ، ﺯﻳﺮﺍ ﺍﻣﻜﺎﻥ ﺩﺍﺭﺩ ﻛﻪ ﺍﺟﺎﺑـﺖ ﺩﺭ ﺩﻝ ﻫﻤﺎﻥ ﺩﻋﺎﻳﯽ ﭘﻨﻬﺎﻥ ﺑﺎﺷﺪ ﻛﻪ ﻣﺎ ﺁﻥ ﺭﺍ ﺟﺪی ﻧﻤﯽ ﮔﻴﺮﻳﻢ. ﻧﻴﺰ ﺗﻮﺻﻴﻪ ﺷﺪﻩ ﻛﻪ ﺩﺭ ﺩﻋﺎ ﻛﺮﺩﻥ ﺍﺻﺮﺍﺭ، ﭘﺎﻓﺸﺎﺭی ﻭ ﺗﺪﺍﻭﻡ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ، ﺯﻳﺮﺍ ﺍﮔﺮ ﻛﺴﯽ ﭘﺲ ﺍﺯ ﻣﺪﺗﯽ ﺍﺯ ﺩﻋﺎ ﺧﺴﺘﻪ ﺷﻮﺩ ﻭ ﺩﺳﺖ ﺍﺯ ﻃﻠﺐ ﺑﺮﺩﺍﺭﺩ، ﺧﻮﺩ ﺭﺳﻤﺎ ﺍﻋﻼﻥ ﻛﺮﺩﻩ ﻛﻪ ﺍﺯ ﺣﻖ ﻣﺘﻌﺎﻝ ﻭ ﻛﻤﮏ ﺍﻭ ﻧﺎﺍﻣﻴﺪ ﺷﺪﻩ ﺍﺳﺖ. ﺧﻼﺻﻪ ﺍﻳﻦ ﻛﻪ ﺭﺳـﻢ ﺩﺭﺳـﺖ ﺩﻋﺎ ﻛﺮﺩﻥ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺍﺯ ﺗﺪﺍﻭﻡ ﺩﺭ ﺁﻥ ﺧﺴﺘﻪ ﻭ ﻧﺎﺍﻣﻴﺪ ﻧﺸـﻮﻳﻢ ﻭ ﻳﻘﻴﻦ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻛﻪ ﺳـﺮﺍﻧﺠﺎﻡ ﺧﻮﺭﺷﻴﺪ ﻋﻨﺎﻳﺎﺕ ﺍﻟﻬﯽ ﺍﺯ ﭘﺲ ﺍﺑﺮﻫﺎ ﺧﻮﺍﻫﺪ ﺩﺭﺧﺸﻴﺪ.

ﻫﻤﻴـﻦ ﻭﺿﻌﻴﺖ ﺩﺭ ﻣﻮﺭﺩ ﺗﻮﺳـﻞ ﺑـﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭ ﺗﻤﺴـﮏ ﺑﻪ ﺍﻭ ﻛﻪ ﺧﻠﻴﻔﻪ ﺧﺪﺍﺳـﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺩﺭ ﺭﺍﻩ ﺗﻤﺴـﮏ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻧﻴﺰ ﺍﻳﻤﺎﻥ ﻣﺆﻣﻨﺎﻥ ﺑﺎ ﻣﺤﮏ ﺻﺒﺮ ﺁﺯﻣﻮﺩﻩ ﻣﯽ ﺷﻮﺩ.
