بازگشت

ﻗﺼﻪ ﺁﻥ ﭼﺎﻩ ﭘﺮ ﺍﺯ ﺁﺏ

ﺧﺪﺍی ﻣﺘﻌﺎﻝ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﭼﻪ ﺑﺴﻴﺎﺭ ... ﭼﺎﻫﯽ ﻛﻪ ﻭﺍﻧﻬﺎﺩﻩ ﺷﺪﻩ ﻭ ﭼﻪ ﺑﺴﺎ ﻗﺼﺮ ﺑﻠﻨﺪی ﻛﻪ ﭼﻨﻴﻦ ﺍﺳﺖ.»(15)

ﭼـﻪ ﺑﺴـﻴﺎﺭ ﻛـﻪ ﺧﺪﺍﻭﻧـﺪ ﺩﺭ ﺍﺛـﺮ ﺳـﺘﻤﻜﺎﺭی ﻣـﺮﺩﻡ ﺁﻧﻬـﺎ ﺭﺍ ﺍﺯ ﻧﻌﻤﺖﻫﺎی ﺧﻮﺩ ﻣﺤﺮﻭﻡ ﻣﯽ ﺳـﺎﺯﺩ. ﺍﻳﻦ ﺳﻨﺖ ﺍﻟﻬﯽ ﺍﺳﺖ ﻭ ﺩﺭ ﺗﻤﺎﻡ ﺍﻣﺖﻫﺎ ﺳﺎﺑﻘﻪ ﺩﺍﺷﺘﻪ ﺍﺳﺖ.

ﺧﺪﺍﻭﻧـﺪ ﺩﺭ ﺍﻳﻦ ﺁﻳﻪ ﺷـﺮﻳﻔﻪ ﺑﻪ ﻣﺤﺮﻭﻡ ﺷـﺪﻥ ﺑﺮﺧـﯽ ﺍﺯ ﻣﺮﺩﻡ ﺍﺯ ﻧﻌﻤﺖﻫﺎی ﺧﻮﻳﺶ ﺍﺷـﺎﺭﻩ ﻣﯽ ﻛﻨﺪ ﻭ ﺑﻴﺎﻥ ﻣـﯽ ﺩﺍﺭﺩ ﻛﻪ ﺩﺭ ﭘﯽ ﻣﺤﺮﻭﻡ ﺷـﺪﻥ ﺧﻄﺎﻛﺎﺭﺍﻥ ﺍﺯ ﻧﻌﻤﺖﻫﺎی ﺍﻟﻬﯽ ، ﺑﺴـﻴﺎﺭی ﺍﺯ ﺍﻳﻦ ﻣﻮﻫﺒﺖﻫﺎ ﺩﺭ ﻃﻮﻝ ﺗﺎﺭﻳﺦ ﺑﺪﻭﻥ ﺍﺳـﺘﻔﺎﺩﻩ ﻣﺎﻧﺪﻩ ﺍﺳﺖ؛ ﻓﯽ ﺍﻟﻤﺜﻞ ﭼﺎﻩ ﻫﺎی ﭘﺮﺁﺑﯽ ﻛﻪ ﺩﻳﮕـﺮ ﺭﻫﮕﺬﺭی ﺁﻧﻬـﺎ ﺭﺍ ﭘﻴـﺪﺍ ﻧﻤﯽ ﻛﻨﺪ، ﺑﺮ ﺳـﺮ ﺁﻧﻬﺎ ﻧﻤﯽ ﺁﻳـﺪ، ﺩﻟﻮی ﺩﺭ ﺁﻥ ﻫـﺎ ﻧﻤﯽ ﺍﻓﻜﻨـﺪ ﻭ ﻋﻄﺸـﯽ ﻓﺮﻭﻧﻤﯽ ﻧﺸـﺎﻧﺪ؛ ﻭ ﻗﺼﺮﻫـﺎی ﺑﻠﻨﺪی ﻛﻪ ﭼﺸـﻢ ﻫﺎ ﺭﺍ ﻣﯽ ﻧﻮﺍﺯﻧـﺪ (ﻭ ﻣﺎﻧﻨﺪ ﭼﺎﻩ ﻫـﺎ ﻣﺨﻔﯽ ﻧﻤﺎﻧﺪﻩ ﺍﻧـﺪ) ﺍﻣﺎ ﺑﺎ ﺍﻳـﻦ ﺣﺎﻝ ﻛﺴـﯽ ﺩﺭ ﺁﻧﻬـﺎ ﺳـﻜﻮﻧﺖ ﻧﻤﯽ ﻛﻨﺪ ﻭ ﺩﺭ ﺳـﺎﻳﻪ ی ﺁﻧﻬـﺎ ﺁﺭﺍﻡ ﻧﻤﯽ ﮔﻴﺮﺩ!

ﺩﺭ ﺍﻳـﻦ ﻛﻼﻡ ﺧـﺪﺍ ﺣﻜﺎﻳﺖ ﺍﺯ ﺩﻭ ﭼﻴﺰ ﻫﺴـﺖ. ﺍﻭﻝ ﺣﻜﺎﻳﺖ ﺍﺯ ﻣﺤﺮﻭﻣﻴـﺖ ﻣـﺮﺩﻡ ﺳـﺘﻤﻜﺎﺭ ﺍﺯ ﻧﻌﻤﺖﻫـﺎی ﺧـﺪﺍ ﻭ ﺩﻭﻡ ﺣﻜﺎﻳﺖ ﺍﺯ ﻣﻬﺠﻮﺭ، ﻏﺮﻳﺐ ﻭ ﺑﯽ ﺍﺳﺘﻔﺎﺩﻩ ﻣﺎﻧﺪﻥ ﺁﻥ ﻧﻌﻤﺖﻫﺎ.

ﺍﻳﻨـﮏ ﻣـﺎ ﺍﺯ ﻣﻔﻬـﻮﻡ ﻇﺎﻫـﺮ ﺁﻳـﻪ ﺁﮔﺎﻫﻴـﻢ. ﻇﺎﻫـﺮ ﺁﻳـﻪ ﺍﺷـﺎﺭﻩ ﺑﻪ ﻗﺼﺮﻫـﺎی ﺑﻠﻨـﺪ ﻭ ﭼﺎﻩ ﻫﺎی ﭘﺮﺁﺑـﯽ ﺩﺍﺭﺩ ﻛـﻪ ﺍﺯ ﺍﻗﻮﺍﻡ ﻫﻼک ﺷـﺪﻩ ﺑﻪ ﺟﺎی ﻣﺎﻧﺪﻩ ﺍﺳـﺖ، ﺍﻣﺎ ﺑﺎﻃﻦ ﻭ ﺗﺄﻭﻳﻞ ﺁﻳﻪ ﺣﻜﺎﻳﺖﮔﺮ ﺍﺯ ﭼﻴﺴـﺖ؟ ﺑﻪ ﺗﻌﺒﻴﺮ ﺩﻳﮕﺮ ﻧﻤﻮﻧﻪ ﻋﻴﻨﯽ ﻭ ﻣﺼﺪﺍﻕ ﻛﻼﻡ ﺣﻖ ﺩﺭ ﺍﻣﺖ ﺍﺳـﻼﻡ ﻭ ﻋﺼﺮ ﺣﺎﺿﺮ ﭼﻪ ﭼﻴﺰ ﻳﺎ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺍﺳﺖ؟

ﺩﺭ ﺭﻭﺍﻳـﺎﺕ ﺍﻫﻞﺑﻴﺖ ﺁﻣﺪﻩ ﻛـﻪ : «ﻣﻨﻈﻮﺭ ﺍﺯ ﭼﺎﻫـﯽ ﻛﻪ ﻭﺍﻧﻬﺎﺩﻩ ﺷـﺪﻩ ﺍﻣﺎﻣﯽ ﺍﺳـﺖ ﻛﻪ ﺳـﻜﻮﺕ ﺍﺧﺘﻴﺎﺭ ﻣﯽ ﻛﻨﺪ ﻭ ﻣﻘﺼـﻮﺩ ﺍﺯ ﻗﺼﺮ ﺑﻠﻨﺪی ﻛﻪ ﺧﺪﺍ ﺑﻪ ﺁﻥ ﺍﺷـﺎﺭﻩ ﻓﺮﻣﻮﺩﻩ ﺍﻣﺎﻣﯽ ﺍﺳـﺖ ﻛﻪ ﺳـﺨﻦ ﺧﻮﺩ ﺭﺍ ﺍﺑﺮﺍﺯ ﻣﯽ ﻧﻤﺎﻳﺪ.»(16)

ﺑـﻪ ﻋﻨـﻮﺍﻥ ﻧﻤﻮﻧـﻪ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴـﻦ ﻭ ﺍﻣﺎﻡ ﺣﺴـﻦ ﻣﺠﺘﺒـﯽ ﻋﻠﻴﻬﻤﺎ ﺍﻟﺴـﻼﻡ، ﺩﺭ ﺯﻣﺎﻧﯽ ﻛﻪ ﺣﺎﻛﻢ ﻣﺴـﻠﻤﺎﻧﺎﻥ ﺑﻮﺩﻧﺪ ﻭ ﻣﺮﺩﻡ ﺭﺍ ﻋﻠﻨﺎ ﺑﻪ ﺭﺍﻩ ﺣـﻖ ﺩﻋﻮﺕ ﻣﯽ ﻛﺮﺩﻧﺪ، ﻧﻤﻮﻧﻪ ﻫﺎﻳـﯽ ﺍﺯ ﺁﻥ ﻗﺼﺮﻫﺎی ﻣﺮﺗﻔﻊ ﻭ ﺑﻠﻨﺪ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ. ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﻋﺼﺮ ﺣﻜﻮﻣﺖ ﻳﺰﻳﺪ ﻧﻴﺰ ﻧﻤﻮﻧـﻪ ﺍی ﺩﻳﮕـﺮ ﺍﺯ ﺁﻥ ﻗﺼﺮﻫـﺎی ﺑﻠﻨﺪ ﻭ ﺁﺷـﻜﺎﺭ ﺍﺳـﺖ. ﺁﻥ ﺣﻀﺮﺕ ﺁﺷﻜﺎﺭﺍ ﺳﺨﻦ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﻣﺖ ﺍﺑﺮﺍﺯ ﻛﺮﺩ.

ﺍﺯ ﻃﺮﻓـﯽ ﺩﻳﮕـﺮ ﺍﻣﺎﻣﺎﻧﯽ ﻫﻤﭽﻮﻥ ﺍﻣﺎﻡ ﻫﺎﺩی ، ﺍﻣﺎﻡ ﻋﺴـﻜﺮی ﻭ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﻧﻤﻮﻧﻪ ﻫﺎﻳﯽ ﺍﺯ ﺁﻥ ﭼﺎﻩ ﻫﺎی ﭘﺮ ﺍﺯ ﺁﺏ ﻫﺴﺘﻨﺪ ﻛـﻪ ﻣﺘـﺮﻭک ﻣﺎﻧﺪﻩ ﺍﻧﺪ. ﺍﻳـﻦ ﺑﺰﺭﮔـﻮﺍﺭﺍﻥ ﺭﻭﺯﮔﺎﺭ ﺍﻣﺎﻣـﺖ ﺧﻮﻳﺶ ﺭﺍ ﻋﻤﺪﺗﺎ ﺩﺭ ﺳﻜﻮﺕ ﻭ ﺧﻔﺎ ﮔﺬﺭﺍﻧﺪﻩ ﺍﻧﺪ.

ﺣﻘﻴﻘﺖ ﺩﺭﺩﻧﺎﻛﯽ ﻛﻪ ﺩﺭ ﺁﻳﻪ ﺷـﺮﻳﻔﻪ ﺑﻴﺎﻥ ﺷـﺪﻩ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﻧﻪ ﺗﻨﻬـﺎ ﺍﻳﻦ ﭼﺎﻩ ﻫﺎی ﭘﺮ ﺍﺯ ﺁﺏ ﻣﺘﺮﻭک ﻭ ﺑﯽ ﺍﺳـﺘﻔﺎﺩﻩ ﻣﺎﻧﺪﻩ ﺍﻧﺪ، ﺑﻠﻜﻪ ﺁﻥ ﻗﺼﺮﻫﺎی ﺁﺷﻜﺎﺭ ﻭ ﺑﻠﻨﺪ ﻧﻴﺰ ﭼﻨﻴﻦ ﻣﻬﺠﻮﺭ ﻭ ﺑﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺭﻫﺎ ﺷﺪﻩ ﺍﻧﺪ. ﺩﺭ ﺁﻳـﻪ ﺷـﺮﻳﻔﻪ ﺍﺷـﺎﺭﻩ ﺷـﺪﻩ ﻛﻪ ﻫـﻢ ﺁﻥ ﭼﺎﻩ ﻫـﺎی ﭘـﺮ ﺍﺯ ﺁﺏ ﻭ ﻫﻢ ﺁﻥ ﻗﺼﺮﻫـﺎی ﻣﺮﺗﻔـﻊ، ﻣﻬﺠﻮﺭ ﻭ ﻏﺮﻳﺐ ﻣﺎﻧﺪﻩ ﺍﻧﺪ. ﺳـﺎﺩﻩ ﺍﻧﮕﺎﺭی ﺍﺳـﺖ ﻛـﻪ ﺑﭙﻨﺪﺍﺭﻳـﻢ ﻣﺮﺩﻡ ﺯﻣـﺎﻥ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻳﺎ ﺍﻣﺎﻡ ﺣﺴـﻦ ﻣﺠﺘﺒﯽ ﻳﺎ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﻋﻤﻮﻣﺎ ﺳـﺨﻨﺎﻥ ﺍﺑﺮﺍﺯ ﺷـﺪﻩ ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭﺍﻥ ﺭﺍ ﺑـﻪ ﺟـﺎﻥ ﻣﯽ ﺧﺮﻳﺪﻧﺪ ﻭ ﭼﻮﻥ ﺁﻧﻬـﺎ ﺭﺍ ﺩﺭ ﺻﺤﻨﻪ ﻋﻠﻨـﯽ ﺍﺟﺘﻤﺎﻉ ﻇﺎﻫﺮ ﻣﯽ ﺩﻳﺪﻧﺪ، ﺑﻬﺮﻩ ﻛﺎﻓﯽ ﺭﺍ ﺍﺯ ﺁﻧﺎﻥ ﻣﯽ ﺑﺮﺩﻧﺪ.

ﺁﻳـﺎ ﺍﻃﺮﺍﻓﻴـﺎﻥ ﻋﻠـﯽ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﻧﺼﻴﺤﺖﻫـﺎی ﺍﻭ ﺭﺍ ﺩﺭ ﻛﻨﺎﺭ ﮔﺬﺍﺷـﺘﻦ ﺑﺪﻋﺖﻫﺎی ﺩﻭ ﺧﻠﻴﻔﻪ ﺍﻭﻝ ﭘﺬﻳﺮﻓﺘﻨﺪ؟ ﺁﻳﺎ ﻣﺪﻋﻴﺎﻥ ﻫﻮﺍﺩﺍﺭی ﻣـﻮﻻ ﺑﻴﻨـﺶ ژﺭﻑ ﺍﻭ ﺭﺍ ﺩﺭ ﻧﺒـﺮﺩ ﺑـﺎ ﺳـﭙﺎﻩ ﺩﺭﻭﻍﭘـﺮﺩﺍﺯ ﻣﻌﺎﻭﻳـﻪ ﻛـﻪ ﻗﺮﺁﻥ ﻫـﺎ ﺭﺍ ﺑﺮ ﺳـﺮ ﻧﻴﺰﻩ ﻫﺎ ﻛـﺮﺩﻩ ﺑﻮﺩﻧﺪ، ﺣﺮﻣﺖ ﻧﻬﺎﺩﻧﺪ؟ ﺁﻳﺎ ﺳـﭙﺎﻫﻴﺎﻥ ﺍﻣـﺎﻡ ﻣﺠﺘﺒﯽ ﻫﻨﮕﺎﻣﯽ ﻛـﻪ ﺁﻥ ﻫﺎ ﺭﺍ ﺑﻪ ﺟﻬﺎﺩ ﻓﺮﺍﻣﯽ ﺧﻮﺍﻧﺪ، ﺑﻪ ﺩﺭﺳـﺘﯽ ﺩﻋﻮﺗﺶ ﺭﺍ ﻟﺒﻴﮏ ﮔﻔﺘﻨﺪ؟ ﻭ ﺁﻳﺎ ﺩﺭ ﺁﻥ ﻫﻨﮕﺎﻡ ﻛﻪ ﺳﺴـﺖﻋﻨﺼﺮی ﺁﻥ ﻫﺎ ﻛﺎﺭ ﺭﺍ ﺑـﺮ ﺍﻣـﺎﻡ ﻭ ﻳﺎﺭﺍﻥ ﺑﺎﻭﻓﺎﻳﺶ ﺳـﺨﺖ ﻛﺮﺩ، ﺑﺮ ﺗﺼﻤﻴـﻢ ﺍﻭ ﺩﺭﺑﺎﺭﻩ ﺻﻠـﺢ ﺑـﻪ ﺭﻏﺒﺖ ﮔـﺮﺩﻥ ﻧﻬﺎﺩﻧـﺪ؟ ﺁﻳﺎ ﻧـﺪﺍی ﻳﺎﺭی ﻃﻠﺒﯽ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﺑـﻦ ﻋﻠـﯽ ﺩﺭ ﻛﺮﺑﻼ ﺑﻪ ﺩﺭﺳـﺘﯽ ﭘﺎﺳـﺦ ﺩﺍﺩﻩ ﺷـﺪ؟ ﺁﻳـﺎ ﺯﻳﻦﺍﻟﻌﺎﺑﺪﻳﻦ، ﺑﺎﻗﺮﺍﻟﻌﻠﻮﻡ، ﺍﻣﺎﻡ ﺻﺎﺩﻕ، ﺍﻣﺎﻡ ﻛﺎﻇﻢ، ﺍﻣﺎﻡ ﺭﺿﺎ ﻭ ﺩﻳﮕﺮ ﺍﻣﺎﻣﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴﻼﻡ، ﻛﺎﻧﻮﻥ ﺗﻮﺟﻪ ﻣﺮﺩﻡ ﻋﺼﺮ ﺧﻮﺩ ﺑﻮﺩﻧﺪ؟ ﺁﻳﺎ ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﻛﻪ ﺁﻧﻬﺎ ﻇﺎﻫـﺮ ﻭ ﺁﺷـﻜﺎﺭ ﺑﻮﺩﻧﺪ ﻣـﺮﺩﻡ ﻗﺪﺭ ﺁﻧﻬـﺎ ﺭﺍ ﻣﯽ ﺩﺍﻧﺴـﺘﻨﺪ ﻭ ﺍﺯ ﮔﺮﺩ ﺁﻧﻬﺎ ﭘﺮﺍﻛﻨﺪﻩ ﻧﻤﯽ ﺷﺪﻧﺪ؟ ﻣﮕﺮ ﻣﺮﺩﻡ ﻛﻮﻓﻪ ﻧﺨﺴﺖ، ﺍﺯ ﻳﺰﻳﺪ ﺑﻴﺰﺍﺭ ﻭ ﺑﻪ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﺍﻣﻴـﺪﻭﺍﺭ ﻧﺒﻮﺩﻧﺪ؟ ﻣﮕﺮ ﺣﺴـﻴﻦ ﺑﻦ ﻋﻠﯽ ﺭﺍ ﻧﻌﻤـﺖ ﺧﺪﺍ ﺑﺮﺍی ﺧﻮﺩ ﻧﻤﯽ ﺩﺍﻧﺴـﺘﻨﺪ؟ ﻣﮕـﺮ ﺍﻭ ﺭﺍ ﺑﻪ ﺳـﺮﺯﻣﻴﻦ ﺧﻮﻳـﺶ ﻓﺮﺍﻧﺨﻮﺍﻧﺪﻧﺪ؟ ﺧﻮﺍﻧﺪﻧﺪ ﺍﻣﺎ ﺩﺭ ﺭﺍﻩ ﺑﻬﺮﻩ ﻣﻨﺪ ﺷﺪﻥ ﺍﺯ ﺍﻭ ﻣﺮﺩﺍﻧﻪ ﭘﺎﻳﺪﺍﺭی ﻧﻜﺮﺩﻧﺪ. ﺁﺭی ﺍﻣﺎﻣﺎﻥ ﭼﻪ ﻇﺎﻫﺮ ﺑﺎﺷﻨﺪ ﻭ ﭼﻪ ﻏﺎﻳﺐ ﮔﻨﺞ ﺍﻟﻬﯽ ﺍﻧﺪ ﻭ ﺑﺮﺍی ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﺳـﺎﺣﺖ ﺁﻧﺎﻥ ﺑﺎﻳﺪ ﺍﻣﺘﺤﺎﻧﺎﺗﯽ ﺭﺍ ﭘﺸﺖ ﺳـﺮ ﮔﺬﺍﺷﺖ. ﺍﻣﺎ ﭼﻪ ﺑﺴﻴﺎﺭﻧﺪ ﻛﺴﺎﻧﯽ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﺍﻣﺘﺤﺎﻥ ﻫﺎ ﺍﻳﻤﺎﻥ ﺧﻮﺩ ﺭﺍ ﻣﯽ ﺑﺎﺯﻧﺪ.

ﺍﻣﺮﻭﺯ ﻫﻢ ﺑﺮﺍی ﻣﺎ ﺩﺭ ﻣﺴـﻴﺮ ﺍﺳـﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻭ ﺍﻋﺘﻘـﺎﺩ ﺑـﻪ ﺑﺮﻗﺮﺍﺭ ﺑﻮﺩﻥ ﺍﻣﺎﻣـﺖ ﺍﻭ ﺍﻣﺘﺤﺎﻥ ﺑﺰﺭﮔﯽ ﻗـﺮﺍﺭ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺍﻣﺘﺤـﺎﻥ ﺑـﺎ ﻏﻴﺒـﺖ ﻭ ﭘﻨﻬﺎﻥ ﺯﻳﺴـﺘﯽ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺷـﻜﻞ ﮔﺮﻓﺘﻪ ﺍﺳـﺖ. ﻟﻐـﺰﺵﮔﺎﻩ ﺍﻳﻦ ﺍﻣﺘﺤﺎﻥ ﺁﻧﺠﺎﺳـﺖ ﻛﻪ ﻏﻴﺒﺖ ﺍﻭ ﺭﺍ ﻣﺎﻧـﻊ ﺍﺯ ﺑﺮﻭﺯ ﻛﻠﻴﻪ ﺁﺛﺎﺭ ﻭ ﺑﺮﻛﺎﺕ ﺍﻭ ﺑﺪﺍﻧﻴﻢ ﻭ ﺩﺭ ﻋﺼﺮ ﻏﻴﺒﺖ ﻧﺴﺒﺖ ﺑﻪ ﺗﻮﺳﻞ، ﻣﺮﺍﺟﻌﻪ ﻭ ﺗﻤﺴـﮏ ﺑﻪ ﺍﻳﺸﺎﻥ ﺗﻌﻠﻞ ﻭ ﻛﻮﺗﺎﻫﯽ ﺑﻮﺭﺯﻳﻢ. ﺑﻪ ﻋﻼﻭﻩ ﺑﭙﻨﺪﺍﺭﻳﻢ ﻛﻪ ﺍﮔﺮ ﺍﻣـﺎﻡ ﻣﺎ ﻇﺎﻫﺮ ﺑﻮﺩ ﺑﺪﻭﻥ ﺍﻳﻦ ﻛﻪ ﺍﻣﺘﺤﺎﻧﯽ ﺩﺭ ﻛﺎﺭ ﺑﺎﺷـﺪ ﺍﺯ ﺍﻭ ﺑﻬﺮﻩ ﻣﻨﺪ ﻣﯽ ﺷﺪﻳﻢ.

ﺷﺎﻫﺪ ﺑﺮ ﺑﯽ ﭘﺎﻳﻪ ﺑﻮﺩﻥ ﭘﻨﺪﺍﺭ ﻛﺴﺎﻧﯽ ﻛﻪ ﺷﺮﻁ ﻻﺯﻡ ﻭ ﻛﺎﻓﯽ ﺑﺮﺍی ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻣﺎﻡ ﺭﺍ ﺗﻨﻬﺎ ﻇﻬﻮﺭ ﺍﻭ ﻣﯽ ﺩﺍﻧﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺍﻭﻻ ﻋﻤﻮﻡ ﻣﺮﺩﻡ ﺩﺭ ﺭﻭﺯﮔﺎﺭ ﻇﻬﻮﺭ ﺍﻣﺎﻣﺎﻥ ﭘﻴﺸﻴﻦ ﺁﻥ ﻫﺎ ﺭﺍ ﻭﺍﻧﻬﺎﺩﻧﺪ ﻭ ﺍﺯ ﺍﻳﺸﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺷـﺎﻳﺎﻥ ﻧﺒﺮﺩﻧﺪ. (ﭘﺲ ﻇﺎﻫﺮ ﺑﻮﺩﻥ ﺍﻣﺎﻡ، ﺷـﺮﻁ ﻛﺎﻓﯽ ﺑﺮﺍی ﺍﺳـﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻭ ﻧﻴﺴـﺖ.) ﺛﺎﻧﻴﺎ ﺑﻪ ﺷـﻬﺎﺩﺕ ﺗﺎﺭﻳﺦ، ﺷـﻴﻌﻴﺎﻧﯽ ﻛﻪ ﺩﺭ ﻫﻤﻴﻦ ﺭﻭﺯﮔﺎﺭ ﻏﻴﺒﺖ، ﺗﻮﺳﻼﺕ ﻋﻤﻴﻖ ﻭ ﺟﺪی ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺩﺍﺷﺘﻪ ﺍﻧﺪ ﺍﺯ ﺍﻳﺸﺎﻥ ﺑﻬﺮﻩ ﺑﺮﺩﻩ ﻭ ﻣﯽ ﺑﺮﻧﺪ. (ﭘﺲ ﻇﺎﻫﺮ ﺑﻮﺩﻥ ﺍﻣﺎﻡ، ﺷـﺮﻁ ﻻﺯﻡ ﺑﺮﺍی ﺍﺳـﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻭ ﻧﻴﺴﺖ.) ﺧﻼﺻﻪ ﺁﻧﻜﻪ ﻏﻴﺒﺖ، ﺩﻟﻴﻞ ﻣﻮﺟﻬﯽ ﺑﺮﺍی ﺍﺳﺘﻔﺎﺩﻩ ﻧﻜﺮﺩﻥ ﻣﺎ ﺍﺯ ﺍﻣﺎﻡ ﻧﻴﺴـﺖ، ﭼﻨﺎﻧﻜﻪ ﻇﻬﻮﺭ ﻧﻴﺰ ﻋﺎﻣﻠﯽ ﻗﻬﺮی ﺑﺮﺍی ﻛﺴﺐ ﻓﻴﺾ ﺍﺯ ﺍﻣﺎﻡ ﻧﻤﯽ ﺑﺎﺷﺪ.

ﺁﺭی ﻗﺼـﻪ «ﺁﻥ ﭼـﺎﻩ ﭘـﺮ ﺍﺯ ﺁﺏ ﻛـﻪ ﻭﺍﻧﻬﺎﺩﻩ ﺷـﺪﻩ »، ﻗﺼﻪ ﻏﺮﺑﺖ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻣﺎﺳـﺖ ﻭ ﻗﺼﻪ ﻫﻤﺎﻥ ﻣﺮﺩ ﺩﺍﻧﺎﺳـﺖ ﻛﻪ ﻧﺎﺩﺍﻧـﺎﻥ ﻗﺪﺭ ﺍﻭ ﺭﺍ ﻧﻤﯽ ﺩﺍﻧﻨﺪ ﻭ ﻧﻴﺎﺯﻣﻨﺪﺍﻥ ، ﺍﺯ ﺩﺍﻧﺶ ﺍﻭ ﺳﻮﺩ ﻧﻤﯽ ﺟﻮﻳﻨﺪ.

ﺭﻭ ﻛﻨﻴﻢ ﺑﻪ ﻛﻼﻡ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻛﻪ ﺩﺭ ﺗﻔﺴﻴﺮ ﻭ ﺭﻣﺰﮔﺸﺎﻳﯽ ﺍﺯ ﻛﻼﻡ ﺣﻖ ﺩﺭﺑﺎﺭﻩ ﺁﻥ «ﭼﺎﻩ ﻫﺎی ﻓﺮﻭﻧﻬﺎﺩﻩ ﺷـﺪﻩ » ﻓﺮﻣﻮﺩﻩ ﺍﺳﺖ: «ﻣﻌﻨﺎی ﺁﻳﻪ ﭼﻨﻴﻦ ﺍﺳـﺖ: ﭼﻪ ﺑﺴﻴﺎﺭ ﻋﺎﻟﻤﯽ ﻛﻪ ﻛﺴـﯽ ﺑﻪ ﺍﻭ ﻣﺮﺍﺟﻌﻪ ﻧﻤﯽ ﻛﻨﺪ ﻭ ﺍﺯ ﻋﻠﻢ ﺍﻭ ﺑﻬﺮﻩ ﻧﻤﯽ ﺑﺮﺩ.»

ﺑـﺎﺯ ﻫﻢ ﺗﻜﺮﺍﺭ ﻣﯽ ﻛﻨﻢ: «ﺑﻴﺎﻳﻴﺪ ﺑـﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻧﻤﺎﻥ ﻣﺮﺍﺟﻌﻪ ﻛﻨﻴﻢ ﻭ ﺍﺯ ﻋﻠﻢ، ﻫﺪﺍﻳﺖ ﻭ ﺩﻳﮕﺮ ﺑﺮﻛﺎﺕ ﺍﻭ ﺑﻬﺮﻩ ﺑﺒﺮﻳﻢ.»

ﻭ ﺑـﺎﺯ ﻣﯽ ﮔﻮﻳـﻢ: «ﻣﺮﺍﺟﻌـﻪ ﻣـﺎ ﺑﻪ ﺍﻳﺸـﺎﻥ ﺩﻭ ﻓﺎﻳﺪﻩ ﺩﺭ ﭘـﯽ ﺩﺍﺭﺩ. ﺍﻭﻝ ﺳـﻌﺎﺩﺕ ﺑﺮﺍی ﻣﺎ ﻭ ﺩﻭﻡ ﺭﻓﻊ ﻏﺮﺑﺖ ﺍﺯ ﺍﻭ، ﻛﻪ ﻧﺰﺩ ﺍﻫﻞ ﻣﺤﺒﺖ ﻭ ﻭﻻﻳﺖ ﻓﺎﻳﺪﻩ ﺩﻭﻡ ﻣﻘﺪﻡ ﺑﺮ ﻓﺎﻳﺪﻩ ﻧﺨﺴﺖ ﺍﺳﺖ.»

ﻭ ﻣﯽ ﮔﻮﻳـﻢ: «ﻣﺮﺍﺟﻌﻪ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻣﻮﻛﻮﻝ ﺑﻪ ﺩﻳﺪﺍﺭ ﺣﻀﻮﺭی ﻭ ﻣﻼﻗﺎﺕ ﺟﺴـﻢ ﺷـﺮﻳﻒ ﺍﻳﺸـﺎﻥ ﻧﻴﺴـﺖ. ﺗﻤﺴﮏ ﻭ ﺗﻮﺳﻞ ﺑﻪ ﺍﻣﺎﻡ، ﻧﻴﺎﺯﻣﻨـﺪ ﭘﺎﻳـﺎﻥ ﻳﺎﻓﺘﻦ ﻏﻴﺒﺖ ﻧﻴﺴـﺖ، ﺑﻠﻜﻪ ﺑﻪ ﻓﺮﻣـﻮﺩﻩ ﭘﻴﺎﻣﺒﺮ ﺩﺭ ﻋﺼﺮ ﻏﻴﺒﺖ ﻧﻴﺰ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﻧﻮﺭ ﻭﻻﻳﺖ ﺁﻥ ﺣﻀﺮﺕ ﭘﺮﺗﻮ ﮔﺮﻓﺖ ﻭ ﺑﻬﺮﻩ ﻣﻨﺪ ﺷﺪ.»

ﻭ ﺑـﺎ ﺗـﻮ ﻣﯽ ﮔﻮﻳﻢ: «ﺍی ﺷـﻴﻌﻪ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ! ﺩﺭ ﺗﻤـﺎﻡ ﺗﺼﻤﻴﻤﺎﺕ ﺯﻧﺪﮔـﯽ ﺍﺕ ﻭ ﺑـﻪ ﻫـﺮ ﻛﺎﺭی ﻛـﻪ ﻣﯽ ﺧﻮﺍﻫﯽ ﺍﻗـﺪﺍﻡ ﻛﻨﯽ ﺑﺎ ﺩﻟـﺖ ﻭ ﺑﺎ ﺯﺑﺎﻧـﺖ، ﺧـﻮﺩ ﺭﺍ ﺑﻪ ﺁﻥ ﺍﺑﺮﻣـﺮﺩ ﻣﻠﻜﻮﺗﯽ ﭘﻴﻮﻧﺪ ﺑـﺰﻥ ﻭ ﺑﮕﻮ ﻳﺎ ﺻﺎﺣﺐ ﺍﻟﺰﻣـﺎﻥ ! ﻣـﺮﺍ ﺩﺭﻳﺎﺏ! ﺗﺎ ﺩﺭ ﺍﻳـﻦ ﺗﺼﻤﻴﻢ ﻭ ﺩﺭ ﺍﻳﻦ ﻛﺎﺭ ﻣﻮﻓﻖ ﺑﺎﺷـﻢ ﻭ ﻣﻮﺭﺩ ﻋﻨﺎﻳﺖ ﺣﻖ ﻣﺘﻌﺎﻝ ﻗﺮﺍﺭ ﮔﻴﺮﻡ.»

ﻭ ﻣﯽ ﮔﻮﻳـﻢ: «ﺑﻬﺮﻩ ﺑﺮﺩﻥ ﺍﺯ ﺍﻳﻦ ﻣﻌﺪﻥ ﭘﺮﻣﺎﻳﻪ ﺑﯽ ﺭﻧﺞ ﻭ ﺳـﺨﺘﯽ ﻫـﻢ ﻧﻴﺴـﺖ، ﭘﺲ ﺩﺭ ﺭﺍﻩ ﺭﺳـﻴﺪﻥ ﺑـﻪ ﺍﻳﻦ ﮔﻨـﺞ ﺍﺯ ﺗـﻼﺵ ﻭ ﭘﺎﻳﺪﺍﺭی ﻣﭙﺮﻫﻴﺰ.»(17)

ﺁﺭی ، ﻫﺮﭼـﻪ ﻣﺮﺍﺟﻌـﺎﺕ ﻣـﺎ ﺷـﻴﻌﻴﺎﻥ ﺑﻪ ﺍﻳـﻦ ﻣﻨﺒﻊ ﻧـﻮﺭ ﻭ ﻣﻌﺪﻥ ﺧﻴﺮ، ﺑﻴﺸﺘﺮ ﻭ ﺑﻴﺸﺘﺮ ﺷﻮﺩ ﻭ ﻫﺮﭼﻪ ﺁﺛﺎﺭ ﻭ ﺑﺮﻛﺎﺕ ﻣﺸﺎﻫﺪﻩ ﺷﺪﻩ ﺍﺯ ﺍﻳﻦ ﻣـﻮﻻی ﻛﺮﻳﻢ ﺯﻳﺎﺩﺗﺮ ﮔﺮﺩﺩ، ﺷـﻮﻕ ﻭ ﺭﻏﺒﺖ ﺍﻣﺖ ﺑـﻪ ﺍﻭ ﺭﻭ ﺑﻪ ﻓﺰﻭﻧﯽ ﺧﻮﺍﻫﺪ ﮔﺬﺍﺷﺖ ﻭ ﻫﻤﭙﺎی ﺭﻓﻊ ﻣﺤﺮﻭﻣﻴﺖ ﻣﺎ ﺍﺯ ﺁﻥ ﺣﻀﺮﺕ، ﻏﺮﺑﺖ ﺍﻳﺸـﺎﻥ ﻧﻴﺰ ﺩﺭ ﻣﻴﺎﻥ ﻣﺎ ﻛﻢ ﺧﻮﺍﻫﺪ ﮔﺸـﺖ. ﺍی ﻛﺎﺵ ﺑﻪ ﭼﻨﺎﻥ ﺭﺷـﺪی ﻧﺎﺋـﻞ ﮔﺮﺩﻳﻢ ﻛـﻪ ﺍﺯ ﻣﺤﺮﻭﻣﻴﺖ ﺧﻮﺩﻣﺎﻥ ﺩﻝﺁﺯﺭﺩﻩ ﺑﺎﺷـﻴﻢ ﻭ ﺍﺯ ﻏﺮﺑﺖ ﺍﻣﺎﻣﻤﺎﻥ ﺩﻝﺁﺯﺭﺩﻩ ﺗﺮ!
