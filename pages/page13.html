بازگشت

ﺑﺎ ﭘﺮﻫﻴﺰﻛﺎﺭی ﺑﻪ ﻳﺎﺭی ﺍﻣﺎﻡ ﻣﻬﺪی ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﺸﺘﺎﺑﻴﻢ.

«ﻣﺮﺣﻮﻡ ﺁﻳﺖﺍﻟﻠﻪ ﺳـﻴﺪ ﺑﺎﻗﺮ ﻣﺠﺘﻬﺪ ﺳﻴﺴـﺘﺎﻧﯽ ﭘﺪﺭ ﺁﻳﺖﺍﻟﻠﻪ ﺍﻟﻌﻈﻤﯽ ﺳـﻴﺪﻋﻠﯽ ﺳﻴﺴﺘﺎﻧﯽ، ﺑﺮﺍی ﺗﺸـﺮﻑ ﺑﻪ ﻣﺤﻀﺮ ﻣﺒﺎﺭک ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺧﺘﻢ ﺯﻳﺎﺭﺕ ﻋﺎﺷﻮﺭﺍ ﺭﺍ ﺩﺭ ﭼﻬﻞ ﺟﻤﻌﻪ ﻭ ﭼﻬﻞ ﻣﺴﺠﺪ ﺁﻏﺎﺯ ﻣﯽ ﻛﻨـﺪ. ﺩﺭ ﻳﻜﯽ ﺍﺯ ﺭﻭﺯﻫﺎی ﺁﺧﺮ ﻛﻪ ﻣﺸـﻐﻮﻝ ﺯﻳﺎﺭﺕ ﺑﻮﺩﻩ، ﻧﺎﮔﻬﺎﻥ ﻧـﻮﺭی ﺍﺯ ﺧﺎﻧﻪ ﺍی ﻧﺰﺩﻳﮏ ﺁﻥ ﻣﺴـﺠﺪ ﻧﻈـﺮ ﺍﻭ ﺭﺍ ﺟﻠـﺐ ﻣﯽ ﻛﻨﺪ. ﻧﻮﺭ ﺍﺯ ﺩﺭﻭﻥ ﺁﻥ ﺧﺎﻧـﻪ ﺗﻸﻟـﺆ ﺩﺍﺷـﺖ. ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﻥ ﺣﺮﻛـﺖ ﻣﯽ ﻛﻨﺪ. ﺧﺎﻧﻪ ﻛﻮﭼـﮏ ﻭ ﻓﻘﻴﺮﺍﻧـﻪ ﺍی ﺑـﻮﺩﻩ ﺍﺳـﺖ. ﺩﺭ ﺭﺍ ﻣﯽ ﻛﻮﺑﺪ ﻭ ﺑﺎ ﺍﺟـﺎﺯﻩ ﻭﺍﺭﺩ ﻣﯽ ﺷـﻮﺩ. ﺩﺭ ﻳﻜـﯽ ﺍﺯ ﺍﻃﺎﻕﻫـﺎ ﺍﻣﺎﻡ ﻋﺼـﺮ ﻛﻪ ﺧﺪﺍﻭﻧـﺪ ﻓﺮﺟﺶ ﺭﺍ ﻧﺰﺩﻳﮏ ﺳـﺎﺯﺩ ﺭﺍ ﻣﺸـﺎﻫﺪﻩ ﻣﯽ ﻛﻨﺪ. ﺧﺪﻣﺖ ﻣﯽ ﺭﺳﺪ ﻭ ﺍﺷﮏﺭﻳﺰﺍﻥ ﻋﺮﺽ ﺳـﻼﻡ ﻣﯽ ﻛﻨـﺪ. ﻛﻨـﺎﺭ ﺍﺗﺎﻕ ﺟﻨـﺎﺯﻩ ﺍی ﺭﺍ ﻛـﻪ ﭘﺎﺭﭼﻪ ﺳـﻔﻴﺪی ﺭﻭی ﺁﻥ ﻛﺸـﻴﺪﻩ ﺑﻮﺩﻧـﺪ، ﻣﯽ ﺑﻴﻨـﺪ. ﺁﻥ ﺣﻀﺮﺕ ﺑﻌﺪ ﺍﺯ ﺟﻮﺍﺏ ﺳـﻼﻡ ﻣﯽ ﻓﺮﻣﺎﻳﻨـﺪ: «ﭼـﺮﺍ ﺍﻳﻨﮕﻮﻧﻪ ﺩﻧﺒﺎﻝ ﻣـﻦ ﻣﯽ ﮔﺮﺩی ﻭ ﭼﻨﻴـﻦ ﺭﻧﺞﻫﺎﻳﯽ ﺭﺍ ﺑـﺮ ﺧـﻮﺩ ﻭﺍﺭﺩ ﻣﯽ ﺳـﺎﺯی؟! ﻣﺜﻞ ﺍﻳﻦ ﺑﺎﺷـﻴﺪ (ﻭ ﺍﺷـﺎﺭﻩ ﺑـﻪ ﺟﻨﺎﺯﻩ ﻣﯽ ﻓﺮﻣﺎﻳﻨﺪ) ﺗﺎ ﻣﻦ ﺩﻧﺒﺎﻝ ﺷﻤﺎ ﺑﻴﺎﻳﻢ.» ﺁﻥ ﺣﻀﺮﺕ ﺩﺭ ﺍﺩﺍﻣﻪ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﺍﻳﻦ ﺑﺎﻧﻮ ﺩﺭ ﺩﻭﺭﺍﻥ ﺑﯽ ﺣﺠﺎﺑﯽ (ﺯﻣﺎﻥ ﺭﺿﺎ ﺷـﺎﻩ ﭘﻬﻠﻮی) ﻫﻔﺖ ﺳﺎﻝ ﺍﺯ ﺧﺎﻧﻪ ﺑﻴﺮﻭﻥ ﻧﻴﺎﻣﺪ، ﻣﺒﺎﺩﺍ ﻣﺮﺩ ﻧﺎﻣﺤﺮﻡ ﺍﻭ ﺭﺍ ﺑﺒﻴﻨﺪ.»(37)

ﺗﺄﻣـﻞ ﺩﺭ ﺍﻳـﻦ ﮔﻮﻧﻪ ﺣﻜﺎﻳـﺎﺕ ﻣﺎ ﺭﺍ ﺑـﻪ ﻓﻜﺮ ﻓﺮﻭ ﻣﯽ ﺑـﺮﺩ ﻛﻪ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﻣﺎ ﺭﺍ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﻭﺭ ﺳـﺎﺧﺘﻪ ﺍﺳـﺖ؟ ﻭ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﻣﺎ ﺭﺍ ﺑﻪ ﺍﻳﺸﺎﻥ ﻧﺰﺩﻳﮏ ﻣﯽ ﻛﻨﺪ؟

ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﻧﺎﻣﻪ ﺍی ﻛـﻪ ﺑﻪ ﻋﺎﻟﻢ ﮔﺮﺍﻧﻘﺪﺭ ﺷـﻴﻌﻪ ، ﻣﺮﺣـﻮﻡ ﺷـﻴﺦ ﻣﻔﻴﺪ ﻧﻮﺷـﺘﻪ ﺍﻧﺪ، ﻣﯽ ﻓﺮﻣﺎﻳﻨﺪ: «ﺗﻨﻬﺎ ﭼﻴﺰی ﻛـﻪ ﻣﺎ ﺭﺍ ﺍﺯ ﺁﻧﺎﻥ (ﻳﻌﻨﯽ ﺷـﻴﻌﻴﺎﻧﻤﺎﻥ ) ﭘﻮﺷـﻴﺪﻩ ﻣﯽ ﺩﺍﺭﺩ، ﻛﺎﺭﻫﺎی ﻧﺎﭘﺴـﻨﺪی ﺍﺳﺖ ﻛﻪ ﺍﺯ ﺍﻳﺸـﺎﻥ ﺑﻪ ﻣﺎ ﻣﯽ ﺭﺳـﺪ ﻭ ﺁﻥ ﻛﺎﺭﻫﺎ ﺧﻮﺷـﺎﻳﻨﺪ ﻣﺎ ﻧﻴﺴﺖ ﻭ ﺍﺯ ﺁﻧﺎﻥ ﺍﻧﺘﻈﺎﺭ ﻧﻤﯽ ﺭﻭﺩ. ﻭ ﺍﻟﻠﻪ ﺍﻟﻤﺴﺘﻌﺎﻥ ﻭ ﻫﻮ ﺣﺴﺒﻨﺎ ﻭ ﻧﻌﻢ ﺍﻟﻮﻛﻴﻞ»(38)

ﺁﻥ ﺣﻀـﺮﺕ ﺩﺭ ﺟـﺎی ﺩﻳﮕﺮی ﺍﺯ ﻫﻤﻴـﻦ ﻧﺎﻣﻪ ﻓﺮﻣﻮﺩﻩ ﺍﻧـﺪ: «ﺍﻣﺎ ﺍﮔـﺮ ﺷـﻴﻌﻴﺎﻥ ﻣﺎ ﻛـﻪ ﺧﺪﺍﻭﻧﺪ ﺁﻧـﺎﻥ ﺭﺍ ﺩﺭ ﺍﻃﺎﻋﺘﺶ ﻳـﺎﺭی ﺩﻫﺪ ﺩﺭ ﻭﻓـﺎی ﺑﻪ ﭘﻴﻤﺎﻧﯽ ﻛﻪ ﺍﺯ ﺍﻳﺸـﺎﻥ ﮔﺮﻓﺘﻪ ﺷـﺪﻩ، ﻳﻜﺪﻝ ﻭ ﻣﺼﻤﻢ ﺑﺎﺷـﻨﺪ، ﻧﻌﻤﺖ ﻟﻘﺎی ﻣﺎ ﺍﺯ ﺁﻧﺎﻥ ﺑﻪ ﺗﺄﺧﻴﺮ ﻧﻴﻔﺘﺪ، ﻭ ﺳﻌﺎﺩﺕ ﺩﻳﺪﺍﺭ ﻣﺎ ﺑﺮﺍی ﺁﻧﺎﻥ ﻫﻤﺮﺍﻩ ﺑﺎ ﻣﻌﺮﻓﺘﯽ ﻛﺎﻣﻞ ﻭ ﺩﺭﺳـﺖ ﻧﺴـﺒﺖ ﺑﻪ ﻣﺎ ﺑﻪ ﺳـﺮﻋﺖ ﻧﺼﻴﺒﺸﺎﻥ ﮔﺮﺩﺩ.»(39)

ﻣﺮﺣـﻮﻡ ﻋﻠﯽ ﺑﻦ ﻣﻬﺰﻳﺎﺭ ﺍﻫﻮﺍﺯی ﺍﺯ ﺑﺰﺭﮔﺎﻥ ﺷـﻴﻌﻪ ﻭ ﺷـﻴﻔﺘﮕﺎﻥ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻮﺩ. ﺍﻭ ﺳـﻔﺮﻫﺎی ﻣﺘﻌﺪﺩی ﺑـﻪ ﺣﺞ ﺧﺎﻧﻪ ی ﺧﺪﺍ ﻣﺸـﺮﻑ ﺷﺪ ﺑﻪ ﺍﻳﻦ ﺍﻣﻴﺪ ﻛﻪ ﺩﺭ ﺁﻥ ﻣﺮﺍﺳﻢ ﺭﻭﺣﺎﻧﯽ، ﺷﺮﻑ ﺩﻳﺪﺍﺭ ﺣﺠـﺖ ﺧـﺪﺍ ﻳﻌﻨﯽ ﺣﻀـﺮﺕ ﻭﻟـﯽ ﻋﺼﺮ ﻛـﻪ ﺟﺎﻥ ﻫﺎﻳﻤـﺎﻥ ﻓﺪﺍﻳﺶ ﺑﺎﺩ ﻧﺼﻴﺒﺶ ﺷـﻮﺩ. ﺩﺭ ﺳـﻔﺮ ﺑﻴﺴـﺘﻢ ﺑﻮﺩ ﻛﻪ ﺩﺭ ﻛﻨﺎﺭ ﺧﺎﻧﻪ ﺧﺪﺍ ﻛﺴﯽ ﺍﺯ ﺟﺎﻧﺐ ﺍﻣﺎﻡ ﻋﺼﺮ ﻧﺰﺩ ﺍﻭ ﺁﻣﺪ ﺗﺎ ﺍﻭ ﺭﺍ ﺑﻪ ﻛﻌﺒﻪ ﺁﺭﺯﻭﻫﺎﻳﺶ ﺭﺍﻫﻨﻤﺎﻳﯽ ﻛﻨﺪ. ﻧﻤﺎﻳﻨﺪﻩ ﺍﻣﺎﻡ ﺑﻪ ﻫﻨﮕﺎﻡ ﻣﻼﻗﺎﺕ ﺑﺎ ﻋﻠﯽ ﺑﻦ ﻣﻬﺰﻳﺎﺭ ﺍﺯ ﺍﻭ ﭘﺮﺳـﻴﺪ: ﺗـﻮ ﺑﻪ ﺩﻧﺒﺎﻝ ﻛـﻪ ﻣﯽ ﮔـﺮﺩی؟ ﻓﺮﺯﻧﺪ ﻣﻬﺰﻳـﺎﺭ ﻋﺮﺽ ﻛﺮﺩ: ﻣـﻦ ﺩﺭ ﭘﯽ ﺍﻣﺎﻣﯽ ﻫﺴﺘﻢ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻋﺎﻟﻢ ﭘﻮﺷﺎﻧﺪﻩ ﺍﺳﺖ.

ﻧﻤﺎﻳﻨﺪﻩ ﺍﻣﺎﻡ ﮔﻔﺖ: ﻧﻪ ! (ﺍﻳﻦ ﺗﻌﺒﻴﺮ ﺗﻮ ﺩﺭﺳﺖ ﻧﻴﺴﺖ) ﺍﻭ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻋﺎﻟﻢ ﻧﭙﻮﺷـﺎﻧﺪﻩ ﺍﺳﺖ، ﺑﻠﻜﻪ ﺍﻳﻦ ﺭﻓﺘﺎﺭﻫﺎی ﺯﺷﺖ ﺷﻤﺎﺳﺖ ﻛﻪ ﺍﻭ ﺭﺍ ﺍﺯ ﺷﻤﺎ ﭘﻨﻬﺎﻥ ﻛﺮﺩﻩ ﺍﺳﺖ.

ﺗﺄﻣـﻞ ﺩﺭ ﻛﻼﻡ ﺍﻳـﻦ ﺑﺰﺭﮔـﻮﺍﺭ ﻣـﺎ ﺭﺍ ﺑـﻪ ﺧﺎﻃـﺮﺍﺕ ﺧﻮﺩﻣـﺎﻥ ﺩﺭ ﺯﻧﺪﮔـﯽ ﺑـﺎﺯ ﻣﯽ ﮔﺮﺩﺍﻧـﺪ ﻭ ﺗﺼﻮﻳـﺮی ﺍﺯ ﺭﻓﺘﺎﺭﻫﺎی ﺯﺷـﺘﻤﺎﻥ ﺭﺍ ﭘﻴﺶ ﭼﺸﻤﻤﺎﻥ ﻣﯽ ﺁﻭﺭﺩ. ﺟﺎی ﺩﺍﺭﺩ ﻛﻪ ﺍﺯ ﺧﺪﺍ ﻃﻠﺐ ﻣﻐﻔﺮﺕ ﻛﻨﻴﻢ ﻭ ﺍﺯ ﺍﻭ ﺑﺨﻮﺍﻫﻴﻢ ﻛﻪ ﺑﻪ ﺣﺮﻣﺖ ﻭﻟﯽ ﺍﺵ ﺁﻣﺮﺯﺵ ﺧﻮﺩ ﺭﺍ ﻧﺼﻴﺐ ﻣﺎ ﮔﺮﺩﺍﻧﺪ ﻭ ﺗﻮﻓﻴﻖ ﻃﺎﻋﺖ ﻭ ﺍﺟﺘﻨﺎﺏ ﺍﺯ ﻣﻌﺼﻴﺖ ﺑﻪ ﻣﺎ ﻋﻨﺎﻳﺖ ﻓﺮﻣﺎﻳﺪ.

ﭼﻨـﺪ ﺳـﺎﻋﺖ ﺑﻌـﺪ ﺍﺯ ﻣﻼﻗـﺎﺕ ﺍﺑﻦﻣﻬﺰﻳﺎﺭ ﺑـﺎ ﻧﻤﺎﻳﻨﺪﻩ ﺍﻣـﺎﻡ، ﺍﻭ ﺗﻮﻓﻴﻖ ﻳﺎﻓﺖ ﻛﻪ ﺑﻪ ﻣﺤﻀﺮ ﺁﻥ ﺳـﺮﻭﺭ ﻋﺰﻳﺰ ﻭ ﻣﻮﻻی ﮔﻢﺷـﺪﻩ ﺍﺵ ﻛﻪ ﺳﺎﻝﻫﺎ ﺑﻮﺩ ﺩﺭ ﭘﯽ ﺍﺵ ﻣﯽ ﮔﺸﺖ، ﺑﺎﺭ ﻳﺎﺑﺪ.

ﺷـﺎﻳﺪ ﺍﺑﻦﻣﻬﺰﻳﺎﺭ ﻣﯽ ﺧﻮﺍﺳـﺖ ﻟﺐ ﺑﻪ ﺷـﻜﺎﻳﺖ ﺑﺎﺯ ﻛﻨـﺪ ﻛﻪ ﺍی ﻣـﻮﻻی ﻣﻦ ﭼﺮﺍ ﺍﻳﻨﻘﺪﺭ ﺩﻳﺮ ﻣﺮﺍ ﺑﻪ ﻣﺤﻀـﺮ ﺧﻮﻳﺶ ﺭﺍﻩ ﺩﺍﺩی ﺑﺎ ﺍﻳﻨﻜﻪ ﺳﺎﻟﻴﺎﻥ ﺳﺎﻝ ﺍﺳﺖ ﺍﺯ ﺗﻮ ﺗﻤﻨﺎ ﺩﺍﺭﻡ ﻛﻪ ﺍﻳﻦ ﺷﻴﻌﻪ ﺩﻟﺴﻮﺧﺘﻪ ﺍﺕ ﺭﺍ ﺑﻪ ﻧﺰﺩ ﺧﻮﻳـﺶ ﻓﺮﺍﺧﻮﺍﻧﯽ. ﺍﻣﺎ ﻫﺮﭼـﻪ ﺑﻮﺩ ﺍﺑﻦﻣﻬﺰﻳﺎﺭ ﻓﺮﺻﺖ ﺍﻳﻦ ﺷـﻜﺎﻳﺖ ﺭﺍ ﻧﻴﺎﻓﺖ، ﺯﻳﺮﺍ ﺁﻥ ﺣﻀﺮﺕ ﭘﻴﺶﺩﺳﺘﯽ ﻓﺮﻣﻮﺩ، ﻟﺐ ﺑﻪ ﺳﺨﻦ ﮔﺸﻮﺩ ﻭ ﺭﺍﺯ ﺩﻝ ﻏﻤﮕﻴﻨﺶ ﺭﺍ ﺍﻳﻦ ﭼﻨﻴﻦ ﺑﺎﺯﮔﻮ ﻧﻤﻮﺩ: ﺍی ﺍﺑﻮﺍﻟﺤﺴﻦ! ﻣﺎ ﺷﺐ ﻭ ﺭﻭﺯ ﻣﻨﺘﻈﺮ ﻭﺭﻭﺩ ﺗﻮ ﺑﻮﺩﻳﻢ، ﭼﺮﺍ ﺍﻳﻦ ﻗﺪﺭ ﺩﻳﺮ ﻧﺰﺩ ﻣﺎ ﺁﻣﺪی؟

ﻓﺮﺯﻧـﺪ ﻣﻬﺰﻳـﺎﺭ ﻋﺮﺽ ﻛﺮﺩ: ﺁﻗﺎی ﻣـﻦ! ﺗﺎ ﻛﻨﻮﻥ ﻛﺴـﯽ ﺭﺍ ﻧﻴﺎﻓﺘﻪ ﺑﻮﺩﻡ ﻛﻪ ﺩﻟﻴﻞ ﻭ ﺭﺍﻫﻨﻤﺎی ﻣﻦ ﺑﻪ ﺳﻮی ﺷﻤﺎ ﺑﺎﺷﺪ.

ﺍﻣـﺎﻡ ﻓﺮﻣﻮﺩ: «ﺁﻳﺎ ﻛﺴـﯽ ﺭﺍ ﻧﻴﺎﻓﺘﯽ ﻛﻪ ﺗﻮ ﺭﺍ ﺩﻻﻟـﺖ ﻛﻨﺪ؟!» ﺑﻌﺪ ﺍﻧﮕﺸـﺖ ﻣﺒﺎﺭک ﺭﺍ ﺑﻪ ﺯﻣﻴﻦ ﻛﺸـﻴﺪ ﻭ ﻓﺮﻣﻮﺩ: «ﻧﻪ ، ﻟﻜﻦ ﺷﻤﺎ ﺍﻣﻮﺍﻟﺘﺎﻥ ﺭﺍ ﻓﺰﻭﻧـﯽ ﺑﺨﺸـﻴﺪﻳﺪ ﻭ ﺑـﺮ ﺑﻴﻨﻮﺍﻳـﺎﻥ ﻣﺆﻣﻨﻴﻦ ﺳـﺨﺖ ﮔﺮﻓﺘـﻪ ، ﺁﻧﺎﻥ ﺭﺍ ﺳـﺮﮔﺮﺩﺍﻥ ﻭ ﺑﯽ ﭼـﺎﺭﻩ ﻛﺮﺩﻳﺪ ﻭ ﺭﺍﺑﻄـﻪ ﺧﻮﻳﺸـﺎﻭﻧﺪی ﺭﺍ ﺩﺭ ﺑﻴﻦ ﺧﻮﺩ ﺍﺯ ﺑﻴـﻦ ﺑﺮﺩﻳـﺪ (ﺻﻠـﻪ ﺭﺣـﻢ ﺍﻧﺠـﺎﻡ ﻧﺪﺍﺩﻳﺪ)، ﺩﻳﮕﺮ ﺷـﻤﺎ ﭼـﻪ ﻋﺬﺭی ﺩﺍﺭﻳﺪ؟!»

ﺩﺭ ﺍﻳـﻦ ﻫﻨـﮕﺎﻡ ﺗﻨﻬﺎ ﭼﻴﺰی ﻛﻪ ﭘﺴـﺮ ﻣﻬﺰﻳﺎﺭ ﻣﯽ ﺗﻮﺍﻧﺴـﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻧـﺶ ﻋﺮﺽ ﻛﻨﺪ ﺍﻳـﻦ ﻛﻠﻤﺎﺕ ﺑﻮﺩ: ﺗﻮﺑﻪ ! ﺗﻮﺑـﻪ ! ﻋﺬﺭ ﻣﯽ ﺧﻮﺍﻫﻢ! ﺑﺒﺨﺸﻴﺪ! ﻧﺎﺩﻳﺪﻩ ﺑﮕﻴﺮﻳﺪ!

ﺳـﭙﺲ ﺍﻣﺎﻡ ﻓﺮﻣﻮﺩﻧﺪ: «ﺍی ﭘﺴـﺮ ﻣﻬﺰﻳﺎﺭ! ﺍﮔﺮ ﻧﺒﻮﺩ ﻛﻪ ﺑﻌﻀﯽ ﺍﺯ ﺷـﻤﺎ ﺷﻴﻌﻴﺎﻥ ﺑﺮﺍی ﺑﻌﻀﯽ ﺩﻳﮕﺮ ﺍﺳﺘﻐﻔﺎﺭ ﻣﯽ ﻛﻨﻴﺪ، ﺗﻤﺎﻡ ﻛﺴﺎﻧﯽ ﻛﻪ ﺑﺮ ﺭﻭی ﺯﻣﻴﻦ ﻫﺴﺘﻨﺪ ﻧﺎﺑﻮﺩ ﻣﯽ ﺷﺪﻧﺪ، ﺑﻪ ﺟﺰ ﺧﺎﺻﺎﻥ ﺷﻴﻌﻪ ؛ ﻫﻤﺎﻥ ﻫﺎﻳﯽ ﻛﻪ ﮔﻔﺘﺎﺭﺷﺎﻥ ﺑﺎ ﻛﺮﺩﺍﺭﺷﺎﻥ ﻫﻤﺎﻧﻨﺪ ﺍﺳﺖ.»(40)

ﻫﻤـﻪ ﻣـﺎ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴـﻢ ﺩﺭ ﺣﻜﻢ ﻓﺮﺯﻧـﺪﺍﻥ ﺍﻣﺎﻡ ﺯﻣﺎﻧﻤﺎﻥ ﻫﺴـﺘﻴﻢ. ﻫﻤﻪ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﻛﻪ ﺍﻭ ﻣﺎ ﺭﺍ ﺩﻭﺳـﺖ ﺩﺍﺭﺩ ﻭ ﻫﻤﻪ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﻛﻪ ﺍﻭ ﺍﺯ ﭘﻴﺸـﺮﻓﺖ ﻣﺎ ﺧﻮﺷﺤﺎﻝ ﻭ ﺍﺯ ﺳـﻘﻮﻁ ﻣﺎ ﺩﻝﺍﻓﻜﺎﺭ ﻣﯽ ﮔﺮﺩﺩ. ﭘﻴﺸﺮﻓﺖ ﺩﻳﻨﯽ ﻣﺎ ﺍﺯ ﻳﮏﺳـﻮ ﺯﻣﻴﻨﻪ ﺳـﻌﺎﺩﺕ ﺧﻮﺩ ﻣﺎﺳﺖ ﻭ ﺍﺯ ﺳﻮی ﺩﻳﮕﺮ ﻣﺎﻳﻪ ﺧﻮﺷﺤﺎﻟﯽ ﻭ ﻗﻮﺕ ﻗﻠﺐ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺍﺳﺖ.

ﺑﻨﺎﺑﺮﺍﻳـﻦ ﺍی ﺷـﻴﻌﻪ ﺍی ﻛـﻪ ﺩﻭﺳـﺖ ﺩﺍﺭی ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺧﻮﻳﺶ ﺭﺍ ﻳﺎﺭی ﻛﻨﯽ، ﺑﺎ ﻭﺭﻉ (ﻳﻌﻨﯽ ﺧﻮﻳﺸـﺘﻨﺪﺍﺭی ﺩﺭ ﺑﺮﺍﺑﺮ ﻭﺳﻮﺳـﻪ ﮔﻨﺎﻫﺎﻥ ) ﻭ ﺍﺟﺘﻬـﺎﺩ (ﻳﻌﻨـﯽ ﺗـﻼﺵ ﺩﺭ ﺟﻬﺖ ﻛﺴـﺐ ﺍﻋﻤﺎﻝ ﻧﻴﻜﻮ) ﺑـﻪ ﻳﺎﺭی ﺍﻭ ﺑﺸـﺘﺎﺏ، ﺑـﺎﺭ ﻏﻤـﯽ ﺍﺯ ﺩﻝ ﺍﻭ ﺑﺮﺩﺍﺭ ﻭ ﻧﺴـﻴﻢ ﺷـﺎﺩی ﺩﺭ ﺣﺮﻳـﻢ ﺩﻝ ﺍﻭ ﺑﺮﺍﻧﮕﻴﺰ.

ﺍﺯ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻧﻘﻞ ﺍﺳﺖ ﻛﻪ ﻓﺮﻣﻮﺩﻧﺪ: «ﻣﻦ ﻭ ﭘﺪﺭﻡ ﺍﺑﻮﺟﻌﻔـﺮ (ﻳﻌﻨـﯽ ﺍﻣـﺎﻡ ﺑﺎﻗﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ) ﺑـﺮ (ﺟﻤﻌﯽ ﺍﺯ) ﺷـﻴﻌﻴﺎﻥ ﮔﺬﺷـﺘﻴﻢ ﻛـﻪ ﻣﻴﺎﻥ ﻗﺒـﺮ (ﭘﻴﻐﻤﺒﺮﺍﻛﺮﻡ ﺻﻠﯽ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ ﻭ ﺳـﻠﻢ) ﻭ ﻣﻨﺒﺮ ﺑﻮﺩﻧﺪ. ﺑﻪ ﭘﺪﺭﻡ ﮔﻔﺘﻢ: ﺧﺪﺍﻭﻧﺪ ﻣﺮﺍ ﻓﺪﺍی ﺷﻤﺎ ﮔﺮﺩﺍﻧﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﺟﻤﻌﯽ ﺍﺯ ﺷﻴﻌﻴﺎﻥ ﻭ ﻣﻮﺍﻟﻴﺎﻥ ﺷﻤﺎ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ. ﺁﻥ ﺣﻀﺮﺕ ﭘﺮﺳﻴﺪ: ﺁﻧﻬـﺎ ﻛﺠﺎﻳﻨـﺪ؟ ﮔﻔﺘﻢ: ﺁﻧـﺎﻥ ﺭﺍ ﻣﺎﺑﻴﻦ ﻗﺒﺮ ﻭ ﻣﻨﺒﺮ ﻣﯽ ﺑﻴﻨـﻢ. ﻓﺮﻣﻮﺩ: ﻣﺮﺍ ﻧﺰﺩ ﺁﻧﺎﻥ ﺑﺒﺮ! ﭘﺲ (ﺍﻣﺎﻡ ﺑﺎﻗﺮ) ﺑﻪ ﺁﻧﺎﻥ ﻧﺰﺩﻳﮏ ﺷـﺪ ﻭ ﺑﺮ ﺍﻳﺸـﺎﻥ ﺳـﻼﻡ ﻛﺮﺩ. ﺳـﭙﺲ ﺑﻪ ﺁﻧﺎﻥ ﻓﺮﻣﻮﺩ: ﺑﻪ ﺧﺪﺍ ﺳﻮﮔﻨﺪ ﻛﻪ ﻣﻦ ﭼﻴﺮﮔﯽ ﻭ ﺁﺳﺎﻳﺶ ﺷـﻤﺎ ﺭﺍ ﺩﻭﺳﺖ ﻣﯽ ﺩﺍﺭﻡ. ﺷﻤﺎ ﻫﻢ ﺑﻪ ﻭﺳﻴﻠﻪ ﺧﻮﻳﺸﺘﻨﺪﺍﺭی ﻭ ﺟﺪﻳﺖ (ﺩﺭ ﺩﻳﻨـﺪﺍﺭی) ﻣﺮﺍ ﻳﺎﺭی ﻛﻨﻴﺪ ﻛﻪ ﺁﻧﭽﻪ ﻧﺰﺩ ﺧﺪﺍﻭﻧﺪ ﺍﺳـﺖ ﺑﻪ ﺩﺳـﺖ ﻧﻤﯽ ﺁﻳـﺪ ﻣﮕـﺮ ﺑـﺎ ﺧﻮﻳﺸـﺘﻨﺪﺍﺭی (ﻧﺴـﺒﺖ ﺑﻪ ﮔﻨﺎﻫـﺎﻥ ) ﻭ ﺗـﻼﺵ (ﺩﺭ ﻛﺎﺭﻫـﺎی ﻧﻴﻜـﻮ)، ﻭ ﻫﻨﮕﺎﻣﯽ ﻛـﻪ ﺑﻨﺪﻩ ﺍی ﺭﺍ ﺍﻣـﺎﻡ ﺧﻮﺩ ﺩﺍﻧﺴـﺘﻴﺪ ﺑﻪ ﺍﻭ ﺍﻗﺘﺪﺍ ﻧﻤﺎﻳﻴﺪ! ﺑﻪ ﺧﺪﺍ ﺳـﻮﮔﻨﺪ ﺷـﻤﺎ ﺑﺮ ﺩﻳﻦ ﻣﻦ ﻭ ﺩﻳﻦ ﭘﺪﺭﺍﻧﻢ ﺍﺑﺮﺍﻫﻴﻢ ﻭ ﺍﺳـﻤﺎﻋﻴﻞ ﻫﺴﺘﻴﺪ، ﻫﻤﺎﻥ ﻃﻮﺭ ﻛﻪ ﺍﻳﻨﺎﻥ (ﻳﻌﻨﯽ ﭘﻴﺮﻭﺍﻥ ﺧﻠﻔﺎ) ﺑﺮ ﺩﻳﻦ ﺁﻧﺎﻥ (ﻳﻌﻨﯽ ﺣﺎﻛﻤﺎﻥ ﺯﻣﺎﻥ ) ﻫﺴـﺘﻨﺪ. ﭘﺲ ﺑﺮ ﺷﻤﺎ ﺑﺎﺩ ﻛﻪ ﺍﻳﻦ (ﻣﻜﺘﺐ) ﺭﺍ ﺑﻪ ﻭﺳﻴﻠﻪ ﻭﺭﻉ ﻭ ﺗﻼﺵ ﻳﺎﺭی ﻧﻤﺎﻳﻴﺪ.»(41)

ﺍﺯ ﻣﺠﻤﻮﻋـﻪ ی ﺍﻳﻦ ﺣﻜﺎﻳﺎﺕ ﻭ ﺭﻭﺍﻳـﺎﺕ ﺩﺭﻣﯽ ﻳﺎﺑﻴﻢ ﻛﻪ ﻳﻜﯽ ﺍﺯ ﻣﻬﻢﺗﺮﻳـﻦ ﺭﺍﻩ ﻫـﺎی ﻳﺎﺭی ﺍﻫﻞﺑﻴﺖ ﻛﺴـﺐ ﺍﻋﻤﺎﻝ ﺻﺎﻟـﺢ ﻭ ﺍﺟﺘﻨﺎﺏ ﺷـﺪﻳﺪ ﺍﺯ ﮔﻨﺎﻫـﺎﻥ ﺑـﺰﺭگ ﺍﺳـﺖ. ﺑﺪﻳﻦ ﻭﺳـﻴﻠﻪ ، ﻫﻢ ﻗﻠﺐ ﺁﻧﺎﻥ ﺷـﺎﺩ ﻣﯽ ﺷـﻮﺩ ﻭ ﻫـﻢ ﺁﺑـﺮﻭ ﻭ ﻋـﺰﺕ ﺁﻧـﺎﻥ ﺑﻴﺸـﺘﺮ ﻣﯽ ﺷـﻮﺩ. ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﻫﻤﻴﻦ ﻧﻜﺘـﻪ ﺭﺍ ﺑـﻪ ﺻﺮﺍﺣﺖ ﺑﻪ ﻳـﺎﺭﺍﻥ ﺧﻮﻳـﺶ ﻓﺮﻣﺎﻥ ﺩﺍﺩﻩ ﺍﻧـﺪ. ﺩﺭ ﻧﻬﺞ ﺍﻟﺒﻼﻏﻪ ﻣﯽ ﺧﻮﺍﻧﻴﻢ ﻛـﻪ ﺁﻥ ﺣﻀﺮﺕ ﻓﺮﻣﻮﺩﻧﺪ: «ﻣﺮﺍ ﺑﺎ ﺧﻮﻳﺸـﺘﻦﺩﺍﺭی (ﻧﺴـﺒﺖ ﺑﻪ ﮔﻨﺎﻫﺎﻥ ) ﻭ ﺗﻼﺵ (ﺩﺭ ﻛﺎﺭﻫﺎی ﻧﻴﮏ) ﻭ ﻋﻔﺖ ﻭﺭﺯﻳﺪﻥ ﻭ ﻣﺤﻜﻢ ﺑﻮﺩﻥ ﻳﺎﺭی ﻛﻨﻴﺪ!»(42)

ﺍﻳﻨﮏ ﺷـﻤﺎ ﻛﻪ ﺧﻮﺍﺳـﺘﺎﺭی ﺗﺎ ﺍﺯ ﻳـﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺧﻮﺩ ﻳﻌﻨﯽ ﺍﻣﺎﻡ ﻣﻬﺪی ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺷـﻤﺎﺭ ﺁﻳﯽ، ﺑﺎﻳﺪ ﺑﺪﺍﻧـﯽ ﻛﻪ ﺗﻼﺵ ﺩﺭ ﺭﺍﻩ ﻧﻴﻜـﻮﻛﺎﺭی ﻭ ﻣﻬﺎﺭ ﻧﻔﺲ ﺩﺭ ﺑﺮﺍﺑﺮ ﻫﺮﺯﮔﯽ ﻫﺎ ﺍﺯ ﻣﻬﻢﺗﺮﻳﻦ ﻭﻇﺎﻳﻔﺖ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﺪ. ﺑﻪ ﺧﺼﻮﺹ ﺑﺎﻳﺪ ﺑﺪﺍﻧﯽ ﻛﻪ ﺍﮔﺮ ﺍﻳﻦ ﻛﺎﺭﻫﺎ ﺭﺍ ﺩﺭ ﺑﺴـﺘﺮ ﺗﻮﺟـﻪ ﺑﻪ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺧـﻮﺩ ﻭ ﺍﻧﺘﻈﺎﺭ ﻇﻬـﻮﺭ ﺍﻭ ﺍﻧﺠﺎﻡ ﺩﻫـﯽ، ﺧﺪﺍﻭﻧﺪ ﻧﺎﻣﺖ ﺭﺍ ﺭﺳﻤﺎ ﺩﺭ ﺩﻓﺘﺮ ﻳﺎﺭﺍﻥ ﺁﻥ ﺍﻣﺎﻡ ﻗﺎﺋﻢ ﺛﺒﺖ ﻣﯽ ﻓﺮﻣﺎﻳﺪ. ﭘﺲ ﮔﻮﺍﺭﺍ ﺑﺎﺩ ﺑﺮ ﺗﻮ ﺭﺣﻤﺖ ﺧﺪﺍﻭﻧﺪ ﻭ ﻣﺒﺎﺭک ﺑﺎﺩ ﺗﻮ ﺭﺍ ﻧﺸـﺎﻥ ﺧﺪﻣﺖﮔﺰﺍﺭی ﻭ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ.
