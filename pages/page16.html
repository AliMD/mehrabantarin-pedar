ﺩﻋﺎ ﺑﺮﺍی ﻓﺮﺝ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ

ﻳﻜـﯽ ﺍﺯ ﺧﺪﻣﺖﮔـﺰﺍﺭﺍﻥ ﺁﺳـﺘﺎﻥ ﺍﻣـﺎﻡ ﻋﺼـﺮ ﻣﯽ ﻓﺮﻣـﻮﺩ: «ﻣﺘﻮﺳـﻞ ﻭ ﻣﺘﺬﻛـﺮ ﺑـﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺷـﻮﻳﺪ ﻭ ﺩﺭ ﺣـﻖ ﺍﻭ ﺩﻋﺎ ﻛﻨﻴﺪ! ﺑـﻪ ﺧﺪﺍ، ﺍﻭ ﺁﻗـﺎ ﻭ ﺟﻮﺍﻧﻤﺮﺩ ﺍﺳـﺖ ﻭ ﭼﻨﺪ ﺑﺮﺍﺑﺮ ﺗﻼﻓﯽ ﻣﯽ ﻛﻨﺪ. ﺷـﻤﺎ ﺍﮔﺮ ﺧﺒﺮﺩﺍﺭ ﺷـﻮی ﻛﻪ ﻛﺴﯽ ﭘﺸﺖ ﺳﺮ ﺩﺭﺑﺎﺭﻩ ﺗﻮ ﺩﻋﺎ ﻛﺮﺩﻩ ﻭ ﺣﻤﺎﻳﺘﯽ ﻧﻤﻮﺩﻩ ﺍﮔﺮ ﺍﻧﺪک ﻧﺸـﺎﻥ ﺁﺩﻣﯽ ﻭ ﻭﺟﺪﺍﻥ ﺍﻧﺴـﺎﻧﯽ ﺩﺍﺷـﺘﻪ ﺑﺎﺷـﯽ ﺗﺎ ﺯﻧﺪﻩ ﻫﺴﺘﯽ ﻣﯽ ﺧﻮﺍﻫﯽ ﺍﻳﻦ ﺧﺪﻣﺘﯽ ﺭﺍ ﻛﻪ ﺍﻭ ﭘﺸﺖ ﺳﺮ ﺗﻮ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺩﻭ ﺑﺮﺍﺑﺮ ﺟﺰﺍ ﻭ ﭘﺎﺩﺍﺵ ﺑﺪﻫﯽ.

ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﻭﻝ ﺟﻮﺍﻧﻤﺮﺩ ﻋﺎﻟﻢ ﺍﺳـﺖ. ﻏﻴﺮ ﺍﺯ ﺟﻨﺒﻪ ﺍﻣﺎﻣﺘـﺶ، ﺟﻮﺍﻧﻤﺮﺩی ﻋﺮﺑﯽ ﻋﺠﻴﺒﯽ ﺩﺍﺭﺩ؛ ﺑﺒﻴﻨـﺪ ﻛﻪ ﺗﻮ ﺩﺭﺑﺎﺭﻩ ﺍﻭ ﺩﻋﺎ ﻣﯽ ﻛﻨـﯽ ﻭ ﺯﻳـﺎﺭﺕ ﺁﻝ ﻳﺎﺳـﻴﻦ ﻣﯽ ﺧﻮﺍﻧـﯽ ﻭ ﺍﺯ ﺧـﺪﺍ ﺳـﻼﻣﺖ ﻭ ﻓﺮﺝ ﻭ ﻧﺸـﺎﻁ ﺍﻭ ﺭﺍ ﻣﯽ ﺧﻮﺍﻫـﯽ، ﺁﻳـﺎ ﺑﻌﺪ ﺍﺯ ﺍﻳـﻦ ﻫﻤﻪ ﺩﻋـﺎ ﻭ ﻫﻮﺍﺧﻮﺍﻫﯽ، ﺁﻥ ﺣﻀـﺮﺕ ﻣﯽ ﮔـﺬﺍﺭﺩ ﻛـﻪ ﺷـﻤﺎ ﺩﺭ ﻣﺸـﻜﻼﺕ ﺑﻤﺎﻧـﯽ؟ ﻣﺤـﺎﻝ ﺍﺳﺖ!»(55)

ﮔﻔﺘﺎﺭ ﺍﻳﻦ ﺑﺰﺭﮔﻮﺍﺭ، ﻫﻢ ﺍﺯ ﺳﻮی ﺧﺮﺩ ﺍﻧﺴﺎﻧﯽ ﻣﻮﺭﺩ ﺗﺄﻳﻴﺪ ﺍﺳﺖ ﻭ ﻫـﻢ ﺍﺯ ﺟﺎﻧﺐ ﺷـﺮﻉ ﺍﺳـﻼﻡ. ﺟﻮﺍﻧﻤـﺮﺩی ﺧﺼﻴﺼﻪ ﺍی ﺧﺮﺩﭘﺴـﻨﺪ ﻭ ﺍﺯ ﻭﻳﮋﮔﯽ ﻫـﺎی ﺍﻧﺴـﺎﻥ ﻛﺎﻣـﻞ ﺍﺳـﺖ. ﺍﮔـﺮ ﻛﺴـﯽ ﺑـﻪ ﺷـﺨﺼﻴﺖ ﻛﻤﺎﻝ ﻳﺎﻓﺘـﻪ ﺍﻣـﺎﻡ ﻋﺼـﺮ ﺗﻮﺟﻪ ﻛﻨﺪ ﺩﺭﻣﯽ ﻳﺎﺑﺪ ﺍﻳﺸـﺎﻥ ﺑﻪ ﺟﻬـﺖ ﻛﻤﺎﻝ ﺟﻮﺍﻧﻤﺮﺩی ﺍﺵ ﻳﻘﻴﻨﺎ ﺩﺭﺻﺪﺩ ﺗﻼﻓﯽ ﺑﺮﺍی ﻛﺴـﯽ ﺑﺮﻣﯽ ﺁﻳﺪ ﻛﻪ ﺑﺮﺍی ﺍﻭ ﺩﻋﺎ ﻛﻨﺪ ﻭ ﺍﺯ ﺧﺪﺍﻭﻧﺪ ﺗﻮﺍﻧﺎ ﻭ ﻋﺰﻳﺰ ﺑﺮﺍی ﺍﻭ ﻳﺎﺭی ﺑﻄﻠﺒﺪ.

ﻋـﻼﻭﻩ ﺑﺮ ﻋﻘﻞ، ﺷـﺮﻳﻌﺖ ﺍﺳـﻼﻡ ﻧﻴﺰ ﻛﻪ ﺑـﺮ ﭘﺎﻳﻪ ﻋﻘـﻞ ﺑﻨﺎ ﻧﻬﺎﺩﻩ ﺷـﺪﻩ، ﭘﺎﺳـﺦ ﺑـﻪ ﺍﺣﺴـﺎﻥ ﻭ ﻣﺤﺒـﺖ ﺩﻳﮕـﺮﺍﻥ ﺭﺍ ﺿـﺮﻭﺭی ﻣﯽ ﺩﺍﻧﺪ. ﺧﺪﺍﻭﻧﺪ ﺗﻌﺎﻟﯽ ﻣﯽ ﻓﺮﻣﺎﻳﺪ:

«ﺍی ﻛﺴـﺎﻧﯽ ﻛـﻪ ﺍﻳﻤﺎﻥ ﺁﻭﺭﺩﻩ ﺍﻳﺪ ﭼﻨﺎﻧﭽﻪ ﻛﺴـﯽ ﺷـﻤﺎ ﺭﺍ ﺗﺤﻴﺖ ﮔﻔﺖ، ﺑﻪ ﺍﻭ ﭘﺎﺳﺨﯽ ﺑﻬﺘﺮ ﺑﺪﻫﻴﺪ ﻳﺎ (ﻻﺍﻗﻞ ﭘﺎﺳﺨﯽ) ﺑﺮﺍﺑﺮ ﺑﺎ ﺁﻥ.» ﺑـﻪ ﻳﻘﻴﻦ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻛﻪ ﺧـﺪﺍ ﻓﺮﺟـﺶ ﺭﺍ ﻧﺰﺩﻳﮏ ﺳـﺎﺯﺩ ﺍﻳﻦ ﺩﺳﺘﻮﺭ ﺍﻟﻬﯽ ﺭﺍ ﭘﺎﺱ ﻣﯽ ﺩﺍﺭﺩ ﻭ ﺧﻮﺩ ﺍﻟﮕﻮی ﻋﻤﻞ ﺑﻪ ﻓﺮﻣﺎﻥﻫﺎی ﺍﻟﻬﯽ ﻣﯽ ﺑﺎﺷـﺪ. ﻣﺤﺎﻝ ﺍﺳـﺖ ﻛﻪ ﻣﻦ ﻭ ﺷـﻤﺎ ﺑﺮ ﺍﻭ ﺩﺭﻭﺩ ﻭ ﺳـﻼﻡ ﺑﻔﺮﺳﺘﻴﻢ ﻭ ﺍﻭ ﭘﺎﺳـﺦ ﻣﺎ ﺭﺍ ﻧﺪﻫﺪ ﻭ ﻣﺤﺎﻝ ﺍﺳـﺖ ﻛﻪ ﻣﻦ ﻭ ﺷﻤﺎ ﺑﺮﺍی ﺍﻭ ﺁﺭﺯﻭی ﺳﻼﻣﺖ ﻭ ﻓﺮﺝ (ﻳﻌﻨﯽ ﺭﻓﻊ ﮔﺮﻓﺘﺎﺭی) ﻛﻨﻴﻢ ﻭ ﺍﻭ ﺑﺮﺍی ﺳﻼﻣﺖ ﻭ ﺭﻓﻊ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﻣﺎ ﺩﻋﺎ ﻧﻜﻨﺪ.

ﻋﻼﻭﻩ ﺑﺮ ﺣﻜﻢ ﻋﻘﻞ ﻭ ﺷـﺮﻉ، ﺳﻴﺮﻩ ﻋﻤﻠﯽ ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭ ﻭ ﭘﺪﺭﺍﻥ ﺍﻳﺸﺎﻥ ﻧﻴﺰ ﮔﻮﺍﻩ ﺁﻥ ﺍﺳﺖ ﻛﻪ ﺁﻧﺎﻥ ﺍﺯ ﻫﻮﺍﺩﺍﺭﺍﻥ ﺧﻮﺩ ﻫﻮﺍﺩﺍﺭی ﻣﯽ ﻛﻨﻨﺪ ﻭ ﺑﻪ ﻛﺴﺎﻧﯽ ﻛﻪ ﺑﻪ ﺍﻳﺸﺎﻥ ﺍﺣﺴﺎﻥ ﻛﻨﻨﺪ، ﺍﺣﺴﺎﻥ ﻣﯽ ﻧﻤﺎﻳﻨﺪ، ﭼﻪ ﺁﻥﻛﻪ ﺍﻳﻦ ﺧﺎﻧﻮﺍﺩﻩ، ﻣﻌﺪﻥ ﺟﻮﺩ ﻭ ﺍﺣﺴـﺎﻧﻨﺪ. ﺍﮔﺮ ﻣﺎ ﻛﺎﺭی ﻛﻨﻴﻢ ﻛﻪ ﺩﻟﻬﺎﻳﻤﺎﻥ ﺑﺎ ﺁﻧﻬـﺎ ﭘﻴﻮﻧﺪ ﺑﺨـﻮﺭﺩ، ﺑﺮ ﺍﺛﺮ ﺍﻳﻦ ﭘﻴﻮﻧﺪ، ﺍﺯ ﻧﻮﺭﺍﻧﻴﺖ، ﺣﻼﻭﺕ ﻭ ﺑﺮﻛﺎﺕ ﺍﻳﻦ ﺑﺰﺭﮔﻮﺍﺭﺍﻥ ﺑﻬﺮﻩ ﻣﻨﺪ ﺧﻮﺍﻫﻴﻢ ﺷﺪ.

ﺍﺯ ﺣﻜﺎﻳـﺎﺕ ﺷـﻴﺮﻳﻨﯽ ﻛـﻪ ﻧﺸـﺎﻥﮔﺮ ﻋﻨﺎﻳﺖ ﺍﻳـﻦ ﺑﺰﺭﮔـﻮﺍﺭﺍﻥ ﺑﻪ ﺩﻭﺳﺘﺪﺍﺭﺍﻥ ﺧﻮﻳﺶ ﺍﺳﺖ ﺣﻜﺎﻳﺖ ﻋﺒﺪﺍﻟﺮﺣﻤﻦ ﺍﺻﻔﻬﺎﻧﯽ ﺍﺳﺖ.

ﺩﺭ ﺯﻣﺎﻥ ﻣﺘﻮﻛﻞ ﻋﺒﺎﺳـﯽ ﺷﺨﺼﯽ ﺑﻨﺎﻡ ﻋﺒﺪﺍﻟﺮﺣﻤﻦ ﺩﺭ ﺍﺻﻔﻬﺎﻥ ﺳﻜﻮﻧﺖ ﺩﺍﺷﺖ ﻛﻪ ﺷﻴﻌﻪ ﺑﻮﺩ. ﺍﺯ ﺍﻭ ﭘﺮﺳﻴﺪﻧﺪ: ﭼﻄﻮﺭ ﺑﻪ ﺍﻣﺎﻣﺖ ﺍﻣﺎﻡ ﻫﺎﺩی ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ (ﻛﻪ ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ ﺯﻧﺪﻩ ﺑﻮﺩﻧﺪ) ﻣﻌﺘﻘﺪ ﺷﺪی؟

ﮔﻔـﺖ: ﺟﺮﻳﺎﻧـﯽ ﺭﺍ ﻣﺸـﺎﻫﺪﻩ ﻛﺮﺩﻡ ﻛﻪ ﺳـﺒﺐ ﺷـﻴﻌﻪ ﺷـﺪﻥ ﻣﻦ ﮔﺮﺩﻳـﺪ. ﻣـﻦ ﻣﺮﺩی ﻓﻘﻴﺮ ﻭﻟـﯽ ﺯﺑـﺎﻥﺩﺍﺭ ﻭ ﭘﺮﺟﺮﺃﺕ ﺑـﻮﺩﻡ. ﺑﻪ ﻫﻤﻴﻦ ﺟﻬـﺖ ﺩﺭ ﻳﻜﯽ ﺍﺯ ﺳـﺎﻝ ﻫﺎ ﺍﻫﻞ ﺍﺻﻔﻬﺎﻥ ﻣـﺮﺍ ﺑﺮﮔﺰﻳﺪﻧﺪ ﺗﺎ ﺑﺎ ﮔﺮﻭﻫﯽ ﺑﺮﺍی ﺩﺍﺩﺧﻮﺍﻫﯽ ﻧﺰﺩ ﻣﺘﻮﻛﻞ ﺑﺮﻭﻳﻢ. ﻭﻗﺘﯽ ﺑﻪ ﺁﺳـﺘﺎﻧﻪ ﺩﺭﺑﺎﺭ ﺭﺳـﻴﺪﻳﻢ، ﺩﺳﺘﻮﺭی ﺍﺯ ﺟﺎﻧﺐ ﻣﺘﻮﻛﻞ ﺑﺮﺍی ﺍﺣﻀﺎﺭ ﻋﻠﯽ ﺑﻦ ﻣﺤﻤﺪ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ (ﻳﻌﻨـﯽ ﺍﻣـﺎﻡ ﻫﺎﺩی) ﺻﺎﺩﺭ ﺷـﺪﻩ ﺑـﻮﺩ. ﺑﻪ ﻳﻜـﯽ ﺍﺯ ﺣﺎﺿـﺮﺍﻥ ﮔﻔﺘﻢ: ﺍﻳـﻦ ﻣﺮﺩ ﻛﻪ ﺩﺳـﺘﻮﺭ ﺍﺣﻀﺎﺭﺵ ﺩﺍﺩﻩ ﺷـﺪﻩ ﻛﻴﺴـﺖ؟ ﮔﻔـﺖ: ﺍﻭ ﻳﻜﯽ ﺍﺯ ﺳـﺎﺩﺍﺕ ﻋﻠﻮی ﺍﺳـﺖ ﻛﻪ ﺭﺍﻓﻀﯽ ﻫﺎ (ﻳﻌﻨﯽ ﺷـﻴﻌﻪ ﻫﺎ) ﺑـﻪ ﺍﻣﺎﻣﺘﺶ ﻣﻌﺘﻘﺪ ﻫﺴـﺘﻨﺪ. ﺳـﭙﺲ ﺁﻥ ﺷـﺨﺺ ﮔﻔﺖ ﻣـﻦ ﻣﻄﻤﺌﻨﻢ ﻣﺘـﻮﻛﻞ ﺍﻭ ﺭﺍ ﺑﺮﺍی ﻛﺸﺘﻦ ﺍﺣﻀﺎﺭ ﻛﺮﺩﻩ ﺍﺳﺖ.

ﻋﺒﺪﺍﻟﺮﺣﻤﻦ ﻣﯽ ﮔﻮﻳﺪ: ﮔﻔﺘﻢ ﺍﺯ ﺍﻳﻨﺠﺎ ﻧﻤﯽ ﺭﻭﻡ ﺗﺎ ﺁﻧﻜﻪ ﺑﺒﻴﻨﻢ ﺍﻳﻦ ﻣﺮﺩ ﭼﮕﻮﻧﻪ ﺷـﺨﺼﯽ ﺍﺳـﺖ. ﭘﺲ ﺍﺯ ﻣﺪﺗﯽ ﺩﻳـﺪﻡ ﻛﻪ ﻋﻠﯽ ﺑﻦ ﻣﺤﻤﺪ ﺁﻣﺪ، ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﺳـﻮﺍﺭ ﺑﺮ ﺍﺳـﺐ ﺑﻮﺩ. ﻣﺮﺩﻡ ﻃﺮﻑ ﺭﺍﺳﺖ ﻭ ﭼﭗ ﺍﻭ ﺻﻒ ﻛﺸـﻴﺪﻩ، ﺑﻪ ﺍﻭ ﻣﯽ ﻧﮕﺮﻳﺴﺘﻨﺪ. ﻫﻤﻴﻦ ﻛﻪ ﺍﻭ ﺭﺍ ﺩﻳﺪﻡ، ﻣﺤﺒﺘﺶ ﺩﺭ ﺩﻟـﻢ ﺍﻓﺘﺎﺩ ﻭ ﺩﻟﻢ ﺑﺮﺍﻳﺶ ﺳـﻮﺧﺖ. ﺷـﺮﻭﻉ ﻛـﺮﺩﻡ ﺩﺭ ﺩﻝ ﺑﺮﺍی ﺍﻭ ﺩﻋﺎ ﻛـﺮﺩﻥ ﻛﻪ ﺧﺪﺍ ﺷـﺮ ﻣﺘـﻮﻛﻞ ﺭﺍ ﺍﺯ ﺍﻭ ﺩﻓﻊ ﻛﻨـﺪ. ﺍﻭ ﺩﺭ ﻣﻴﺎﻥ ﻣﺮﺩﻡ ﭘﻴﺶ ﻣﯽ ﺁﻣﺪ ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﺑﻪ ﻳﺎﻝ ﺍﺳﺒﺶ ﻧﻈﺮ ﺩﺍﺷﺖ ﻭ ﺑﻪ ﭼﭗ ﻭ ﺭﺍﺳﺖ ﻧﮕﺎﻩ ﻧﻤﯽ ﻛﺮﺩ ﻭ ﻣﻦ ﺩﺭ ﺩﻝ ﺑﻪ ﺩﻋﺎی ﺧﻮﺩ ﻣﺸـﻐﻮﻝ ﺑﻮﺩﻡ. ﻫﻤﻴﻦ ﻛﻪ ﺍﻳﺸـﺎﻥ ﻣﻘﺎﺑﻞ ﻣﻦ ﺭﺳﻴﺪ، ﺳﺮﺵ ﺭﺍ ﺑﻠﻨﺪ ﻛﺮﺩ. ﺭﻭ ﺑﻪ ﻣﻦ ﻧﻤﻮﺩﻩ، ﮔﻔﺖ:

«استجاب الله دعاک، و طوّل عمرک، وکثّر مالک و ولدک. »

ﻳﻌﻨﯽ«ﺧﺪﺍ ﺩﻋﺎﻳﺖ ﺭﺍ ﻣﺴﺘﺠﺎﺏ ﻛﻨﺪ، ﻋﻤﺮﺕ ﺭﺍ ﻃﻮﻻﻧﯽ ﻧﻤﺎﻳﺪ، ﻭ ﻣﺎﻝ ﻭ ﻓﺮﺯﻧﺪﺍﻧﺖ ﺭﺍ ﺯﻳﺎﺩ ﺳﺎﺯﺩ.»

ﺳـﻴﺪ ﺣـﻼﻭی ، ﻗﺼﻴـﺪﻩ ی ﺷـﻜﻮﺍﺋﻴﻪ ﺍی ﮔﻔﺘﻪ ﺑـﻮﺩ ﻭ ﺩﺭ ﺁﻥ ﺧﻄـﺎﺏ ﺑـﻪ ﺁﻥ ﺣﻀـﺮﺕ ﻋﺮﺿـﻪ ﺩﺍﺷـﺖ: ﻳﺎﺑـﻦ ﺍﻟﻌﺴـﻜﺮی، ﺗﺎ ﻛﯽ ﭘﻨﻬﺎﻧـﯽ؟ ﺷـﻴﻌﻪ ﺩﺭ ﺭﻧﺞ ﻭ ﺯﺣﻤﺖ ﻫﺴـﺘﻨﺪ ﻭ ﻫﻤﻪ ﮔﺮﻓﺘﺎﺭی ﺷـﻴﻌﻴﺎﻥ ﺭﺍ ﺑﺮﺷـﻤﺮﺩﻩ ﺑـﻮﺩ ... ﺷـﺐ، ﺧﻮﺑـﺎﻥ ﻭ ﭘﺎﻛﺎﻧـﯽ ﺩﺭ ﻧﺠﻒ ﺍﺷـﺮﻑ ﺩﺭ ﻋﺎﻟﻢ ﺭﺅﻳﺎ ﺷـﺮﻓﻴﺎﺏ ﺧﺪﻣﺖ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺷـﺪﻩ ﺑﻮﺩﻧﺪ ﻛﻪ ﺣﻀﺮﺕ ﻓﺮﻣﻮﺩﻧﺪ: ﺑﺮﻭﻳﺪ ﺑﻪ ﺳـﻴﺪ ﺑﮕﻮﻳﻴﺪ: ﺍﻳﻦ ﻗﺪﺭ ﺩﻝ ﻣﺮﺍ ﻣﺴﻮﺯﺍﻥ، ﺍﻳـﻦ ﻗـﺪﺭ ﮔﺮﻓﺘﺎﺭی ﺷـﻴﻌﻴﺎﻧﻢ ﺭﺍ ﻣﺸـﻤﺎﺭ، ﺩﻟﻢ ﺍﺯ ﺷـﻨﻴﺪﻥ ﺁﻥﻫﺎ ﻛﺒﺎﺏ ﻣﯽ ﺷﻮﺩ. «ﻓﺈﻥ ﺍﻷﻣﺮ ﻟﻴﺲ ﺑﻴﺪی»؛ (ﻳﻌﻨﯽ ﻇﻬﻮﺭ ﺑﻪ ﺩﺳﺖ ﻣﻦ ﻧﻴﺴﺖ) ﻛﺎﺭ، ﺩﺳـﺖ ﺧﺪﺍﺳـﺖ. ﺑﮕﻮ: ﺩﻋـﺎ ﻛﻨﻨﺪ ﻭ ﺍﺯ ﺧﺪﺍ ﻓﺮﺝ ﻣـﺮﺍ ﺑﺨﻮﺍﻫﻨﺪ ﻭ ﺧﺪﺍ ﻓﺮﺝ ﻣﺮﺍ ﺑﺮﺳـﺎﻧﺪ؛ ﺩﺳـﺖ ﻣﺮﺍ ﺑﺎﺯ ﻛﻨﺪ ﺗﺎ ﮔﺮﻓﺘﺎﺭی ﺩﻭﺳﺘﺎﻧﻢ ﺭﺍ ﺑﺮﻃﺮﻑ ﻧﻤﺎﻳﻢ.»(58)

ﺑﻪ ﻫﺮ ﺗﻘﺪﻳﺮ ﺳـﺨﻦ ﺩﺭ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﻧﻤﺎﺯ ﻭ ﺑﻪ ﺧﺼﻮﺹ ﻫﻨﮕﺎﻡ ﻗﻨﻮﺕ ﺩﺭ ﻧﻤﺎﺯ، ﻭﻗﺖ ﺧﻮﺑﯽ ﺑﺮﺍی ﺩﻋﺎ ﻛﺮﺩﻥ ﺑﺮ ﻓﺮﺝ ﻣﻮﻻﺳﺖ.

ﺩﺭ ﺑﺮﺧﯽ ﺭﻭﺍﻳﺎﺕ ﺁﻣﺪﻩ ﻛﻪ ﺩﺭﺳﺖ ﭘﺲ ﺍﺯ ﭘﺎﻳﺎﻥ ﻫﺮ ﻧﻤﺎﺯ ﻧﻴﺰ ﺑﻨﺪﻩ ﺩﻋﺎی ﻣﺴـﺘﺠﺎﺑﯽ ﺩﺍﺭﺩ. ﺍﺯ ﺍﻳﻦ ﻟﺤﻈﻪ ﻧﻴﺰ ﻧﺒﺎﻳﺪ ﺩﺭ ﺟﻬﺖ ﺩﻋﺎ ﺑﺮﺍی ﺁﻥ ﻣﺤﺒﻮﺏ ﻏﻔﻠﺖ ﻛﻨﻴﻢ. ﺳﺠﺪﻩ ﻧﻴﺰ ﺍﺯ ﻣﻮﺍﺿﻊ ﻗﺮﺏ ﺑﻨﺪﻩ ﺑﻪ ﺧﺪﺍ ﺍﺳﺖ ﻭ ﺩﻋﺎ ﺑﺮﺍی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﺳﺠﺪﻩ ﻧﻴﺰ ﺳﻔﺎﺭﺵ ﺷﺪﻩ ﺍﺳﺖ.

ﻳﮏ ﻧﻜﺘﻪ ﻣﻬﻢ ﺩﺭﺑﺎﺭﻩ ﻧﻤﺎﺯ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺩﻋﺎی ﺍﻧﺴﺎﻥ ﺩﺭ ﻗﻨﻮﺕ ﻭ ﺳـﺠﺪﻩ ﻭ ﭘﺲ ﺍﺯ ﻧﻤﺎﺯ ﻭﻗﺘﯽ ﺍﺛﺮ ﺧﺎﺹ ﭘﻴﺪﺍ ﻣﯽ ﻛﻨﺪ ﻛﻪ ﺍﻧﺴﺎﻥ ﻭﺍﻗﻌﺎ ﺍﺯ ﻧﻤـﺎﺯ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﺮﺩﺑﺎﻥ ﻋﺮﻭﺝ ﺑﻪ ﺳـﻤﺖ ﺧﺪﺍ ﺍﺳـﺘﻔﺎﺩﻩ ﻛﺮﺩﻩ ﺑﺎﺷـﺪ ﻭ ﻭﺍﻗﻌﺎ ﺩﻟﺶ ﺑﻪ ﺧﺪﺍی ﻣﺘﻌﺎﻝ ﻧﺰﺩﻳﮏ ﺷـﺪﻩ ﺑﺎﺷـﺪ. ﺍﮔﺮ ﻧﻤﺎﺯ ﺍﻧﺴـﺎﻥ ﭼﻨﻴـﻦ ﺑﺎﺷـﺪ، ﻓﺮﺻـﺖ ﺧﻮﺑﯽ ﺑـﺮﺍی ﺩﻋﺎی ﺑﺮ ﻓـﺮﺝ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺧﻮﺍﻫﺪ ﺷـﺪ. ﺍﻣﺎ ﺍﮔﺮ ﺍﻧﺴـﺎﻥ ﺩﺭ ﻧﻤﺎﺯ ﺧﻮﺩ ﺗﻨﻬﺎ ﺑﻪ ﻳﮏ ﺳﺮی ﺣﺮﻛﺎﺕ ﻓﻴﺰﻳﻜﯽ ﺑﺴﻨﺪﻩ ﻛﻨﺪ ﻭ ﻗﻠﺒﺶ ﻣﺘﻮﺟﻪ ﺧﺪﺍ ﻧﺸﻮﺩ، ﻗﺎﻋﺪﺗﺎ ﺣﺎﻝ ﻧﻤﺎﺯ ﺍﻭ ﺑـﺎ ﺍﺣﻮﺍﻝ ﺩﻳﮕﺮﺵ ﺗﻔﺎﻭﺕ ﻧﺪﺍﺭﺩ. ﺩﺭ ﺑﺮﺧﯽ ﺭﻭﺍﻳﺎﺕ ﺁﻣﺪﻩ ﺁﻥ ﺑﺨﺸـﯽ ﺍﺯ ﻧﻤﺎﺯ ﺍﻧﺴـﺎﻥ ﻣﻘﺒﻮﻝ ﺧﺪﺍ ﻣﯽ ﺷـﻮﺩ ﻛﻪ ﻗﻠﺐ ﺍﻧﺴـﺎﻥ ﻃﯽ ﺁﻥ ﻣﺘﻮﺟﻪ ﺧﺪﺍ ﺑﺎﺷﺪ.

ﻣـﺎﺩﺭی ﺭﺍ ﺩﺭ ﻧﻈـﺮ ﺑﮕﻴﺮﻳـﺪ ﻛـﻪ ﻓﺮﺯﻧـﺪﺵ ﺩﺭ ﺣﺎﻝ ﺍﻏﻤـﺎﺀ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳـﺖ ﻭ ﺑﺎ ﻣﺮگ ﺩﺳـﺖ ﻭ ﭘﻨﺠـﻪ ﻧﺮﻡ ﻣﯽ ﻛﻨﺪ. ﭘﺰﺷـﻜﺎﻥ ﺑﻪ ﺍﻳﻦ ﻣﺎﺩﺭ ﮔﻔﺘﻪ ﺍﻧﺪ ﻛﻪ ﻣﺎ ﻫﺮﭼﻪ ﺩﺭ ﺗﻮﺍﻥ ﺩﺍﺷـﺘﻪ ﺍﻳﻢ ﺑﺮﺍی ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺍﻳﻢ، ﺍﺯ ﺍﻳﻦ ﺑﻪ ﺑﻌﺪ ﻛﺎﺭی ﺍﺯ ﺩﺳـﺖ ﻣﺎ ﺑﺮ ﻧﻤﯽ ﺁﻳﺪ ﻭ ﺗﻨﻬﺎ ﺧﺪﺍﺳـﺖ ﻛﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﻫﻮﺷﻴﺎﺭی ﻭ ﺯﻧﺪﮔﯽ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎﺯﮔﺮﺩﺍﻧﺪ.

ﺍﻳـﻦ ﻣـﺎﺩﺭ ﺗﺼﻤﻴـﻢ ﻣﯽ ﮔﻴﺮﺩ ﺑﻪ ﺳـﺮﺍﻍ ﺧـﺪﺍ ﺑـﺮﻭﺩ ﻭ ﺍﺯ ﺍﻭ ﺑﺮﺍی ﻓﺮﺯﻧـﺪﺵ ﻛﻤـﮏ ﺑﺨﻮﺍﻫﺪ. ﻣﺎﺩﺭ ﺷـﻨﻴﺪﻩ ﺍﺳـﺖ ﻛـﻪ ﺑـﺮﺍی ﺍﺛﺮﺑﺨﺶ ﺷـﺪﻥ ﺩﻋﺎ، ﺧﻮﺏ ﺍﺳـﺖ ﻛﻪ ﺍﻧﺴـﺎﻥ ﺩﻭ ﺭﻛﻌﺖ ﻧﻤﺎﺯ ﺣﺎﺟﺖ ﺑﺨﻮﺍﻧﺪ ﻭ ﺩﺭ ﭼﻨﻴـﻦ ﺣﺎﻟـﯽ ﺑﺎ ﺧﺪﺍ ﺭﺍﺯ ﻭ ﻧﻴﺎﺯ ﻛﻨﺪ. ﺍﻭ ﺑﺮﺍی ﻭﺿﻮ ﻭ ﻧﻤﺎﺯ ﺍﻗﺪﺍﻡ ﻣﯽ ﻛﻨـﺪ، ﺍﻣﺎ ﺁﻳـﺎ ﺗﻮﺟﻪ ﺍﻭ ﺑﻪ ﺧﻀﻮﻉ ﻭ ﺧﺸـﻮﻉ ﺩﺭ ﺑﺮﺍﺑﺮ ﺧـﺪﺍ ﺩﺭ ﺍﻳﻦ ﻧﻤﺎﺯ ﺑﺎ ﻧﻤﺎﺯﻫﺎی ﺩﻳﮕﺮﺵ ﻓﺮﻕ ﻧﺪﺍﺭﺩ؟ ﻣﺴﻠﻤﺎ ﭼﺮﺍ!

ﺍﻳـﻦ ﻧﻤـﺎﺯ، ﻧﻤﺎﺯ ﻳـﮏ ﺑﻨﺪﻩ ﻧﻴﺎﺯﻣﻨﺪ ﺍﺳـﺖ ﻛـﻪ ﺑﺎ ﻋﺠـﺰ ﻭ ﻻﺑﻪ ﻭ ﺍﻟﺘﻤﺎﺱ ﺻﻮﺭﺕ ﻣﯽ ﭘﺬﻳﺮﺩ. ﺩﺭ ﺍﻳﻦ ﻧﻤﺎﺯ ﺑﻪ ﭘﺎﺧﺎﺳـﺘﻦ، ﺗﻌﻈﻴﻢ ﻛﺮﺩﻥ، ﺑـﻪ ﺳـﺠﺪﻩ ﺍﻓﺘﺎﺩﻥ ﻭ ﺩﺳـﺖ ﺑﻪ ﺩﻋـﺎ ﺑﺮﺩﺍﺷـﺘﻦ ﻣﻌﻨﺎی ﺩﻳﮕـﺮی ﺩﺍﺭﺩ. ﻧﻤﺎﺯﮔـﺰﺍﺭ ﺩﺭ ﺍﻳـﻦ ﻧﻤﺎﺯ ﻭﻗﺘﯽ ﺗﻜﺒﻴﺮﻩ ﺍﻻﺣﺮﺍﻡ ﻣﯽ ﮔﻮﻳﺪ ﻭ ﺩﺳـﺘﻬﺎﻳﺶ ﺭﺍ ﺑﻪ ﻧﺸـﺎﻧﻪ ﻧﻔﯽ ﻭ ﭘﺸـﺖ ﮔﻮﺵ ﺍﻧﺪﺍﺧﺘﻦ ﻫﺮﭼﻪ ﻏﻴﺮ ﺧﺪﺍ ﺍﺳﺖ، ﺑﺎﻻ ﻣﯽ ﺑﺮﺩ، ﺣﻘﻴﻘﺘﺎ ﺑﻪ ﻣﻌﻨﺎی ﻋﻤﻠﺶ ﺗﻮﺟﻪ ﻣﯽ ﻛﻨﺪ. ﭼﻨﻴﻦ ﻧﻤﺎﺯی ﺍﺳـﺖ ﻛﻪ ﺍﻧﺴـﺎﻥ ﺭﺍ ﺑﻪ ﺳـﻮی ﺧﺪﺍ ﺳـﻴﺮ ﻣﯽ ﺩﻫـﺪ ﻭ ﺩﻋـﺎی ﺍﻭ ﺭﺍ ﺩﺭ ﻣﻨﺰﻟﮕﺎﻩ ﻗﺮﺏ ﺍﻟﻬﯽ ﺑﻪ ﻣﺤﻀﺮ ﺧﺪﺍﻭﻧﺪ ﻋﺮﺿﻪ ﻣﯽ ﺩﺍﺭﺩ.

ﺍﺯ ﻣﺎ ﺧﻮﺍﺳﺘﻪ ﺷﺪﻩ ﻛﻪ ﻫﻤﻴﺸﻪ ﭼﻨﻴﻦ ﻧﻤﺎﺯی ﺑﺨﻮﺍﻧﻴﻢ ﻟﻴﻜﻦ ﭼﻮﻥ ﻣـﺎ ﻫﻤﻮﺍﺭﻩ ﺁﺭﺯﻭﻫﺎی ﺑﺰﺭگ ﻧﺪﺍﺭﻳﻢ، ﻧﻤﺎﺯ ﺧﻮﺩ ﺭﺍ ﺍﺻﻼﺡ ﻧﻤﯽ ﻛﻨﻴﻢ. ﺍﮔﺮ ﻛﺴـﯽ ﺑﺮﺍی ﻫـﺮ ﻧﻤﺎﺯﺵ ﻳﮏ ﺁﺭﺯﻭی ﺑﺰﺭگ ﺩﺍﺷـﺘﻪ ﺑﺎﺷـﺪ، ﻫﻤﻪ ﻧﻤﺎﺯﻫﺎﻳﺶ ﺑﺎﺗﻮﺟﻪ ﻭ ﺧﺎﻟﺼﺎﻧﻪ ﻣﯽ ﺷﻮﺩ.

ﮔﻔﺘﻴﻢ ﻛﻪ ﺑﻨﺪﻩ ﻧﻴﺎﺯﻣﻨﺪ ﻧﻤﺎﺯﺵ ﺑﺎ ﻋﺠﺰ ﻭ ﻻﺑﻪ ﻭ ﺍﻟﺘﻤﺎﺱ ﺻﻮﺭﺕ ﻣﯽ ﭘﺬﻳﺮﺩ. ﺩﺭ ﻧﻤﺎﺯ ﭼﻨﻴﻦ ﺑﻨﺪﻩ ﺍی ﺑﻪ ﭘﺎﺧﺎﺳﺘﻦ، ﺗﻌﻈﻴﻢ ﻛﺮﺩﻥ، ﺑﻪ ﺳﺠﺪﻩ ﺍﻓﺘﺎﺩﻥ ﻭ ﺩﺳـﺖ ﺑﻪ ﺩﻋﺎ ﺑﺮﺩﺍﺷـﺘﻦ ﻫﻤﺮﺍﻩ ﺑﺎ ﺭﻭﺡ ﻭ ﺗﻮﺟﻪ ﻗﻠﺒﯽ ﺍﺳﺖ ﻭ ﭼﻨﻴﻦ ﺷﺨﺼﯽ ﻃﯽ ﻧﻤﺎﺯ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﻧﺰﺩﻳﮏ ﻣﯽ ﺷﻮﺩ ﻭ ﻻﻳﻖ ﻣﻨﺎﺟﺎﺕ ﺑـﺎ ﺍﻭ ﻣﯽ ﮔـﺮﺩﺩ. ﺍﮔـﺮ ﻓـﺮﺝ ﻭ ﺭﻓﻊ ﮔﺮﻓﺘـﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﻏﺪﻏـﻪ ﻭ ﻧﻴـﺎﺯ ﻫـﺮ ﺭﻭﺯ ﻣﺎ ﺑﺎﺷـﺪ ﻭ ﺍﮔﺮ ﺑﺎﻭﺭ ﺩﺍﺷـﺘﻪ ﺑﺎﺷـﻴﻢ ﻛﻪ ﻓﺮﺝ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺩﺳـﺖ ﺧﺪﺍﺳـﺖ، ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﺍﺯ ﻫﺮ ﻓﺮﺻﺘﯽ ﺳـﻮﺩ ﺧﻮﺍﻫﻴﻢ ﺟﺴـﺖ ﻛﻪ ﻧﻤـﺎﺯ ﺑﺎ ﺗﻮﺟﻬﯽ ﺑﺨﻮﺍﻧﻴﻢ، ﺑـﻪ ﺩﺭﮔﺎﻩ ﺧﺪﺍ ﻋﺮﺽ ﺑﻨﺪﮔﯽ، ﺧﻀﻮﻉ ﻭ ﺧﺸـﻮﻉ ﻛﻨﻴﻢ، ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻭ ﺑﺎ ﺍﺩﺏ ﺑﺎﻳﺴﺘﻴﻢ، ﺑﺎ ﺍﺩﺏ ﺗﻌﻈﻴـﻢ ﻛﻨﻴﻢ ﻭ ﺑﺎ ﺍﻟﺘﻤﺎﺱ ﺑﻪ ﺧـﺎک ﺑﻴﻔﺘﻴﻢ ﻭ ﺁﻥ ﻗﺪﺭ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺗﻜﺮﺍﺭ ﻛﻨﻴﻢ ﻛﻪ ﻣﻮﺭﺩ ﭘﺴﻨﺪ ﺍﻭ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻳﻢ. ﺁﻧﮕﺎﻩ ﺩﺳﺘﯽ ﺑﻪ ﻧﻴﺎﻳﺶ ﺑﻠﻨﺪ ﻛﻨﻴﻢ ﻭ ﺍﺯ ﺍﻭ ﺳـﻼﻣﺖ ﻭ ﻓـﺮﺝ ﺍﻣﺎﻡ ﺯﻣﺎﻧﻤـﺎﻥ ﺭﺍ ﺑﺨﻮﺍﻫﻴﻢ. ﻻﺯﻣـﻪ ﺗﻤﺎﻡ ﺍﻳﻦ ﺣﺎﻻﺕ ﻧﻴﺰ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺭﺷﺘﻪ ﻣﺤﺒﺘﻤﺎﻥ ﺑﺎ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻣﺤﻜﻢ ﻭ ﻏﺮﺑﺖ ﺍﻳﺸـﺎﻥ ﺑـﺮﺍی ﻣـﺎ ﺩﺭﺩﺁﻭﺭ ﺷـﺪﻩ ﺑﺎﺷـﺪ. ﺁﻥ ﻭﻗـﺖ ﺩﻝ ﻣﺎ ﺁﺗـﺶ ﺧﻮﺍﻫﺪ ﮔﺮﻓﺖ ﻭ ﺭﺍﺯ ﻭ ﻧﻴﺎﺯ ﻣﺎ ﺣﻘﻴﻘﯽ ﻭ ﺍﺯ ﺳـﺮ ﺳـﻮﺯ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺩﻋﺎﻳﯽ ﻛﻪ ﻫﻤﺮﺍﻩ ﺑﺎ ﭼﻨﻴﻦ ﺳﻮﺯ ﻭ ﮔﺪﺍﺯی ﺑﺎﺷﺪ، ﺍﺛﺮ ﺧﻮﺍﻫﺪ ﻛﺮﺩ.

ﺩﻻ ﺑﺴﻮﺯ ﻛﻪ ﺳﻮﺯ ﺗﻮ ﻛﺎﺭﻫﺎ ﺑﻜﻨﺪ ﺩﻋﺎی ﻧﻴﻤﻪ ﺷﺒﯽ ﺩﻓﻊ ﺻﺪ ﺑﻼ ﺑﻜﻨﺪ

ﻳـﺎﺭی ﺍﻣﺎﻡ ﻋﺼـﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺗﻨﻬﺎ ﻣﺤـﺪﻭﺩ ﺑﻪ ﺍﻳﻦ ﻧﻴﺴـﺖ ﻛﻪ ﺷـﺨﺺ، ﺗﻮﺍﻥ ﻭ ﺗﻮﺷـﻪ ﺧـﻮﺩﺵ ﺭﺍ ﺑﺮﺍی ﻳـﺎﺭی ﺁﻥ ﺣﻀﺮﺕ ﺻﺮﻑ ﻛﻨـﺪ، ﺑﻠﻜﻪ ﺍﮔﺮ ﻛﺴـﯽ ﺗﻮﺍﻥ ﻭ ﻗـﺪﺭﺕ ﺧﺪﺍی ﺑﺰﺭگ ﺭﺍ ﺑـﻪ ﻳﺎﺭی ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻓـﺮﺍ ﺑﺨﻮﺍﻧﺪ ﻭ ﺑﺎ ﺩﻋـﺎی ﺧﺎﻟﺼﺎﻧﻪ ﺧﻮﻳﺶ ﺍﺯ ﺧـﺪﺍ ﺑﺨﻮﺍﻫﺪ ﻛﻪ ﻣﺸـﻜﻼﺕ ﺁﻥ ﺣﻀﺮﺕ ﻭ ﺩﻭﺳـﺘﺎﻥ ﺍﻭ ﺭﺍ ﺑﻪ ﻓﻀـﻞ ﻭ ﻛﺮﻣﺶ ﺑﺮﻃﺮﻑ ﺳـﺎﺯﺩ، ﺍﻳﻦ ﺷـﺨﺺ ﻧﻴﺰ ﺑﺎﻋﺚ ﺍﻓﺰﺍﻳﺶ ﻗﺪﺭﺕ ﻭ ﻋﺰﺕ ﺁﻥ ﻣﻮﻻ ﺷﺪﻩ ﻭ ﺍﺯ ﻣﺸـﻜﻼﺕ ﺁﻥ ﺣﻀﺮﺕ ﻛﺎﺳﺘﻪ ﺍﺳﺖ. ﺑﻪ ﭼﻨﻴﻦ ﺷﺨﺼﯽ ﻧﻴﺰ ﻳﺎﻭﺭ ﺍﻣﺎﻡ ﻋﺼﺮ ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ.

ﭘﺲ ﺧﻼﺻﻪ ی ﺍﻳﻦ ﻣﻘﺎﻝ ﺁﻥ ﺑﻮﺩ ﻛﻪ:

ﺍﻭﻻ ﺩﻋـﺎی ﺑﺮ ﺳـﻼﻣﺖ ﻭ ﻓﺮﺝ ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﻧﺴـﺒﺖ ﺑﻪ ﺩﻋﺎی ﺑﺮ ﺧﻮﺩ ﺗﺮﺟﻴﺢ ﺩﻫﻴﻢ ﻭ ﺑﺮﺍی ﺁﻥ ﺍﻫﻤﻴﺖ ﻗﺎﺋﻞ ﺷﻮﻳﻢ.

ﺛﺎﻧﻴﺎ ﺑﺪﺍﻧﻴﻢ ﻛﺴـﯽ ﻛﻪ ﺑﺮﺍی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺧﺎﻟﺼﺎﻧﻪ ﺩﻋﺎ ﻣﯽ ﻛﻨﺪ، ﺍﻳﻦ ﻛﺎﺭﺵ ﺍﺯ ﻧﻈﺮ ﺁﻥ ﺟﻮﺍﻧﻤﺮﺩ ﭘﻮﺷـﻴﺪﻩ ﻧﻤﯽ ﻣﺎﻧﺪ ﻭ ﺑﯽ ﺗﺮﺩﻳﺪ ﭘﺎﺳﺦ ﭘﺮﻣﺤﺒﺖ ﺍﻭ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﺧﻮﺍﻫﺪ ﻛﺮﺩ.

ﺛﺎﻟﺜـﺎ ﺑـﺮﺍی ﻛﺎﺭﺳـﺎﺯ ﺷـﺪﻥ ﺩﻋﺎﻫﺎﻳﻤـﺎﻥ ﺯﻣﺎﻥﻫـﺎ، ﻣﻜﺎﻥﻫـﺎ ﻭ ﻣﻮﻗﻌﻴﺖﻫـﺎی ﻭﻳـﮋﻩ ﺭﺍ ﺍﺯ ﺩﺳـﺖ ﻧﺪﻫﻴﻢ ﻭ ﺑﻪ ﺧﺼـﻮﺹ ﺯﻣﺎﻥﻫﺎ ﻭ ﻣﻜﺎﻥﻫـﺎی ﻣﺮﺑﻮﻁ ﺑﻪ ﺣﻀـﺮﺕ ﺍﺑﺎﻋﺒﺪﺍﻟﻠﻪ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭ ﻧﻴﺰ ﻧﻤﺎﺯ ﺭﺍ ﻫﻤﻮﺍﺭﻩ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.
