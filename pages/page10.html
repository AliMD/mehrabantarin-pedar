ﺁﻥ ﻭﻗﺖ ﻏﺼﻪ ﺩﺍﺭ ﻭ ﻣﻬﻤﻮﻡ ﻭ ﻏﻢﺯﺩﻩ ﻭ ﮔﺮﻳﺎﻥ ﻣﯽ ﺷﻮﻳﺪ؟

ﺑﺒﻴﻨﻴﺪ! ﺍﮔﺮ ﺍﻳﻦ ﺣﺎﻟﺖ ﺩﺭ ﺷـﻤﺎ ﻧﻴﺴـﺖ ﻭ ﺳـﺎﻝ ﻭ ﻣـﺎﻩ ﻭ ﻫﻔﺘﻪ ﻭ ﺭﻭﺯﻫـﺎ ﻣﯽ ﮔـﺬﺭﺩ ﻭ ﺍﺻـﻼ ﺑﻪ ﻓﻜﺮ ﺍﻭ ﻧﻴﺴـﺘﻴﺪ ﺗﺎ ﭼﻪ ﺑﺮﺳـﺪ ﺑﻪ ﺍﻳﻨﻜﻪ ﻏﺼﻪ ﺍﺵ ﺭﺍ ﺑﺨﻮﺭﻳﺪ ﻭ ﺍﺷﮏ ﺑﺮﻳﺰﻳﺪ ﻭ ﺩﺭ ﻓﺮﺍﻗﺶ ﮔﺮﻳﻪ ﻛﻨﻴﺪ ﭘﺲ ﺑﻪ ﺍﻭ ﻣﺤﺒﺖ ﻧﺪﺍﺭﻳﺪ ﻭ ﺍﮔﺮ ﺍﺩﻋﺎی ﻣﺤﺒﺖ ﻛﻨﻴﺪ، ﺩﺭﻭﻍ ﮔﻔﺘﻪ ﺍﻳﺪ... ﺑﺎ ﺍﻳﻦ ﻧﺸﺎﻧﻪ ، ﺣﺴﺎﺏ ﻛﺎﺭ ﺧﻮﺩ ﺭﺍ ﺑﻜﻨﻴﺪ.»

ﺁﻥ ﺩﻟﺴـﻮﺧﺘﻪ ﻣﯽ ﻓﺮﻣﻮﺩ: «ﻳﻜﯽ ﺍﺯ ﺁﺛﺎﺭ ﻣﺤﺒﺖ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺁﺩﻡ ﺩﺍﺋـﻢ ﺑﻪ ﻳﺎﺩ ﻣﺤﺒﻮﺑﺶ ﺍﺳـﺖ ﻭ ﺩﻟﺶ ﻣﯽ ﺧﻮﺍﻫﺪ ﺍﺳـﻢ ﺍﻭ ﺭﺍ ﺑﺒﺮﺩ ﻭ ﺍﺯ ﺍﻭ ﻳﺎﺩ ﺷـﻮﺩ ﻭ ﺳـﺨﻦ ﮔﻔﺘﻪ ﺷـﻮﺩ. ﺣﻀﺮﺕ ﻳﻌﻘﻮﺏ ﻛﻪ ﺑﻪ ﻳﻮﺳـﻒ ﻣﺤﺒﺖ ﺩﺍﺷـﺖ - ﺩﻟﺶ ﻣﯽ ﺧﻮﺍﺳﺖ ﺩﺍﺋﻤﺎ ﺍﺳﻢ ﻳﻮﺳﻒ ﺑﺮﺩﻩ ﺷﻮﺩ؛ ﺍﺯ ﺩﺭ ﻭ ﺩﻳﻮﺍﺭ، ﺯﻥ ﻭ ﻣﺮﺩ، ﺩﻭﺳـﺖ ﻭ ﺩﺷـﻤﻦ، ﺧﻮﺩی ﻭ ﺑﻴﮕﺎﻧﻪ ، ﻧﺸـﺎﻥ ﻭ ﺍﺳـﻢ ﻳﻮﺳـﻒ ﺑﻴﺎﻳﺪ. ﺯﻟﻴﺨﺎ ﻧﻴﺰ ﻛﻪ ﻳﻮﺳـﻒ ﺭﺍ ﺩﻭﺳـﺖ ﺩﺍﺷـﺖ ﻣﯽ ﺧﻮﺍﺳـﺖ ﺩﺍﺋﻤـﺎ ﺑﺤﺚ ﻳﻮﺳـﻒ ﭘﻴﺶ ﺁﻳﺪ. ﺷـﻤﺎ ﻧﻴﺰ ﻛـﻪ ﺑﭽﻪ ﺧﻮﺩ ﺭﺍ ﺩﻭﺳـﺖ ﻣﯽ ﺩﺍﺭﻳـﺪ ﺩﻟﺘﺎﻥ ﻣﯽ ﺧﻮﺍﻫﺪ ﻫﻤﻴﺸـﻪ ﺻﺤﺒـﺖ ﺍﺯ ﺍﻭ ﭘﻴﺶ ﺁﻳﺪ. ﻣﺨﺼﻮﺻﺎ ﺍﮔﺮ ﺍﻭ ﺩﻛﺘﺮ ﻭ ﻣﻬﻨﺪﺱ ﻫﻢ ﺷـﺪﻩ ﺑﺎﺷـﺪ، ﺩﺍﺋﻤﺎ ﺩﻟﺘﺎﻥ ﻣﯽ ﺧﻮﺍﻫـﺪ ﺑﮕﻮﻳﻴـﺪ: ﺁﻗﺎی ﻣﻬﻨـﺪﺱ ﭼﻨﻴﻦ ﻛﺮﺩﻧﺪ؛ ﺁﻗـﺎی ﺩﻛﺘﺮ ﭼﻨﺎﻥ ﮔﻔﺘﻨﺪ! ﺍﻧﺴـﺎﻥ ﺩﻟـﺶ ﻣﯽ ﺧﻮﺍﻫﺪ ﻣﺤﺒﻮﺑﺶ ﺩﺭ ﻣﻴﺎﻥ ﻣـﺮﺩﻡ، ﻣﺤﺘﺮﻡ ﻭ ﻣﻌـﺰﺯ ﻭ ﻣﻌﻈﻢ ﺑﺎﺷـﺪ ﻭ ﺟﻠﻮﻩ ﺍﺵ ﺩﺍﺋﻤﺎ ﺯﻳﺎﺩ ﺑﺸـﻮﺩ. ﺷـﻤﺎ ﺭﺍ ﺑﻪ ﺧﺪﺍ ﻗﺴﻢ ﺁﻳﺎ ﻧﺴﺒﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺘﺎﻥ ﻫﻤﻴﻦﻃﻮﺭ ﻫﺴﺘﻴﺪ؟

ﺍﮔﺮ ﺍﻭ ﺭﺍ ﺩﻭﺳـﺖ ﺑﺪﺍﺭﻳﺪ، ﺩﻟﺘﺎﻥ ﻣﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺋﻤﺎ ﺍﺳﻢ ﺣﻀﺮﺕ ﺑـﺮﺩﻩ ﺷـﻮﺩ؛ ﺟـﻼﻝ ﻭ ﺷـﻮﻛﺖ ﻭ ﻋﻈﻤﺘﺶ ﺑﻪ ﮔـﻮﺵ ﻣﺮﺩﻡ ﺑﺮﺳـﺪ ﺗﺎ ﻣﺮﺩﻡ ﺑﻪ ﺍﻭ ﻋﻼﻗﻪ ﻣﻨﺪ ﺑﺸـﻮﻧﺪ؛ ﻣﻬﺮﺑﺎﻥ ﺑﺸـﻮﻧﺪ ﻭ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ، ﺗﺮﻭﻳﺞ ﺁﻥ ﺣﻀﺮﺕ ﻣﯽ ﻛﻨﻴﺪ ﻛﻪ ﭼﻨﻴﻦ ﺑﺸﻮﺩ.

ﺗﺮﻭﻳﺞ ﺣﻀﺮﺕ ﻫﻢ ﻣﻨﺤﺼﺮ ﺑﻪ ﺯﺑﺎﻥ ﻭ ﻗﻠﻢ ﻧﻴﺴـﺖ؛ ﭘﻮﻝ ﺍﺯ ﻫﻤﻪ ﺑﺎﻻﺗـﺮ ﺍﺳـﺖ. ﺭﻭﺯﻫﺎی ﺟﻤﻌﻪ ، ﺳـﻔﺮﻩ ﭘﻬـﻦ ﻛﻨﻴﺪ؛ ﺑﻪ ﻧـﺎﻡ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ. ﭼﻨﺪ ﺗﺎ ﺍﺯ ﺍﻫﻞ ﻋﻠﻢ، ﻣﺨﺼﻮﺻﺎ ﺳﺎﺩﺍﺕ ﻋﻠﻤﺎ ﺭﺍ ﺗﺤﺖ ﻋﻨﻮﺍﻥ ﺩﻋﺎی ﻧﺪﺑﻪ ﺩﻋﻮﺕ ﻛﻨﻴﺪ. ﺷﻴﺮ ﻭ ﭼﺎی ﻣﻔﺼﻠﯽ ﺑﺪﻫﻴﺪ ﻭ ﺑﻌﺪ ﺍﺯ ﺩﻋﺎ، ﭼﺎﺷـﺘﯽ ﺑﺪﻫﻴﺪ ﺗﺎ ﻳﻜﯽ ﺩﻭ ﺳـﺎﻋﺖ ﺩﻳﮕﺮ ﺑﺤﺚ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﺸﻮﺩ. ﺑﻌﺪ ﻧﺎﻫﺎﺭ ﺑﺪﻫﻴﺪ. ﺍﻳﻦ ﺧﻮﺩ، ﺗﻌﻈﻴﻢ ﻭ ﺗﺠﻠﻴﻞ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻭ ﺗﺸﻮﻳﻖ ﻣﺮﺩﻡ ﺑﻪ ﺳﺎﺣﺖ ﻭﻻﻳﺖ ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭ ﺍﺳﺖ.

ﺁﻳﺎ ﻫﻴـﭻ ﻭﻗﺖ ﺍﻳﻦﻃﻮﺭ ﺑﻮﺩﻩ ﺍﻳـﺪ؟ ﺁﻥ ﻃﻮﺭ ﻛﻪ ﺩﻟﺘـﺎﻥ ﻣﯽ ﺧﻮﺍﻫﺪ ﻓﺮﺯﻧﺪﺗـﺎﻥ ، ﺑﺰﺭگ ﻭ ﻣﻌﺮﻭﻑ ﻭ ﻣﻮﺻﻮﻑ ﺑﺸـﻮﺩ ﻭ ﻣـﺮﺩﻡ ﺑﻪ ﺍﻭ ﺗﻮﺟﻪ ﭘﻴـﺪﺍ ﻛﻨﻨﺪ، ﻳﮏﺩﻫﻢ ﻳﺎ ﻳﮏﭘﻨﺠﺎﻫﻢ ﺍﻳﻦ، ﺩﺭ ﻓﻜـﺮ ﺗﺮﻭﻳﺞ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺘﺎﻥ ﺑﻮﺩﻩ ﺍﻳﺪ؟ ﺍﮔﺮ ﺑﻮﺩﻩ ﺍﻳﺪ، ﺍﺯ ﺷـﻤﺎ ﻣﻠﺘﻤﺲ ﺩﻋﺎﻳﻢ ﻭ ﺍﮔﺮ ﻧﺒﻮﺩﻩ ﺍﻳﺪ، ﺑﺮﻭﻳﺪ ﺑﺮﺍی ﺧﻮﺩ ﻓﻜﺮی ﺑﻜﻨﻴﺪ.

ﺭﺍﺑﻄـﻪ ﻭ ﭘﻴﻮﻧـﺪ ﻛـﻪ ﺑﻪ ﻟﻔﻆ ﻧﻤﯽ ﺷـﻮﺩ ﻛﻪ ﺻﺮﻓـﺎ ﺑﻌـﺪ ﺍﺯ ﻧﻤﺎﺯ ﺑﺎ ﺍﻧﮕﺸـﺖ، ﺳـﻪ ﺗﺎ ﺍﺷﺎﺭﻩ ﻳﻜﯽ ﺑﻪ ﺍﻣﺎﻡ ﺣﺴﻴﻦ، ﻳﻜﯽ ﺑﻪ ﺍﻣﺎﻡ ﺭﺿﺎ ﻭ ﻳﻜﯽ ﺑﻪ ﺻﺎﺣﺐ ﺍﻟﺰﻣﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴﻼﻡ ﺑﻜﻨﻴﺪ ﻭ ﺗﻤﺎﻡ ﺷﻮﺩ. ﺑﺎﻳﺪ ﺩﻝ ﻣﺘﻮﺟﻪ ﺑﺎﺷـﺪ. ﺍﮔـﺮ ﺩﻟﺖ ﺑﻪ ﻃﺮﻑ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺭﻓﺘﻪ ﻭ ﭘﻴﻮﻧﺪ ﺧـﻮﺭﺩﻩ ﻛﻪ ﻫﻴﭻ، ﻭ ﺍﮔـﺮ ﭘﻴﻮﻧﺪ ﻧﺨﻮﺭﺩﻩ، ﺑﺮﻭ ﺑﺮﺍی ﺧﻮﺩﺕ ﻓﻜـﺮی ﺑﻜﻦ. ﺍﮔﺮ ﻣﯽ ﮔﻮﻳﯽ ﻗﻠﺒـﺖ ﺑـﺎ ﺁﻥ ﺣﻀـﺮﺕ ﭘﻴﻮﻧﺪ ﺧـﻮﺭﺩﻩ ﺍﺳـﺖ، ﻧﺸـﺎﻧﻪ ﺍﺵ ﻭ ﻣﻴﻮﻩ ﻫﺎی ﺷـﻴﺮﻳﻨﺶ ﻛـﻮ؟... ﻛﺠﺎﺳـﺖ ﺁﻥ ﻣﻴﻮﻩ ﺷـﻴﺮﻳﻨﯽ ﻛﻪ ﺑﺮ ﺍﺛـﺮ ﭘﻴﻮﻧﺪ ﺗﻮ ﺑﻪ ﺷـﺠﺮﻩ ﻻﻫﻮﺗﯽ ﺍﻣـﺎﻡ ﻋﺼﺮ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻩ ﺑﺎﺷـﺪ؟ ﺁﻥ ﺭﺍ ﻧﺸـﺎﻥ ﺑﺪﻩ! ﺍﺯ ﻛﺠﺎی ﺩﺭ ﻭ ﺩﻳﻮﺍﺭ ﻭﺟﻮﺩﺕ، ﺗﺠﻠﯽ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻧﻤﺎﻳﺎﻥ ﺷـﺪﻩ ﺍﺳـﺖ؟ ﺍﺯ ﭘﻮﻟﺖ؟ ﺍﺯ ﻗﺪﻣـﺖ؟ ﺍﺯ ﺯﺑﺎﻧﺖ؟ ﺍﺯ ﻛﺠـﺎ؟ ﺍﮔﺮ ﭘﻴﻮﻧﺪ ﻧﮕﺮﻓﺘﻪ ﺍﺳﺖ، ﺩﺭ ﺭﻓﺘﺎﺭﺕ ﺗﺠﺪﻳﺪ ﻧﻈﺮ ﻛﻦ.

ﺁﺭی ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣﺎﻣـﯽ -ﻛـﻪ ﻋﻤـﻮﻡ ﺍﻧﺴـﺎﻥ ﻫﺎ ﻧﺴـﺒﺖ ﺑـﻪ ﺍﻭ ﻭ ﺑﺰﺭﮔـﯽ ﺍﺵ ﻏﺎﻓﻠﻨـﺪ- ﻳﻜـﯽ ﺍﺯ ﻭﻇﺎﻳـﻒ ﺩﻭﺳـﺘﺪﺍﺭﺍﻥ ﻭ ﻫـﻮﺍﺩﺍﺭﺍﻥ ﺁﻥ ﺣﻀـﺮﺕ ﺍﺳـﺖ ﻭ ﻫـﺮ ﻳﮏ ﺍﺯ ﻣﺎ ﺑﺎﻳـﺪ ﻭﻇﻴﻔﻪ ﺧﻮﺩ ﺑﺪﺍﻧـﺪ ﺗﺎ ﺑﻪ ﻫﺮ ﻭﺟـﻪ ﻣﻤﻜـﻦ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺍﺯ ﺧﻮﺩ ﺣﺮﻛﺘﯽ ﻧﺸـﺎﻥ ﺩﻫـﺪ. ﺑﺪﺍﻧﻴﻢ ﻛﻪ ﺍﮔﺮ ﺑـﻪ ﺍﻳﻦ ﻛﺎﺭ ﺗﻮﻓﻴﻖ ﻳﺎﻓﺘﻴﻢ ﻭ ﻧﺎﻣﻤﺎﻥ ﺩﺭ ﺷـﻤﺎﺭ ﻳـﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭ ﺧﺪﻣﺖﮔﺰﺍﺭﺍﻥ ﺍﻭ ﺛﺒﺖ ﺷـﺪ، ﺩﺭ ﺩﻧﻴﺎ ﻭ ﺁﺧﺮﺕ ﻧﺰﺩ ﺭﺳـﻮﻝ ﺧﺪﺍ ﺭﻭﺳﻔﻴﺪ ﺧﻮﺍﻫﻴﻢ ﺑﻮﺩ ﻭ ﻣﻮﺭﺩ ﻋﻨﺎﻳﺖ ﺧﺎﺹ ﺧﺪﺍﻭﻧﺪ ﺗﻌﺎﻟﯽ ﻗﺮﺍﺭ ﺧﻮﺍﻫﻴﻢ ﮔﺮﻓﺖ.

ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺳـﺎﺩﻩ ﺗﺮﻳﻦ ﺷـﻜﻞﻫﺎ ﺍﻧﺠﺎﻡ ﺩﺍﺩ. ﻣﺜـﻼ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺩﺭ ﺍﺗـﺎﻕ ﻛﺎﺭ ﻳـﺎ ﭘﺬﻳﺮﺍﻳـﯽ ﻣﻨـﺰﻝ ﻳـﺎ ﺣﺘـﯽ ﺍﺗﺎﻕ ﺧﺼﻮﺻﯽ ﻣـﺎﻥ ﺗﺎﺑﻠﻮﻳﯽ ﻛﻪ ﺣـﺎﻭی ﻳﻜﯽ ﺍﺯ ﻧﺎﻡﻫﺎی ﺯﻳﺒﺎی ﺍﻭ ﺑﺎﺷـﺪ، ﻧﺼـﺐ ﻛﻨﻴـﻢ ﻭ ﺑﺎ ﺍﻳﻦ ﻛﺎﺭ ﻧﺸـﺎﻥ ﺩﻫﻴـﻢ ﻛﻪ ﻋﺰﻳﺰﺗﺮﻳﻦ ﻛﺴـﯽ ﻛﻪ ﺑﺮﺍی ﺍﻫـﻞ ﺍﻳﻦ ﺧﺎﻧـﻪ ﻳﺎ ﺍﻳﻦ ﺍﺗﺎﻕ، ﻭﺟـﻮﺩ ﺩﺍﺭﺩ، ﺻﺎﺣﺐ ﺍﻟﺰﻣﺎﻥ ﺍﺳـﺖ. ﺑﻪ ﺭﺥ ﺧﻮﺩﻣﺎﻥ ﻭ ﺩﻳﮕﺮﺍﻥ ﺑﻜﺸـﻴﻢ ﻛﻪ ﺍﻭ ﺑﻪ ﻛﻠﯽ ﻏﺮﻳﺐ ﻧﻴﺴـﺖ، ﻳﺎﻭﺭﺍﻥ ﻭ ﻫﻮﺍﺩﺍﺭﺍﻧـﯽ ﻫﻢ ﺩﺍﺭﺩ. ﺁﺭی، ﻫﺮ ﺳـﻠﻄﺎﻥ ﺳﭙﻪ ﺳـﺎﻻﺭ ﺩﺭ ﺳـﭙﺎﻩ ﺧﻮﺩ ﻫﻢ ﺳـﺮﺩﺍﺭ ﺩﺍﺭﺩ ﻭ ﻫﻢ ﺳـﻴﺎﻫﯽ ﻟﺸﮕﺮ. ﻣﺎ ﺍﮔﺮ ﻻﻳﻖ ﺳﺮﺩﺍﺭی ﺑﺮﺍی ﺍﻭ ﻧﺒﺎﺷﻴﻢ ﻻﺍﻗﻞ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﻛﺎﺭی ﻛﻨﻴﻢ ﻛﻪ ﻫﻮﺍﺩﺍﺭ ﻭ ﺳﻴﺎﻫﯽ ﻟﺸﮕﺮ ﺍﻭ ﺑﻪ ﺷﻤﺎﺭ ﺁﻳﻴﻢ.

ﺭﻭﺯی ﺑﻪ ﻣﻨﺎﺳـﺒﺘﯽ ﺩﺭ ﺍﺗـﺎﻕ ﺧﺼﻮﺻﯽ ﻳﻜﯽ ﺍﺯ ﺩﻭﺳـﺘﺎﻧﻢ ﻭﺍﺭﺩ ﺷـﺪﻡ. ﺩﻳﺪﻡ ﺑﺎﻻی ﺳـﺮﺵ ﺗﺎﺑﻠﻮﻳـﯽ ﺯﻳﺒﺎ ﻧﺼﺐ ﻛﺮﺩﻩ ﻭ ﻧـﺎﻡ ﺩﻝﺍﻧﮕﻴﺰ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻳﻌﻨﯽ «ﻗﺎﺋـﻢ ﺁﻝ ﻣﺤﻤﺪ» ﺩﺭ ﺁﻥ ﻗـﺎﺏ ﺑﻪ ﺧﻂ ﺧﻮﺵ ﻧﻮﺷـﺘﻪ ﺷﺪﻩ ﺍﺳـﺖ. ﻣﻦ ﺍﻟﺒﺘﻪ ﺍﺯ ﺍﻳﻦ ﺩﺳـﺖ ﺗﺎﺑﻠﻮﻫﺎ ﺯﻳﺎﺩ ﺩﻳﺪﻩ ﺑـﻮﺩﻡ ﻭﻟﯽ ﺩﺭ ﺁﻧﺠﺎ ﺩﻳﺪﻡ ﻛﻪ ﺁﻥ ﺩﻭﺳـﺖ ﻋﺰﻳﺰ، ﺧﻮﺩﺵ ﻫﻢ ﻳﮏ ﺑﻴﺖ ﺷـﻌﺮ ﻧﻮﺷـﺘﻪ ﻭ ﺁﻥ ﺭﺍ ﭘﺎﺋﻴﻦ ﻧﺎﻡ ﺍﻣﺎﻡ ﻋﺼﺮ ﻛﻪ ﺟﺎﻥ ﻫﺎﻳﻤﺎﻥ ﻓﺪﺍﻳﺶ ﺑﺎﺩ ﭼﺴـﺒﺎﻧﺪﻩ ﺍﺳـﺖ. ﺷـﻌﺮ ﺭﺍ ﻛﻪ ﺧﻮﺍﻧﺪﻡ ﻣﻨﻘﻠـﺐ ﮔﺮﺩﻳـﺪﻡ ﻭ ﺍﻛﻨﻮﻥ ﻛﻪ ﺳﺎﻝﻫﺎ ﺍﺯ ﺁﻥ ﻣﺎﺟﺮﺍ ﻣﯽ ﮔﺬﺭﺩ، ﻫﻨﻮﺯ ﺁﻥ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺭﻡ ﻭ ﻫﺮ ﺑﺎﺭ ﻛﻪ ﺁﻥ ﺻﺤﻨﻪ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﻣﯽ ﺁﻭﺭﻡ، ﺩﻟﻢ ﻣﯽ ﻟﺮﺯﺩ. ﺁﻥ ﺷﻌﺮ ﭼﻨﻴﻦ ﺑﻮﺩ:

ﺍﻭ ﺑﺎ ﺍﻳﻦ ﺷـﻌﺮ ﭘﺮﻣﺤﺘﻮﺍ ﺗﻮﺍﻧﺴـﺘﻪ ﺑﻮﺩ ﺑﻪ ﺧﻮﺩﺵ ﻭ ﺑﻪ ﻣﻦ ﺑﮕﻮﻳﺪ ﻛﻪ ﻣﻦ ﺍﻳﻦ ﺗﺎﺑﻠﻮ ﺭﺍ ﻧﻪ ﺑﺪﻭﻥ ﻫﺪﻑ ﻭ ﺍﺯ ﺳﺮ ﻳﮏ ﻋﺎﺩﺕ ﻓﺮﻫﻨﮕﯽ ﺑﺎﻻی ﺳﺮ ﺧﻮﺩﻡ ﻧﺼﺐ ﻛﺮﺩﻩ ﺍﻡ، ﺑﻠﻜﻪ ﺍﺯ ﺍﻳﻦ ﻛﺎﺭ ﻫﺪﻑ ﻭ ﭘﻴﺎﻣﯽ ﺩﺍﺷﺘﻪ ﺍﻡ. ﭘﻴـﺎﻡ ﻣـﻦ ﺍﻳﻦ ﺍﺳـﺖ ﻛـﻪ ﻣﻦ ﺳـﺮ ﻭ ﺟﺎﻥ ﺧـﻮﺩ ﺭﺍ ﺍﺭﺯﺍﻧـﯽ ﻗﺪﻭﻡ ﻣﺒﺎﺭک ﻣﻬﺪی ﻓﺎﻃﻤﻪ ﻭ ﻗﺎﺋﻢ ﺁﻝﻣﺤﻤﺪ ﻣﯽ ﺩﺍﻧﻢ ﻭ ﺑﻪ ﺍﻳﻦ ﻋﻘﻴﺪﻩ ﺍﻓﺘﺨﺎﺭ ﻣﯽ ﻛﻨﻢ.

ﺁﺭی ﭼﻪ ﺧﻮﺏ ﺍﺳـﺖ ﻛﻪ ﺍﻧﺴـﺎﻥ ﻫﺮﻛﺎﺭی ﺭﺍ ﻛﻪ ﺍﻧﺠﺎﻡ ﻣﯽ ﺩﻫﺪ، ﻧﻪ ﺍﺯ ﺭﻭی ﻋﺎﺩﺕ ﺑﻠﻜﻪ ﻫﻤﺮﺍﻩ ﺑﺎ ﻗﺼﺪ ﻭ ﻧﻴﺖ ﺑﺎﺷـﺪ. ﮔﺎﻫﯽ ﭼﺴﺒﺎﻧﺪﻥ ﭘﻮﺳـﺘﺮ ﻭ ﻧﺼـﺐ ﺗﺎﺑﻠـﻮ ﻭ ﻛﺎﺭﻫﺎﻳـﯽ ﺍﺯ ﺍﻳـﻦ ﻗﺒﻴﻞ ﻫﻢ ﺑـﺮﺍی ﻣﺎ ﻋﺎﺩﺕ ﻣﯽ ﺷـﻮﺩ ﻭ ﭼﻪ ﺧﻮﺏ ﻛـﻪ ﻫﻤﻴﻦ ﻛﺎﺭﻫﺎی ﻧﻴﮏ ﺭﺍ ﻧﻴـﺰ ﺑﺎ ﻧﻴﺖ ﻭ ﻗﺼﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ.

ﺳـﺨﻦ ﺩﺭﺑـﺎﺭﻩ ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑـﻮﺩ ﻭ ﺍﻳﻨﻜﻪ ﺩﻭﺳـﺘﺪﺍﺭ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﺎﻳﺪ ﺍﺯ ﻫﺮ ﺭﺍﻫﯽ ﺑﺮﺍی ﺑﺰﺭﮔﺪﺍﺷﺖ ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﺑﻬﺮﻩ ﺑﮕﻴﺮﺩ. ﻧﺼﺐ ﺗﺎﺑﻠﻮی ﻣﺸﺘﻤﻞ ﺑﺮ ﻧﺎﻡ ﺁﻥ ﺣﻀﺮﺕ ﺗﻨﻬﺎ ﻧﻤﻮﻧﻪ ﺍی ﺍﺯ ﺍﻳﻦ ﻗﺒﻴﻞ ﺍﺳﺖ.

ﻧﻤﻮﻧﻪ ﺍی ﺩﻳﮕﺮ ﺍﺯ ﺗﺠﻠﻴﻞ ﻫﻤﻴﻦ ﺍﺳـﺖ ﻛـﻪ ﺧﻴﻠﯽ ﻫﺎ ﻭﻗﺘﯽ ﻧﺎﻡ ﺁﻥ ﺣﻀﺮﺕ ﻭ ﭘﺪﺭﺍﻥ ﺑﺰﺭﮔﻮﺍﺭﺵ ﺑﺮﺩﻩ ﻣﯽ ﺷـﻮﺩ، ﺻﻠﻮﺍﺕ ﻣﯽ ﻓﺮﺳﺘﻨﺪ ﻳﺎ ﺑﺮ ﺍﻭ ﺳـﻼﻡ ﻣﯽ ﻛﻨﻨﺪ ﻭ ﻣﺜـﻼ ﻣﯽ ﮔﻮﻳﻨﺪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻛـﻪ ﺩﺭﻭﺩ ﺧﺪﺍ ﺑﺮ ﺍﻭ ﺑﺎﺩ، ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻛـﻪ ﺟﺎﻧﻬﺎی ﻣﺎ ﻓﺪﺍﻳﺶ ﺑﺎﺩ، ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻛﻪ ﺧﺪﺍ ﻓﺮﺟﺶ ﺭﺍ ﻧﺰﺩﻳﮏ ﺳﺎﺯﺩ.

ﺑـﻪ ﭘﺎ ﺧﺎﺳـﺘﻦ ﺑﻪ ﻫﻨﮕﺎﻡ ﺫﻛﺮ ﻧﺎﻡ ﺧـﺎﺹ ﺁﻥ ﺣﻀﺮﺕ (ﻳﻌﻨﯽ ﻗﺎﺋﻢ ﻛـﻪ ﺑـﻪ ﻣﻌﻨﺎی ﺑﻪ ﭘـﺎ ﺧﻴﺰﻧﺪﻩ ﻭ ﻗﻴﺎﻡ ﻛﻨﻨﺪﻩ ﺍﺳـﺖ.) ﻧﻴﺰ ﺷـﻴﻮﻩ ﺍی ﺩﻳﮕﺮ ﺑﺮﺍی ﺗﺠﻠﻴﻞ ﺍﺯ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ، ﭼﻨﺎﻧﻜﻪ ﻧﻘﻞ ﺷﺪﻩ ﺭﻭﺯی ﺩﺭ ﻣﺠﻠﺲ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺯ ﺣﻀﺮﺕ ﺻﺎﺣﺐ ﺍﻟﺰﻣﺎﻥ ﻳﺎﺩ ﺷـﺪ، ﭘﺲ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﻪ ﺧﺎﻃﺮ ﺗﻌﻈﻴﻢ ﻭ ﺍﺣﺘﺮﺍﻡ ﺍﺳﻢ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﭘﺎ ﺍﻳﺴﺘﺎﺩﻧﺪ.(22)

ﺍﺣﺘﺮﺍﻡ ﺑﻪ ﺁﻥ ﺣﻀﺮﺕ ﺩﺭ ﻧﺸﺴـﺖ ﻭ ﺑﺮﺧﺎﺳـﺖ ﻭ ﺭﻋﺎﻳﺖ ﺍﺩﺏ ﻣﺤﻀـﺮ ﺁﻥ ﺣﻀﺮﺕ ﺑـﻪ ﺷـﻴﻮﻩ ﻫﺎی ﻣﺨﺘﻠـﻒ ﺍﻣﻜﺎﻥ ﭘﺬﻳﺮ ﻣﯽ ﺑﺎﺷـﺪ. ﺭﻭﺷـﻦ ﺍﺳـﺖ ﻛﻪ ﻣﺎ ﺑﻪ ﻭﺳـﻴﻠﻪ ﺧﻀﻮﻉ ﺑﺪﻧﯽ ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻳﺸﺎﻥ ، ﺗﻘﺮﺏ ﺑـﻪ ﺳـﻮی ﺧﺪﺍﻳـﯽ ﺭﺍ ﻣﯽ ﺟﻮﻳﻴـﻢ ﻛـﻪ ﺍﻭ ﺭﺍ ﺁﻓﺮﻳﺪﻩ ﻭ ﺑـﺰﺭگ ﻭ ﻣﺤﺘﺮﻡ ﺩﺍﺷﺘﻪ ﺍﺳﺖ. ﺩﺭ ﺩﻋﺎﻳﯽ ﻣﯽ ﺧﻮﺍﻧﻴﻢ: «ﺑﺎﺭ ﺧﺪﺍﻳﺎ! ﻣﻦ ﺗﻘﺮﺏ ﻣﯽ ﺟﻮﻳﻢ ﺑـﻪ ﺩﺭﮔﺎﻩ ﺗﻮ ﺑـﺎ ﺩﻟﯽ ﺧﺎﺿﻊ ﻭ ﺑـﻪ ﺩﺭﮔﺎﻩ ﻭﻟﻴﺖ ﺑﺎ ﺑﺪﻧـﯽ ﻓﺮﻭﺗﻦ، ﻭ ﺑﻪ ﺍﻣﺎﻣﺎﻥ ﻫﺪﺍﻳﺖ ﻛﻨﻨﺪﻩ ﺑﺎ ﻗﻠﺒﯽ ﻣﺘﻮﺍﺿﻊ»(23)

ﺑﺮﮔﺰﺍﺭی ﻣﺮﺍﺳـﻢ ﺧﺎﺹ ﺩﺭ ﺭﻭﺯﻫﺎﻳﯽ ﻛﻪ ﻣﻨﺘﺴـﺐ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺳـﺖ ﺑﻪ ﺧﺼﻮﺹ ﻧﻴﻤﻪ ﺷـﻌﺒﺎﻥ ، ﺟﻤﻌﻪ ﻫﺎ ﻭ ﻋﻴﺪﻏﺪﻳﺮ ﻧﻤﻮﻧﻪ ﻫﺎی ﺩﻳﮕﺮی ﺍﺯ ﺑﺰﺭﮔﺪﺍﺷﺖ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ.

ﻧﺎﻡ ﮔﺬﺍﺭی ﻓﺮﺯﻧﺪﺍﻥ ﺑﺎ ﺍﺳـﺘﻔﺎﺩﻩ ﺍﺯ ﻧﺎﻡﻫﺎی ﻣﻨﺘﺴـﺐ ﺑﻪ ﺍﻳﺸﺎﻥ ﻧﻴﺰ ﻧﻮﻋـﯽ ﺍﺑـﺮﺍﺯ ﻋﻼﻗﻪ ﻭ ﺗﺠﻠﻴـﻞ ﺍﺯ ﺁﻥ ﻋﺰﻳﺰ ﺩﺭ ﺟﺎﻣﻌﻪ ﺍﺳـﺖ. ﻧﺎﻡﻫﺎﻳﯽ ﭼـﻮﻥ ﻣﻬـﺪی، ﻣﻬﺪﻳﻪ ، ﺍﻣﻴﺮﻣﻬـﺪی (ﻳﻌﻨﯽ ﻛﺴـﯽ ﻛﻪ ﺍﻣﻴـﺮﺵ ﻣﻬﺪی ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﺍﺳـﺖ)، ﻣﺤﻤﺪﻣﻬـﺪی، ﻣﻬﺪﻳـﺎﺭ (ﻳﻌﻨﯽ ﻛﺴـﯽ ﻛﻪ ﻳﺎﺭ ﻣﻬـﺪی ﺍﺭﻭﺍﺣﻨﺎﻓﺪﺍﻩ ﺍﺳـﺖ) ﻭ ﺻﺎﻟـﺢ ﻭ ﺻﺎﻟﺤﻪ (ﭼﺮﺍﻛـﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺍﺑﺎﺻﺎﻟﺢ ﺍﺳـﺖ) ﻭ ﻧﻈﺎﻳﺮ ﺁﻧﻬﺎ ﺍﺯ ﻧﺎﻡﻫﺎی ﻣﺮﺗﺒﻂ ﺑﺎ ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﻣﯽ ﺑﺎﺷـﺪ. ﺩﺭ ﻧﺎﻡ ﮔﺬﺍﺭی ﺩﺧﺘﺮﺍﻥ ﻣﯽ ﺗﻮﺍﻥ ﻋﻼﻭﻩ ﺑﺮ ﺍﺳـﺘﻔﺎﺩﻩ ﺍﺯ ﻧﺎﻡﻫﺎﻳـﯽ ﭼﻮﻥ ﻣﻬﺪﻳﻪ ﻭ ﺻﺎﻟﺤـﻪ ﺍﺯ ﻧﺎﻡﻫﺎی ﻣﺎﺩﺭ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻬﻤﺎ ﺍﻟﺴـﻼﻡ ﻫﻤﭽﻮﻥ ﻧﺮﺟﺲ (ﻳﺎ ﻫﻤﺎﻥ ﻧﺮﮔﺲ)، ﻣﻠﻴﻜﺎ (ﻳﺎ ﻫﻤﺎﻥ ﻣﻠﻴﻜﻪ ) ﻭ ﺳﻮﺳﻦ ﺑﻬﺮﻩ ﮔﺮﻓﺖ.

ﺗﺮﻭﻳـﺞ ﺍﻳـﻦ ﻧﺎﻡﻫـﺎ ﺩﺭ ﺟﺎﻣﻌـﻪ ﻧﺸـﺎﻧﻪ ﺗﻮﺟـﻪ ﺑـﻪ ﺻﺎﺣﺒـﺎﻥ ﺍﻳـﻦ ﻧﺎﻡﻫﺎﺳـﺖ ﻭ ﺑﻴﺎﻧﮕـﺮ ﺁﻥ ﺍﺳـﺖ ﻛﻪ ﺍﻣﺎﻡ ﻋﺼـﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﭼﻨﻴﻦ ﺟﺎﻣﻌـﻪ ﺍی ﭘﺎﻳـﮕﺎﻩ ﻭ ﻫﻮﺍﺩﺍﺭﺍﻧﯽ ﺩﺍﺭﺩ. ﺑـﻪ ﻋﻜﺲ ﺟﺎﻣﻌﻪ ﺍی ﻛـﻪ ﺩﺭ ﺁﻥ ﺑـﺪﻭﻥ ﺩﻟﻴﻞ ﻣﻮﺟﻪ ، ﺧﺒﺮی ﺍﺯ ﺍﻳﻦﮔﻮﻧﻪ ﻧﺎﻡﻫﺎ ﻧﺒﺎﺷـﺪ، ﺟﺎﻣﻌﻪ ﺍی ﻏﺎﻓﻞ ﻧﺴﺒﺖ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﺩ.

ﺳـﺮﻭﺩﻥ ﺷـﻌﺮ ﺑـﺎ ﻣﺤﺘﻮﺍﻫـﺎی ﺑﻠﻨﺪ ﺩﺭﺑـﺎﺭﻩ ﺁﻥ ﺣﻀـﺮﺕ ﺟﻠﻮﻩ ﺩﻳﮕـﺮی ﺍﺯ ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺳـﺖ. ﺍﺯ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﻛـﻪ ﻓﺮﻣﻮﺩﻧـﺪ: «ﻫﺮﻛﺲ ﺩﺭﺑﺎﺭﻩ ﻣـﺎ ﻳﮏ ﺑﻴﺖ ﺷﻌﺮ ﺑﮕﻮﻳﺪ ﺧﺪﺍﻭﻧﺪ ﺑﺮﺍی ﺍﻭ ﺑﻴﺘﯽ (ﻳﻌﻨﯽ ﺧﺎﻧﻪ ﺍی) ﺩﺭ ﺑﻬﺸﺖ ﺧﻮﺍﻫﺪ ﺳﺎﺧﺖ.»(24)

ﺗﻮﺻﻴـﻪ ﺷـﺪﻩ ﻛﻪ ﻣﺤﺘﻮﺍی ﺍﺷـﻌﺎﺭ ﻣـﺎ ﺗﻨﻬﺎ ﺑﻪ ﺍﺑﺮﺍﺯ ﺷـﻮﻕ ﺟﻬﺖ ﺩﻳـﺪﺍﺭ ﺁﻥ ﺣﻀـﺮﺕ ﻣﺤﺪﻭﺩ ﻧﺸـﻮﺩ. ﭼﻪ ﺧﻮﺏ ﺍﺳـﺖ ﻛﻪ ﺑﺨﺸـﯽ ﺍﺯ ﺍﺷـﻌﺎﺭ ﻣﻬﺪﻭی ﺩﺭ ﻭﺻـﻒ ﻛﻤﺎﻻﺕ ﻭ ﺑﻴﺎﻥ ﻣﻨﺎﻗﺐ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺑﺎﺷـﺪ. ﺑﺨﺸـﯽ ﺩﻳﮕﺮ ﺩﺭ ﺗﺸـﻮﻳﻖ ﺑﻪ ﺗﻤﺴـﮏ ﻭ ﺗﻮﺳـﻞ ﺑـﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﺎﺷـﺪ. ﺑﺨﺸـﯽ ﻫﻢ ﺩﺭ ﺟﻬﺖ ﺗﺮﻏﻴﺐ ﺍﻣﺖ ﺑﻪ ﺩﻋﺎی ﺑﺮﺍی ﻇﻬﻮﺭ ﺁﻥ ﺣﻀﺮﺕ ﺍﺧﺘﺼـﺎﺹ ﻳﺎﺑـﺪ. ﺍﻳﻦﮔﻮﻧﻪ ﻣﻀﺎﻣﻴﻦ ﺩﺭ ﺍﺷـﻌﺎﺭ ﮔﺬﺷـﺘﮕﺎﻥ ﻣﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻟﮕﻮی ﺭﺍﻩ ﻣﺎ ﺑﺎﺷﺪ.

ﻋﺎﻟﻢ ﻧﺎﻣﺪﺍﺭ ﺷـﻴﻌﯽ ﻣﺮﺣﻮﻡ ﻣﻼ ﻣﺤﺴﻦ ﻓﻴﺾ ﻛﺎﺷﺎﻧﯽ، ﺩﻳﻮﺍﻧﯽ ﺑﻪ ﻧﺎﻡ ﺷـﻮﻕ ﻣﻬﺪی ﺩﺍﺭﺩ. ﺍﻳﺸـﺎﻥ ﺩﺭ ﺍﻳﻦ ﺩﻳﻮﺍﻥ ﺑﺮﺧﯽ ﺍﺯ ﺍﺷـﻌﺎﺭ ﻧﺎﺏ ﺧﻮﺍﺟﻪ ﺣﺎﻓﻆ ﺭﺍ ﺑﺮﮔﺰﻳﺪﻩ ﻭ ﺑﻪ ﺁﻧﻬﺎ ﺭﻧﮓ ﻭ ﺑﻮی ﻣﻬﺪﻭی ﺩﺍﺩﻩ ﺍﺳﺖ. ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣـﺎﻡ ﻋﺼﺮ ﺩﺭ ﺍﺷـﻌﺎﺭ ﻣﺰﺑﻮﺭ ﻣـﻮﺝ ﻣﯽ ﺯﻧﺪ ﻭ ﻧﻤﻮﻧـﻪ ﺍی ﺍﺯ ﺑﺰﺭﮔﺪﺍﺷﺖ ﺍﻣﺎﻡ ﻋﺼﺮ ﻭ ﺑﻴﺎﻥ ﻣﻨﺎﻗﺐ ﺍﻳﺸﺎﻥ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ.

ﺩﻭﺭی ﮔﺰﻳـﺪﻥ ﻭ ﻫـﻮﺍﺩﺍﺭی ﻧﻜـﺮﺩﻥ ﺍﺯ ﻛﺴـﺎﻧﻴﻜﻪ ﺑـﺎ ﺍﻣﺎﻡ ﻋﺼﺮ ﻛﺎﺭی ﻧﺪﺍﺭﻧﺪ ﻧﻴﺰ ﻧﻮﻋﯽ ﺟﺎﻧﺒﺪﺍﺭی ﺍﺯ ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺳﺖ.

ﺑﺮﺍﺋـﺖ ﻭ ﺑﻴـﺰﺍﺭی ﺟﺴـﺘﻦ ﻭ ﻣﺨﺎﻟﻔـﺖ ﺑﺎ ﺩﺷـﻤﻨﺎﻥ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻭ ﺗﻀﻌﻴـﻒ ﺟﻨـﺎﺡ ﺁﻧﺎﻥ ﺷـﻜﻞ ﺩﻳﮕـﺮی ﺍﺯ ﻫﻮﺍﺧﻮﺍﻫـﯽ ﺁﻥ ﺣﻀﺮﺕ ﺍﺳﺖ.

ﺍﻳـﻦ ﮔﻮﻧـﻪ ﺗﺠﻠﻴﻞﻫـﺎ ﺍﺯ ﻋﻬـﺪﻩ ﻫـﺮ ﻳـﮏ ﺍﺯ ﻣـﺎ ﺑـﺮ ﻣﯽ ﺁﻳـﺪ ﻭ ﻣﺎ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺑﺎ ﺍﻳﻦ ﺷﻴﻮﻩ ﻫﺎ ﺑﻪ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﺸﺘﺎﺑﻴﻢ.
