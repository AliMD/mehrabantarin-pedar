بازگشت

ﻗﺼﺪ ﻧﻜﻮﻫﺶ ﻧﻴﺴﺖ، ﺑﻴﺎﻳﻴﺪ ﻛﺎﺭ ﺳﺎﺯﻧﺪﻩ ﺍی ﺑﻜﻨﻴﻢ!

ﻃﺮﺡ ﺍﻳﻦ ﻣﻄﻠﺐ ﻛﻪ «ﺍﻣﺎﻡ ﺯﻣﺎﻥ، ﻏﺮﻳﺐ ﺍﺳﺖ.» ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﺩﻭ ﻫﺪﻑ ﺻﻮﺭﺕ ﮔﻴﺮﺩ. ﺍﻳﻦ ﻛﺎﺭ ﻣﻤﻜﻦ ﺍﺳـﺖ ﺑﻪ ﻗﺼﺪ ﻧﻜﻮﻫﺶ «ﺧﻮﺩﻣﺎﻥ» ﻳﺎ «ﺩﻳﮕـﺮﺍﻥ» ﺑﺎﺷـﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨـﺎ ﻛﻪ ﺩﺭ ﺟﺴـﺘﺠﻮی ﻛﺴـﺎﻧﯽ ﺑﺮﺁﻳﻴﻢ ﻛﻪ ﺯﻣﻴﻨﻪ ﺳـﺎﺯ ﺍﻳﻦ ﻏﺮﺑﺖ ﺷـﺪﻩ ﺍﻧﺪ ﻭ ﺳـﭙﺲ ﺁﻧﻬـﺎ ﺭﺍ ﻣﻼﻣﺖ ﻛﻨﻴـﻢ. ﺍﻟﺒﺘﻪ ﺗﻤﺎﻡ ﻛﺴـﺎﻧﯽ ﻛﻪ ﻋﺎﻣﻞ ﺟﺪﺍﺷـﺪﻥ ﺟﺎﻣﻌﻪ ﺑﺸـﺮی ﺍﺯ ﺁﻥ ﺍﻣﺎﻡ ﺁﺳـﻤﺎﻧﯽ ﻫﺴﺘﻨﺪ، ﺷﺎﻳﺴﺘﻪ ﻣﻼﻣﺖ ﻭ ﻧﻜﻮﻫﺶﺍﻧﺪ، ﻟﻴﻜﻦ ﻗﺼﺪ ﻣﺎ ﺩﺭ ﺍﻳﻨﺠﺎ ﻃﺮﺡ ﺷﻜﺎﻳﺖ ﻭ ﻣﻼﻣﺖ ﻛﺴﯽ ﻧﻴﺴﺖ.

ﻫـﺪﻑ، ﺩﺭ ﺍﻳﻨﺠﺎ ﻧﻪ ﺳـﺮﺯﻧﺶ ﻋﺎﻣﻼﻥ ﺑﺮﻭﺯ ﻣﺸـﻜﻞ ﺑﻠﻜﻪ ﺍﻗﺪﺍﻡ ﺟﻬﺖ ﺭﻓﻊ ﻣﺸـﻜﻞ ﻳـﺎ ﻻﺍﻗﻞ ﻛﻢ ﻛﺮﺩﻥ ﺷـﺪﺕ ﺁﻥ ﺍﺳـﺖ. ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﻛﻨﻴﻢ ﻛﻪ ﺑﺴـﻴﺎﺭی ﺍﺯ ﻣﺮﺩﻡ ﺭﻭﺯﮔﺎﺭ ﻣﺎ ﺑﺎ ﺍﻳﻦ ﻧﻌﻤﺖ ﺍﻟﻬﯽ ﺭﻭی ﺳـﺘﻴﺰ ﻧﺪﺍﺭﻧـﺪ، ﺑﻠﻜﻪ ﺗﻨﻬـﺎ ﺍﺯ ﺍﻭ ﻏﺎﻓﻞﺍﻧﺪ. ﺁﻧﻬﺎ ﺍﺳﺎﺳـﺎ ﻧﻤﯽ ﺩﺍﻧﻨﺪ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﭼﻨﻴـﻦ ﻧﻌﻤﺘـﯽ ﺭﺍ ﺑﺮﺍی ﺁﻧﻬﺎ ﻗـﺮﺍﺭ ﺩﺍﺩﻩ ﺍﺳـﺖ ﻭ ﻫﻤﻴﻦ ﻧﺪﺍﻧﺴﺘﻦﺷـﺎﻥ ﺍﺳـﺖ ﻛﻪ ﺑﻪ ﻏﺮﺑﺖ ﺁﻥ ﻣﻮﻻ ﺩﺍﻣﻦ ﻣﯽ ﺯﻧﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﻛﺎﺭی ﻧﺪﺍﺭﻳﻢ ﻛﻪ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺩﺭ ﻧﺪﺍﻧﺴـﺘﻦ ﻭ ﻏﻔﻠﺘﺸـﺎﻥ ﺗﻘﺼﻴﺮ ﺩﺍﺭﻧﺪ ﻳﺎ ﻧﺪﺍﺭﻧﺪ. ﺑﺎﺯ ﺗﺄﻛﻴﺪ ﻣﯽ ﻛﻨﻢ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﻣﻘﺎﻡ ﺩﺭ ﺻﺪﺩ ﻣﻼﻣﺖ ﻭ ﻧﻜﻮﻫﺶ ﻛﺴﯽ ﻧﻴﺴﺘﻴﻢ ﻭ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﻘﺼﺮ ﻧﻤﯽ ﮔﺮﺩﻳﻢ.

ﺁﻧﭽـﻪ ﺑﺎﻳـﺪ ﺑﺪﺍﻧﻴﻢ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺍﻛﻨـﻮﻥ ﺍﻳﻦ ﺟﻬـﻞ ﻭ ﺑﯽ ﺧﺒﺮی ﺍﺗﻔـﺎﻕ ﺍﻓﺘـﺎﺩﻩ ﻭ ﻓﺮﺍﮔﻴـﺮ ﺷـﺪﻩ ﺍﺳـﺖ. ﻫﻤﭽﻨﻴﻦ ﺑﺎﻳـﺪ ﺑﺪﺍﻧﻴـﻢ ﻛﻪ ﺍﻳﻦ ﺟﻬﻞ ﻓﺮﺍﮔﻴﺮ، ﻗﺎﺑﻞ ﺩﺭﻣﺎﻥ ﺍﺳﺖ. ﺩﻭﺍی ﺟﻬﻞﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﻗﺒﻴﻞ ﺗﺒﻠﻴﻎ ﻭ ﭘﻴﺎﻡﺭﺳﺎﻧﯽ ﺍﺳﺖ. ﺍﮔﺮ ﻣﺎ ﭘﻴﺎﻡ ﻭﺟﻮﺩ ﭼﻨﻴﻦ ﻧﻌﻤﺘﯽ ﺭﺍ ﺑﻪ ﻣﺮﺩﻡ ﺟﻬﺎﻥ ﺑﺮﺳـﺎﻧﻴﻢ ﺑﺴـﻴﺎﺭی ﺍﺯ ﺁﻧـﺎﻥ ﺑﻪ ﺟﻬﺖ ﻧﻴﺎﺯ ﻭ ﺷـﻮﻗﯽ ﻛﻪ ﺑـﻪ ﺍﻭ ﺩﺍﺭﻧﺪ ﺑﻪ ﺳـﻤﺖ ﺍﻭ ﻣﺘﻮﺟﻪ ﻣﯽ ﺷـﻮﻧﺪ. ﺩﺭ ﺍﺛﺮ ﭘﻴﺎﻡﺭﺳﺎﻧﯽ ﺟﻬﻞ ﺍﺯ ﺑﻴﻦ ﻣﯽ ﺭﻭﺩ، ﮔﺮﺍﻳﺶ ﺑﻪ ﻣﻨﺒﻊ ﺧﻴﺮ ﻓﺮﺍﮔﻴﺮ ﻣﯽ ﮔﺮﺩﺩ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﺯ ﺑﺎﺭ ﺳﻨﮕﻴﻦ ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﻛﺎﺳﺘﻪ ﻣﯽ ﺷﻮﺩ.

ﻫـﺮ ﻳﮏ ﺍﺯ ﻣﺎ ﺑﺎﻳﺪ ﺍﻳﻦ ﭘﻴﺎﻡ ﺭﺍ ﺑﻪ ﻛﺴـﺎﻧﯽ ﻛﻪ ﻣﺨﺎﻃﺒﻤﺎﻥ ﻫﺴـﺘﻨﺪ ﻣﻨﺘﻘﻞ ﻛﻨﻴﻢ ﻛﻪ : «ﺧﺪﺍﻭﻧﺪ ﺑﺮﺍی ﻣﺮﺩﻡ ﻫﺮ ﺯﻣﺎﻥ ﺍﻣﺎﻣﯽ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺗﺎ ﭘﻨﺎﻩ ﺁﻥ ﻫﺎ ﺑﺎﺷﺪ. ﺍﻭ ﻫﻤﭽﻮﻥ ﭼﺸﻤﻪ ﺁﺏ ﮔﻮﺍﺭﺍﻳﯽ ﺍﺳﺖ، ﺑﺮﺍی ﺗﺸﻨﮕﯽ ﻫﺎی ﺗﻤﺎﻡ ﺍﻧﺴﺎﻥ ﻫﺎی ﻛﻤﺎﻝﺟﻮ.»

ﺍﮔـﺮ ﺍﻳﻦ ﭘﻴﺎ مرﺳـﺎﻧﯽ ﺑـﻪ ﺩﺭﺳـﺘﯽ ﺍﻧﺠﺎﻡ ﺷـﻮﺩ ﺧﻮﺍﻫﻴـﻢ ﺩﻳﺪ ﻛﻪ ﮔﺮﻭﻩ ﻫﺎﻳﯽ ﺍﺯ ﻣﺮﺩﻡ ﺑﺎ ﺷﻮﻕ ﻓﺮﺍﻭﺍﻥ ﺑﻪ ﺍﻳﻦ ﭼﺸﻤﻪ ﺣﻴﺎﺕ ﺭﻭی ﺧﻮﺍﻫﻨﺪ ﺁﻭﺭﺩ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ، ﺍﺯ ﮔﺴـﺘﺮﻩ ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﻛﺎﺳﺘﻪ ﻣﯽ ﺷﻮﺩ. ﺍﻟﺒﺘﻪ ﺷـﻴﻄﺎﻥ ﺩﺭ ﺑﺮﺍﺑـﺮ ﺍﻳـﻦ ﭘﺪﻳـﺪﻩ ﻣﺒـﺎﺭک ﺑﯽ ﺗﻔـﺎﻭﺕ ﻧﺨﻮﺍﻫﺪ ﻧﺸﺴـﺖ. ﺍﻭ ﻧﻴـﺰ ﺑـﻪ ﺗﻼﺷـﯽ ﮔﺴـﺘﺮﺩﻩ ﺩﺭ ﺟﻬﺖ ﻋﻘﻴﻢ ﺳـﺎﺧﺘﻦ ﺗﻼﺵ ﺷـﻴﻌﻴﺎﻥ ﺩﺳـﺖ ﺧﻮﺍﻫـﺪ ﺯﺩ ﻭ ﻣﻌﺮﻛﻪ ﺍی ﺑﻪ ﭘﺎ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﻣﻬﻢ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﻣﺎ ﺍﺯ ﺭﻭﻳﺎﺭﻭﻳﯽ ﺑﺎ ﻟﺸـﻜﺮ ﺷـﻴﻄﺎﻥ ﻧﻬﺮﺍﺳـﻴﻢ ﺯﻳﺮﺍ ﺩﺭ ﺍﻳـﻦ ﻛﺎﺭﺯﺍﺭ ﺗﻨﻬﺎ ﻧﺨﻮﺍﻫﻴـﻢ ﺑﻮﺩ. ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﭼﻮﻥ ﺑـﻪ ﻳﺎﺭی ﻭﻟﯽ ﺧﺪﺍ ﺁﻣﺪﻩ ﺍﻳﻢ، ﺧﺪﺍ ﻧﻴﺰ ﺑﻪ ﻳﺎﺭی ﻣﺎ ﺧﻮﺍﻫﺪ ﺁﻣﺪ ﻭ ﺩﺷـﻤﻦ ﻫﺮ ﭼﻨﺪ ﺗﻮﺍﻧﺎ ﻭ ﺣﺎﺿﺮﻳﺮﺍﻕ ﺑﺎﺷـﺪ، ﺩﺭ ﺑﺮﺍﺑﺮ ﺧﺪﺍﻭﻧﺪ ﺁﺳﻤﺎﻥ ﻭ ﺯﻣﻴﻦ ﺧ ﺮﺩ ﻭ ﻧﺎﭼﻴﺰ ﺍﺳﺖ.

ﺍﻳﻨﮏ ﺑﻴﺎﻳﻴﺪ ﺍﺯ ﻣﻴﺎﻥ ﻣﺮﺩﻡ ﺩﻧﻴﺎ ﺗﻮﺟﻬﻤﺎﻥ ﺭﺍ ﺑﺮﺭﻭی ﺟﺎﻣﻌﻪ ﺷﻴﻌﻪ ﻣﺘﻤﺮﻛﺰ ﻛﻨﻴﻢ. ﻳﻜﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﺍﻓﺘﺨﺎﺭﺍﺕ ﺷـﻴﻌﻪ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺧﻮﺩ ﻳﻌﻨﯽ ﺍﻳﻦ ﮔﻨﺞ ﺧﺪﺍﺩﺍﺩﻩ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳﺪ ﻭ ﺍﺯ ﻭﺟﻮﺩﺵ ﺑﺎﺧﺒﺮ ﺍﺳـﺖ. ﺷﻴﻌﻴﺎﻥ ﺩﺭﺳﺖ ﺩﺭ ﻟﺤﻈﺎﺕ ﺣﺴﺎﺳـﯽ ﻛﻪ ﺩﻳﮕﺮﺍﻥ ﺩﺭ ﺯﻳﺮ ﺑﺎﺭ ﻣﺸـﻜﻼﺕ ﻣﺎﺩی ﻭ ﻣﻌﻨﻮی ﺑﻪ ﺯﺍﻧﻮ ﺩﺭ ﻣﯽ ﺁﻳﻨـﺪ، ﺍﺯ ﻳﺎﺭی ﺁﻥ ﺣﻀﺮﺕ ﺑﻬﺮﻩ ﻣﻨﺪ ﺷﺪﻩ ﻭ ﻣﯽ ﺷﻮﻧﺪ.

ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻏﺮﺑﺖ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺩﺭ ﻣﻴﺎﻥ ﺷـﻴﻌﻴﺎﻥ ﺍﺯ ﺟﻬﺘﯽ ﻛﻤﺘﺮ ﺍﺯ ﻏﺮﺑـﺖ ﺍﻳﺸـﺎﻥ ﺩﺭ ﻣﻴـﺎﻥ ﺩﻳﮕﺮﺍﻥ ﺍﺳـﺖ. ﭼـﺮﺍ؟ ﭼﻮﻥ ﺷـﻴﻌﻪ ﺍﻭ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳـﺪ ﻭ ﻗـﺪﺭﺵ ﺭﺍ ﺑﻴﺸـﺘﺮ ﻣﯽ ﺩﺍﻧﺪ. ﺍﻣـﺎ ﺁﻳﺎ ﻣﻴﺰﺍﻥ ﻗﺪﺭﺷﻨﺎﺳـﯽ ﺷﻴﻌﻪ ﻧﺴﺒﺖ ﺑﻪ ﮔﻮﻫﺮ ﻭﺟﻮﺩ ﺍﻣﺎﻡ ﻋﺼﺮ ﻗﺎﺑﻞ ﺍﺭﺗﻘﺎ ﻧﻴﺴﺖ؟ ﺁﻳﺎ ﻣﻤﻜﻦ ﻧﻴﺴﺖ ﻛﻪ ﻣﺎ ﺑﺎ ﺗﻮﺟﻪ ﺩﺍﺩﻥ ﻳﻜﺪﻳﮕﺮ ﻧﺴﺒﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻭ ﺑﻪ ﺍﻳﻨﻜﻪ ﺍﺯ ﺍﻳﺸـﺎﻥ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﻣﯽ ﺗﻮﺍﻥ ﺑﻬﺮﻩ ﺑﺮﺩ، ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﺭﺍ ﻛﺎﻫﺶ ﺩﻫﻴﻢ؟

ﭼﻨـﺪ ﺩﺭﺻﺪ ﺷـﻴﻌﻴﺎﻥ ﺩﺭ ﻋﻤﻮﻡ ﺍﻣـﻮﺭ ﺯﻧﺪﮔﻴﺸـﺎﻥ ﻣﺘﻮﺟﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻭ ﻣﺘﻤﺴ ـﮏ ﺑﻪ ﺍﻳﺸﺎﻥ ﻫﺴـﺘﻨﺪ؟ ﺑﺎﻳﺪ ﭘﺬﻳﺮﻓﺖ ﻛﻪ ﺩﺭﺻﺪ ﺯﻳﺎﺩی ﺍﺯ ﻣﺎ ﺷـﻴﻌﻴﺎﻥ ﺩﺭ ﻋﻤﻮﻡ ﺍﻣﻮﺭ ﺯﻧﺪﮔﯽ ﻣﺘﻮﺟﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻭ ﻣﺘﻤﺴ ـﮏ ﺑﻪ ﺍﻳﺸـﺎﻥ ﻧﻴﺴـﺘﻴﻢ ﻭ ﺍﻳﻦ ﺩﺭﺻﺪ ﭘﺎﻳﻴﻦ ﻫﺮﮔﺰ ﻗﺎﻧﻊﻛﻨﻨﺪﻩ ﻧﻴﺴـﺖ. ﺍﻳﻦ ﺗﻤﺴـﮏ ﻭ ﺗﻮﺳـﻼﺕ ﺍﻧـﺪک ﺑـﻪ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺩﺭ ﺷـﺮﺍﻳﻄﯽ ﺍﺳـﺖ ﻛﻪ ﺧﺪﺍﻭﻧـﺪ ﺁﻥ ﺣﻀﺮﺕ ﺭﺍ ﭘﻨـﺎﻩ ﻣﺎ ﺩﺭ ﺗﻤﺎﻡ ﺍﻣـﻮﺭ ﺯﻧﺪﮔﻴﻤﺎﻥ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺍﺳـﺖ.

ﻣﺎ ﺷﻴﻌﻴﺎﻥ ﺑﺎﻳﺪ ﭘﺎ ﺑﻪ ﭘﺎی ﮔﺴﺘﺮﺵ ﺗﻼﺵﻫﺎی ﺷﺨﺼﯽ ﻣﺎﻥ ﺑﺮﺍی ﺑﻬﺒﻮﺩ ﺯﻧﺪﮔﯽ ﺩﻧﻴﻮی ﻭ ﺣﻴﺎﺕ ﺍﺧﺮﻭی ﻣﺎﻥ، ﺭﻭﺡ ﺗﻮﺳـﻞ ﻭ ﺗﻤﺴﮏ ﺑﻪ ﺍﻣـﺎﻡ ﻋﺼﺮ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺯﻧﺪﻩ ﻛﻨﻴﻢ. ﺍﮔﺮ ﭼﻨﻴﻦ ﻛﻨﻴﻢ ﺁﻥ ﻭﻗﺖ ﺍﺳـﺖ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﺯﻧﺪﮔﯽ ﻣﺎ ﺣﻀﻮﺭ ﻣﻌﻨﻮی ﻣﯽ ﻳﺎﺑﺪ ﻭ ﻧﻘﺶ ﻣﯽ ﺁﻓﺮﻳﻨـﺪ. ﺑﺎﻳـﺪ ﻫﻤﻮﺍﺭﻩ ﺩﺭ ﺩﻝ ﻭ ﺑﺮ ﺯﺑﺎﻧﻤﺎﻥ ﺫﻛـﺮ ﻳﺎ ﺻﺎﺣﺐ ﺍﻟﺰﻣﺎﻥ ﺟﺎﺭی ﺑﺎﺷﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺯ ﺟﺎﻧﺐ ﻣﻠﻜﻮﺕ ﻳﺎﺭی ﺧﻮﺍﻫﻴﻢ ﺷﺪ ﻭ ﺯﻧﺪﮔﻴﻤﺎﻥ ﻣﺘﺤﻮﻝ ﻣﯽ ﺷـﻮﺩ، ﺭﺍﻩ ﺧﻴﺮ ﻭ ﻧﻴﻜﯽ ﭘﻴﺶ ﭘﺎﻳﻤﺎﻥ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷـﺪ ﻭ ﺍﺯ ﺍﺭﺗﻜﺎﺏ ﺑﺪی ﻫﺎ ﻭ ﺯﺷـﺘﯽ ﻫﺎ ﻣﺼﻮﻥ ﺧﻮﺍﻫﻴﻢ ﻣﺎﻧﺪ. ﻫﺮ ﻗﺪﺭ ﺍﻳﻦ ﺗﻮﺳـﻞ ﻭ ﺗﻤﺴﮏ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﻣﺎ ﺷﺪﻳﺪﺗﺮ ﺑﺎﺷﺪ، ﺗﻮﻓﻴﻘﻤـﺎﻥ ﺩﺭ ﻛﺎﺭﻫـﺎی ﻧﻴﮏ ﻭ ﺩﻭﺭی ﺍﺯ ﺍﻣﻮﺭ ﻧﺎﭘﺴـﻨﺪ ﺑﻴﺸـﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺩﺭﺳﺖ ﺍﺳﺖ ﻛﻪ ﻣﺎ ﺷﻴﻌﻴﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮ ﻣﺮﺩﻡ ﺩﻧﻴﺎ ﺍﻳﻦ ﺍﻣﺘﻴﺎﺯ ﺭﺍ ﺩﺍﺭﻳـﻢ ﻛـﻪ ﺍﻣـﺎﻡ ﺯﻣﺎﻧﻪ ﺧﻮﻳـﺶ ﻳﻌﻨﯽ ﺍﻳﻦ ﮔﻨـﺞ ﭘﺮﻧﻌﻤـﺖ ﺍﻟﻬﯽ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳـﻴﻢ، ﻭ ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﻫﻢ ﺑﺎﻳﺪ ﺧﺪﺍ ﺭﺍ ﺷـﺎﻛﺮ ﺑﺎﺷﻴﻢ، ﻟﻴﻜﻦ ﺑﺎﻳﺪ ﺑﭙﺬﻳﺮﻳﻢ ﻛﻪ ﻣﯽ ﺑﺎﻳﺴـﺖ ﺧﻴﻠﯽ ﺑﻴﺶ ﺍﺯ ﺁﻧﭽﻪ ﻫﺴـﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻣﺮﺍﺟﻌﻪ ، ﺗﻮﺳﻞ ﻭ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﮔﺴﺘﺮﺵ ﺗﻮﺟﻪ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻛﻪ ﺩﺭﻭﺩ ﺧﺪﺍ ﺑﺮ ﺍﻭ ﺑﺎﺩ ﺩﻭ ﻓﺎﻳﺪﻩ ﺩﺭ ﭘﯽ ﺩﺍﺭﺩ: ﻓﺎﻳﺪﻩ ﺍﻭﻟﺶ ﺑﻬﺘﺮ ﺷـﺪﻥ ﻭ ﺍﺭﺗﻘﺎی ﺳﻄﺢ ﺯﻧﺪﮔﯽ ﻛﺴﺎﻧﯽ ﺍﺳﺖ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﻥ ﺣﻀﺮﺕ ﭘﻴﻮﻧﺪ ﻣﯽ ﺩﻫﻨﺪ ﻭ ﻓﺎﻳﺪﻩ ﺩﻭﻣﺶ ﺧﺎﺭﺝ ﺷﺪﻥ ﺁﻥ ﺩﺍﻧﺎی ﻏﺮﻳﺐ ﺍﺯ ﺍﻧﺰﻭﺍ ﻭ ﻛﻢ ﺷﺪﻥ ﻏﺮﺑﺖ ﺍﻳﺸﺎﻥ ﺍﺳﺖ. ﮔﻔﺘﻴﻢ ﻛﻪ ﺑﺨﺸـﯽ ﺍﺯ ﻏﺮﺑﺖ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﻪ ﻫﻤﻴﻦ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻜﺮﺩﻥ ﻣﺎ ﺍﺯ ﺍﻳﺸـﺎﻥ ﺑﺮﻣﯽ ﮔﺮﺩﺩ. ﺷـﻴﻌﻪ ﻳﺎ ﻫﺮ ﺷـﺨﺺ ﺩﻳﮕﺮ، ﺍﮔـﺮ ﺍﺯ ﻧﻌﻤﺖ ﻭﺟﻮﺩ ﺍﻳﺸـﺎﻥ ﺑﻴﺸـﺘﺮ ﺁﮔﺎﻩ ﺑﺎﺷﺪ، ﻃﺒﻴﻌﺘﺎ ﺑﻴﺸـﺘﺮ ﺑﻪ ﺁﻥ ﺣﻀﺮﺕ ﺭﻏﺒﺖ ﻭ ﺍﺷـﺘﻴﺎﻕ ﻣﯽ ﻭﺭﺯﺩ ﻭ ﻫﻤﻴﻦ ﺭﻏﺒﺖ ﺑﻴﺸـﺘﺮ ﺑﻪ ﺍﻳﺸـﺎﻥ ﺑﺎﻋﺚ ﻛﻢ ﺷﺪﻥ ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﮔﺸـﺘﻪ ، ﺯﻣﻴﻨﻪ ﺗﺠﻠﻴﻞ ﺍﺯ ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭ ﺭﺍ ﺩﺭ ﺟﺎﻣﻌﻪ ﻓﺮﺍﻫﻢ ﻣﯽ ﺁﻭﺭﺩ.

ﺧﻼﺻـﻪ ﺁﻧﻜـﻪ ﺍﮔـﺮ ﻣﯽ ﺧﻮﺍﻫﻴـﺪ ﺍﻭ ﺭﺍ ﻳـﺎﺭی ﻛﻨﻴـﺪ ﻳﻜـﯽ ﺍﺯ ﺭﺍﻩ ﻫـﺎی ﭘﻴﺶ ﺭﻭی ﺷـﻤﺎ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺧﺒﺮ ﺍﻭ، ﻳﻌﻨـﯽ ﺧﺒﺮ ﺣﻀﻮﺭ ﺍﻭ ﻭ ﺧﺒـﺮ ﺑـﺮﻛﺎﺕ ﻭﺟـﻮﺩ ﺍﻭ ﺭﺍ ﺑـﻪ ﺩﻳﮕـﺮﺍﻥ ﺑﺮﺳـﺎﻧﻴﺪ. ﺑـﻪ ﺩﻳﮕـﺮﺍﻥ ﺑﮕﻮﺋﻴـﺪ ﻛﻪ : «ﺧﺪﺍﻭﻧﺪ، ﺍﻣﺎﻣﯽ ﺯﻧﺪﻩ ﻭ ﺣﺎﺿـﺮ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﻨﺎﻩ ﺑﺮﺍی ﻣﺎ ﺑﺮﮔﺰﻳﺪﻩ ﻭ ﺑﻪ ﻣﺎ ﺩﺳـﺘـﻮﺭ ﺩﺍﺩﻩ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺳـﺎﻳﻪ ﺍﻭ ﻗﺮﺍﺭ ﺩﻫﻴﻢ ﺗﺎ ﺍﺯ ﺩﺭﻣﺎﻧﺪﮔﯽ ﻫﺎ ﻧﺠﺎﺕ ﻳﺎﺑﻴـﻢ ﻭ ﺑﻪ ﺳـﻌﺎﺩﺕ ﻫـﺎی ﺩﻧﻴـﺎ ﻭ ﺁﺧـﺮﺕ ﺩﺳـﺖ ﻳﺎﺑﻴـﻢ.»

ﺑﻪ ﺩﻭﺳﺘﺎﻥ ﻭ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺧﻮﺩ ﺑﮕﻮﺋﻴﺪ ﻛﻪ : «ﻣﺎ ﺍﻣﺎﻣﯽ ﺯﻧﺪﻩ ﺩﺍﺭﻳﻢ ﻛﻪ ﻧﺰﺩ ﺑﻴﻨﺎﺩﻻﻥ ﻫﻤﻮﺍﺭﻩ ﺣﺎﺿﺮ ﺍﺳـﺖ ﻭ ﭼﻮﻥ ﺧﺎﻟﺼﺎﻧﻪ ﻓﺮﺍﺧﻮﺍﻧﺪﻩ ﺷـﻮﺩ، ﻣﯽ ﺁﻳـﺪ ﻭ ﺭﺍﻩ ﻣﯽ ﻧﻤﺎﻳـﺪ، ﻣﺸـﻜﻞ ﻣﯽ ﮔﺸـﺎﻳﺪ ﻭ ﻏـﻢ ﻭ ﺭﻧـﺞ ﻣﯽ ﺯﺩﺍﻳـﺪ.»

ﺑـﻪ ﺁﻧﻬـﺎ ﺑﮕﻮﺋﻴـﺪ: «ﺍﻣـﺎﻡ ﺷـﻤﺎ ﺯﻧـﺪﻩ ﺍﺳـﺖ ﻭ ﻧـﺰﺩ ﺍﻭ ﺑـﺮﻛﺎﺕ ﻓﺮﺍﻭﺍﻧـﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻋﻠﻢ ﻫﺴـﺖ، ﻫﺪﺍﻳﺖ ﻫﺴـﺖ، ﺷـﻔﺎ ﻫﺴـﺖ، ﺷـﻔﺎﻋﺖ ﻫﺴﺖ، ﺣﻤﺎﻳﺖ ﻫﺴـﺖ، ﺍﻧﺲ ﻭ ﺭﻓﺎﻗﺖ ﻫﺴﺖ، ﻫﻢ ﻏﻨﺎ ﻭ ﻫﻢ ﻗﻨﺎﻋﺖ ﻫﺴـﺖ، ﻳﻌﻨﯽ ﺑﻪ ﺷـﻤﺎ ﻫﻢ ﻏﻨﺎ ﻭ ﻫﻢ ﻗﻨﺎﻋﺖ ﻣﯽ ﺑﺨﺸـﺪ، ﺁﺭی ﻧـﺰﺩ ﺍﻭ ﺗﻤـﺎﻡ ﺍﻳﻦ ﮔﻨﺞﻫـﺎ ﻭ ﻫﺰﺍﺭﺍﻥ ﻫﺰﺍﺭ ﻧﻌﻤـﺖ ﺩﻳﮕﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﭘﺲ ﺑﻪ ﺍﻭ ﻣﺘﻮﺳـﻞ ﺷـﻮﻳﺪ ﻭ ﺩﺍﻣﻦﻫﺎی ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻋﻄﺎﻳﺎی ﺍﻭ ﭘـﺮﻛﻨﻴـﺪ!»

ﻣﻄﻤﺌﻦ ﺑﺎﺷـﻴﺪ ﻛﻪ ﺍﮔﺮ ﺍﻳﻦ ﭘﻴﺎﻡ ﺭﺍ ﺑﻪ ﺩﺭﺳﺘﯽ ﺑﻪ ﻳﻜﺪﻳﮕﺮ ﻣﻨﺘﻘﻞ ﻛﻨﻴـﻢ ﻭ ﭘﯽ ﺩﺭ ﭘﯽ ﺁﻥ ﺭﺍ ﺑﻪ ﻫﻢ ﻳـﺎﺩﺁﻭﺭی ﻧﻤﺎﻳﻴﻢ ﺗﻮﺟﻪ ﻋﻤﻮﻣﯽ ﻣﺎﻥ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻴﺸـﺘﺮ ﺧﻮﺍﻫﺪ ﺷﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ، ﺍﻳﻦ ﻛﻌﺒﻪ ﻓﺮﺍﻣﻮﺵ ﺷـﺪﻩ ﺑﻴﺶ ﺍﺯ ﭘﻴﺶ ﺷـﺎﻫﺪ ﺯﺍﺋﺮﺍﻧﯽ ﭘﺮﺷـﻮﺭ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﻏﺒﺎﺭ ﻏﺮﺑﺘﺶ ﺑﺎ ﮔﻼﺏ ﺗﻮﺳـﻞ ﻣﺮﺩﻡ ﺯﺩﻭﺩﻩ ﺧﻮﺍﻫﺪ ﮔﺸـﺖ.
