بازگشت

ﺁﻥ ﻣﺮﺩ ﺩﺍﻧﺎ، ﻛﻪ ﻗﺪﺭﺵ ﺭﺍ ﻧﻤﯽ ﺩﺍﻧﻴﻢ.

ﭘﻴﺎﻣﺒـﺮ ﺧﺪﺍ ﺻﻠﯽ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ ﻓﺮﻣﻮﺩﻧﺪ: «ﺑﻪ ﺳـﻪ ﻛﺲ ﺭﺣﻢ ﻛﻨﻴﺪ. ﺍﻭﻝ، ﺑﻪ ﺷﺨﺼﯽ ﻛﻪ ﻗﺒﻼ ﻋﺰﻳﺰ ﺑﻮﺩﻩ ﻭ ﺍﻳﻨﮏ ﺧﻮﺍﺭ ﮔﺸﺘﻪ ﺍﺳﺖ. ﺩﻭﻡ، ﺑـﻪ ﻛﺴـﯽ ﻛﻪ ﺛﺮﻭﺗﻤﻨـﺪ ﺑـﻮﺩﻩ ﻭ ﺍﻛﻨﻮﻥ ﺩﭼﺎﺭ ﻓﻘﺮ ﻭ ﺗﻨﮕﺪﺳـﺘﯽ ﺷـﺪﻩ . ﺳـﻮﻡ، ﺑـﻪ ﻋﺎﻟﻤﯽ ﻛـﻪ ﺩﺭ ﻣﻴـﺎﻥ ﻣﺮﺩﻣﯽ ﻧﺎﺩﺍﻥ ﻗـﺮﺍﺭ ﮔﺮﻓﺘـﻪ ، ﻭ ﻗﺪﺭﺵ ﺩﺍﻧﺴﺘﻪ ﻧﻤﯽ ﺷﻮﺩ.»

ﺍﮔﺮ ﺍﺯ ﺷـﻤﺎ ﺑﭙﺮﺳـﻨﺪ ﺁﻳﺎ ﺩﺍﻧﺸـﻤﻨﺪی ﺭﺍ ﻣﯽ ﺷﻨﺎﺳـﻴﺪ ﻛﻪ ﺩﺭ ﻣﻴﺎﻥ ﻣـﺮﺩﻡ ﺑﺎﺷـﺪ، ﻭﻟﯽ ﺁﻧﻬـﺎ ﭼﻨﺎﻥ ﻛﻪ ﺑﺎﻳﺪ ﻭ ﺷـﺎﻳﺪ ﻗـﺪﺭ ﺍﻭ ﺭﺍ ﻧﺪﺍﻧﻨﺪ، ﭼﻪ ﻛﺴﯽ ﺭﺍ ﻣﻌﺮﻓﯽ ﻣﯽ ﻛﻨﻴﺪ؟ ﺁﻳﺎ ﺷﺨﺼﺎ ﻓﺮﺩ ﺩﺍﻧﺎﻳﯽ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳﻴﺪ ﻛﻪ ﺩﺭ ﺍﺛﺮ ﻋﺪﻡ ﺗﻮﺟﻪ ﻣﺮﺩﻡ ﺑﻪ ﺍﻭ ﮔﻮﺷﻪ ﮔﻴﺮ ﺷﺪﻩ ﺑﺎﺷﺪ؟ ﺁﻳﺎ ﺗﺎ ﻛﻨﻮﻥ ﺍﺯ ﻏﺮﺑﺖ ﻭ ﻣﻈﻠﻮﻣﻴﺖ ﻋﺎﻟﻤﯽ ﻣﻄﻠﻊ ﺷـﺪﻩ ﺍﻳﺪ ﻭ ﺩﻝ ﻭ ﺟﺎﻧﺘﺎﻥ ﺑﺮ ﻏﺮﺑﺖ ﺍﻭ ﺁﺗﺶ ﮔﺮﻓﺘﻪ ﻭ ﺳﻮﺧﺘﻪ ﺍﺳﺖ؟

ﺍﮔﺮ ﺍﺯ ﻣﻦ ﺩﺭﺑﺎﺭﻩ ﭼﻨﻴﻦ ﺷـﺨﺼﯽ ﺑﭙﺮﺳـﻴﺪ ﻣﯽ ﮔﻮﻳﻢ: «ﺩﺍﻧﺎﺗﺮ ﺍﺯ ﻫﻤـﻪ ﺩﺍﻧﺎﻫـﺎ، ﻋﺎﻟﻢ ﺗﺮ ﺍﺯ ﻫﻤﻪ ﻋﻠﻤـﺎ ﻭ ﺁﮔﺎﻩ ﺗﺮ ﺍﺯ ﻫﻤﻪ ﺁﮔﺎﻫـﺎﻥ، ﺍﻣﺮﻭﺯ ﻫﻤﺎﻥ ﻛﺴـﯽ ﺍﺳـﺖ ﻛﻪ ﺧﺪﺍ ﺍﻭ ﺭﺍ ﺍﺯ ﺳﺮﭼﺸﻤﻪ ﻋﻠﻮﻡ ﺁﺳﻤﺎﻧﯽ ﺧﻮﻳﺶ ﺳـﻴﺮﺍﺏ ﻛﺮﺩﻩ ﻭ ﺍﻭ ﺭﺍ ﺩﺭ ﻣﻴﺎﻥ ﺑﻨﺪﮔﺎﻥ ﺧﻮﻳﺶ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﻤﺎﻳﻨﺪﻩ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺍﺳﺖ.»

ﻣـﻦ ﻣﯽ ﮔﻮﻳـﻢ: «ﺍﻭ ﻫﻤﺎﻥ ﭼﺸـﻤﻪ ﺣﻴﺎﺕ، ﻣﻬﺪی ﺍﻣـﺖ، ﺩﺍﻧﺎی ﺩﺍﻧﺎﻳـﺎﻥ، ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻣﺎﺳـﺖ، ﻛﻪ ﻓﺮﺯﻧـﺪ ﮔﺮﺍﻣﯽ ﺍﻣﺎﻡ ﻋﺴـﻜﺮی ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺍﺳﺖ.»

ﻣﯽ ﮔﻮﻳـﻢ: «ﺑـﺎ ﺍﻳﻨﻜـﻪ ﺍﻭ ﺩﺍﻧﺎی ﺍﺳـﺮﺍﺭ ﭘﻨﻬـﺎﻥ، ﻭ ﺁﮔﺎﻩ ﺍﺯ ﺗﻤـﺎﻡ ﺭﺍﺯﻫـﺎی ﺻﻌــﻮﺩ ﺑـﻪ ﻣﻠﻜــﻮﺕ ﺁﺳـﻤﺎﻥ ﻭ ﺗﻮﺍﻧـﺎ ﺑـﻪ ﺩﺭﻣـﺎﻥ ﺗﻤـﺎﻡ ﺩﺭﺩﻫﺎی ﺑﺸـﺮ ﺍﺳــﺖ ﻭ ﺑﺎ ﺍﻳﻨﻜﻪ ﺍﻧﺴـﺎﻥ ﻫـﺎ ﺑﻪ ﺷـﻜـﻞ ﮔﺴـﺘﺮﺩﻩ ﺍی ﺑـﻪ ﺩﺍﻧـﺶ ﺍﻭ ﻧﻴـﺎﺯ ﺩﺍﺭﻧــﺪ، ﺑــﻪ ﺍﻭ ﭘﺸـﺖ ﻛــﺮﺩﻩ ﻭ ﺍﺯ ﺍﻭ ﻏﺎﻓــﻞ ﻣﺎﻧﺪﻩ ﺍﻧــﺪ.»

ﻣﯽ ﮔﻮﻳـﻢ: «ﺳﺎﻝﻫﺎﺳـﺖ ﻛـﻪ ﻣـﺮﺩﻡ، ﺍﻳﻦ ﭼﺸـﻤﻪ ﺣﻴـﺎﺕ ﺭﺍ ﺩﺭ ﻇﻠﻤﺎﺕ ﺩﻧﻴﺎی ﺁﺷﻔﺘﻪ ﻭ ﭘﺮ ﺳﺘﻤﺸﺎﻥ ﮔﻢ ﻛﺮﺩﻩ ﺍﻧﺪ، ﺍﺯ ﻧﻌﻤﺖ ﻭﺟﻮﺩﺵ ﺑﯽ ﺑﻬـﺮﻩ ﻣﺎﻧﺪﻩ ﺍﻧـﺪ ﻭ ﺩﺭ ﺍﻳﻦ ﺳـﺎﻟﻴﺎﻥ ﺩﺭﺍﺯ ﺍﻭ ﺗﻨﻬﺎﻳﯽ ﻭ ﻏﺮﺑﺖ ﺭﺍ ﺗﺠﺮﺑﻪ ﻣﯽ ﻛﻨﺪ.»

ﻣﯽ ﮔﻮﻳﻢ: «ﺍﻭ ﻫﻤﺎﻥ ﻛﺴـﯽ ﺍﺳـﺖ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺩﺭﺑﺎﺭﻩ ﺍﻭ ﻓﺮﻣﻮﺩ ﻛـﻪ ﻛﻤـﺎﻝ ﻣﻮﺳـﯽ ﺩﺍﺭﺩ ﻭ ﻧﻮﺭﺍﻧﻴـﺖ ﻋﻴﺴـﯽ ﻭ ﺩﺭ ﻋﻴـﻦ ﺣـﺎﻝ ﺻﺒـﺮ ﺍﻳﻮﺏ.»(7)

ﻣﯽ ﮔﻮﻳﻢ: «ﺍﻭ ﻫﻤﺎﻥ ﻣﻬﺪی ﻗﺎﺋﻢ ﺍﺳـﺖ ﻛﻪ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﺩﺭ ﺧﻄﺒﻪ ﺁﻳﻨﺪﻩ ﺳـﺎﺯ ﻏﺪﻳـﺮ ﺩﺭﺑـﺎﺭﻩ ﺍﻭ ﻓﺮﻣـﻮﺩ ﻛـﻪ ﺍﻭ ﺗﻤـﺎﻡ ﺩﺍﻧﺶﻫﺎ ﺭﺍ ﺑـﻪ ﺍﺭﺙ ﻣﯽ ﺑﺮﺩ ﻭ ﺑﻪ ﻫﻤﻪ ﻋﻠﻮﻡ ﺍﺣﺎﻃﻪ ﺩﺍﺭﺩ.»

ﻣﯽ ﮔﻮﻳﻢ: «ﺍﻭ ﻫﻤﺎﻥ ﺑﺰﺭگﻣﺮﺩی ﺍﺳـﺖ ﻛﻪ ﻧﻴﺎی ﺍﻭ ﺍﻣﻴﺮﻣﺆﻣﻨﺎﻥ ﺩﺭﺑـﺎﺭﻩ ﺍﺵ ﻓﺮﻣـﻮﺩ ﻛﻪ ﺁﻏﻮﺵ ﭘﻨﺎﻫﺶ ﺑﻴﺶ ﺍﺯ ﻫـﺮ ﭘﻨﺎﻫﮕﺎﻫﯽ ﺑﻪ ﺭﻭی ﭘﻨﺎﻩ ﺟﻮﻳﺎﻥ ﮔﺸـﺎﺩﻩ ﺍﺳـﺖ ﻭ ﺩﺍﻧﺸـﺶ ﺑﻴﺶ ﺍﺯ ﺩﺍﻧﺶ ﻫﻤﻪ ﺷﻤﺎﺳﺖ ﻭ ﺩﻟﺴﻮﺯی ﺍﺵ ﺍﺯ ﻫﻤﻪ ﺷﻤﺎ ﻓﺮﺍﮔﻴﺮﺗﺮ ﺍﺳﺖ.»(8)

ﻭ ﻣﯽ ﮔﻮﻳﻢ: «ﺍﻭ ﻫﻤﺎﻥ ﻛﺴـﯽ ﺍﺳـﺖ ﻛﻪ ﺍﻣﺎﻡ ﻛﺎﻇﻢ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻏﺮﺑﺘـﺶ ﺭﺍ ﭘﻴﺶ ﺍﺯ ﻭﻻﺩﺗﺶ ﭘﻴﺶﮔﻮﻳﯽ ﻛـﺮﺩﻩ ﻭ ﺩﺭﺑﺎﺭﻩ ﺍﺵ ﻓﺮﻣﻮﺩﻩ : ﺍﻭ ﻫﻤـﺎﻥ ﻃﺮﺩﺷـﺪﻩ ﺗﻨﻬـﺎ ﻭ ﻏﺮﻳـﺐ ﺍﺳـﺖ ﻛـﻪ ﺍﺯ ﻧﺰﺩﻳﻜﺎﻧـﺶ ﻏﺎﻳﺐ ﻣﯽ ﺷﻮﺩ.»(9)

ﻣـﻦ ﻣﯽ ﮔﻮﻳـﻢ: «ﻭﻗﺘﯽ ﺑـﻪ ﺩﺭﻣﺎﻧﺪﮔﯽ ﻫﺎﻳـﯽ ﻛـﻪ ﺩﺭ ﺩﻭﺭی ﺍﺯ ﺍﻭ ﮔﺮﻳﺒﺎﻥ ﮔﻴﺮ ﻣﺎ ﺷـﺪﻩ ﻣﯽ ﻧﮕﺮﻳﻢ، ﺩﻟﻤﺎﻥ ﻣﯽ ﺳـﻮﺯﺩ ﺍﺯ ﺍﻳﻨﻜـﻪ ﻛﻪ ﭼﺮﺍ ﻣﺎ ﺍﻧﺴﺎﻥ ﻫﺎ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﺩﺭ ﻧﺰﺩﻳﻜﯽ ﻳﮏ ﭼﺸﻤﻪ ﺣﻴﺎﺕ ﺑﺨﺶ ﺍﺯ ﺗﺸﻨﮕﯽ ﺟـﺎﻥ ﺑﺪﻫﻴﻢ، ﺍﺯ ﺑﻴﻦ ﺑﺮﻭﻳﻢ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺑﻪ ﺭﺷـﺪ ﻣﻄﻠﻮﺑﯽ ﻛﻪ ﺷﺎﻳﺴـﺘﻪ ﻣﺎﺳﺖ ﻧﺮﺳﻴﻢ؟»

ﺍﻣـﺎ ﺩﻝ ﻣـﻦ ﺗﻨﻬـﺎ ﺑﺮﺍی ﺍﺯ ﺩﺳـﺖ ﺭﻓﺘـﻦ ﺧﻮﺩﻣﺎﻥ ﻧﻤﯽ ﺳـﻮﺯﺩ، ﺑﻠﻜـﻪ ﺩﻟـﻢ ﺑﻴﺸـﺘﺮ ﺑـﺮﺍی ﺁﻥ ﭼﺸـﻤﻪ ﺣﻴـﺎﺕ ﻣﯽ ﺳـﻮﺯﺩ ﻛﻪ ﻛﺴـﯽ ﺑﻪ ﺍﻭ ﺭﻏﺒـﺖ ﻧﺸـﺎﻥ ﻧﻤﯽ ﺩﻫـﺪ. ﺩﻟﻢ ﺑـﺮﺍی ﺁﻥ ﭼﺎﻩ ﭘﺮﺁﺏ ﻣﯽ ﺳـﻮﺯﺩ ﻛﻪ ﻣﻌﻄـﻞ ﻭ ﺑﯽ ﻧﺸـﺎﻥ ﻭ ﻏﺮﻳـﺐ ﻭ ﺩﻭﺭﺍﻓﺘـﺎﺩﻩ ﻣﺎﻧـﺪﻩ ﺍﺳـﺖ. ﺩﻟﻢ ﺑﺮﺍی ﺁﻥ ﻣـﺮﺩ ﺩﺍﻧــﺎ ﻣﯽ ﺳـﻮﺯﺩ ﻛﻪ ﻣـﺎ ﻧﻴﺎﺯﻣﻨﺪﺍﻥ ﺩﺍﻧـﺶ، ﺍﻭ ﺭﺍ ﺑﻪ ﺭﺍﺳـﺘﯽ ﻃﻠـﺐ ﻧﻤﯽ ﻛﻨﻴﻢ. («ﻣﺎ» ﻛﻪ ﻣﯽ ﮔﻮﻳﻢ، ﻣﻨﻈﻮﺭﻡ ﻫﻤﻪ «ﻣﺎﺳـﺖ» ﻳﻌﻨﯽ ﺗﻤـﺎﻡ ﻣــﺮﺩﻡ ﺩﻧﻴـﺎ.) ﺑﺸـﺮ ﺁﻧﻘـﺪﺭ ﺧـﻮﺩﺵ ﺭﺍ ﺍﺯ ﺍﻭ ﺟﺪﺍ ﺳـﺎﺧﺘﻪ ﻛﻪ ﺑﺴـﻴﺎﺭی ﺍﺯ ﻣـﺎ ﺁﺩﻡﻫﺎ ﺣﺘــﯽ ﻧﻤﯽ ﺩﺍﻧﻴــﻢ ﻛـﻪ ﺍﻭ ﻭﺟــﻮﺩ ﺩﺍﺭﺩ ﻭ ﺩﺭ ﻛﻨـﺎﺭﻣﺎﺳـﺖ.

ﻣـﻦ ﺩﻟﻢ ﻧـﻪ ﺗﻨﻬﺎ ﺑﺮ ﺧﻮﺩﻣﺎﻥ ﺑﻠﻜﻪ ﺑﺮ ﺍﻭ ﻣﯽ ﺳـﻮﺯﺩ ﻭ ﺍﺯ ﺗﻨﻬﺎﻳﯽ ﻭ ﻏﺮﺑﺖ ﺷـﺪﻳﺪﺵ ﻛﻪ ﻋﻠﻴﺮﻏﻢ ﺩﺍﻧﺎﻳﯽ ﻭ ﺑﺮﻛﺎﺕ ﻭﺟﻮﺩ ﺍﻭﺳـﺖ، ﻏﺼﻪ ﻣﯽ ﺧﻮﺭﻡ.

ﻭ ﭘﻴﺎﻣﺒﺮ ﻫﻢ ﺑﻪ ﻣﺎ ﮔﻔﺘﻪ ﺑﻮﺩ ﺑﻪ ﺁﻥ ﺩﺍﻧﺎﻳﯽ ﻛﻪ ﺩﺭ ﻣﻴﺎﻥ ﻧﺎﺩﺍﻧﺎﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﻗﺪﺭ ﺍﻭ ﺩﺍﻧﺴـﺘﻪ ﻧﻤﯽ ﺷـﻮﺩ ﺭﺣـﻢ ﻛﻨﻴﺪ، ﻳﻌﻨـﯽ ﺩﻟﻬﺎﻳﺘﺎﻥ ﺑﺮﺍی ﻏﺮﺑﺘﺶ ﺑﺴﻮﺯﺩ.
