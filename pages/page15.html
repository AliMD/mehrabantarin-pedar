ﺧﺪﺍﻭﻧﺪ ﻣﺘﻌﺎﻝ ﺩﺭ ﺿﻤﻦ ﺍﻳﻦ ﺁﻳﻪ ﺗﺼﺮﻳﺢ ﻣﯽ ﻓﺮﻣﺎﻳﺪ ﻛﻪ ﺷﺎﻳﺴـﺘﻪ ﻧﻴﺴـﺖ ﻣﺆﻣﻨﺎﻥ، ﺭﺍﺣﺖ ﻭ ﻋﺎﻓﻴﺖ ﺧـﻮﺩ ﺭﺍ ﺑﺮ ﺭﺍﺣﺖ ﻭ ﻋﺎﻓﻴﺖ ﭘﻴﺎﻣﺒﺮ ﺧـﺪﺍ ﺗﺮﺟﻴﺢ ﺩﻫﻨـﺪ. ﻭﺟﺪﺍﻥ ﻣﺎ ﻧﻴﺰ ﺑﻪ ﻣﺎ ﺣﻜﻢ ﻣﯽ ﻛﻨـﺪ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﻣـﻮﻻ ﻭ ﻭﻟﯽ ﻧﻌﻤﺖ ﺧﻮﻳـﺶ ﺗﺮﺟﻴﺢ ﻧﺪﻫﻴﻢ. ﺍﻳﻦ ﺭﻭﺵ ﺧﺪﺍﭘﺴـﻨﺪﺍﻧﻪ ﻭ ﺧﺮﺩﭘﺴـﻨﺪ ﻳﻌﻨﯽ ﻣﻘﺪﻡ ﺩﺍﺷـﺘﻦ ﻣﻮﻻ ﺑﺮ ﺧﻮﺩ، ﺷـﺮﻁ ﺍﻳﻤﺎﻥ ﻭﺍﻗﻌﯽ ﻭ ﻧﺸـﺎﻧﻪ ﻣﺤﺒـﺖ ﺣﻘﻴﻘﯽ ﺑﻪ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﺍﺳـﺖ. ﻫﻤﻴـﻦ ﻭﺿﻌﻴﺖ ﺩﺭﺑﺎﺭﻩ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻳﻌﻨﯽ ﻓﺮﺯﻧﺪ ﺑﺮﮔﺰﻳﺪﻩ ﭘﻴﺎﻣﺒﺮ ﻭ ﺟﺎﻧﺸﻴﻦ ﺯﻧﺪﻩ ﺍﻭ ﻧﻴﺰ ﺟﺎﺭی ﺍﺳﺖ.

ﺭﺳـﻮﻝ ﺧﺪﺍ ﻛﻪ ﺩﺭﻭﺩ ﭘﺮﻭﺭﺩﮔﺎﺭ ﺑـﺮ ﺍﻭ ﻭ ﺧﺎﻧﺪﺍﻧﺶ ﺑﺎﺩ ﺧﻮﺩ ﻓﺮﻣﻮﺩﻩ ﺍﻧـﺪ: «ﻫﻴﭻ ﺑﻨﺪﻩ ﺍی ﺑﻪ (ﻛﻤﺎﻝ) ﺍﻳﻤﺎﻥ ﺩﺳـﺖ ﻧﻴﺎﺑـﺪ ﻣﮕﺮ ﺍﻳﻨﻜﻪ ﻣﻦ، ﺩﺭ ﻧﺰﺩ ﺍﻭ ﺍﺯ ﺧﻮﺩﺵ ﻣﺤﺒﻮﺑﺘﺮ ﺑﺎﺷﻢ، ﻭ ﺧﺎﻧﻮﺍﺩﻩ ﺍﻡ ﺍﺯ ﺧﺎﻧﻮﺍﺩﻩ ﺍﺵ ﻧﺰﺩ ﺍﻭ ﻣﺤﺒﻮﺑﺘﺮ ﺑﺎﺷﻨﺪ.»(51)

ﺁﻧﻬـﺎ ﻛـﻪ ﺩﺭ ﻃﺮﻳـﻖ ﻣﺤﺒـﺖ ﺍﻣـﺎﻡ ﻋﺼـﺮ ﺍﺭﻭﺍﺣﻨـﺎ ﻓﺪﺍﻩ ﭘﻴﺸـﺘﺎﺯ ﺑﻮﺩﻩ ﺍﻧـﺪ، ﺑﺰﺭﮔﺘﺮﻳـﻦ ﺭﺍﺯ ﺗﻘـﺮﺏ ﺑـﻪ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺭﺍ ﻣﺤﺒﺖ ﺩﺍﺷـﺘﻦ ﺑﻪ ﺍﻭ ﺩﺍﻧﺴـﺘﻪ ﺍﻧﺪ. ﺍﺯ ﺑﺮﺗﺮﻳﻦ ﻧﺸـﺎﻧﻪ ﻫﺎی ﻣﺤﺒﺖ ﻭﺍﻗﻌﯽ ﻧﻴـﺰ ﺗﺮﺟﻴﺢ ﺩﺍﺩﻥ ﻣﺤﺒـﻮﺏ ﻭ ﺧﻮﺍﺳـﺘﻪ ﻫﺎی ﺍﻭ ﺑـﺮ ﺧﻮﺍﺳـﺘﻪ ﻫﺎی ﺧﻮﺩ ﺍﺳـﺖ. ﺗﻨﻬﺎ ﺍﮔﺮ ﺍﻳـﻦ ﺣﺎﻟﺖ ﺑﺮ ﺍﻧﺴـﺎﻥ ﺣﺎﻛﻢ ﺷـﻮﺩ، ﺷـﺨﺺ ﺣﺎﺿﺮ ﺧﻮﺍﻫﺪ ﺷـﺪ ﻛﻪ ﻳـﺎﺭ ﻓﺪﺍﻛﺎﺭ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺶ ﮔﺮﺩﺩ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺍﺳـﺖ ﻛﻪ ﺍﻧﺴـﺎﻥ ﺟﺎﻥ ﻭ ﻣـﺎﻝ ﻭ ﺧﺎﻧـﻮﺍﺩﻩ ﻭ ﻋﺰﻳﺰﺍﻥ ﺧـﻮﺩ ﺭﺍ ﻓﺪﻳـﻪ ﺁﺯﺍﺩی ﻣﺤﺒﻮﺑﺶ ﺍﺯ ﺭﻧﺞ ﻭ ﻏﻢ ﻣﯽ ﺳـﺎﺯﺩ. ﻳـﺎﺭی ﻓﺪﺍﻛﺎﺭﺍﻧﻪ ﻧﺴـﺒﺖ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺯ ﺳﺮﭼﺸـﻤﻪ ﭼﻨﻴﻦ ﻣﺤﺒﺘﯽ ﺍﺳﺖ ﻛﻪ ﺳﻴﺮﺍﺏ ﻣﯽ ﺷﻮﺩ ﻭ ﺍﺯ ﻧﻮﺭ ﭼﻨﻴﻦ ﺍﺭﺍﺩﺗﯽ ﺍﺳﺖ ﻛﻪ ﭘﺮﺗﻮ ﻣﯽ ﮔﻴﺮﺩ.

ﻣﺤـﺐ ﺩﻟﺴـﻮﺧﺘﻪ ﺍی ﻛـﻪ ﭼﻨﻴـﻦ ﺩﺭ ﻣﺤﺒـﻮﺏ ﺧﻮﺩ ﺫﻭﺏ ﺷـﺪﻩ ﺑﺎﺷـﺪ، ﻫﺮﮔﺎﻩ ﺑﺨﻮﺍﻫﺪ ﻧﻌﻤﺘﯽ ﺭﺍ ﺑﻪ ﺳـﻮی ﺧﻮﺩ ﺟﻠـﺐ ﻛﻨﺪ ﻳﺎ ﺑﻼﻳﯽ ﺭﺍ ﺍﺯ ﺧـﻮﺩ ﺩﻓـﻊ ﻧﻤﺎﻳـﺪ، ﺗﻮﻗـﻒ ﻣﯽ ﻛﻨـﺪ. ﻧﮕﺎﻫـﯽ ﺑﻪ ﻣﺤﺒـﻮﺏ ﺧﻮﺩ ﻣﯽ ﺍﻧﺪﺍﺯﺩ، ﻭﺿﻌﻴﺖ ﻭ ﺷـﺮﺍﻳﻂ ﺍﻭ ﺭﺍ ﻣﯽ ﺑﻴﻨﺪ ﻭ ﺍﻭ ﺭﺍ ﺩﺭ ﺑﺮﺧﻮﺭﺩﺍﺭی ﺍﺯ ﻧﻌﻤﺖ ﻭ ﺍﻳﻤﻨﯽ ﺍﺯ ﺑﻼﻳﺎ ﺑﺮ ﺧﻮﺩ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﺪ.

ﺍﻭﺝ ﻣﺮﺍﺗـﺐ ﺍﻳﻤـﺎﻥ (ﺍﻳﻤﺎﻧـﯽ ﻛـﻪ ﺍﺯ ﭼﻨﻴﻦ ﺷـﻮﺭی ﺑﺮﺧـﻮﺭﺩﺍﺭ ﺍﺳـﺖ)، ﺩﺭ ﺻﺤﻨﻪ ﻛﺮﺑﻼی ﺳـﺎﻝ 61 ﻫﺠﺮی ﺟﻠﻮﻩ ﮔﺮ ﺷـﺪﻩ ﻭ ﺑﺮﺍی ﻛﺴﺎﻧﯽ ﻛﻪ ﻣﯽ ﺧﻮﺍﻫﻨﺪ ﻳﺎﻭﺭ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺸﺎﻥ ﺑﺎﺷﻨﺪ ﺑﻬﺘﺮﻳﻦ ﺍﻟﮕﻮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﺩ.

ﻋﺒـﺎﺱ ﺑـﻦ ﻋﻠﯽ ﻧﻤﻮﻧﻪ ﺍی ﺑـﺎﺭﺯ ﺍﺯ ﻳﺎﺭﺍﻥ ﺑﺎﻭﻓﺎﻳﯽ ﺍﺳـﺖ ﻛﻪ ﺍﻣﺎﻡ ﺧـﻮﺩ ﺭﺍ ﺑـﺮ ﺧـﻮﺩ ﺗﺮﺟﻴـﺢ ﻣﯽ ﺩﻫﻨـﺪ ﻭ ﺩﺭ ﺣﺎﻟـﯽ ﻛـﻪ ﺧﻮﺍﺳـﺘﻪ ﻫﺎی ﺍﻣﺎﻣﺸـﺎﻥ ﺑﺮﺁﻭﺭﺩﻩ ﻧﺸﺪﻩ، ﺧﻮﺍﺳﺘﻪ ﻫﺎی ﺧﻮﺩ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﻧﻤﯽ ﻛﻨﻨﺪ. ﺩﺭ ﺁﻥ ﻟﺤﻈﻪ ﻫﺎی ﭘﺎﻳﺎﻧﯽ ﻭﻗﺘﯽ ﻟﻬﻴﺐ ﺁﺗﺶ ﺍﺯ ﺁﺳﻤﺎﻥ ﻣﯽ ﺑﺎﺭﻳﺪ ﻭ ﻋﻄﺶ ﺍﺯ ﺩﺭﻭﻥ ﺷﻌﻠﻪ ﻣﯽ ﻛﺸﻴﺪ، ﻋﺒﺎﺱ ﺑﻦ ﻋﻠﯽ ﺑﺮﺍی ﺁﻭﺭﺩﻥ ﺁﺏ، ﺳﭙﺎﻩ ﺩﺷﻤﻦ ﺭﺍ ﺷـﻜﺎﻓﺖ ﻭ ﭘـﺎی ﺑـﻪ ﻧﻬـﺮ ﺁﺏ ﻧﻬﺎﺩ. ﺩﺭ ﺍﻭﺝ ﺗﺸـﻨﮕﯽ ﺩﺳـﺖ ﺑﻪ ﺯﻳﺮ ﺁﺏ ﺑﺮﺩ ﺗﺎ ﺑﻨﻮﺷـﺪ ﺍﻣـﺎ ﭼﻮﻥ ﺁﺏ ﺭﺍ ﺑﻪ ﻧﺰﺩﻳﮏ ﻟﺐﻫﺎی ﺧﻮﺩ ﺭﺳـﺎﻧﺪ، ﺗﺸـﻨﮕﯽ ﻣﻮﻻﻳﺶ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺁﻭﺭﺩ. ﺁﺏ ﺭﺍ ﺑﺮ ﺁﺏ ﺭﻳﺨﺖ ﻭ ﻣﺸـﮏ ﺭﺍ ﭘﺮ ﻛﺮﺩ ﻭ ﺗﺸـﻨﻪ ﺑﻪ ﺳـﻮی ﺧﻴﻤﻪ ﮔﺎﻩ ﺑﺮﺍﺩﺭ ﺭﻭﺍﻧﻪ ﺷﺪ. ﺍﻳﻦ ﺻﺤﻨﻪ ﺗﺎﺭﻳﺨﯽ ﺗﻨﻬـﺎ ﻳـﮏ ﻧﻤﻮﻧﻪ ﺍﺯ ﻓﺪﺍﻛﺎﺭی ﻫـﺎی ﺧﺎﻧﺪﺍﻥ ﻭ ﻳﺎﺭﺍﻥ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﻋﺮﺻﻪ ﻛﺮﺑﻼﺳﺖ.

ﺷـﻴﻌﻴﺎﻥ ﺭﺍﺳـﺘﻴﻦ ﭘﻴـﺶ ﺍﺯ ﺣﺎﺩﺛﻪ ﻛﺮﺑـﻼ ﻭ ﭘﺲ ﺍﺯ ﺁﻥ ﻧﻴـﺰ ﺑﺎ ﺍﻳﻦ ﮔﻮﻧﻪ ﻭﻓﺎﺩﺍﺭی ﻫﺎ ﻭ ﻓﺪﺍﻛﺎﺭی ﻫﺎ ﺑﻪ ﻣﻘﺎﻡ ﻗﺮﺏ ﺍﻣﺎﻣﺎﻥ ﺭﺍﻩ ﻳﺎﻓﺘﻪ ﻭ ﻣﻮﺭﺩ ﻋﻨﺎﻳـﺖ ﺧﺎﺹ ﺧﺪﺍﻭﻧﺪ ﻗـﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ. ﺑـﻪ ﻋﻨﻮﺍﻥ ﻣﺜـﺎﻝ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﺟﻨﺎﺏ ﺳـﻠﻤﺎﻥ ﻳـﺎﺩ ﻛﺮﺩ. ﺍﮔﺮ ﺳـﻠﻤﺎﻥ (ﻛﻪ ﺍﻣﺎﻡ ﺻـﺎﺩﻕ ﺍﻭ ﺭﺍ ﺩﺭﻳﺎﻳﯽ ﺑﯽ ﻛﺮﺍﻧﻪ ﺧﻮﺍﻧﺪﻩ ﺍﺳـﺖ) ﺳﻠﻤﺎﻥ ﺷﺪ ﻭ ﺩﺭ ﺑﺎﻻﺗﺮﻳﻦ ﺩﺭﺟﻪ ﺍﻳﻤﺎﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺖ، ﺍﺯ ﺭﻫﮕﺬﺭ ﻫﻤﻴﻦ ﺍﺭﺍﺩﺕ ﺑﻪ ﻣﻮﻻی ﺧﻮﺩ ﻭ ﻣﻘﺪﻡ ﺩﺍﺷﺘﻦ ﺍﻭ ﺑﺮ ﺧﻮﺩﺵ ﺑﻮﺩﻩ ﺍﺳـﺖ ﻭ ﺍﮔﺮ ﺩﺭ ﺭﻭﺯﮔﺎﺭ ﻣﺎ ﻧﻴﺰ ﻛﺴـﯽ ﺑﺨﻮﺍﻫﺪ ﻫﻤﭽﻮﻥ ﺳـﻠﻤﺎﻥ ﺩﺭ ﻧـﺰﺩ ﻣﻮﻻﻳـﺶ ﻣﻘﺮﺏ ﺷـﻮﺩ، ﺑﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﺷـﻴﻮﻩ ﺑﻪ ﻣﻮﻻی ﺧﻮﺩ ﻣﻬﺮ ﺑﻮﺭﺯﺩ.

ﻣﻨﺼـﻮﺭ ﺑـﺰﺭﺝ ﺍﺯ ﺷـﻴﻌﻴﺎﻥ ﺍﻣﺎﻡ ﺻـﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﮔﻮﻳﺪ: ﺑﻪ ﺣﻀـﺮﺕ ﺍﺑﯽ ﻋﺒﺪﺍﻟﻠـﻪ ﺍﻣﺎﻡ ﺻـﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻋﺮﺿﻪ ﺩﺍﺷـﺘﻢ: ﺍی ﺁﻗﺎی ﻣﻦ، ﭼﻪ ﺑﺴﻴﺎﺭ ﺍﺯ ﺷﻤﺎ ﻳﺎﺩ ﺳﻠﻤﺎﻥ ﻓﺎﺭﺳﯽ ﺭﺍ ﻣﯽ ﺷﻨﻮﻡ؟ ﻓﺮﻣﻮﺩ: «ﻧﮕﻮ ﺳـﻠﻤﺎﻥ ﻓﺎﺭﺳـﯽ ﺑﻠﻜﻪ ﺑﮕﻮ ﺳـﻠﻤﺎﻥ ﻣﺤﻤﺪی! ﺁﻳﺎ ﻣﯽ ﺩﺍﻧﯽ ﭼﺮﺍ ﺑﺴﻴﺎﺭ ﺍﺯ ﺍﻭ ﻳﺎﺩ ﻣﯽ ﻛﻨﻢ؟» ﻋﺮﺽ ﻛﺮﺩﻡ: ﻧﻪ، ﻓﺮﻣﻮﺩ: «ﺑﺮﺍی ﺳﻪ ﺻﻔﺖ ﻛﻪ ﺩﺭ ﺍﻭ ﺑﻮﺩ. ﺍﻭﻝ ﻣﻘﺪﻡ ﺩﺍﺷـﺘﻦ ﺧﻮﺍﺳـﺘﻪ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﺑﺮ ﺧﻮﺍﺳـﺘﻪ ﺧﻮﺩﺵ. ﺩﻭﻡ ﺩﻭﺳـﺖ ﺩﺍﺷـﺘﻦ ﻓﻘﺮﺍ، ﻭ ﺑﺮﮔﺰﻳﺪﻥ ﺁﻥﻫﺎ ﺑﺮ ﺍﻫﻞ ﺛﺮﻭﺕ ﻭ ﻣﻜﻨﺖ، ﻭ ﺳـﻮﻡ ﺩﻭﺳـﺖ ﺩﺍﺷﺘﻦ ﻋﻠﻢ ﻭ ﻋﻠﻤﺎ. ﻫﻤﺎﻧﺎ ﺳﻠﻤﺎﻥ ﺑﻨﺪﻩ ی ﺻﺎﻟﺢ ﺩﺭﺳﺘﻜﺎﺭ ﻣﺴﻠﻤﺎﻧﯽ ﺑﻮﺩ، ﻭ ﺍﺯ ﻣﺸﺮﻛﺎﻥ ﻧﺒﻮﺩ.»(52)

ﻭ ﺍﻣﺎ ﺩﻭ ﺍﻣﺮ ﺍﺯ ﺍﻣﻮﺭی ﻛﻪ ﻣﯽ ﺗﻮﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺭﺍ ﺩﺭ ﺁﻥﻫﺎ ﺑﺮ ﺧﻮﺩ ﺗﺮﺟﻴﺢ ﺩﺍﺩ ﻭ ﻣﻘﺪﻡ ﺩﺍﺷﺖ، ﻳﻜﯽ ﺩﻋﺎ ﻭ ﻧﻴﺎﻳﺶ ﻭ ﺩﻳﮕﺮی ﺻﺪﻗﻪ ﺍﺳﺖ، ﭼﻨﺎﻧﻜﻪ ﺷﺎﻋﺮ ﮔﻔﺘﻪ ﺍﺳﺖ:

ﺍﻫﻞ ﻭﻻ ﭼﻮ ﺭﻭی، ﺑﻪ ﺳﻮی ﺧﺪﺍ ﻛﻨﻨﺪ ﺍﻭﻝ ﺑﻪ ﺟﺎﻥ ﮔﻤﺸﺪﻩ ﺧﻮﺩ ﺩﻋﺎ ﻛﻨﻨﺪ

ﺍﮔـﺮ ﻣﺤﺐ ﺩﻟﺴـﻮﺧﺘﻪ ﺩﺭ ﺣﺎﻟﺘﯽ ﻗﺮﺍﺭ ﺑﮕﻴـﺮﺩ ﻛﻪ ﻣﻮﺿﻊ ﺍﺟﺎﺑﺖ ﺩﻋﺎﺳـﺖ، ﺩﻋﺎی ﺑﺮ ﻣﻮﻻ ﺭﺍ ﺑﺮ ﺩﻋﺎی ﺑﺮ ﺧـﻮﺩ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﺪ ﻭ ﺍﮔﺮ ﺑـﺮﺍی ﺍﻭ ﺍﻣـﻜﺎﻥ ﺻﺪﻗـﻪ ﺩﺍﺩﻥ ﻭ ﻛﻤﮏ ﺑﻪ ﻓﻘﺮﺍ ﭘﻴﺶ ﺁﻳـﺪ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑﻪ ﻧﻴﺎﺑﺖ ﻭ ﺍﺯ ﻃﺮﻑ ﻣﺤﺒﻮﺏ ﺧﻮﺩ ﺍﻧﺠﺎﻡ ﻣﯽ ﺩﻫﺪ.

ﺩﺭ ﺁﺋﻴـﻦ ﺍﺳـﻼﻡ ﺳـﻔﺎﺭﺵ ﻓﺮﺍﻭﺍﻥ ﺷـﺪﻩ ﻛﻪ ﺑﻪ ﻧﻴﺎﺯﻣﻨـﺪﺍﻥ ﻛﻤﮏ ﻛﻨﻴﻢ ﻭ ﺍﺻﻄﻼﺣﺎ ﺑﻪ ﺁﻧﻬﺎ ﺻﺪﻗﻪ ﺑﺪﻫﻴﻢ. ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﺑﻪ ﻣﺎ ﺧﺒﺮ ﺩﺍﺩﻩ ﻛﻪ ﺻﺪﻗﻪ ﺭﻓﻊ ﺑﻼ ﻣﯽ ﻛﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺗﻘﺪﻳﺮ ﺭﺍ ﺑﻪ ﻧﻔﻊ ﻛﺴﯽ ﻛﻪ ﺻﺪﻗﻪ ﻣﯽ ﺩﻫﺪ ﺗﻐﻴﻴﺮ ﻣﯽ ﺩﻫﺪ ﻭ ﺑﻼﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺑﻨﺎ ﺑﺮ ﻧﻘﺸﻪ ﺗﻘﺪﻳﺮ ﺑﺮ ﺳﺮ ﺭﺍﻩ ﺍﻭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ، ﺍﺯ ﺍﻭ ﺩﻭﺭ ﻣﯽ ﺳﺎﺯﺩ.

ﻫﻤـﻪ ﻣـﺎ ﻧﺴـﺒﺖ ﺑـﻪ ﺻﺪﻗـﻪ ﻭ ﺗﺄﺛﻴـﺮ ﺁﻥ ﻣﻌﺘﻘـﺪ ﻫﺴـﺘﻴﻢ ﻭ ﺑﺮﺍی ﺳـﻼﻣﺖ ﺧـﻮﺩ ﻭ ﻋﺰﻳﺰﺍﻧﻤﺎﻥ ﺻﺪﻗـﻪ ﻣﯽ ﺩﻫﻴﻢ. ﻣـﺎ ﺩﺍﺩﻥ ﺻﺪﻗﻪ ﺭﺍ ﻧﻪ ﻓﻘـﻂ ﺑـﺮﺍی ﺭﻓﻊ ﺑـﻼ ﺍﺯ ﺧﻮﺩﻣﺎﻥ ﺑﻠﻜﻪ ﺑـﺮﺍی ﺭﻓﻊ ﺑـﻼ ﺍﺯ ﻫﺮ ﻛﺲ ﻛﻪ ﺑـﺮﺍی ﻣﺎ ﻋﺰﻳﺰ ﺑﺎﺷـﺪ ﺍﻧﺠـﺎﻡ ﻣﯽ ﺩﻫﻴﻢ. ﺑﻪ ﺧﺼﻮﺹ ﭘـﺪﺭﺍﻥ ﻭ ﻣﺎﺩﺭﺍﻥ ﺑـﺮﺍی ﻓﺮﺯﻧﺪﺍﻧﺸـﺎﻥ ﻭ ﺍﺯ ﻃﺮﻑ ﺁﻥﻫﺎ ﺻﺪﻗـﻪ ﻣﯽ ﺩﻫﻨﺪ. ﺗﺠﺮﺑﻪ ﻧﺸـﺎﻥ ﺩﺍﺩﻩ ﺍﺯ ﺁﻧﺠﺎ ﻛﻪ ﻓﺮﺩ ﺩﺭ ﺻﺪﻗﻪ ﺩﺍﺩﻥ ﺳﺨﺖﺗﺮﻳﻦ ﺭﻳﺎﺿﺖﻫﺎ ﺭﺍ (ﻳﻌﻨﯽ ﺩﻝ ﻛﻨـﺪﻥ ﺍﺯ ﻣـﺎﻝ ﺭﺍ) ﺍﻧﺠـﺎﻡ ﻣﯽ ﺩﻫﺪ، ﺑﻪ ﺭﺍﺣﺘﯽ ﻭ ﺑﺮﺍی ﻫﺮﻛﺴـﯽ ﺑﻪ ﺩﺍﺩﻥ ﺻﺪﻗﻪ ﺍﻗﺪﺍﻡ ﻧﻤﯽ ﻛﻨﺪ. ﻋﻤﻮﻡ ﻣﺎ ﺗﻨﻬﺎ ﺣﺎﺿﺮﻳﻢ ﻛﻪ ﺑﺮﺍی ﺳﻼﻣﺖ ﻭ ﻋﺎﻓﻴﺖ ﻛﺴـﺎﻧﯽ ﻣﺮﺗﺒﺎ ﺻﺪﻗﻪ ﺩﻫﻴﻢ ﻛﻪ ﺑﻪ ﺷـﺪﺕ ﺩﻭﺳﺘﺸـﺎﻥ ﺩﺍﺭﻳﻢ. ﺷﺎﻳﺪ ﺻﺪﻗﻪ ﺭﺍ ﺍﺯ ﺁﻥ ﺟﻬﺖ ﺻﺪﻗﻪ ﮔﻔﺘﻪ ﺍﻧﺪ ﻛﻪ ﻧﺸﺎﻥ ﺻﺪﻕ ﻭ ﺭﺍﺳﺘﯽ ﺍﺩﻋﺎﻫﺎی ﻣﺎﺳﺖ.

ﺁﺭی ﭘﺮﺩﺍﺧﺖ ﺻﺪﻗﻪ ﺑﺮﺍی ﺳـﻼﻣﺖ ﻭ ﻋﺎﻓﻴﺖ ﻛﺴـﯽ ﺍﺯ ﺑﻼ، ﺁﻥ ﻫﻢ ﻧﻪ ﻳﮏ ﺑﺎﺭ ﻭ ﺩﻭ ﺑﺎﺭ ﺑﻠﻜﻪ ﻫﺮ ﺭﻭﺯ ﻭ ﻫﻤﻴﺸـﻪ، ﻧﺸـﺎﻧﻪ ﺭﺍﺳـﺖ ﺑﻮﺩﻥ ﺍﺩﻋﺎی ﻣﺎ ﺩﺭ ﻣﺤﺒﺖ ﺑﻪ ﺍﻭﺳﺖ.

ﺑﻪ ﺁﻧﻬﺎ ﻛﻪ ﺩﻭﺳـﺖ ﺩﺍﺭﻧﺪ ﻣﺤﺒﺖ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺸـﺎﻥ ﻧﺸﺎﻥ ﺩﻫﻨﺪ ﺗﻮﺻﻴﻪ ﺷـﺪﻩ ﻛﻪ ﺑﻪ ﺻﻮﺭﺕ ﻣﺪﺍﻭﻡ ﺑﺮﺍی ﺳـﻼﻣﺖ ﻭ ﻋﺎﻓﻴﺖ ﺍﻭ ﺑﻪ ﻓﻘﺮﺍ ﺻﺪﻗﻪ ﺩﻫﻨﺪ.

ﺑـﻪ ﺁﻧﻬﺎ ﻛـﻪ ﻣﯽ ﺧﻮﺍﻫﻨﺪ ﻓﺪﺍﻳﯽ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺑﺎﺷـﻨﺪ ﻭ ﺧﻮﺩ ﺭﺍ ﺳـﭙﺮ ﺑـﻼی ﺁﻥ ﺣﻀـﺮﺕ ﺳـﺎﺯﻧﺪ، ﮔﻔﺘﻪ ﺷـﺪﻩ ﻛﻪ ﺑﺮﻃـﺮﻑ ﺳـﺎﺧﺘﻦ ﺑﻼ ﺍﺯ ﻭﺟـﻮﺩ ﺍﻭ ﺗﻨﻬـﺎ ﺑﻪ ﺍﻳﻦ ﻧﻴﺴـﺖ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﻫﺪﻑ ﺁﻥ ﺑﻼﻫـﺎ ﻛﻨﻴﻢ، ﺑﻠﻜﻪ ﺧﻴﻠﯽ ﺍﻭﻗﺎﺕ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﻣﺎﻝ ﻭ ﺩﺍﺭﺍﻳﯽ ﺧﻮﺩ ﺭﺍ ﺳﭙﺮ ﺑﻼی ﺁﻥ ﺣﻀﺮﺕ ﮔﺮﺩﺍﻧﻴﻢ.

ﺍﮔـﺮ ﺗﺎ ﺩﻳـﺮﻭﺯ ﺍﺯ ﻛﻤﮏ ﺑﻪ ﻓﻘـﺮﺍ ﻏﺎﻓﻞ ﺑﻮﺩﻩ ﺍﻳﺪ، ﺍﺯ ﺍﻣـﺮﻭﺯ ﺑﻪ ﺍﻳﻦ ﻛﺎﺭ ﻣﺒـﺎﺩﺭﺕ ﻛﻨﻴـﺪ ﻭ ﺑﺪﺍﻧﻴـﺪ ﻛـﻪ ﺧﺪﺍﻭﻧـﺪ ﺑﻪ ﻭﺍﺳـﻄﻪ ﺻﺪﻗـﻪ ﺩﺍﺩﻥ، ﺷـﻤﺎﺭ ﺯﻳﺎﺩی ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎ ﺭﺍ ﺍﺯ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﺑﺮﻃﺮﻑ ﻣﯽ ﺳﺎﺯﺩ. ﺍﻣﺎ ﻫﺮﮔﺎﻩ ﺍﺭﺯﺵ ﻭ ﺍﺛﺮ ﺻﺪﻗﻪ ﺭﺍ ﺩﺍﻧﺴـﺘﻴﺪ ﻭ ﺗﺼﻤﻴﻢ ﮔﺮﻓﺘﻴﺪ ﻛﻪ ﺑﻪ ﻛﺴـﯽ ﻛﻤـﮏ ﻣﺎﻟﯽ ﻛﻨﻴﺪ، ﻣﺤﺒﻮﺏ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺁﻭﺭﻳﺪ ﻭ ﺑﺪﺍﻧﻴﺪ ﻛﻪ ﻗﻠﺐ ﺍﻭ ﻃﻮﻓﺎﻧﯽ ﺗﺮﻳـﻦ ﻗﻠﺐ ﺩﺭ ﺍﻳﻦ ﺭﻭﺯﮔﺎﺭ ﺍﺳـﺖ ﻛﻪ ﺑـﺮ ﺁﻥ ﺩﺍﻍ ﻭ ﺩﺭﺩ ﻗﻠﻮﺏ ﺗﻤﺎﻡ ﺷﻴﻌﻴﺎﻥ ﺑﻠﻜﻪ ﻋﻤﻮﻡ ﺑﻨﺪﮔﺎﻥ ﺧﺪﺍ ﺳﻨﮕﻴﻨﯽ ﻣﯽ ﻛﻨﺪ.

ﺩﺳﺖ ﻛﻪ ﺩﺭ ﺟﻴﺒﺖ ﺑﺮﺩی ﺗﺎ ﭼﻴﺰی ﺑﺮﺍی ﺻﺪﻗﻪ ﺩﺍﺩﻥ ﭘﻴﺪﺍ ﻛﻨﯽ، ﻧﮕﺎﻫـﺖ ﺭﺍ ﺑـﻪ ﻗﻠـﺐ ﺩﺭﺩﻣﻨﺪ ﻭ ﺑﻼﻛﺸـﻴﺪﻩ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻛـﻪ ﺧﺪﺍﻭﻧﺪ ﺳـﻼﻣﺘﺶ ﺑﺪﺍﺭﺩ ﻣﺘﻮﺟﻪ ﺳـﺎﺯ ﻭ ﺑﻪ ﺁﻭﺍی ﺩﻟﻨﺸـﻴﻦ ﻣﺤﺒﺖ ﺍﻭ ﭘﺎﺳـﺦ ﻣﺜﺒﺖ ﺑﺪﻩ! ﺑﮕﻮ ﺧﺪﺍﻳﺎ ﺷـﺎﻫﺪ ﺑﺎﺵ ﻛﻪ ﻣﻮﻻی ﻏﺮﻳﺐ ﻭ ﺁﻗﺎی ﻣﻬﺮﺑﺎﻧﻢ ﺭﺍ ﺑﺮ ﺧﻮﺩﻡ ﻭ ﺧﺎﻧﻮﺍﺩﻩ ﺍﻡ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﻢ ﻭ ﺍﻳﻦ ﺻﺪﻗﻪ ﺭﺍ ﺑﻪ ﺟﺎی ﺁﻥﻛﻪ ﺍﺯ ﺑﺎﺑـﺖ ﺧـﻮﺩﻡ ﻭ ﺧﺎﻧﻮﺍﺩﻩ ﺍﻡ ﺑﭙﺮﺩﺍﺯﻡ، ﺍﺯ ﻃـﺮﻑ ﺍﻭ ﭘﺮﺩﺍﺧﺖ ﻣﯽ ﻛﻨﻢ. ﺧﺪﺍﻳﺎ ﺑﻼ ﺭﺍ ﺍﺯ ﺍﻭ ﻭ ﻋﺰﻳﺰﺍﻥ ﺍﻭ ﺩﻭﺭ ﺑﮕﺮﺩﺍﻥ ﻭ ﺩﻟﺶ ﺭﺍ ﺷﺎﺩ ﺳﺎﺯ!

ﺑـﺪﺍﻥ ﻛﻪ ﺍﮔﺮ ﺍﻳﻦﮔﻮﻧـﻪ ﺻﺎﺩﻗﺎﻧﻪ ﺍﻭ ﺭﺍ ﺑﺮ ﺧـﻮﺩ ﺗﺮﺟﻴﺢ ﺩﻫﯽ، ﺍﻭ ﻧﻴـﺰ ﺑﻪ ﺍﻗﺘﻀﺎی ﺟﻮﺍﻧﻤـﺮﺩی ﺍﺵ ﺗﻮ ﺭﺍ ﺩﻭ ﭼﻨﺪﺍﻥ ﻣﻮﺭﺩ ﻟﻄﻒ ﺧﻮﻳﺶ ﻗﺮﺍﺭ ﺧﻮﺍﻫﺪ ﺩﺍﺩ.

ﺩﺭ ﺁﺋﻴـﻦ ﻣﺎ ﻛﺎﺭﻫﺎﻳﯽ ﺍﺯ ﻗﺒﻴﻞ ﺻﺪﻗﻪ ﺩﺍﺩﻥ، ﺍﻧﺠﺎﻡ ﺣﺞ ﻣﺴـﺘﺤﺒﯽ ﻭ ﻧﻤﺎﺯﮔـﺰﺍﺭﺩﻥ ﺑـﻪ ﻧﻴﺎﺑـﺖ ﺍﺯ ﺩﻳﮕـﺮﺍﻥ ﻣﺠﺎﺯ ﺩﺍﻧﺴـﺘﻪ ﺷـﺪﻩ ﻭ ﺣﺘﯽ ﺑﻪ ﺍﻳﻦﮔﻮﻧﻪ ﻛﺎﺭﻫﺎ ﺳﻔﺎﺭﺵ ﺷﺪﻩ ﺍﻳﻢ.

ﻋﻠـﯽ ﺑـﻦ ﺍﺑﯽ ﺍﻟﺤﻤﺰﻩ ﮔﻮﻳـﺪ: ﺑﻪ ﺣﻀـﺮﺕ ﺍﺑﻮﺍﺑﺮﺍﻫﻴﻢ (ﻣﻮﺳـﯽ ﺑـﻦ ﺟﻌﻔـﺮ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ) ﻋﺮﺿـﻪ ﺩﺍﺷـﺘﻢ: ﺁﻳـﺎ ﻣﯽ ﺗﻮﺍﻧـﻢ ﺍﺯ ﻃـﺮﻑ ﺧﻮﻳﺸـﺎﻭﻧﺪﺍﻥ ﻭ ﺩﻭﺳـﺘﺎﻥ ﺯﻧﺪﻩ ﻳـﺎ ﻣﺮﺩﻩ ﺍﻡ ﺣﺞ ﺑﻪ ﺟـﺎی ﺁﻭﺭﻡ ﻳﺎ ﻧﻤﺎﺯ ﺑﺨﻮﺍﻧـﻢ ﻳﺎ ﺻﺪﻗﻪ ﺑﺪﻫﻢ؟ ﺍﻣﺎﻡ ﻓﺮﻣﻮﺩﻧﺪ: «ﺁﺭی، ﺍﺯ ﺳـﻮی ﺁﻧﻬﺎ ﺻﺪﻗﻪ ﺑـﺪﻩ ﻭ ﻧﻤـﺎﺯ ﺑﮕﺰﺍﺭ ﻭ ﺑﻪ ﺳـﺒﺐ ﺍﻳـﻦ ﻛﻪ ﺍﻳـﻦ ﻛﺎﺭ ﺗﻮ ﭘﻴﻮﻧﺪ ﺑـﺎ ﺁﻥﻫﺎ ﻧﻴﺰ ﻫﺴﺖ ﺑﺮﺍﻳﺖ ﭘﺎﺩﺍﺵ ﺩﻳﮕﺮی ﻫﻢ ﻣﻨﻈﻮﺭ ﺧﻮﺍﻫﺪ ﺷﺪ.»(53)

ﺣـﺎﻝ ﻛﻪ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﺟﺎﻧﺐ ﺩﻳﮕﺮی ﺑﻪ ﺍﻳﻦﮔﻮﻧﻪ ﻛﺎﺭﻫﺎ ﭘﺮﺩﺍﺧﺖ، ﺁﻥ ﺩﻳﮕـﺮی ﺭﺍ ﺩﻭﺳـﺖ ﺩﺍﺷـﺘﻨﯽ ﺗﺮﻳﻦ ﻣﺤﺒـﻮﺏ، ﻧﺰﺩﻳﮏﺗﺮﻳـﻦ ﻓﺮﺯﻧﺪ ﭘﻴﺎﻣﺒﺮ، ﺑﺮﺗﺮﻳﻦ ﻭ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻏﺮﻳﺐﺗﺮﻳﻦ ﺑﻨﺪﻩ ﺧﺪﺍ ﺩﺭ ﻋﺼﺮ ﺣﺎﺿﺮ ﻗـﺮﺍﺭ ﺑـﺪﻩ. ﺍﻭ ﺭﺍ ﺑﺮﮔﺰﻳﻦ ﻭ ﺑﺎ ﺍﺣﺴـﺎﻥ ﺑﻪ ﺍﻭ، ﺧﻮﺩ ﺭﺍ ﻧﺰﺩ ﺭﺳـﻮﻝ ﺧﺪﺍ ﻧﻴﺰ ﻣﻘﺮﺏ ﺳﺎﺯ.

ﺟﻠﺐ ﻋﻨﺎﻳﺎﺕ ﺣﻖ ﻣﺘﻌﺎﻝ ﻭ ﺑﻼ ﮔﺮﺩﺍﻧﺪﻥ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻭ ﻋﺰﻳﺰﺍﻥ ﺍﻭ ﺑﻪ ﻭﺳﻴﻠﻪ ﺻﺪﻗﻪ، ﺍﺯ ﻣﺼﺎﺩﻳﻖ ﺑﺎﺭﺯ ﻳﺎﺭی ﻭ ﺣﻤﺎﻳﺖ ﺍﺯ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﺪ ﻭ ﻫﺮ ﺷـﻴﻌﻪ ﺍی ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﭼﻨﻴﻦ ﻛﺎﺭﻫﺎﻳﯽ ﻣﺒﺎﺩﺭﺕ ﻭﺭﺯﺩ.

ﺩﺭ ﭘﺎﻳﺎﻥ ﺍﻳﻦ ﻣﻘﺎﻝ، ﺑﺨﺸﯽ ﺍﺯ ﻭﺻﻴﺖ ﻋﺎﻟﻢ ﻧﺎﻣﺪﺍﺭ ﻭ ﻳﺎﻭﺭ ﺩﻟﺴﻮﺯ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻣﺮﺣﻮﻡ ﺳـﻴﺪ ﺑﻦ ﻃﺎﻭﻭﺱ ﺭﺍ ﻣﺮﻭﺭ ﻣﯽ ﻛﻨﻴﻢ. ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭ ﺍﻳﻦ ﻭﺻﻴﺖ ﺭﺍ ﺩﺭ ﺭﻭﺯﮔﺎﺭی ﻛﻪ ﺑﻴﺶ ﺍﺯ ﻫﻔﺖﺻﺪ ﺳـﺎﻝ ﺑﺎ ﻣﺎ ﻓﺎﺻﻠﻪ ﺩﺍﺭﺩ، ﺑﺮﺍی ﻓﺮﺯﻧﺪ ﺩﻟﺒﻨﺪ ﺧﻮﻳﺶ ﻧﻮﺷـﺘﻪ ﺍﺳـﺖ. ﺩﻗﺖ ﺩﺭ ﺗﺎﺭﻳـﺦ ﺩﻳﺮﻳﻦ ﺍﻳـﻦ ﻭﺻﻴﺖﻧﺎﻣﻪ ﺑﻪ ﻣﺎ ﻳﺎﺩﺁﻭﺭ ﻣﯽ ﺷـﻮﺩ ﻛﻪ ﺭﺍﻩ ﻭ ﺭﺳـﻢ ﺩﻭﺳـﺘﯽ ﻭ ﻓـﺪﺍﻛﺎﺭی ﺑـﺮﺍی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧـﻪ ﺑﺪﻋﺘﯽ ﻧﻮ ﺩﺭ ﺗﺸﻴﻊ، ﺑﻠﻜﻪ ﺷﻴﻮﻩ ﺍی ﺩﻳﺮﻳﻦ ﻭ ﺑﺎﺍﺻﺎﻟﺖ ﺍﺳﺖ. ﺷﻴﻮﻩ ﺍی ﻛﻪ ﺳﺎﺑﻘﻪ ﺍﺵ ﭼﻨﺎﻥ ﻛﻪ ﮔﺬﺷـﺖ ﺑﻪ ﻛﺮﺑﻼی ﺳـﺎﻝ 61 ﻫﻢ ﻣﯽ ﺭﺳﺪ. ﻛﺮﺑﻼﻳﯽ ﻛﻪ ﺩﺭ ﺁﻥ ﻋﺒﺎﺱ ﺑﻦ ﻋﻠﯽ ﻭ ﺩﻳﮕﺮ ﻳﺎﺭﺍﻥ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﻧﺸـﺎﻥ ﺩﺍﺩﻧﺪ ﻛﻪ ﺣﻘﻴﻘﺘﺎ ﺍﻣـﺎﻡ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺧﻮﺩ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﻨﺪ. ﺑﻠﻜﻪ ﺍﻳﻦ ﺷـﻴﻮﻩ ﺑﻪ ﺳـﺎﻝ ﻫﺎی ﺣﻴﺎﺕ ﺭﺳـﻮﻝ ﺧـﺪﺍ ﺑﺎﺯ ﻣﯽ ﮔﺮﺩﺩ ﺁﻧﺠـﺎ ﻛﻪ ﻣﻘﺪﺍﺩ ﺑﺎﻭﻓﺎ ﭘﻴـﺶ ﺍﺯ ﺁﻏﺎﺯ ﻧﺒـﺮﺩ ﺑـﺪﺭ ﻛﻪ ﭘﻴﻜﺎﺭی ﺳـﺨﺖ ﻭ ﻧﺎﺑﺮﺍﺑﺮ ﺑـﻮﺩ، ﺑﻪ ﺭﺳـﻮﻝ ﺧﺪﺍ ﻋﺮﺿﻪ ﺩﺍﺷﺖ:
