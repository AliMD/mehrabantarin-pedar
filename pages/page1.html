ﺁﻥ ﻣﺮﺩ ﺩﺍﻧﺎ، ﻛﻪ ﻗﺪﺭﺵ ﺭﺍ ﻧﻤﯽ ﺩﺍﻧﻴﻢ.

۱) ﺳـﺘﺎﻳﺶ ﻭ ﺳـﭙﺎﺱ ﺑﯽ ﻛﺮﺍﻥ ﺍﺯ ﺁﻥ ﺧﺪﺍی ﺑﺎ ﻋﺰﺕ ﻭ ﺟﻼﻝ ﺍﺳـﺖ ﻛﻪ ﻫﻴﭻ ﻛﺲ ﺍﻭ ﺭﺍ ﻣﺎﻧﻨﺪ ﻧﻴﺴﺖ ﻭ ﻫﻤﻪ ﻛﺲ ﻣﺪﻳﻮﻥ ﻋﻨﺎﻳﺎﺕ ﻭ ﺍﻟﻄﺎﻑ ﺍﻭﻳﻨـﺪ؛ ﻭ ﺩﺭﻭﺩ ﺑﯽ ﭘﺎﻳﺎﻥ ﺑﺮ ﻣﺤﻤﺪ ﻣﺼﻄﻔﯽ ، ﺭﺳـﻮﻝ ﺷﺎﻳﺴـﺘﻪ ﻭ ﺑﻨﺪﻩ ﭘﺴـﻨﺪﻳﺪﻩ ﺍﺵ، ﻭ ﺑﺮ ﺧﺎﻧﺪﺍﻥ ﭘﺎک ﻭ ﺗﺎﺑﻨﺎک ﺁﻥ ﺣﻀﺮﺕ ﺑﺎﺩ، ﻛﻪ ﺣﺒﻴﺐ ﺭﺏ ﺍﻧﺪ ﻭ ﻃﺒﻴﺐ ﺧﻠﻖ.

۲) ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﻭ ﺑﻨﺪﮔﺎﻥ ﺑﺮﮔﺰﻳـﺪﻩ ﺍﻭ، ﺍﺯ ﻣﻔﺎﻫﻴﻢ ﺑﻠﻨﺪ ﺩﻳﻨﯽ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﺁﻳﻴﻦ ﺍﺳﻼﻡ ﺑﺮ ﺁﻥ ﺗﺄﻛﻴﺪ ﻓﺮﺍﻭﺍﻥ ﺷﺪﻩ ﺍﺳﺖ. ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﺁﻳـﺎﺕ ﻗﺮﺁﻥ، ﻣﺆﻣﻨـﺎﻥ ﺭﺍ ﻓﺮﻣﻮﺩﻩ ﺗﺎ ﺧﺪﺍی ﺧﻮﺩ ﺭﺍ ﻳﺎﺭی ﺭﺳـﺎﻧﻨﺪ. ﻣﺸـﺘﻘ ﺎﺕ ﻭﺍژﻩ «ﻧﺼﺮﺕ» ﻛﻪ ﺑﻪ ﻣﻌﻨﺎی «ﻳﺎﺭی ﺭﺳﺎﻧﺪﻥ» ﺍﺳﺖ، ﺑﺎﺭﻫﺎ ﺩﺭ ﻛﺘـﺎﺏ ﺧـﺪﺍ ﺁﻣﺪﻩ ﻭ ﻃـﯽ ﺁﻥ ﮔﺎﻩ ﺍﺷـﺎﺭﻩ ﺑﻪ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺧﺪﺍ ﺑﻪ ﺑﻨﺪﮔﺎﻧـﺶ ﻭ ﮔﺎﻩ ﺍﺷـﺎﺭﻩ ﺑـﻪ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻨﺪﮔﺎﻥ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﺷـﺪﻩ ﺍﺳﺖ.

۲) ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﻭ ﺑﻨﺪﮔﺎﻥ ﺑﺮﮔﺰﻳـﺪﻩ ﺍﻭ، ﺍﺯ ﻣﻔﺎﻫﻴﻢ ﺑﻠﻨﺪ ﺩﻳﻨﯽ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﺁﻳﻴﻦ ﺍﺳﻼﻡ ﺑﺮ ﺁﻥ ﺗﺄﻛﻴﺪ ﻓﺮﺍﻭﺍﻥ ﺷﺪﻩ ﺍﺳﺖ. ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﺁﻳـﺎﺕ ﻗﺮﺁﻥ، ﻣﺆﻣﻨـﺎﻥ ﺭﺍ ﻓﺮﻣﻮﺩﻩ ﺗﺎ ﺧﺪﺍی ﺧﻮﺩ ﺭﺍ ﻳﺎﺭی ﺭﺳـﺎﻧﻨﺪ. ﻣﺸـﺘﻘ ﺎﺕ ﻭﺍژﻩ «ﻧﺼﺮﺕ» ﻛﻪ ﺑﻪ ﻣﻌﻨﺎی «ﻳﺎﺭی ﺭﺳﺎﻧﺪﻥ» ﺍﺳﺖ، ﺑﺎﺭﻫﺎ ﺩﺭ ﻛﺘـﺎﺏ ﺧـﺪﺍ ﺁﻣﺪﻩ ﻭ ﻃـﯽ ﺁﻥ ﮔﺎﻩ ﺍﺷـﺎﺭﻩ ﺑﻪ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺧﺪﺍ ﺑﻪ ﺑﻨﺪﮔﺎﻧـﺶ ﻭ ﮔﺎﻩ ﺍﺷـﺎﺭﻩ ﺑـﻪ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻨﺪﮔﺎﻥ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﺷـﺪﻩ ﺍﺳﺖ.

۳) ﺩﺭک ﻣﻔﻬـﻮﻡ ﻳـﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺧـﺪﺍ ﺑـﻪ ﺑﻨﺪﮔﺎﻥ ﺩﺷـﻮﺍﺭ ﻧﻴﺴـﺖ، ﻟﻴﻜﻦ ﻣﻔﻬﻮﻡ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻨﺪﮔﺎﻥ ﺑﻪ ﺧﺪﺍ ﺑﺮﺍی ﺷـﻤﺎﺭی ﻏﺮﻳﺐ ﻭ ﺷـﮕﻔﺖﺍﻧﮕﻴﺰ ﻣﯽ ﻧﻤﺎﻳﺪ. ﭘﻴﭽﻴﺪﮔﯽ ﺍﻣﺮ ﺩﺭ ﺁﻥ ﺍﺳـﺖ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﻭ ﺑﻪ ﺗﺒﻊ ﺍﻭ، ﻧﻤﺎﻳﻨﺪﮔﺎﻥ ﻭﻳﮋﻩ ﭘﺮﻭﺭﺩﮔﺎﺭ، ﻭﻟﯽ ﻧﻌﻤﺖ ﺑﻨﺪﮔﺎﻥ ﻫﺴـﺘﻨﺪ، ﺗﻤﺎﻡ ﺗﻮﺍﻥﻫﺎ ﻭ ﻗﺪﺭﺕ ﻫﺎ ﺍﺯ ﻧﺎﺣﻴﻪ ﺁﻧﺎﻥ ﺑﻪ ﺟﺎﻧﺐ ﺧﻠﻖ ﺳـﺮﺍﺯﻳﺮ ﻣﯽ ﺷـﻮﺩ ﻭ ﺩﺭ ﺑﺎﻃﻦ ﺍﻣﺮ ﺑﻪ ﻛﻠﯽ ﺑﯽ ﻧﻴﺎﺯ ﺍﺯ ﺩﻳﮕﺮﺍﻥ ﻣﯽ ﺑﺎﺷـﻨﺪ. ﭘﺲ ﭼﮕﻮﻧﻪ ﺍﺳـﺖ ﻛﻪ ﻓﺮﻭﺩﺳـﺘﺎﻥ ﻛﻪ ﺩﺭ ﺗﻤﺎﻡ ﺗﻮﻓﻴﻘﺎﺕ ﺧﻮﻳﺶ ﻭﺍﻣـﺪﺍﺭ ﺁﻧﺎﻥﺍﻧﺪ، ﺑﺘﻮﺍﻧﻨﺪ ﺑﻪ ﺍﻳﺸﺎﻥ ﻳﺎﺭی ﺭﺳﺎﻧﻨﺪ؟

۴) ﻃـﺮﺡ ﺍﻳﻦ ﭘﺮﺳـﺶ ﺩﺭ ﺁﻏـﺎﺯ ﻛﺘﺎﺏ ﺍﺯ ﺁﻥ ﺭﻭ ﻣﻬﻢ ﺑـﻮﺩ ﻛﻪ ﭼﻮﻥ ﺍﺯ ﺷـﻴﻌﻴﺎﻥ ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﺩﻋﻮﺕ ﻣﯽ ﮔـﺮﺩﺩ ﻛﻪ ﺑﻪ ﻳـﺎﺭی ﺍﻭ ﺍﻗـﺪﺍﻡ ﻧﻤﺎﻳﻨﺪ، ﮔﺎﻩ ﺍﻳﻦ ﭘﺎﺳـﺦ ﺷﻨﻴﺪﻩ ﻣﯽ ﺷـﻮﺩ ﻛﻪ ﻣﮕﺮ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺑﻪ ﻳﺎﺭی ﻣﺎ ﻧﻴﺎﺯﻣﻨﺪ ﺍﺳـﺖ؟ ﺍﻭ ﻭﻟﯽ ﻧﻌﻤﺖ ﻣﺎﺳـﺖ، ﭘﺲ ﻣﺎ ﺭﺍ ﻧﺮﺳﺪ ﻛﻪ ﺑﺨﻮﺍﻫﻴﻢ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﺍﻗﺪﺍﻡ ﻛﻨﻴﻢ. ﺍﮔﺮ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﮔﻴﺮی ﺩﺭﺳﺖ ﺑﺎﺷﺪ، ﭘﺲ ﺑﺎﻳﺪ ﺩﺭ ﺍﻣﻜﺎﻥ ﻳـﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻪ ﺧﺪﺍ ﻧﻴﺰ ﺗﺮﺩﻳﺪ ﻛﺮﺩ، ﺣﺎﻝ ﺁﻧﻜـﻪ ﺧﺪﺍﻭﻧﺪ، ﺗﺒﺎﺭک ﻭ ﺗﻌﺎﻟـﯽ ﺑـﻪ ﻣﺎ ﺍﻣﺮ ﻓﺮﻣﻮﺩﻩ ﻛﻪ ﺑﻪ ﺍﻭ ﻳﺎﺭی ﺭﺳـﺎﻧﻴﻢ ﻳﺎ ﺑﻪ ﺍﻭ ﻗﺮﺽ ﺑﺪﻫﻴﻢ ﻳﺎ... .

۵) ﺩﺭ ﺗﻮﺿﻴﺢ ﻣﻔﺎﻫﻴﻤﯽ ﺍﺯ ﻗﺒﻴﻞ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻨﺪﮔﺎﻥ ﺑﻪ ﺧﺪﺍﻭﻧﺪ، ﻳـﺎ ﻗﺮﺽ ﺩﺍﺩﻥ ﺑﻨـﺪﮔﺎﻥ ﺑﻪ ﺧﺪﺍ، ﺑﺎﻳـﺪ ﺑﻪ ﺍﻳﻦ ﻧﻜﺘﻪ ﻇﺮﻳـﻒ ﺗﻮﺟﻪ ﻛﺮﺩ ﻛـﻪ ﺧﺪﺍی ﻣﺘﻌـﺎﻝ ﻧﻈﺎﻡ ﺧﻠﻘﺖ ﺭﺍ ﺑﺮ ﭘﺎﻳﻪ ﺍﺧﺘﻴﺎﺭ ﻭ ﺣﺮﻳ ﺖ ﺑﺸـﺮ ﻧﻬﺎﺩﻩ ﺍﺳـﺖ ﻭ ﺍﻧﺴـﺎﻥﻫﺎ ﻧﻈـﺮ ﺑـﻪ ﺣﺮﻳ ﺖ ﻭ ﻗـﺪﺭﺕ ﺧﻮﻳـﺶ ﺩﺭ ﻗﺒـﺎﻝ ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩﺷـﺎﻥ ﺑـﺎ ﺣﺮﻣﺖﻫـﺎ ﻭ ﻣﻘﺪﺳـﺎﺕ ﺍﻟﻬـﯽ ﺁﺯﻣـﻮﺩﻩ ﻣﯽ ﺷـﻮﻧﺪ. ﺍﻧﺴـﺎﻥﻫﺎ ﻫـﺮ ﻳﮏ ﺩﺍﺭﺍی ﺣﺼـﻪ ﺍی ﺍﺯ ﻗـﺪﺭﺕ ﻭ ﺍﺧﺘﻴﺎﺭ ﻫﺴـﺘﻨﺪ ﻭ ﺑﺮ ﺍﻳﻦ ﭘﺎﻳﻪ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﭘﺎﺳـﺪﺍﺭ ﺣﺮﻣﺖﻫﺎی ﺍﻟﻬﯽ ﻳﺎ ﺣﺮﻣﺖﺷﻜـﻦ ﻧﺴﺒﺖ ﺑـﻪ ﺣﺪﻭﺩ ﻭ ﻣﻘﺪﺳـﺎﺕ ﺍﻟﻬﯽ ﮔﺮﺩﻧـﺪ. ﺧﺪﺍﻭﻧﺪ ﺑﺮﺍی ﻧﻤﺎﻳﺎﻥ ﺳـﺎﺧﺘﻦ ﺩﺭﺟﺎﺕ ﺳـﻌﺎﺩﺕ ﻭ ﺷـﻘﺎﻭﺕ ﺍﻧﺴـﺎﻥﻫﺎ ﺩﺭ ﺑﺮﻫﻪ ﻫﺎﻳﯽ ﺍﺯ ﺗﺎﺭﻳﺦ ﺣﺘﯽ ﺍﺯ ﺗﺎﺧـﺖ ﻭ ﺗﺎﺯ ﻧﺎﺑﻜﺎﺭﺍﻥ ﺑﺸـﺮ ﺑﺮ ﻣﻘﺪﺱﺗﺮﻳﻦ ﻣﻘﺪﺳـﺎﺕ ﺧﻮﻳﺶ ﻧﻴﺰ ﺟﻠﻮﮔﻴـﺮی ﻧﻤﯽ ﻛﻨﺪ. ﺑﺎﺭﻫﺎ ﺷـﺪﻩ ﻛﻪ ﻧﺎﺑﻜﺎﺭﺍﻥ ﺑـﺮ ﺩﻳﻦ ﺧﺪﺍ، ﺑﺮ ﻛﺘﺎﺏ ﺧﺪﺍ، ﺑﺮ ﭘﻴﺎﻣﺒﺮﺍﻥ ﺧﺪﺍ ﻭ ﺑﺮ ﺟﺎﻧﺸﻴﻨﺎﻥ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﺗﺎﺧﺘﻪ ﺍﻧﺪ ﻭ ﺧﺪﺍﻭﻧﺪ، ﺧﻮﺩ ﺑﻪ ﺻﻮﺭﺕ ﻣﺴـﺘﻘﻴﻢ ﻭ ﺳـﺮﻳﻊ ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻳﺸﺎﻥ ﻧﺎﻳﺴـﺘﺎﺩﻩ ﻭ ﺑﺎ ﺁﻧﻬﺎ ﻣﻘﺎﺑﻠﻪ ﻧﻜﺮﺩﻩ ﺍﺳﺖ. ﺩﺭ ﺍﺛﺮ ﻫﻤﻴﻦ ﺣﻤﻠﻪ ﻫﺎ ﻭ ﺣﺮﻣﺖﺷﻜﻨﯽ ﻫﺎ، ﺑﺎﺭﻫﺎ ﻭ ﺑﺎﺭﻫﺎ، ﻋﺰﺕ، ﺍﻣﻨﻴﺖ ﻭ ﻣﻮﻗﻌﻴﺖ ﺍﺟﺘﻤﺎﻋﯽ ﻗﺪ ﻳﺴﺎﻥ ﺍﻟﻬﯽ ﺩﺭ ﻣﺤﺎﻕ ﻏﺮﺑﺖ، ﻣﻬﺠﻮﺭﻳﺖ ﻭ ﺍﻧﺰﻭﺍ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ.

ﺩﺭ ﻇـﺮﻑ ﭼﻨﻴﻦ ﺷـﺮﺍﻳﻄﯽ ﺧﺪﺍﻭﻧﺪ ﺑﻨـﺪﮔﺎﻥ ﻣﺆﻣﻦ ﺧﻮﻳﺶ ﺭﺍ ﺑﻪ ﻳﺎﺭی ﺧﻮﻳﺶ (ﻳﻌﻨﯽ ﻳﺎﺭی ﻗﺪ ﻳﺴﺎﻥ ﺧﻮﻳﺶ) ﻓﺮﺍ ﻣﯽ ﺧﻮﺍﻧﺪ، ﺑﺪﻭﻥ ﺍﻳﻦ ﻛﻪ ﺑﻪ ﺁﻧﻬﺎ ﻧﻴﺎﺯی ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺩﺭ ﺣﻘﻴﻘـﺖ ﺍﻭ ﺑـﺎ ﮔﺸـﻮﺩﻥ ﺑـﺎﺏ ﺧﺪﻣﺖ ﻭ ﻧﺼـﺮﺕ، ﺑﻪ ﻋﺮﺻﻪ ﺭﺷـﺪ ﻭ ﺗﻌﺎﻟﯽ ﻣﺎ ﺑﻨﺪﮔﺎﻥ ﺭﻭﻧﻖ ﺩﺍﺩﻩ ﻭ ﺑﺮ ﻛﺴـﺎﻧﯽ ﻛﻪ ﺗﻮﻓﻴﻖ ﻳﺎﺭی ﺣﻖ ﺭﺍ ﺑﻴﺎﺑﻨـﺪ ﻣﻨ ـﺖ ﻧﻬﺎﺩﻩ ﺍﺳـﺖ. ﺍﺯ ﺍﻳﻦ ﺭﻭ ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑـﻪ ﺧﺪﺍ ﻛﻪ ﺑﻪ ﻣﻌﻨﺎی ﻳﺎﺭی ﺭﺳـﺎﻧﺪﻥ ﺑﻪ ﻣﻘﺮﺑـﺎﻥ ﺩﺭﮔﺎﻩ ﺍﻟﻬﯽ ﻭ ﺣﻔﺎﻇﺖ ﺍﺯ ﻣﻘﺮﺭﺍﺕ ﺍﻟﻬﯽ ﺍﺳـﺖ، ﻧﻪ ﺑﻪ ﻣﻌﻨﺎی ﻧﻴﺎﺯ ﺁﻧﻬﺎ ﺑﻪ ﻳﺎﺭی ﻣﺎ ﺑﻠﻜﻪ ﺑﻪ ﺟﻬﺖ ﺭﺷـﺪ ﻭ ﺗﻌﺎﻟﯽ ﺧﻮﺩ ﻣﺎ ﻣﯽ ﺑﺎﺷﺪ.

۶) ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﻛﻨﺎﺭ ﻗـﺮﺁﻥ، ﺑﺰﺭگﺗﺮﻳﻦ ﻣﺼﺪﺍﻕ ﺯﻧﺪﻩ ﺍﺯ ﻣﺠﻤﻮﻋﻪ ﻣﻘﺪﺳـﺎﺕ ﺍﻟﻬﯽ ﺍﺳـﺖ ﻛﻪ ﻫﻢ ﻣﻮﺭﺩ ﻏﻔﻠﺖ ﺩﻭﺳﺘﺎﻧﺶ ﻭ ﻫﻢ ﺩﺭ ﻣﻌﺮﺽ ﺣﻤﻠﻪ ﺩﺷـﻤﻨﺎﻧﺶ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳـﺖ ﻭ ﺷﺪﺕ ﻏﺮﺑﺖ، ﻣﻈﻠﻮﻣﻴـﺖ ﻭ ﻧﺎﺍﻣﻨـﯽ ، ﺍﻳﺸـﺎﻥ ﺭﺍ ﺑـﺮﺍی ﺳـﺎﻟﻴﺎﻥ ﺩﺭﺍﺯی ﺍﺳـﺖ ﻛـﻪ ﺗﺎ ﺳﺮﺣﺪ ﻏﻴﺒﺖ ﻭ ﭘﻨﻬﺎﻥﺯﻳﺴﺘﯽ ﺑﻪ ﺍﻧﺰﻭﺍ ﻛﺸﺎﻧﺪﻩ ﺍﺳﺖ.

ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻳﻜﯽ ﺍﺯ ﺩﻭ ﺣﻘﻴﻘﺖ ﮔﺮﺍﻥﺑﻬﺎﻳﯽ ﺍﺳـﺖ ﻛـﻪ ﭘﻴﺎﻣﺒﺮ ﺍﻛﺮﻡ ﺻﻠﯽ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ ﻭ ﺳـﻠﻢ ﺁﻧﻬـﺎ ﺭﺍ ﺩﺭ ﻣﻴﺎﻥ ﺍﻣﺖ ﺑﻪ ﻳﺎﺩﮔﺎﺭ ﮔﺬﺍﺷﺘﻪ ﻭ ﻣﺎ ﺭﺍ ﺑﻪ ﺗﻤﺴﮏ ﺑﻪ ﺁﻧﻬﺎ ﺳﻔﺎﺭﺵ ﻛﺮﺩﻩ ﺍﺳﺖ.

ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ، ﻳـﺎﺩﮔﺎﺭ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﻭ ﺗﻤـﺎﻡ ﺍﻣﺎﻣﺎﻥ ﭘـﺎک ﺍﺯ ﻋﺘﺮﺕ ﭘﻴﺎﻣﺒﺮ ﺍﺳﺖ ﻛﻪ ﺍﻣﺮﻭﺯ ﺩﺭ ﻣﻴﺎﻥ ﻣﺎ ﺯﻧﺪﮔﯽ ﻣﯽ ﻛﻨﺪ ﻭ ﭼﻪ ﺑﺴﻴﺎﺭ ﻛﻪ ﻣﺎ ﺍﺯ ﺍﻭ ﺩﻭﺭ ﻭ ﻧﺴﺒﺖ ﺑﻪ ﻏﺼﻪ ﻫﺎ ﻭ ﺩﺭﺩﻫﺎی ﺍﻭ ﺑﯽ ﺗﻔﺎﻭﺗﻴﻢ.

ﺍﻣﺮﻭﺯﻩ ، ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻣﺎ، ﻫﻤﺎﻥ ﻗﺪﻳﺲ ﺁﺳـﻤﺎﻧﯽ ﺍﺳـﺖ ﻛﻪ ﺯﻣﻴﻨﻴﺎﻥ ﺩﺭ ﻗﺒﺎﻝ ﺗﻮﺟﻪ ﺑﻪ ﺍﻭ ﻳﺎ ﺑﯽ ﺗﻮﺟﻬﯽ ﺑﻪ ﺍﻳﺸﺎﻥ ﺁﺯﻣﺎﻳﺶ ﻣﯽ ﺷﻮﻧﺪ.

۷) ﻛﺘﺎﺏ ﺣﺎﺿﺮ ﺑﺎ ﻋﻨﻮﺍﻥ «ﺑﻴﺎﻳﻴﺪ ﺍﻭ ﺭﺍ ﻳﺎﺭی ﻛﻨﻴﻢ» ﺭﺍﻫﻨﻤﺎﻳﯽ ﺍﺳـﺖ ﻛﺎﺭﺑﺮﺩی ﺑﺮﺍی ﻳﺎﺭی ﺭﺳﺎﻧﺪﻥ ﺑﻪ ﺧﺪﺍ ﺍﺯ ﻃﺮﻳﻖ ﻳﺎﺭی ﺭﺳﺎﻧﺪﻥ ﺑﻪ ﻭﻟﯽ ﺑﺮﮔﺰﻳﺪﻩ ﺍﻭ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ.

ﺍﻳﻦ ﻛﺘﺎﺏ ﺑﺮﺍی ﻛﺴـﺎﻧﯽ ﻧﻮﺷـﺘﻪ ﺷﺪﻩ ﻛﻪ ﺩﻭﺳـﺖ ﺩﺍﺭﻧﺪ ﻧﺎﻣﺸﺎﻥ ﺩﺭ ﺩﻳﻮﺍﻥ ﺍﻟﻬﯽ ﺩﺭ ﻓﻬﺮﺳـﺖ ﻳﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻣﻜﺘﻮﺏ ﺷﻮﺩ.

ﺍﻳﻦ ﻛﺘﺎﺏ ﺑﻪ ﺍﻳﻦ ﺍﻣﻴﺪ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﻛﻪ :

ﺷـﻴﻌﻴﺎﻥ ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﺭﺍ ﺑﻴﺸـﺘﺮ ﺩﺭک ﻛﻨﻨﺪ ﻭ ﺻﺎﺣﺒﺎﻥ ﺩﺭﺩ، ﺳـﻮﺯ ﻭ ﺷﻮﺭ ﺑﺮﺍی ﻳﺎﺭی ﺍﻭ ﮔﺮﺩﻧﺪ، ﺯﻳﺮﺍ ﺑﺴﻴﺎﺭ ﺷﺪﻩ ﻛﻪ ﺩﻳﺪﻩ ﺍﻳﻢ ﺷﻴﻌﻴﺎﻥ ﺣﻀﺮﺗﺶ ﺩﺭک ﺩﺭﺳﺘﯽ ﺍﺯ ﻏﺮﺑﺖ ﺍﻭ ﻧﺪﺍﺭﻧﺪ ﻭ ﺍﺯ ﻏﺼﻪ ﻫﺎی ﺟﺎﻧﻜﺎﻩ ﺍﻭ ﺑﻪ ﻛﻠﯽ ﺑﯽ ﺧﺒﺮﻧﺪ.

ﺻﻔﺤﺎﺕ ﺍﻳﻦ ﻛﺘﺎﺏ، ﺑﻪ ﺍﻳﻦ ﺁﺭﺯﻭ ﺑﻪ ﻫﻢ ﭘﻴﻮﺳﺘﻪ ﺍﻧﺪ ﻛﻪ :

ﺭﺍﻫﻨﻤﺎﻳﯽ ﺑﺎﺷﻨﺪ ﺑﺮﺍی ﺷﻴﻌﻴﺎﻥ ﻏﻴﻮﺭ ﺍﻣﺎﻡ ﻋﺼﺮ؛ ﺁﻧﻬﺎ ﻛﻪ ﺩﻭﺳﺖ ﻣﯽ ﺩﺍﺭﻧـﺪ ﺍﻣـﺎﻡ ﺧـﻮﺩ ﺭﺍ ﻳﺎﺭی ﻛﻨﻨـﺪ، ﺍﻣـﺎ ﺍﺯ ﺭﺍﻩ ﻫﺎی ﻋﻤﻠـﯽ ﺁﻥ ﺁﮔﺎﻩ ﻧﻴﺴـﺘﻨﺪ. ﺩﺭ ﺍﻳـﻦ ﻧﻮﺷـﺘﻪ ﺑﺮﺧـﯽ ﺍﺯ ﺷـﻴﻮﻩ ﻫﺎی ﺧﺪﻣﺖ ﺑﻪ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﻭ ﻳـﺎﺭی ﺍﻳﺸـﺎﻥ ﺩﺭ ﺩﻭﺭﺍﻥ ﻏﻴﺒﺖ ﺑﻴﺎﻥ ﮔﺮﺩﻳﺪﻩ ﺍﺳـﺖ؛ ﺷﻴﻮﻩ ﻫﺎﻳﯽ ﻛﻪ ﺑﺮﺍی ﻋﻤﻮﻡ ﺷﻴﻌﻴﺎﻥ ﻗﺎﺑﻞ ﺍﻧﺠﺎﻡ ﺍﺳﺖ.

ﻛﻠﻤﺎﺕ ﺍﻳﻦ ﻛﺘﺎﺏ، ﺩﺳﺖ ﺑﻪ ﺩﺳﺖ ﻫﻢ ﺩﺍﺩﻧﺪ، ﺑﻪ ﺍﻳﻦ ﺍﻣﻴﺪ ﻛﻪ :

ﺑـﻪ ﺷـﻤﺎﺭ ﻳﺎﺭﺍﻥ ﻣﻮﻻﻳﻤﺎﻥ ﻛـﻪ ﺩﺭﻭﺩ ﻭﺧﺪﺍ ﺑﺮ ﺍﻭ ﺑـﺎﺩ ﺍﻓﺰﻭﺩﻩ ﮔـﺮﺩﺩ، ﺍﺯ ﻏﺮﺑـﺖ ﺣﻀﺮﺗـﺶ ﻛﺎﺳـﺘﻪ ﺷـﻮﺩ، ﺭﻭﺡ ﺗﺸـﻴﻊ ﻛـﻪ ﻫﻤﺎﻧـﺎ ﮔﺮﺩﻳـﺪﻥ ﭘﺮﻭﺍﻧﻪ ﺳـﺎﻥ ﺑﻪ ﮔﺮﺩ ﺷـﻤﻊ ﻭﺟﻮﺩ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺳـﺖ، ﺩﺭ ﻛﺎﻟﺒـﺪ ﺟﺎﻣﻌـﻪ ﺷـﻴﻌﻪ ﺑﻴـﺶ ﺍﺯ ﭘﻴـﺶ ﺩﻣﻴﺪﻩ ﺷـﻮﺩ ﻭ ﻳﺎﺭی ﻭ ﻓـﺪﺍﻛﺎﺭی ﺑﺮﺍی ﺁﻥ ﻣﻮﻻ ﺑﻪ ﺷـﻜﻠﯽ ﮔﺴـﺘﺮﺩﻩ ، ﺩﺭ ﺗﻤـﺎﻡ ﻛﻮﭼﻪ ﻫﺎ ﻭ ﺧﺎﻧﻪ ﻫﺎی ﺷﻬﺮ ﻭ ﺩﻳﺎﺭ ﺷﻴﻌﻴﺎﻥ ﻧﻤﺎﻳﺎﻥ ﮔﺮﺩﺩ.

۸) ﻧﻮﺷـﺘﻪ ﺣﺎﺿـﺮ ﺑـﺎ ﻋﻨﺎﻳـﺖ ﺑﻪ ﻣﻮﺿـﻮﻉ ﻣﻬـﻢ ﺧﻮﻳـﺶ، ﺍﺯ ﺗﻌﺎﻟﻴﻢ ﺑﻠﻨـﺪ ﺧـﺪﺍ ﻭ ﺭﺳـﻮﻝ، ﻭ ﮔﻔﺘـﺎﺭ ﺍﺋﻤـﻪ ﺑﺰﺭﮔـﻮﺍﺭ ﻭ ﻧﻴـﺰ ﺳـﻔﺎﺭﺵﻫﺎ ﻭ ﺭﻫﻨﻤﻮﺩﻫـﺎی ﺑﺰﺭﮔﺎﻥ ﺷـﻴﻌﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺁﻛﻨﺪﻩ ﺷﺪﻩ ﺍﺳﺖ.

ﺍﺯ ﺳﻮی ﺩﻳﮕﺮ ﺳﻌﯽ ﺑﺮ ﺁﻥ ﺑﻮﺩﻩ ﻛﻪ ﺩﺭ ﺁﻥ ﺍﺯ ﺑﻪ ﻛﺎﺭﮔﻴﺮی ﻋﺒﺎﺭﺍﺕ ﻭ ﻟﻐـﺎﺕ ﺩﺷـﻮﺍﺭ ﻭ ﻣﺘـﻮﻥ ﻋﺮﺑﯽ ﺍﺟﺘﻨﺎﺏ ﺷـﻮﺩ ﻭ ﺁﻥ ﺣﻘﺎﻳـﻖ ﺑﻠﻨﺪ ﺩﺭ ﻗﺎﻟﺐ ﻛﻠﻤﺎﺕ ﺳﺎﺩﻩ ﺑﻪ ﻣﺨﺎﻃﺐ ﮔﺮﺍﻣﯽ ﺍﻟﻘﺎ ﮔﺮﺩﺩ.

ﺗﺬﻛـﺮ ﺍﻳـﻦ ﻧﻜﺘـﻪ ﺿﺮﻭﺭی ﺍﺳـﺖ ﺗـﺎ ﺳﺎﺩﻩ ﻧﻮﻳﺴـﯽ ﻣﻮﺟـﻮﺩ ﺩﺭ ﻧﮕﺎﺭﺵ ﻛﺘﺎﺏ، ﺧﻮﺍﻧﻨﺪﻩ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﺍﻫﻤﻴﺖ ﻣﻮﺿﻮﻉ ﻭ ﻣﺤﺘﻮﺍی ﺁﻥ، ﻭ ﺗﻌﻤﻖ ﻭ ﺩﻭﺑﺎﺭﻩ ﺧﻮﺍﻧﯽ ﻋﺒﺎﺭﺍﺕ ﺁﻥ ﺑﯽ ﺗﻮﺟﻪ ﻧﺴﺎﺯﺩ.

ﺑﻪ ﻫﻤﻴﻦ ﺭﻭی، ﺑﺮﺍی ﺑﻪ ﺧﺎﻃﺮ ﺳـﭙﺮﺩﻩ ﺷـﺪﻥ ﻭ ﻧﻬﺎﺩﻳﻨﻪ ﮔﺮﺩﻳﺪﻥ ﻧﻜﺎﺕ ﻭ ﺩﺳـﺘﻮﺭﺍﻟﻌﻤﻞﻫﺎی ﺍﻳﻦ ﺍﺛﺮ، ﺗﻮﺻﻴﻪ ﻣﯽ ﺷـﻮﺩ ﻛـﻪ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﺧﻮﺩ، ﻋﺒـﺎﺭﺍﺕ ﻛﻠﻴﺪی ﺁﻥ ﺭﺍ ﻋﻼﻣﺖﮔﺰﺍﺭی ﻓﺮﻣﻮﺩﻩ ﻭ ﻣﻮﺭﺩ ﻣﻄﺎﻟﻌﻪ ﻣﺠﺪﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﮔﻮ ﺍﻳﻨﻜﻪ ﺑﻪ ﻫﻨﮕﺎﻡ ﺻﻔﺤﻪ ﺁﺭﺍﻳﯽ ﺑﺮﺧﯽ ﺍﺯ ﺟﻤﻼﺕ ﻭ ﻋﺒﺎﺭﺍﺕ ﺑﺎ ﺣﺮﻭﻑ ﺿﺨﻴﻢ ﻣﻮﺭﺩ ﺗﺄﻛﻴﺪ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ.

۹) ﻣﻬﻢﺗﺮﻳﻦ ﺳـﺒﺒﯽ ﻛﻪ ﺑﺎﻋﺚ ﺷـﺪ، ﺷـﻴﻮﻩ ﻧﻮﺷـﺘﺎﺭ ﻛﺘﺎﺏ ﺭﺍ ﺑﻪ ﺯﺑﺎﻥ ﮔﻔﺘـﺎﺭ ﻧﺰﺩﻳـﮏ ﻛﻨﻴـﻢ، ﺁﻥ ﺑـﻮﺩ ﻛﻪ ﻋﻤـﻮﻡ ﺧﺪﻣﺘﮕـﺰﺍﺭﺍﻥ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻭ ﺩﻟﺴـﻮﺧﺘﮕﺎﻥ ﻏﺮﺑـﺖ ﺍﻳﺸـﺎﻥ ﺑﺘﻮﺍﻧﻨـﺪ ﺗﻌﺎﺑﻴـﺮ ﺁﻥ ﺭﺍ ﺩﺭ ﮔﻔﺘﮕﻮﻫﺎی ﺭﻭﺯﻣـﺮﻩ ﺧـﻮﺩ ﺑـﻪ ﻛﺎﺭﮔﻴﺮﻧـﺪ ﻭ ﺁﻧﻬـﺎ ﺭﺍ ﺗﻮﺷـﻪ ﺭﺍﻫﯽ ﺩﺭ ﻣﺴـﻴﺮ ﻳﺎﺭی ﻣﻮﻻی ﻏﺮﻳﺐ ﺧﻮﻳﺶ ﻗﺮﺍﺭ ﺩﻫﻨﺪ.

ﺍﻣﻴﺪ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﻣﺘﻌﺎﻝ ﻓﻮﻕ ﺭﻏﺒﺖ ﻭ ﻓﺮﺍﺗﺮ ﺍﺯ ﺍﻧﮕﻴﺰﻩ ﻭ ﻧﻴﺖ ﻣﺎ، ﺑﻪ ﺍﻳﻦ ﺗﻼﺵ ﻧﺎﻗﺎﺑﻞ ﺑﺮﻛﺖ ﻭ ﺗﺄﺛﻴﺮ ﺑﺨﺸﺪ.

۱۰) ﺩﺭ ﭘﺎﻳـﺎﻥ ﺑﺎﻳﺪ ﺍﻗـﺮﺍﺭ ﻛﻨﻢ ﻛﻪ ﺗﺪﻭﻳﻦ ﻛﺘـﺎﺏ ﺣﺎﺿﺮ ﺩﺭ ﻋﻴﻦ ﺗﻤﺎﻡ ﻛﺎﺳـﺘﯽ ﻫﺎﻳﺶ ﻫﺮﮔﺰ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﺮﺍﻳﻢ ﻣﻴﺴـﺮ ﻧﺒﻮﺩ ﻭ ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻨﻜﻪ ﺁﻥ ﺭﺍ ﺑـﻪ ﺗﻮﻓﻴـﻖ ﺧﺪﺍی ﻣﺘﻌـﺎﻝ ﻣﯽ ﺩﺍﻧﻢ، ﻫﻤـﻜﺎﺭی ﻭ ﭘﺸـﺘﻴﺒﺎﻧﯽ ﺟﻤﻌﯽ ﺍﺯ ﺍﺭﺍﺩﺗﻤﻨـﺪﺍﻥ ﺑﯽ ﺭﻳﺎی ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﻣﻬﻢﺗﺮﻳﻦ ﻋﺎﻣﻞ ﺷـﻜﻞﮔﻴﺮی ﻭ ﺑﻪ ﺳﺎﺯی ﺁﻥ ﻣﯽ ﺑﻴﻨﻢ.

ﺩﺭ ﭘـﯽ ﺗﺄﻟﻴـﻒ ﺍﻳﻦ ﺍﺛـﺮ ﺍﺯ ﺧﺪﺍی ﻣﺘﻌـﺎﻝ، ﺳﭙﺎﺳـﮕﺰﺍﺭﻡ ﻛﻪ ﺗﻤﺎﻡ ﻧﻴﻜﻮﻳﯽ ﻫﺎ ﺭﺍ ﻭﺍﻣﺪﺍﺭ ﺍﻭ ﻫﺴﺘﻴﻢ.

ﺍﺯ ﭘﻴﺎﻣﺒﺮ ﺍﻛﺮﻡ ﺻﻠﻮﺍﺕ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ ﻭ ﺳﻠﻢ ﻭ ﺍﻫﻞﺑﻴﺖ ﺍﻳﺸﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ، ﺳﭙﺎﺳـﮕﺰﺍﺭﻡ ﻛﻪ ﻫﺮ ﺩﺍﻧﺶ ﺁﺳﻤﺎﻧﯽ ﻛﻪ ﺩﺭ ﻫﺮ ﺩﻓﺘﺮی ﮔﺮﺩ ﺁﻳﺪ، ﻓﺮﻭﻍ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻓﺮﻭﺯﺵ ﺁﻧﺎﻥ ﺑﺮﮔﺮﻓﺘﻪ ﺍﺳﺖ.

ﺍﺯ ﺍﻣـﺎﻡ ﻋﺼـﺮ - ﻛـﻪ ﺩﺭﻭﺩ ﺧـﺪﺍ ﺑـﺮ ﺍﻭ ﻭ ﺑـﺮ ﭘﺪﺭﺍﻧـﺶ ﺑـﺎﺩ - ﺳﭙﺎﺳـﮕﺰﺍﺭﻡ، ﻛﻪ ﺍﻳﻦ ﻛﻤﺘﺮﻳﻦ ﺭﺍ ﺍﺯ ﺑﻪ ﻗﻠـﻢ ﺁﻭﺭﺩﻥ ﻧﺎﻡ ﺑﺰﺭگ ﺧﻮﻳﺶ ﻣﺤﺮﻭﻡ ﻧﺴﺎﺧﺖ.

ﺳـﭙﺲ ﺧﻮﺩ ﺭﺍ ﻣﺪﻳﻮﻥ ﺗﻤﺎﻡ ﺭﺍﻭﻳﺎﻥ ﺍﺧﺒﺎﺭ ﻭ ﺁﺛﺎﺭ ﭘﻴﺎﻣﺒﺮ ﮔﺮﺍﻣﯽ ﻭ ﺍﻫﻞﺑﻴﺖ ﺍﻳﺸﺎﻥ ﻭ ﻣﺮﻫﻮﻥ ﺟﻤﻠﻪ ﻋﺎﻟﻤﺎﻥ ﺑﺰﺭﮔﻮﺍﺭ ﻭ ﺑﺎﺗﻘﻮﺍﻳﯽ ﻣﯽ ﺩﺍﻧﻢ ﻛﻪ ﻭﺍﺳـﻄﻪ ﻣﺎ ﺩﺭ ﻛﺴـﺐ ﻓﻴـﻮﺽ ﺭﺑ ﺎﻧﯽ ﻭ ﻋﻠﻮﻡ ﺁﺳـﻤﺎﻧﯽ ﺍﻫﻞﺑﻴﺖ ﻭ ﻣﺮﺑﯽ ﻭ ﺍﻟﮕﻮی ﺟﺎﻣﻌﻪ ﺷـﻴﻌﻪ ﺩﺭ ﺟﻬﺖ ﺧﺪﻣﺘﮕﺰﺍﺭی ﻭ ﺁﺳﺘﺎﻧﻪ ﺑﻮﺳﯽ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻮﺩﻩ ﺍﻧﺪ.

ﻫﻤﭽﻨﻴـﻦ ﺍﺯ ﺩﻭﺳـﺘﺎﻥ ﺑﺰﺭﮔـﻮﺍﺭی ﻛـﻪ ﭘﻴﺸـﻨﻬﺎﺩ ﻭ ﺗﺮﻏﻴﺒﺐ ﺁﻧﻬﺎ ﺑـﻪ ﻣﻜﺘـﻮﺏ ﺷـﺪﻥ ﭘﻴـﺎﻡ ﺧﻴﺰﺵ ﺑـﺮﺍی ﻳـﺎﺭی ﺍﻣـﺎﻡ ﻋﺼـﺮ ﺍﻧﺠﺎﻣﻴﺪ، ﺳﭙﺎﺳﮕﺰﺍﺭﻡ.

ﺍﺯ ﺍﺳـﺎﺗﻴﺪ ﺑﺰﺭﮔﻮﺍﺭﻡ ﺟﻨـﺎﺏ ﺁﻗﺎی ﺍﺻﻐﺮ ﺻﺎﺩﻗﯽ ﻭ ﺟﻨﺎﺏ ﺁﻗﺎی ﺩﻛﺘﺮ ﻗﻨﺪی ﻛﻪ ﺍﻳﻦ ﻗﻠﻢ ﺷﻜﺴﺘﻪ ﺭﺍ ﻣﻮﺭﺩ ﻣﺪﺍﻗﻪ ﺧﻮﻳﺶ ﻗﺮﺍﺭ ﺩﺍﺩﻧﺪ ﻭ ﺑﺎ ﺗﻮﺻﻴﻪ ﻫﺎی ﺧﻮﺩ ﺁﻥ ﺭﺍ ﻧﻮﺍ ﺑﺨﺸﻴﺪﻧﺪ، ﺳﭙﺎﺳﮕﺰﺍﺭﻡ.

ﺍﺯ ﻣـﺎﺩﺭ ﺑﺰﺭﮔـﻮﺍﺭ ﻭ ﺗﻼﺷـﮕﺮﻡ ﻭ ﺍﺯ ﻫﻤﺴـﺮ ﺷـﻜﻴﺒﺎ ﻭ ﻫﻤﺮﺍﻫﻢ ﻛﻪ ﭘﺸـﺘﻴﺒﺎﻧﯽ ﻭ ﻫﻤﺪﻟﯽ ﺍﻳﺸـﺎﻥ ﻫﻤﻮﺍﺭﻩ ﻣﺮﺍ ﺩﺭ ﺟﻬـﺖ ﺧﺪﻣﺎﺕ ﺩﻳﻨﯽ ﻛﺎﻣﻴﺎﺏ ﺳﺎﺧﺘﻪ ، ﺻﻤﻴﻤﺎﻧﻪ ﻗﺪﺭﺩﺍﻧﯽ ﻣﯽ ﻛﻨﻢ.
