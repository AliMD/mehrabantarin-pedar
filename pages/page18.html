بازگشت

ﻣﺎﻧﻨﺪ ﭘﻴﺮﻣـﺮﺩ ﻗﻔﻞﺳـﺎﺯ

ﻳﻜـﯽ ﺍﺯ ﺷـﻴﻌﻴﺎﻥ، ﺑﻪ ﻧﺎﻡ ﺍﺑﻮﻫـﺎﺭﻭﻥ ﮔﻮﻳﺪ: ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑـﻪ ﭼﻨﺪ ﺗﻦ ﺍﺯ ﻛﺴـﺎﻧﯽ ﻛﻪ ﺑﻪ ﺧﺪﻣﺘﺶ ﺭﺳـﻴﺪﻩ ﺑﻮﺩﻧﺪ، ﻓﺮﻣﻮﺩ: «ﺷـﻤﺎ ﭼﺮﺍ ﺑﻪ ﻣﺎ ﺗﻮﻫﻴﻦ ﻣﯽ ﻛﻨﻴﺪ!؟» ﭘﺲ ﺷـﺨﺼﯽ ﺍﺯ ﺍﻫﻞ ﺧﺮﺍﺳﺎﻥ ﺍﺯ ﺟﺎی ﺑﺮﺧﺎﺳـﺖ ﻭ ﻋﺮﺿﻪ ﺩﺍﺷـﺖ: «ﻣﺎ ﺑﻪ ﺧﺪﺍ ﭘﻨﺎﻩ ﻣﯽ ﺑﺮﻳﻢ ﺍﺯ ﺍﻳﻨﻜﻪ ﺷﻤﺎ ﺭﺍ ﺳـﺒﮏ ﺑﺸﻤﺎﺭﻳﻢ، ﻳﺎ ﭼﻴﺰی ﺍﺯ ﺍﻭﺍﻣﺮ ﻭ ﺩﺳـﺘﻮﺭﺍﺗﺖ ﺭﺍ ﻧﺎﺩﻳﺪﻩ ﮔﻴﺮﻳﻢ.» ﺁﻥ ﺣﻀﺮﺕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩ: «ﺍﻣﺎ ﺗﻮ ﺧﻮﺩ ﻳﻜﯽ ﺍﺯ ﻛﺴﺎﻧﯽ ﻫﺴﺘﯽ ﻛـﻪ ﺑـﻪ ﻣﻦ ﺗﻮﻫﻴﻦ ﻛﺮﺩﻩ ﺍﺳـﺖ.» ﻋﺮﺽ ﻛﺮﺩ: «ﭘﻨﺎﻩ ﺑﻪ ﺧـﺪﺍ ﻣﯽ ﺑﺮﻡ ﺍﺯ ﺍﻳﻨﻜﻪ ﺑﻪ ﺷـﻤﺎ ﺗﻮﻫﻴﻦ ﻛﺮﺩﻩ ﺑﺎﺷـﻢ!» ﻓﺮﻣﻮﺩ: «ﻭﺍی ﺑﺮ ﺗﻮ! ﺁﻳﺎ ﻧﺸـﻨﻴﺪی ﻛـﻪ ﻓﻼﻧـﯽ ﺩﺭ ﺣﺎﻟﻴﻜـﻪ ﺩﺭ ﻧﺰﺩﻳﻜﯽ ﺟﺤﻔـﻪ ﺑﻮﺩﻳﺪ (ﺟﺤﻔـﻪ ﻳﻜﯽ ﺍﺯ ﻣﻴﻘﺎﺕ ﻫـﺎی ﺣﺞ ﺍﺳـﺖ) ﺑﻪ ﺗـﻮ ﻣﯽ ﮔﻔﺖ: ﺑـﻪ ﻣﻘﺪﺍﺭ ﻳﮏ ﻣﻴـﻞ ﺭﺍﻩ (ﺗﻘﺮﻳﺒﺎ ﻣﻌﺎﺩﻝ ﭼﻬﺎﺭ ﻛﻴﻠﻮﻣﺘﺮ) ﻣﺮﺍ ﺳـﻮﺍﺭ ﻛﻦ، ﻛﻪ ﺑﻪ ﺧﺪﺍ ﺳﻮﮔﻨﺪ ﻣﻦ ﺧﺴـﺘﻪ ﻭ ﻣﺎﻧﺪﻩ ﺷﺪﻩ ﺍﻡ. ﺑﻪ ﺧﺪﺍ ﺗﻮ ﺳـﺮﺕ ﺭﺍ ﻫﻢ ﺑﺮﺍی ﺍﻭ ﺑﻠﻨﺪ ﻧﻜﺮﺩی ﻭ ﺍﻳـﻦ ﭼﻨﻴـﻦ ﺑﻪ ﺍﻭ ﺗﻮﻫﻴـﻦ ﻛـﺮﺩی. ﺑﺪﺍﻧﻴﺪ ﻫﺮ ﻛﺲ ﻣﺆﻣﻨﯽ ﺭﺍ ﺳـﺒﮏ ﺷﻤﺎﺭﺩ ﻣﺎ ﺭﺍ ﺳﺒﮏ ﺷﻤﺮﺩﻩ ﻭ ﺣﺮﻣﺖ ﺧﺪﺍی ﻋﺰﻭﺟﻞ ﺭﺍ ﺿﺎﻳﻊ ﻧﻤﻮﺩﻩ ﺍﺳﺖ.»(68)

ﻳﻜﯽ ﺍﺯ ﺍﻣﻮﺭی ﻛﻪ ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﻧﺴـﺒﺖ ﺑﻪ ﺁﻥ ﺑﺴـﻴﺎﺭ ﺣﺴـﺎﺱ ﻫﺴـﺘﻨﺪ، ﺭﺳﻴﺪﮔﯽ ﺑﻪ ﺷﻴﻌﻴﺎﻥ ﺍﺳـﺖ. ﻫﺮ ﻛﺲ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﺍﻣﺮ ﺩﻗﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﺭﺳﻴﺪﮔﯽ ﺑﻪ ﺷﻴﻌﻴﺎﻥ ﺧﺎﻟﺼﺎﻧﻪ ﻭ ﻓﺪﺍﻛﺎﺭﺍﻧﻪ ﺑﻜﻮﺷـﺪ، ﺑﻪ ﻃﻮﺭ ﻓﻮﻕﺍﻟﻌﺎﺩﻩ ﺍی ﺩﺭ ﻧﻈـﺮ ﺍﻫﻞﺑﻴﺖ ﻭ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴﻼﻡ ﻋﺰﻳﺰ ﻣﯽ ﺷﻮﺩ ﻭ ﻫﺮ ﻛﺲ ﺭﺳﻴﺪﮔﯽ ﺑﻪ ﺷﻴﻌﻴﺎﻥ ﺩﺭ ﻧﻈﺮﺵ ﻣﻬﻢ ﻧﺒﺎﺷـﺪ ﻭ ﻧﺴـﺒﺖ ﺑﻪ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺁﻧﻬـﺎ ﺑﯽ ﺗﻮﺟﻪ ﻭ ﻻﺍﺑﺎﻟﯽ ﺑﺎﺷـﺪ، ﺍﺯ ﭼﺸـﻢ ﺍﻫﻞﺑﻴﺖ ﻭ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻣﯽ ﺍﻓﺘﺪ ﻭ ﻣﻮﺭﺩ ﺧﺸـﻢ ﻭ ﻏﻀﺐ ﺍﻳﺸـﺎﻥ ﻗﺮﺍﺭ ﻣﯽ ﮔﻴﺮﺩ.

ﺍﺯ ﺍﻳـﻦﺭﻭ ﻳﻜـﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﺭﺍﻫﻬﺎی ﺗﻘﺮﺏ ﺑﻪ ﺳـﺎﺣﺖ ﻣﻘﺪﺱ ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﺭﺳـﻴﺪﮔﯽ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﮔﺮﻓﺘـﺎﺭ ﻭ ﺑﺮﺧﻮﺭﺩ ﺧﻴﺮﺧﻮﺍﻫﺎﻧـﻪ، ﻣﻨﺼﻔﺎﻧـﻪ ﻭ ﺟﻮﺍﻧﻤﺮﺩﺍﻧﻪ ﺑﺎ ﺍﻳﺸـﺎﻥ ﺍﺳـﺖ. ﺍﮔﺮ ﻛﺴـﯽ ﺩﺭ ﺍﻳـﻦ ﻭﺍﺩی ﮔﺎﻡ ﻧﻬـﺪ، ﺑﯽ ﺗﺮﺩﻳـﺪ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﺰﺩﻳﮏ ﺧﻮﺍﻫـﺪ ﺷـﺪ ﻭ ﺁﻥ ﺣﻀـﺮﺕ ﺍﻭ ﺭﺍ ﻣـﻮﺭﺩ ﻋﻨﺎﻳـﺖ ﻭﻳﮋﻩ ﺧﻮﻳـﺶ ﻗﺮﺍﺭ ﻣﯽ ﺩﻫﺪ. ﺍﺯ ﺭﻭﺍﻳﺎﺕ ﺑﺮﻣﯽ ﺁﻳﺪ ﻛﻪ ﺍﮔﺮ ﻛﺴـﯽ ﺑﻪ ﻳﺎﺭی ﺷـﻴﻌﻴﺎﻥ ﺑﺸـﺘﺎﺑﺪ ﻭ ﺁﻧﺎﻥ ﺭﺍ ﺍﻛﺮﺍﻡ ﻛﻨﺪ، ﮔﻮﻳﯽ ﺑﻪ ﻳﺎﺭی ﺍﻣﺎﻡ ﺷﻴﻌﻴﺎﻥ ﺷﺘﺎﻓﺘﻪ ﻭ ﺍﻭ ﺭﺍ ﺍﻛﺮﺍﻡ ﻛﺮﺩﻩ ﺍﺳﺖ.

ﻛﺴـﺎﻧﯽ ﻛﻪ ﺩﺭ ﻓﻜـﺮ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺷـﻴﻌﻴﺎﻥ ﻧﻴﺴـﺘﻨﺪ ﻭ ﺑﻪ ﺿﺮﺏ ﭼﻠﻪ ﻧﺸﻴﻨﯽ ﻫﺎ ﻭ ﺣﻠﻘﻪ ﻫﺎی ﺫﻛﺮ ﻭ ﺭﻳﺎﺿﺖﻛﺸﯽ ﻫﺎ ﻣﯽ ﺧﻮﺍﻫﻨﺪ ﺻﻔﺎی ﺑﺎﻃـﻦ ﭘﻴﺪﺍ ﻛﻨﻨﺪ ﻭ ﻣـﻮﺭﺩ ﻋﻨﺎﻳﺖ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻗﺮﺍﺭ ﮔﻴﺮﻧﺪ، ﺑﺎﻳﺪ ﺑﺪﺍﻧﻨﺪ ﻛﻪ ﺑﻪ ﺑﻴﺮﺍﻫﻪ ﻣﯽ ﺭﻭﻧﺪ.

ﻧﻘﻞ ﺍﺳﺖ ﻛﻪ ﻳﻜﯽ ﺍﺯ ﺩﺍﻧﺸﻤﻨﺪﺍﻥ، ﺁﺭﺯﻭی ﺯﻳﺎﺭﺕ ﺣﻀﺮﺕ ﺑﻘﻴﺔ ﺍﻟﻠـﻪ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﺭﺍ ﺩﺍﺷـﺖ ﻭ ﺍﺯ ﻋﺪﻡ ﻣﻮﻓﻘﻴﺖ ﺧـﻮﺩ، ﺭﻧﺞ ﻣﯽ ﺑﺮﺩ. ﻣﺪﺗﻬـﺎ ﺭﻳﺎﺿﺖ ﻛﺸـﻴﺪ ﻭ ﺁﻥ ﭼﻨﺎﻥ ﻛـﻪ ﺩﺭ ﻣﻴﺎﻥ ﻃـﻼﺏ ﺣﻮﺯﻩ ﻧﺠﻒ ﻣﺸـﻬﻮﺭ ﺍﺳﺖ، ﺷـﺐﻫﺎی ﭼﻬﺎﺭﺷﻨﺒﻪ ﺑﻪ «ﻣﺴﺠﺪ ﺳـﻬﻠﻪ» ﻣﯽ ﺭﻓﺖ ﻭ ﺑـﻪ ﻋﺒﺎﺩﺕ ﻣﯽ ﭘﺮﺩﺍﺧﺖ، ﺗﺎ ﺷـﺎﻳﺪ ﺗﻮﻓﻴﻖ ﺩﻳﺪﺍﺭ ﺁﻥ ﻣﺤﺒﻮﺏ ﻋﺎﺷـﻘﺎﻥ ﻧﺼﻴﺒﺶ ﮔﺮﺩﺩ.

ﻣﺪﺕ ﻫﺎ ﻛﻮﺷـﻴﺪ ﻭﻟﯽ ﺑﻪ ﻧﺘﻴﺠﻪ ﻧﺮﺳـﻴﺪ. ﺳـﭙﺲ ﺑﻪ ﻋﻠﻮﻡ ﻏﺮﻳﺒﻪ ﻭ ﺍﺳـﺮﺍﺭ ﺣﺮﻭﻑ ﻭ ﺍﻋﺪﺍﺩ ﻣﺘﻮﺳـﻞ ﺷـﺪ، ﭼﻠﻪ ﻫﺎ ﻧﺸﺴـﺖ ﻭ ﺭﻳﺎﺿﺖﻫﺎ ﻛﺸـﻴﺪ، ﺍﻣﺎ ﺑﺎﺯ ﻫﻢ ﻧﺘﻴﺠـﻪ ﺍی ﻧﮕﺮﻓﺖ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺷـﺐﺑﻴﺪﺍﺭی ﻫﺎی ﻓﺮﺍﻭﺍﻥ ﻭ ﻣﻨﺎﺟﺎﺕ ﻫﺎی ﺳﺤﺮﮔﺎﻫﺎﻥ، ﺻﻔﺎی ﺑﺎﻃﻨﯽ ﺩﺭ ﺍﻭ ﺍﻳﺠﺎﺩ ﻛﺮﺩﻩ ﺑﻮﺩ، ﮔﺎﻫﯽ ﻧﻮﺭی ﺑﺮ ﺩﻟﺶ ﻣﯽ ﺗﺎﺑﻴﺪ ﻭ ﺣﻘﺎﻳﻘﯽ ﺭﺍ ﻣﯽ ﺩﻳﺪ ﻭ ﺩﻗﺎﻳﻘﯽ ﺭﺍ ﻣﯽ ﺷﻨﻴﺪ.

ﺭﻭﺯی ﺩﺭ ﻳﻜـﯽ ﺍﺯ ﺍﻳـﻦ ﺣﺎﻻﺕ ﻣﻌﻨﻮی ﺑﻪ ﺍﻭ ﮔﻔﺘﻪ ﺷـﺪ: «ﺩﻳﺪﻥ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﺮﺍی ﺗﻮ ﻣﻤﻜﻦ ﻧﻴﺴـﺖ، ﻣﮕـﺮ ﺁﻧﻜﻪ ﺑﻪ ﻓﻼﻥ ﺷﻬﺮ ﺳﻔﺮ ﻛﻨﯽ.» ﺑﻪ ﻋﺸﻖ ﺩﻳﺪﺍﺭ، ﺭﻧﺞ ﺍﻳﻦ ﻣﺴﺎﻓﺮﺕ ﺭﺍ ﺑﺮ ﺧﻮﺩ ﻫﻤﻮﺍﺭ ﻛـﺮﺩ ﻭ ﭘﺲ ﺍﺯ ﭼﻨﺪ ﺭﻭﺯ ﺑﻪ ﺁﻥ ﺷـﻬﺮ ﺭﺳـﻴﺪ. ﺩﺭ ﺁﻧﺠـﺎ ﻧﻴﺰ ﭼﻠﻪ ﮔﺮﻓﺖ ﻭ ﺑﻪ ﺭﻳﺎﺿﺖ ﻣﺸـﻐﻮﻝ ﺷـﺪ. ﺭﻭﺯ ﺳـﯽ ﻭﻫﻔﺘﻢ ﻭ ﻳﺎ ﺳﯽ ﻭﻫﺸـﺘﻢ ﺑﻪ ﺍﻭ ﮔﻔﺘﻨـﺪ: «ﺍﻵﻥ ﺣﻀﺮﺕ ﺑﻘﻴﺔ ﺍﻟﻠﻪ ﺍﺭﻭﺍﺣﻨـﺎ ﻓﺪﺍﻩ ﺩﺭ ﺑﺎﺯﺍﺭ ﺁﻫﻨﮕﺮﺍﻥ، ﺩﺭ ﻣﻐﺎﺯﻩ ﭘﻴﺮﻣﺮﺩ ﻗﻔﻞﺳـﺎﺯی ﻧﺸﺴـﺘﻪ ﺍﻧﺪ. ﻫﻢ ﺍﻛﻨﻮﻥ ﺑﺮﺧﻴـﺰ ﻭ ﺑﻪ ﺧﺪﻣﺖ ﺣﻀﺮﺕ ﺷـﺮﻓﻴﺎﺏ ﺷﻮ!» ﺑﺎ ﺍﺷﺘﻴﺎﻕ ﺍﺯ ﺟﺎ ﺑﺮﺧﺎﺳﺖ. ﺑﻪ ﺩﻛﺎﻥ ﭘﻴﺮﻣﺮﺩ ﺭﻓـﺖ. ﻭﻗﺘـﯽ ﺭﺳـﻴﺪ، ﺩﻳـﺪ ﺣﻀـﺮﺕ ﻭﻟﯽ ﻋﺼـﺮ ﻋﺠـﻞ ﺍﻟﻠـﻪ ﺗﻌﺎﻟﯽ ﻓﺮﺟﻪ ﺍﻟﺸـﺮﻳﻒ ﺁﻧﺠﺎ ﻧﺸﺴـﺘﻪ ﺍﻧﺪ، ﺑﺎ ﭘﻴﺮﻣﺮﺩ ﮔﺮﻡ ﮔﺮﻓﺘﻪ ﺍﻧﺪ ﻭ ﺳـﺨﻨﺎﻥ ﻣﺤﺒﺖﺁﻣﻴـﺰ ﻣﯽ ﮔﻮﻳﻨـﺪ. ﻫﻤﻴـﻦ ﻛﻪ ﺳـﻼﻡ ﻛـﺮﺩ، ﺁﻥ ﺣﻀﺮﺕ ﭘﺎﺳـﺦ ﻓﺮﻣﻮﺩﻧﺪ ﻭ ﺍﺷﺎﺭﻩ ﺑﻪ ﺳﻜﻮﺕ ﻛﺮﺩﻧﺪ.

ﺩﺭ ﺍﻳـﻦ ﺣﺎﻝ، ﺩﻳـﺪ ﭘﻴﺮﺯﻧﯽ ﻧﺎﺗـﻮﺍﻥ ﻭ ﻗﺪﺧﻤﻴـﺪﻩ، ﻋﺼﺎﺯﻧﺎﻥ ﺁﻣﺪ ﻭ ﺑـﺎ ﺩﺳـﺖ ﻟـﺮﺯﺍﻥ ﻗﻔﻠـﯽ ﺭﺍ ﻧﺸـﺎﻥ ﺩﺍﺩ ﻭ ﮔﻔـﺖ: ﺍﮔﺮ ﻣﻤﻜﻦ ﺍﺳـﺖ ﺑـﺮﺍی ﺭﺿـﺎی ﺧﺪﺍ ﺍﻳﻦ ﻗﻔﻞ ﺭﺍ ﺑﻪ ﻣﺒﻠﻎ ﺳـﻪ ﺷـﺎﻫﯽ ﺑﺨﺮﻳـﺪ ﻛﻪ ﻣﻦ ﺑﻪ ﺳـﻪ ﺷـﺎﻫﯽ ﭘﻮﻝ ﻧﻴـﺎﺯ ﺩﺍﺭﻡ. ﭘﻴﺮﻣﺮﺩ ﻗﻔﻞ ﺭﺍ ﮔﺮﻓﺖ ﻭ ﻧـﮕﺎﻩ ﻛﺮﺩ ﻭ ﺩﻳﺪ ﺑﯽ ﻋﻴﺐ ﻭ ﺳـﺎﻟﻢ ﺍﺳـﺖ (ﻭ ﺗﻨﻬﺎ ﻣﺸﻜﻠﺶ ﺁﻥ ﺍﺳـﺖ ﻛﻪ ﻛﻠﻴﺪ ﻧﺪﺍﺭﺩ.) ﮔﻔـﺖ: ﺧﻮﺍﻫـﺮﻡ! ﺍﻳﻦ ﻗﻔﻞ (ﺑﺎ ﻛﻠﻴﺪﺵ ﺩﻩ ﺷـﺎﻫﯽ ﻣﯽ ﺍﺭﺯﺩ ﻭ ﺍﻻﻥ ﻛﻪ ﻛﻠﻴﺪ ﻧﺪﺍﺭﺩ) ﻫﺸﺖ ﺷﺎﻫﯽ ﺍﺭﺯﺵ ﺩﺍﺭﺩ، ﺯﻳﺮﺍ ﻗﻴﻤﺖ ﺳﺎﺧﺖ ﻛﻠﻴﺪ ﺁﻥ، ﺑﻴﺶ ﺍﺯ ﺩﻭ ﺷﺎﻫﯽ ﻧﻴﺴﺖ. ﺷﻤﺎ ﺍﮔﺮ ﺩﻭ ﺷﺎﻫﯽ ﺑﻪ ﻣﻦ ﺑﺪﻫﻴﺪ، ﻣﻦ ﻛﻠﻴﺪ ﺍﻳﻦ ﻗﻔﻞ ﺭﺍ ﻣﯽ ﺳـﺎﺯﻡ ﻭ ﺑﺎ ﺩﺍﺷـﺘﻦ ﻛﻠﻴﺪ ﺩﻩ ﺷـﺎﻫﯽ، ﻗﻴﻤﺖ ﺁﻥ ﺧﻮﺍﻫﺪ ﺑﻮﺩ!

ﭘﻴﺮﺯﻥ ﮔﻔﺖ: ﻧﻪ، ﺑﻪ ﺁﻥ ﻧﻴﺎﺯی ﻧﺪﺍﺭﻡ، ﺷﻤﺎ ﺍﻳﻦ ﻗﻔﻞ ﺭﺍ ﺳﻪ ﺷﺎﻫﯽ ﺍﺯ ﻣﻦ ﺑﺨﺮﻳﺪ، ﺷﻤﺎ ﺭﺍ ﺩﻋﺎ ﻣﯽ ﻛﻨﻢ.

ﭘﻴﺮﻣﺮﺩ ﺑﺎ ﻛﻤﺎﻝ ﺳـﺎﺩﮔﯽ ﮔﻔﺖ: ﺧﻮﺍﻫﺮﻡ! ﺗﻮ ﻣﺴﻠﻤﺎﻧﯽ، ﻣﻦ ﻫﻢ ﻛﻪ ﻣﺴﻠﻤﺎﻧﻢ. ﭼﺮﺍ ﻣﺎﻝ ﻣﺴﻠﻤﺎﻥ ﺭﺍ ﺍﺭﺯﺍﻥ ﺑﺨﺮﻡ ﻭ ﺣﻖ ﻛﺴﯽ ﺭﺍ ﺿﺎﻳﻊ ﻛﻨـﻢ؟ ﺍﻳﻦ ﻗﻔﻞ ﺍﻛﻨﻮﻥ ﻫﺸـﺖ ﺷـﺎﻫﯽ ﺍﺭﺯﺵ ﺩﺍﺭﺩ، ﻣـﻦ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻢ ﻣﻨﻔﻌﺖ ﺑﺒﺮﻡ، ﺑﻪ ﻫﻔﺖ ﺷﺎﻫﯽ ﻣﯽ ﺧﺮﻡ، ﺯﻳﺮﺍ ﺩﺭ ﻣﻌﺎﻣﻠﻪ ﻫﺸﺖ ﺷﺎﻫﯽ، ﺑﻴﺶ ﺍﺯ ﻳﮏ ﺷـﺎﻫﯽ ﻣﻨﻔﻌﺖ ﺑﺮﺩﻥ، ﺑﯽ ﺍﻧﺼﺎﻓﯽ ﺍﺳﺖ. ﺍﮔﺮ ﻣﯽ ﺧﻮﺍﻫﯽ ﺑﻔﺮﻭﺷـﯽ، ﻣﻦ ﻫﻔﺖ ﺷـﺎﻫﯽ ﻣﯽ ﺧـﺮﻡ ﻭ ﺑﺎﺯ ﺗﻜـﺮﺍﺭ ﻣﯽ ﻛﻨـﻢ: ﻗﻴﻤﺖ ﻭﺍﻗﻌﯽ ﺁﻥ ﻫﺸـﺖ ﺷﺎﻫﯽ ﺍﺳﺖ، ﭼﻮﻥ ﻣﻦ ﻛﺎﺳﺐ ﻫﺴﺘﻢ ﻭ ﺑﺎﻳﺪ ﻧﻔﻌﯽ ﺑﺒﺮﻡ، ﻳﮏ ﺷﺎﻫﯽ ﺍﺭﺯﺍﻥﺗﺮ ﻣﯽ ﺧﺮﻡ!

ﭘﻴـﺮﺯﻥ ﺑـﺎﻭﺭ ﻧﻤﯽ ﻛـﺮﺩ ﻛﻪ ﺍﻳﻦ ﻣـﺮﺩ ﺣﻘﻴﻘـﺖ ﺭﺍ ﺑـﻪ ﺍﻭ ﻣﯽ ﮔﻮﻳﺪ ﻭ ﻗﺼـﺪ ﻣﺴـﺨﺮﻩ ﻛـﺮﺩﻥ ﺍﻭ ﺭﺍ ﻧـﺪﺍﺭﺩ. ﻧﺎﺭﺍﺣﺖ ﺷـﺪﻩ ﺑـﻮﺩ ﻭ ﺑﺎ ﺧﻮﺩ ﻣﯽ ﮔﻔـﺖ: ﻣـﻦ ﺧـﻮﺩﻡ ﻣﯽ ﮔﻮﻳـﻢ ﻫﻴـﭻ ﻛﺲ ﺑـﻪ ﺳـﻪ ﺷـﺎﻫﯽ ﺭﺍﺿﯽ ﻧﺸـﺪﻩ ﺍﺳـﺖ. ﺍﻟﺘﻤﺎﺱ ﻛﺮﺩﻡ ﻛﻪ ﺳﻪ ﺷﺎﻫﯽ ﺧﺮﻳﺪﺍﺭی ﻛﻨﻨﺪ ﻭﻟﯽ ﻗﺒﻮﻝ ﻧﻜﺮﺩﻧﺪ. ﻋﻠﺖ ﺍﻳﻨﻜﻪ ﺍﺻﺮﺍﺭ ﺩﺍﺷﺘﻢ ﻛﻪ ﺁﻥ ﺭﺍ ﺳﻪ ﺷﺎﻫﯽ ﺑﺨﺮﻧﺪ ﻫﻢ ﺍﻳﻦ ﺑﻮﺩ ﻛﻪ ﻣﻘﺼﻮﺩ ﻣﻦ ﺑﺎ ﺩﻭ ﺷﺎﻫﯽ ﺍﻧﺠﺎﻡ ﻧﻤﯽ ﮔﻴﺮﺩ ﻭ ﭘﻮﻝ ﻣﻮﺭﺩ ﺍﺣﺘﻴﺎﺝ ﻣﻦ ﺳﻪ ﺷﺎﻫﯽ ﺍﺳﺖ.

ﭘﻴﺮﻣـﺮﺩ ﻫﻔﺖ ﺷـﺎﻫﯽ ﺑـﻪ ﺁﻥ ﺯﻥ ﺩﺍﺩ ﻭ ﻗﻔﻞ ﺭﺍ ﺧﺮﻳـﺪ. ﻫﻤﻴﻦ ﻛﻪ ﭘﻴـﺮﺯﻥ ﺭﻓـﺖ ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺁﻥ ﻋﺎﻟـﻢ ﻓﺮﻣﻮﺩﻧﺪ: «ﺁﻗـﺎی ﻋﺰﻳﺰ! ﺩﻳـﺪی ﻭ ﺍﻳﻦ ﻣﻨﻈﺮﻩ ﺭﺍ ﺗﻤﺎﺷـﺎ ﻛﺮﺩی؟! ﺍﻳﻦﻃﻮﺭ ﺷـﻮﻳﺪ، ﺗﺎ ﻣﺎ ﺳـﺮﺍﻍ ﺷـﻤﺎ ﺑﻴﺎﻳﻴﻢ. ﭼﻠﻪ ﻧﺸـﻴﻨﯽ ﻻﺯﻡ ﻧﻴﺴـﺖ! ﺑﻪ ﺟﻔﺮ ﻣﺘﻮﺳﻞ ﺷﺪﻥ ﺳﻮﺩی ﻧﺪﺍﺭﺩ! ﻋﻤﻞ ﺳﺎﻟﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻭ ﻣﺴﻠﻤﺎﻥ ﺑﺎﺷﻴﺪ ﺗﺎ ﻣﻦ ﺑﺘﻮﺍﻧﻢ ﺑﺎ ﺷﻤﺎ ﻫﻤﻜﺎﺭی ﻛﻨﻢ! ﺍﺯ ﻫﻤﻪ ﺍﻳﻦ ﺷـﻬﺮ، ﻣﻦ ﺍﻳﻦ ﭘﻴﺮﻣﺮﺩ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻛﺮﺩﻩ ﺍﻡ، ﺯﻳﺮﺍ ﺍﻳﻦ ﻣﺮﺩ، ﺩﻳﻦﺩﺍﺭ ﺍﺳـﺖ ﻭ ﺧﺪﺍ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳـﺪ. ﺍﻳﻦ ﻫﻢ ﺍﻣﺘﺤﺎﻧﯽ ﻛـﻪ ﺩﺍﺩ. ﺍﺯ ﺍﻭﻝ ﺑـﺎﺯﺍﺭ، ﺍﻳـﻦ ﭘﻴﺮﺯﻥ ﻋﺮﺽ ﺣﺎﺟﺖ ﻛـﺮﺩ ﻭ ﭼﻮﻥ ﺍﻭ ﺭﺍ ﻣﺤﺘـﺎﺝ ﻭ ﻧﻴﺎﺯﻣﻨـﺪ ﺩﻳﺪﻧﺪ، ﻫﻤﻪ ﺩﺭ ﻣﻘﺎﻡ ﺁﻥ ﺑﻮﺩﻧـﺪ ﻛﻪ ﺍﺭﺯﺍﻥ ﺑﺨﺮﻧﺪ ﻭ ﻫﻴﭻ ﻛﺲ ﺣﺘﯽ ﺳـﻪ ﺷﺎﻫﯽ ﻧﻴﺰ ﺧﺮﻳﺪﺍﺭی ﻧﻜﺮﺩ ﻭ ﺍﻳﻦ ﭘﻴﺮﻣﺮﺩ ﺑﻪ ﻫﻔﺖ ﺷﺎﻫﯽ ﺧﺮﻳﺪ. ﻫﻔﺘﻪ ﺍی ﺑﺮ ﺍﻭ ﻧﻤﯽ ﮔﺬﺭﺩ، ﻣﮕﺮ ﺁﻧﻜﻪ ﻣﻦ ﺳﺮﺍﻍ ﺍﻭ ﻣﯽ ﺁﻳﻢ ﻭ ﺍﺯ ﺍﻭ ﺩﻟﺠﻮﻳﯽ ﻭ ﺍﺣﻮﺍﻟﭙﺮﺳﯽ ﻣﯽ ﻛﻨﻢ.»

ﻣﺒﺎﻧﯽ ﺩﻳﻨﯽ ﻣﺎ ﻣﺤﺘﻮﺍی ﺍﻳﻦ ﺣﻜﺎﻳﺖ ﺭﺍ ﺗﺄﻳﻴﺪ ﻣﯽ ﻛﻨﺪ. ﺑﻪ ﺭﺍﺳـﺘﯽ ﺍﮔـﺮ ﻫﺮ ﻳﮏ ﺍﺯ ﻣﺎ ﺑﺮﺍی ﺷـﻴﻌﻴﺎﻥ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺣﺮﻣﺖ ﻗﺎﺋﻞ ﺷـﻮﺩ، ﺁﻧﻬﺎ ﺭﺍ ﻓﺮﺯﻧـﺪﺍﻥ ﺭﻭﺣﺎﻧـﯽ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﺪﺍﻧـﺪ ﻭ ﻫﻤﺎﻧﻄﻮﺭ ﻛﻪ ﺩﺭ ﻣـﻮﺭﺩ ﻓﺮﺯﻧـﺪﺍﻥ ﺧـﻮﺩ ﺍﻧﺼـﺎﻑ ﺑﻪ ﺧـﺮﺝ ﻣﯽ ﺩﻫـﺪ، ﺩﺭ ﻣـﻮﺭﺩ ﺁﻧﻬﺎ ﻧﻴﺰ ﻣﻨﺼﻒ ﺑﺎﺷـﺪ، ﺟﺎﻣﻌﻪ ﺷـﻴﻌﻪ ﺑﻪ ﭼﻪ ﺷـﻜﻞ ﻭ ﺻﻮﺭﺗﯽ ﺩﺭ ﺧﻮﺍﻫﺪ ﺁﻣﺪ؟ ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﺍﺳﻼﻡ ﻭ ﻣﺴﻠﻤﺎﻧﯽ ﭼﻪ ﺭﻭﻧﻘﯽ ﭘﻴﺪﺍ ﺧﻮﺍﻫﺪ ﻛﺮﺩ؟ ﺍﻳﻤﺎﻥ ﺣﻘﻴﻘﯽ ﺑﻪ ﭼﻪ ﺷـﻜﻮﻓﺎﻳﯽ ﻭ ﺩﺭﺧﺸﺸﯽ ﺧﻮﺍﻫﺪ ﺭﺳﻴﺪ؟ ﺑﻴﺎﻳﻴﻢ ﺑﺎ ﮔﺴـﺘﺮﺵ ﻓﺮﻫﻨﮓ ﻣﺴـﻠﻤﺎﻧﯽ ﻭ ﺟﻮﺍﻧﻤـﺮﺩی ﻭ ﺗﻮﺟﻪ ﻭﻳـﮋﻩ ﺑﻪ ﺑﻨﺪﮔﺎﻥ ﺧـﺪﺍ ﻛﺎﺭی ﻛﻨﻴﻢ ﻛﻪ ﺟﺎﻣﻌﻪ ﺷـﻴﻌﻪ ﺑﻪ ﺭﺷـﺪ ﻭ ﺗﻌﺎﻟـﯽ ﻣﻌﻨﻮی ﻧﺰﺩﻳﮏ ﺷﻮﺩ ﻭ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺳﻼﻡ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﺧﻮﺷﺤﺎﻝ ﻭ ﺭﺍﺿﯽ ﮔﺮﺩﺩ.

ﺁﺭﺯﻭی ﻫﻤـﻪ ﻣـﺎ ﺍﻳـﻦ ﺍﺳـﺖ ﻛـﻪ ﺭﻭﺯﮔﺎﺭ ﺣﻜﻮﻣـﺖ ﺣﻀـﺮﺕ ﻣﻬﺪی ﺭﺍ ﺷـﺎﻫﺪ ﺑﺎﺷـﻴﻢ. ﻳﻜﯽ ﺍﺯ ﺩﻻﻳﻞ ﺍﻳﻦ ﺁﺭﺯﻭ ﺁﻥ ﺍﺳـﺖ ﻛﻪ ﺩﺭ ﺁﻥ ﺭﻭﺯﮔﺎﺭ، ﻓﻀﺎﻳﻞ ﺍﺧﻼﻗﯽ ﺑﻪ ﺷـﻜﻠﯽ ﮔﺴـﺘﺮﺩﻩ ﻭ ﻛﺎﻣﻞ ﻭﺟﻮﺩ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ ﻭ ﻋﻤﻮﻡ ﺍﻧﺴﺎﻥﻫﺎ ﺩﺭ ﺍﺛﺮ ﺗﺮﺑﻴﺖ ﻭ ﻋﻨﺎﻳﺖ ﺣﻀﺮﺕ ﻭﻟﯽ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﺷﺎﻳﺴﺘﻪ ﻧﺎﻡ ﺍﻧﺴﺎﻧﻴﺖ ﻣﯽ ﺷﻮﻧﺪ. ﺍﺯ ﺧﺪﺍﻭﻧﺪ ﺗﻤﻨﺎ ﻣﯽ ﻛﻨﻴﻢ ﻛﻪ ﺩﺳﺖ ﺁﻥ ﻣﺮﺑﯽ ﺍﻧﺴﺎﻧﻴﺖ ﺭﺍ ﻫﺮﭼﻪ ﺯﻭﺩﺗﺮ ﺩﺭ ﺗﺮﺑﻴﺖ ﺑﺸﺮ ﺑﺎﺯ ﺳﺎﺯﺩ ﻭ ﻣﺎ ﺭﺍ ﺗﻮﻓﻴﻖ ﺩﻫﺪ ﺗﺎ ﺍﺯ ﻛﺴﺎﻧﯽ ﺑﺎﺷﻴﻢ ﻛﻪ ﺩﺳﺖ ﻋﻨﺎﻳﺖ ﺍﻭ ﺑﺮ ﺳﺮﺷﺎﻥ ﻛﺸﻴﺪﻩ ﻣﯽ ﺷﻮﺩ ﻭ ﻋﻘﻞﻫﺎﻳﺸﺎﻥ ﺑﺎﺭﻭﺭ ﻣﯽ ﮔﺮﺩﺩ.

ﻣﺨﻔـﯽ ﻧﻤﺎﻧﺪ ﻛﻪ ﺭﻭﺯﮔﺎﺭ ﻇﻬﻮﺭ، ﺭﻭﺯﮔﺎﺭ ﮔﺴـﺘﺮﺵ ﻫﻤﻪ ﺟﺎﻧﺒﻪ ﺍﻳﻦ ﻛﻤﺎﻻﺕ ﺍﺳـﺖ، ﻟﻜﻦ ﺩﺭ ﻋﺼﺮ ﺣﺎﺿﺮ ﻧﻴﺰ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻣﺘﻮﺳـﻞ ﺷـﺪ ﻭ ﺩﺭ ﺳـﺎﻳﻪ ﻋﻨﺎﻳﺎﺕ ﺍﻭ ﺑﻪ ﻛﻤﺎﻝ ﺍﻧﺴـﺎﻧﻴﺖ ﺩﺳـﺖ ﻳﺎﻓﺖ. ﺩﺭﺳـﺖ ﺍﺳـﺖ ﻛـﻪ ﺍﻛﻨـﻮﻥ ﺭﻭﺯﮔﺎﺭ ﻏﺮﺑـﺖ ﻛﻤﺎﻻﺕ ﺍﻧﺴـﺎﻧﯽ ﺣﺘـﯽ ﺩﺭ ﻣﻴﺎﻥ ﻣﺆﻣﻨﺎﻥ ﺍﺳـﺖ، ﺍﻣﺎ ﻫﻤـﻮﺍﺭﻩ ﺩﺭ ﻫﻤﻴﻦ ﺟﻮﺍﻣﻊ ﺁﻟـﻮﺩﻩ ﻧﻴـﺰ، ﻣﺆﻣﻨﺎﻥ ﻛﺎﻣﻠﯽ ﻫﺴـﺘﻨﺪ ﻛﻪ ﺍﺯ ﻓﻴﺾ ﺍﻣﺎﻡ ﺯﻧـﺪﻩ ﺯﻣﺎﻥ ﺧﻮﺩ ﺑﻬﺮﻩ ﻣﻨﺪ ﻣﯽ ﮔﺮﺩﻧﺪ ﻭ ﺧﻮﺩ ﺩﻟﻴﻠﯽ ﺑﺮ ﺗﻌﻄﻴﻞ ﻧﺸـﺪﻥ ﻧﻈﺎﻡ ﺍﻣﺎﻣﺖ ﺍﻟﻬﯽ ﻫﺴـﺘﻨﺪ. ﭼﻪ ﺍﺷـﻜﺎﻟﯽ ﺩﺍﺭﺩ ﻛﻪ ﻣﺎ ﻧﻴﺰ ﺑﺎ ﺗﻮﺳـﻞ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓـﺪﺍﻩ ﺗﻮﻓﻴﻖ ﻳﺎﺑﻴﻢ ﻛﻪ ﺍﺯ ﺁﻧﺎﻥ ﺑﺎﺷـﻴﻢ؟ ﻳﻜﯽ ﺍﺯ ﻧﺸـﺎﻧﻪ ﻫﺎی ﺍﻳﻦ ﻣﺆﻣﻨﺎﻥ ﻛﺎﻣـﻞ، ﺁﻥ ﺍﺳـﺖ ﻛـﻪ ﻧـﺰﺩ ﺁﻧﻬـﺎ ﺭﻋﺎﻳـﺖ ﺣﺮﻣﺖ ﻛﺴـﯽ ﻛﻪ ﺑـﻪ ﺍﻧﺪک ﺩﺭﺟـﻪ ﺍی ﺑﻪ ﺭﺷـﺘﻪ ﻭﻻﻳﺖ ﺍﻫﻞﺑﻴﺖ ﻣﺘﺼﻞ ﺍﺳـﺖ، ﺑﺮ ﭘـﻮﻝ ﻭ ﺛﺮﻭﺕ (ﺍﻳﻦ ﻣﻌﺸﻮﻕ ﺩﻧﻴﺎﺩﺍﺭﺍﻥ) ﺑﺮﺗﺮی ﺩﺍﺭﺩ. ﺧﻮﺷﺎ ﺑﻪ ﺣﺎﻝ ﺁﻧﺎﻥ.

ﻳﻜـﯽ ﺍﺯ ﻳـﺎﺭﺍﻥ ﺍﻣـﺎﻡ ﺻـﺎﺩﻕ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﮔﻮﻳﺪ: ﺑـﻪ ﺣﻀﺮﺕ ﺻـﺎﺩﻕ ﻋﺮﺿﻪ ﺩﺍﺷـﺘﻢ ﻛﻪ ﻣﯽ ﺗﺮﺳـﻢ ﻣـﺎ ﻣﺆﻣﻦ ﻧﺒﺎﺷـﻴﻢ. ﺁﻥ ﺣﻀﺮﺕ ﻓﺮﻣﻮﺩ: «ﭼﺮﺍ ﭼﻨﻴﻦ ﺑﺎﺷﺪ؟»

ﻋـﺮﺽ ﻛـﺮﺩﻡ: ﺯﻳـﺮﺍ ﻛﻪ ﻣـﺎ ﺩﺭ ﻣﻴـﺎﻥ ﺧـﻮﺩ ﻛﺴـﯽ ﺭﺍ ﻧﻤﯽ ﺑﻴﻨﻴﻢ ﻛـﻪ ﺑﺮﺍﺩﺭ ﺩﻳﻨﻴـﺶ ﻧـﺰﺩ ﺍﻭ ﺍﺯ ﺩﺭﻫﻢ ﻭ ﺩﻳﻨﺎﺭﺵ ﺍﺭﺯﺷـﻤﻨﺪﺗﺮ ﺑﺎﺷـﺪ، ﻭ ﻣﯽ ﺑﻴﻨﻴـﻢ ﺩﻳﻨـﺎﺭ ﻭ ﺩﺭﻫﻢ ﻧﺰﺩ ﻣﺎ ﻋﺰﻳﺰﺗﺮ ﺍﺳـﺖ ﺍﺯ ﺑـﺮﺍﺩﺭی ﻛﻪ ﻭﻻﻳﺖ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻴﻦ ﻣﺎ ﻭ ﺍﻭ ﭘﻴﻮﻧﺪ ﺑﺮﻗﺮﺍﺭ ﻛﺮﺩﻩ ﺍﺳـﺖ! ﺁﻥ ﺣﻀﺮﺕ ﻓﺮﻣﻮﺩ: «ﻧﻪ؛ ﺷـﻤﺎ ﻣﺆﻣﻨﻴﺪ، ﻭﻟﯽ ﺍﻳﻤﺎﻧﺘـﺎﻥ ﺑﻪ ﻛﻤﺎﻝ ﻧﺨﻮﺍﻫﺪ ﺭﺳـﻴﺪ ﺗﺎ ﺁﻧﮕﺎﻩ ﻛﻪ ﻗﺎﺋﻢ ﻣﺎ ﺑﻪ ﭘﺎﺧﻴﺰﺩ. ﺩﺭ ﺁﻥ ﻫﻨﮕﺎﻡ ﺍﺳـﺖ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﻋﻘﻞﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺟﻤﻊ ﺧﻮﺍﻫﺪ ﺳـﺎﺧﺖ ﻭ ﻣﺆﻣﻨﺎﻥ ﻛﺎﻣﻠﯽ ﺧﻮﺍﻫﻴﺪ ﺷﺪ. ﺩﺭ ﻋﻴـﻦ ﺣـﺎﻝ ﺍﮔﺮ ﺯﻣﺎﻧﯽ ﺯﻣﻴﻦ ﺑﻪ ﻛﻠﯽ ﺍﺯ ﻣﺆﻣﻨﺎﻥ ﻛﺎﻣﻞ ﺧﺎﻟﯽ ﮔﺮﺩﺩ، ﺩﺭ ﺁﻥ ﺻﻮﺭﺕ ﺧﺪﺍﻭﻧﺪ ﻣﺎ ﺭﺍ ﺑﻪ ﺳﻮی ﺧﻮﺩ ﺑﺎﻻ ﺧﻮﺍﻫﺪ ﺑﺮﺩ.»(69)
