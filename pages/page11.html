بازگشت

ﺍﺣﻴﺎی ﺍﻣﺮ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ

ﻳﻜﯽ ﺍﺯ ﺭﺍﻩ ﻫﺎی ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ، «ﺍﺣﻴﺎی ﺍﻣﺮ ﺍﻳﺸـﺎﻥ » ﺍﺳـﺖ. ﺍﺣﻴـﺎی ﺍﻣـﺮ ﺍﻫﻞﺑﻴﺖ ﺑـﻪ ﻣﻌﻨﯽ ﺯﻧﺪﻩ ﺳـﺎﺧﺘﻦ ﻣﻜﺘﺐ ﺍﻳﺸـﺎﻥ (ﻳﻌﻨﯽ ﻣﻜﺘﺐ ﺷـﻴﻌﻪ ) ﻣﯽ ﺑﺎﺷـﺪ. ﻣﻜﺘﺐ ﺷـﻴﻌﻪ ﺛﻤﺮﻩ ﻧﮕﺎﻫﯽ ﺧﺎﺹ ﻭ ﺧﺎﻟﺺ ﺍﺯ ﺍﻧﺤﺮﺍﻓﺎﺕ، ﺑﻪ ﺁﻣﻮﺯﻩ ﻫﺎی ﭘﻴﺎﻣﺒﺮ ﺍﻛﺮﻡ ﺍﺳـﺖ. ﺗﺸﻴﻊ ﭼﻴﺰی ﻧﻴﺴﺖ ﺟﺰ ﻫﻤﺎﻥ ﺍﺳﻼﻡ ﻧﺎﺏ ﻛﻪ ﺭﺳﻮﻝ ﺧﺪﺍ ﺍﺯ ﺟﺎﻧﺐ ﺣﻖ ﺑﺮﺍی ﺑﺸﺮ ﺑـﻪ ﺍﺭﻣﻐﺎﻥ ﺁﻭﺭﺩﻩ ﺍﺳـﺖ ﻭ ﺍﻫﻞﺑﻴـﺖ ﺩﺭ ﮔﺬﺭ ﺯﻣـﺎﻥ ﺁﻥ ﺭﺍ ﺍﺯ ﺗﺤﺮﻳﻒ ﺣﻔﺎﻇﺖ ﻛﺮﺩﻩ ﻭ ﺑﻪ ﺩﺳﺖ ﻣﺎ ﺳﭙﺮﺩﻩ ﺍﻧﺪ.

ﭘـﺲ ﺍﺯ ﺭﺣﻠـﺖ ﭘﻴﺎﻣﺒـﺮ ﺍﺳـﻼﻡ ﺻﻠـﯽ ﺍﻟﻠﻪ ﻋﻠﻴـﻪ ﻭ ﺁﻟـﻪ ، ﺟﺎﻣﻌﻪ ﺍﺳـﻼﻣﯽ ﺩﭼـﺎﺭ ﻓﺘﻨﻪ ﮔﺸـﺖ ﻭ ﻋﺪﻩ ﺍی ﺍﺯ ﻛﺴـﺎﻧﻴﻜﻪ ﺩﺭ ﻇﺎﻫـﺮ ﺍﺩﻋﺎی ﻣﺴـﻠﻤﺎﻧﯽ ﺩﺍﺷـﺘﻨﺪ، ﻭﺻﺎﻳﺎی ﭘﻴﺎﻣﺒﺮ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﺭﻫﺒـﺮی ﺟﺎﻣﻌﻪ ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻨـﺪ ﻭ ﺑـﻪ ﻧﺎﺣﻖ ﺑﺮ ﻣﺴـﻨﺪ ﺍﺩﺍﺭﻩ ﺍﻣﺖ ﺗﻜﻴﻪ ﺯﺩﻧﺪ. ﺍﻳـﻦ ﮔﺮﻭﻩ ﺑﺮﺍی ﻣﺨﻔﯽ ﻛﺮﺩﻥ ﻣﺎﻫﻴﺖ ﻏﺎﺻﺒﺎﻧﻪ ﺣﻜﻮﻣﺖ ﺧﻮﻳﺶ، ﻧﻘﻞ ﻭ ﻧﺸﺮ ﺍﺣﺎﺩﻳﺚ ﭘﻴﺎﻣﺒـﺮ ﺭﺍ ﺑﺮﺍی ﺳـﺎﻝﻫﺎ ﻣﻤﻨﻮﻉ ﻛﺮﺩﻧﺪ ﻭ ﺩﺭ ﺍﻳـﻦ ﺩﻭﺭﻩ ﺗﻨﻬﺎ ﺑﻪ ﻋﺪﻩ ﺍی ﺧـﺎﺹ ﻛﻪ ﻣﯽ ﺗﻮﺍﻥ ﺁﻥ ﻫـﺎ ﺭﺍ ﺭﻭﺣﺎﻧﻴﺎﻥ ﺩﺭﺑﺎﺭی ﻧﺎﻡ ﻧﻬـﺎﺩ، ﺍﺟﺎﺯﻩ ﻧﻘﻞ ﺍﺣﺎﺩﻳﺚ ﺩﺍﺩﻩ ﺷـﺪ. ﺍﻳﻦ ﺭﺍﻭﻳﺎﻥ ﺩﺭﺑﺎﺭی ﺑﺎ ﺳﻴﺎﺳـﺖ ﺳﺎﻧﺴﻮﺭ، ﻛﺘﻤﺎﻥ ﺣﻘﺎﻳـﻖ، ﺗﺤﺮﻳـﻒ ﺍﻟﻔﺎﻅ ﻭ ﺟﻌـﻞ ﺍﺣﺎﺩﻳﺚ ﺩﺭﻭﻏﻴـﻦ ﺩﺭ ﻣﺪﺡ ﺣﻜﺎﻡ ﺯﻣـﺎﻥ ، ﺍﺳـﻼﻡ ﺭﺍ ﻫﻤﭽـﻮﻥ ﭘﻮﺳـﺘﻴﻨﯽ ﻭﺍﺭﻭﻧﻪ ﺑـﻪ ﺗﻦ ﺟﺎﻣﻌـﻪ ﻛﺮﺩﻧﺪ. ﻣﻜﺘﺒـﯽ ﺭﺍ ﻛﻪ ﺍﻳﻦ ﻋﺎﻟﻤﺎﻥ ﺩﺭﺑـﺎﺭی ﭘﯽ ﺍﻓﻜﻨﺪﻧﺪ، ﻣﯽ ﺗﻮﺍﻥ ﻣﻜﺘﺐ ﺧﻠﻔﺎ (ﻳﻌﻨﯽ ﻣﺪﺭﺳﻪ ﺣﺎﻛﻤﺎﻥ ﻭ ﻗﺪﺭﺕ ﻣﺪﺍﺭﺍﻥ ) ﻧﺎﻣﻴﺪ.

ﺩﺭ ﻣﻘﺎﺑـﻞ ﺍﻳـﻦ ﺳـﻴﺮ ﺍﻧﺤﺮﺍﻓﯽ، ﺍﺯ ﻫﻤـﺎﻥ ﺁﻏﺎﺯ، ﺟﺮﻳـﺎﻥ ﺩﻳﮕﺮی ﻭﺟﻮﺩ ﺩﺍﺷـﺖ ﻛﻪ ﻫﻤﻮﺍﺭﻩ ﺣﻴﺎﺕ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺑﺴـﺘﺮ ﺗﺎﺭﻳﺦ ﺣﻔﻆ ﻛﺮﺩﻩ ﺍﺳـﺖ. ﺍﻳـﻦ ﺟﺮﻳﺎﻥ ﺗﻮﺳـﻂ ﺍﻫﻞ ﺑﻴﺖ ﭘـﺎک ﭘﻴﺎﻣﺒﺮ ﺭﻫﺒﺮی ﻣﯽ ﺷـﺪ ﻭ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﻳﺎ ﻣﻜﺘﺐ ﺗﺸﻴﻊ ﻧﺎﻡ ﺩﺍﺭﺩ.

ﺷـﻴﻌﻴﺎﻥ ﻳﻌﻨﯽ ﭘﻴﺮﻭﺍﻥ ﻣﻜﺘﺐ ﺗﺸـﻴﻊ ﺑﻨﺎ ﺑﺮ ﺗﻌﺎﻟﻴﻢ ﭘﻴﺎﻣﺒﺮ ﻣﻌﺘﻘﺪﻧﺪ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺑﺮﺍی ﺣﻔﻆ ﺍﺳﻼﻡ ﺍﺻﻴﻞ ﺩﺭ ﺍﻣﺘﺪﺍﺩ ﺗﺎﺭﻳﺦ، ﺩﻭﺍﺯﺩﻩ ﺍﻧﺴﺎﻥ ﻣﻌﺼـﻮﻡ ﻭ ﻭﺍﻻﻣﻘﺎﻡ ﺭﺍ ﺑﺮﮔﺰﻳﺪﻩ ﺍﺳـﺖ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﻳﻜﯽ ﭘﺲ ﺍﺯ ﺩﻳﮕﺮی ﻭﺍﺭﺙ ﻋﻠـﻮﻡ ﻗـﺮﺁﻥ ، ﺣﺎﻣـﻞ ﺗﻌﺎﻟﻴـﻢ ﭘﻴﺎﻣﺒـﺮ ﻭ ﺣﺎﻓﻆ ﻛﺘﺎﺏ ﻭ ﺳـﻨﺖ ﺳﺎﺧﺘﻪ ﺍﺳﺖ.

ﺷـﻴﻌﻴﺎﻥ ﻣﻌﺘﻘﺪﻧـﺪ ﻛـﻪ ﻫﻴﭽـﮕﺎﻩ ﺯﻣﻴـﻦ ﺍﺯ ﺍﻳﻦ ﻋﺎﻟﻤـﺎﻥ ﻣﻌﺼﻮﻡ ﺧﺎﻟـﯽ ﻧﺸـﺪﻩ ﻭ ﻧﻤﯽ ﺷـﻮﺩ ﻭ ﺁﻧﻬﺎ ﻫﻤﭽﻮﻥ ﺳﻠﺴـﻠﻪ ﺍی ﺑﻪ ﻫﻢ ﭘﻴﻮﺳـﺘﻪ ﺣﺎﻣﯽ ﺍﺳﻼﻡ ﻭ ﺣﺎﻓﻆ ﺍﺻﺎﻟﺖ ﺁﻥ ﺑﻮﺩﻩ ﻭ ﻫﺴﺘﻨﺪ.

ﺷـﻴﻌﻴﺎﻥ ﺍﻋﺘﻘـﺎﺩ ﺩﺍﺭﻧﺪ ﻛﻪ ﺍﻳﻦ ﺍﻓﺮﺍﺩ، ﺗﻮﺳـﻂ ﭘﻴﺎﻣﺒـﺮ ﺍﻛﺮﻡ ﺻﻠﯽ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ ﻣﻌﻴﻦ ﻭ ﻣﻌﺮﻓﯽ ﺷـﺪﻩ ﺍﻧﺪ. ﺍﻭﻟﻴﻦ ﺍﻳﺸـﺎﻥ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻋﻠﯽ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻭ ﺁﺧﺮﻳﻦ ﺍﻳﺸﺎﻥ ﺍﻣﺎﻡ ﻣﻬﺪی ﻋﺠﻞ ﺍﻟﻠﻪ ﺗﻌﺎﻟﯽ ﻓﺮﺟﻪ ﺍﻟﺸﺮﻳﻒ ﻣﯽ ﺑﺎﺷﺪ.

ﺷﻴﻌﻴﺎﻥ ﻣﻌﺘﻘﺪﻧﺪ ﻫﻤﮕﯽ ﺍﻳﻦ ﺑﺰﺭﮔﻮﺍﺭﺍﻥ ﺑﺮﺍی ﺭﻭﺷﻦ ﻧﮕﻪ ﺩﺍﺷﺘﻦ ﭼـﺮﺍﻍ ﺁﺋﻴﻦ، ﭼﻨﺎﻧﻜﻪ ﺷﺎﻳﺴـﺘﻪ ﺑـﻮﺩ ﺗﻼﺵ ﻛﺮﺩﻩ ﻭ ﺑﺎ ﻧﺸـﺮ ﻋﻠﻮﻡ ﻧﺒﻮی ﻭ ﺗﺮﺑﻴـﺖ ﻋﺎﻟﻤﺎﻧـﯽ ﻭﺍﻻﻣﻘـﺎﻡ ﻣﻜﺘﺒﯽ ﺭﺍ ﭘﯽ ﺭﻳﺨﺘﻪ ﺍﻧﺪ ﻛـﻪ ﻫﻴﭻ ﮔﺎﻩ ﺍﺯ ﺻﻔﺤﻪ ﺗﺎﺭﻳﺦ ﻣﺤﻮ ﻧﺸـﺪﻩ ﻭ ﻧﺨﻮﺍﻫﺪ ﺷـﺪ. ﺑﺪﻳﻦ ﺭﻭی ﺗﺸـﻴﻊ ﭼﻴﺰی ﺟﺰ ﻫﻤﺎﻥ ﺍﺳﻼﻡ ﭘﻴﺎﻣﺒﺮ ﻧﻴﺴﺖ.

ﺑﺎﺭ ﺣﻔﻆ ﻭ ﺣﺮﺍﺳﺖ ﺍﺯ ﺍﺳﻼﻡ ﺷﻴﻌﯽ ﺭﺍ ﺩﺭ ﻫﺮ ﺯﻣﺎﻥ ، ﺩﺭ ﺩﺭﺟﻪ ی ﺍﻭﻝ ﺍﻣـﺎﻡ ﺯﻧـﺪﻩ ی ﺁﻥ ﺯﻣﺎﻥ ﺑﻪ ﺩﻭﺵ ﻣﯽ ﻛﺸـﺪ. ﺍﻭﺳـﺖ ﻛـﻪ ﺣﻘﻴﻘﺖ ﺍﺳـﻼﻡ ﺭﺍ ﺑﺮﺍی ﭘﻴﺮﻭﺍﻥ ﺧﻮﻳﺶ ﺑﻪ ﻃﻮﺭ ﻣﺴﺘﻘﻴﻢ ﻳﺎ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﺁﺷﻜﺎﺭ ﻣﯽ ﺳﺎﺯﺩ ﻭ ﺑﺎﻃﻞ ﺑﻮﺩﻥ ﺑﺪﻋﺖﻫﺎ ﺭﺍ ﺑﺮﺍی ﻛﺴﺎﻧﯽ ﻛﻪ ﺑﺎ ﺍﻭ ﭘﻴﻮﻧﺪ ﺩﺍﺭﻧﺪ، ﺑﺮﻣﻼ ﻣﯽ ﻛﻨﺪ. (ﺗﺄﻛﻴﺪ ﻣﯽ ﻛﻨﻢ! ﺑﺮﺍی ﻛﺴـﺎﻧﯽ ﻛﻪ ﺑﺎ ﺍﻭ ﭘﻴﻮﻧﺪ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﺍﻭ ﻣﺘﻮﺳﻠﻨﺪ! ﻭ ﺗﻨﻬﺎ ﻧﺎﻡ ﺷﻴﻌﻪ ﺭﺍ ﺑﺮ ﺧﻮﺩ ﻧﺪﺍﺭﻧﺪ!)

ﺩﺭ ﺭﺗﺒـﻪ ﺑﻌﺪ ﺍﺯ ﺍﻣـﺎﻡ ﺯﻧﺪﻩ ﺯﻣﺎﻥ ، ﺍﻳﻦ ﺭﺍﻭﻳـﺎﻥ ﺍﺣﺎﺩﻳﺚ ﺍﻫﻞﺑﻴﺖ ﻭ ﻋﺎﻟﻤﺎﻥ ﺩﻳﻦ ﻫﺴـﺘﻨﺪ ﻛﻪ ﺑﻪ ﭘﺸﺘﻴﺒﺎﻧﯽ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺧﻮﻳﺶ ﺑﺎﺭ ﺣﺮﺍﺳﺖ ﻭ ﺣﻔﺎﻇـﺖ ﺍﺯ ﻣﻜﺘـﺐ ﺭﺍ ﺑـﻪ ﺩﻭﺵ ﺩﺍﺭﻧﺪ. ﻋﺎﻟﻤﺎﻥ ﭘﺮﻫﻴـﺰﻛﺎﺭی ﻛﻪ ﺑﺎ ﺭﻭﺍﻳﺎﺕ ﺍﻫﻞﺑﻴﺖ ﻭ ﻣﻜﺘﺐ ﺍﻳﺸـﺎﻥ ﺁﺷـﻨﺎﻳﻨﺪ، ﻳﺎﻭﺭﺍﻥ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﺯﻧﺪﻩ ﻧﮕﻪ ﺩﺍﺷﺘﻦ ﺗﺸﻴﻊﺍﻧﺪ.

ﭘﺲ ﺍﺯ ﻋﺎﻟﻤﺎﻥ ﻭ ﺣﺎﻓﻈﺎﻥ ﺍﺣﺎﺩﻳﺚ، ﺷﻴﻌﻴﺎﻧﯽ ﻛﻪ ﺑﺎ ﻋﻠﻤﺎ ﺭﻓﺖ ﻭ ﺁﻣﺪ ﺩﺍﺭﻧﺪ ﻭ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﺭﺍ ﻣﯽ ﺁﻣﻮﺯﻧﺪ ﻭ ﻳﻜﺪﻳﮕﺮ ﺭﺍ ﺑﻪ ﺁﻥ ﺳﻔﺎﺭﺵ ﻣﯽ ﻛﻨﻨﺪ، ﻛﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﺭﺍ ﺯﻧﺪﻩ ﻧﮕﻪ ﻣﯽ ﺩﺍﺭﻧﺪ.

ﺍﺯ ﺁﻥ ﺟﺎ ﻛﻪ ﻣﺴـﺌﻮﻟﻴﺖ ﺯﻧﺪﻩ ﻧﮕﻪ ﺩﺍﺷـﺘﻦ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﺩﺭ ﺍﺻﻞ ﺑﺮﻋﻬﺪﻩ ﺍﻣﺎﻡ ﺯﻧﺪﻩ ﻋﺼﺮ ﺍﺳـﺖ، ﺗﻤﺎﻡ ﻛﺴﺎﻧﯽ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ (ﻳﻌﻨﯽ ﺯﻧﺪﻩ ﺳـﺎﺧﺘﻦ ﻭ ﺭﻭﻧﻖ ﺑﺨﺸـﻴﺪﻥ ﺑـﻪ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ) ﺗﻼﺵ ﻭ ﻓﺪﺍﻛﺎﺭی ﻣﯽ ﻛﻨﻨﺪ (ﺍﻋﻢ ﺍﺯ ﺩﺍﻧﺸﻮﺭﺍﻥ ﻭ ﺩﺍﻧﺶﭘﮋﻭﻫﺎﻥ ﻣﻜﺘﺐ ﺷﻴﻌﻪ ) ﻳﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ.

ﺍﺋﻤﻪ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﺍﺯ ﻣﻜﺘﺐ ﻧـﺎﺏ ﺧﻮﻳﺶ ﺑﺎ ﺍﺻﻄﻼﺡ «ﺍﻣﺮﻧﺎ» (ﻳﻌﻨﯽ ﺍﻣﺮ ﻣﺎ) ﻳﺎﺩ ﻛﺮﺩﻩ ﺍﻧﺪ ﻭ ﺑﻪ ﺑﻴﺎﻥ ﻫﺎی ﻣﺨﺘﻠﻒ ﻓﺮﻣﻮﺩﻩ ﺍﻧﺪ ﻫﺮﻛﺴﯽ ﻛـﻪ ﺍﻣـﺮ ﻣﺎ (ﻳﻌﻨﯽ ﻣﻜﺘـﺐ ﻣﺎ) ﺭﺍ ﺯﻧـﺪﻩ ﻛﻨﺪ، ﻣﻮﺭﺩ ﺭﺣﻤـﺖ ﻭ ﻋﻨﺎﻳﺖ ﺧﺎﺹ ﭘﺮﻭﺭﺩﮔﺎﺭ ﺍﺳﺖ.

ﮔﻔﺘﻴﻢ ﻛﻪ ﺯﻧﺪﻩ ﻧﮕﻪ ﺩﺍﺷﺘﻦ ﻣﻜﺘﺐ ﺩﺭ ﺍﺻﻞ ﺑﺮ ﻋﻬﺪﻩ ﺍﻣﺎﻡ ﻫﺮ ﺯﻣﺎﻥ ﺍﺳﺖ ﻭ ﺩﺭ ﻫﻤﻴﻦ ﺭﺍﺳﺘﺎ ﻋﺎﻟﻤﯽ ﻫﻢ ﻛﻪ ﻋﻠﻢ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺍﻫﻞﺑﻴﺖ ﮔﺮﻓﺘﻪ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﻳﮕﺮ ﺷﻴﻌﻴﺎﻥ ﺍﻧﺘﻘﺎﻝ ﻣﯽ ﺩﻫﺪ، ﺍﺯ ﻛﺴﺎﻧﯽ ﺍﺳﺖ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺭﺍ ﺩﺭ ﺯﻧﺪﻩ ﻧﮕﻬﺪﺍﺷﺘﻦ ﻣﻜﺘﺐ ﺗﺸﻴﻊ ﻳﺎﺭی ﻣﯽ ﻛﻨﺪ. ﻫﻤﭽﻨﻴﻦ ﺷﻴﻌﻴﺎﻧﯽ ﻛـﻪ ﺑﺎ ﻋﺎﻟﻤﺎﻥ ﭘﺮﻫﻴﺰﻛﺎﺭ ﺭﻓﺖ ﻭ ﺁﻣﺪ ﻣﯽ ﻛﻨﻨـﺪ ﻭ ﺗﻌﺎﻟﻴﻢ ﺍﻫﻞﺑﻴﺖ ﺭﺍ ﺍﺯ ﺁﻧﺎﻥ ﻣﯽ ﺁﻣﻮﺯﻧﺪ، ﻳﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭ ﺍﺣﻴﺎی ﻣﻜﺘﺐ ﺗﺸـﻴﻊ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺭﻭﻧﺪ.

ﺍﻳﻨﺎﻥ ﺑﺎ ﺗﺸـﻜﻴﻞ ﺟﻠﺴـﺎﺕ ﻭ ﺩﻋﻮﺕ ﺍﺯ ﻋﺎﻟﻤﺎﻥ ﻭ ﺩﺭﺧﻮﺍﺳﺖ ﺍﺯ ﺁﻧـﺎﻥ ﺑـﺮﺍی ﻧﻘﻞ ﺭﻭﺍﻳﺎﺕ ﺍﻫﻞﺑﻴـﺖ ﻭ ﻧﺘﻴﺠﻪ ﮔﻴﺮی ﻫﺎی ﺩﺭﺳـﺖ ﺍﺯ ﺁﻥ ﺭﻭﺍﻳـﺎﺕ، ﺩﺭ ﺍﺣﻴﺎی ﺍﻣﺮ ﺍﻫﻞﺑﻴﺖ ﻧﻘﺶ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﻧﻮﺑﻪ ﺧﻮﺩ ﺍﺯ ﻳﺎﺭﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ.

ﭘﺲ ﺭﻭﺷـﻦ ﮔﺸـﺖ ﻛـﻪ ﻳﻜـﯽ ﺍﺯ ﻣﻬﻢﺗﺮﻳـﻦ ﺭﺍﻩ ﻫﺎی ﻳـﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ، ﺯﻧﺪﻩ ﺳـﺎﺧﺘﻦ ﻣﻜﺘﺐ ﺍﻳﺸـﺎﻥ ﺍﺳـﺖ ﻛﻪ ﺍﻳﻦ ﻛﺎﺭ ﺍﺯ ﻳﮏ ﺳـﻮ ﺑﺎ ﺗـﻼﺵ ﻧﺎﻗـﻼﻥ ﺍﺣﺎﺩﻳﺚ ﻭ ﻋﺎﻟﻤﺎﻥ ﻭ ﻣﺒﻠﻐﺎﻥ ﺧﺒﺮﻩ ﻭ ﺍﺯ ﺳـﻮی ﺩﻳﮕﺮ ﺑﺎ ﻣﺮﺍﺟﻌﻪ ﻣﺎ ﺑﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﻓﺮﺍﺩ ﻭ ﺗﺸﻜﻴﻞ ﺟﻠﺴﺎﺕ ﻭ ﻧﺸﺴﺖﻫﺎی ﻣﺪﺍﻭﻡ ﺟﻬﺖ ﺁﻣﻮﺧﺘﻦ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﺷﺪﻧﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺍﺯ ﺧﺪﺍﻭﻧـﺪ ﻣﯽ ﺧﻮﺍﻫﻴـﻢ ﻛﻪ ﺑﻪ ﻋﺎﻟﻤﺎﻥ ﻣﺎ ﺗﻮﻓﻴـﻖ ﺩﻫﺪ ﺗﺎ ﺑﻴﺶ ﺍﺯ ﭘﻴﺶ ﺑﻪ ﻧﻘﻞ ﺭﻫﻨﻤﻮﺩﻫﺎی ﺍﻫﻞﺑﻴﺖ ﻭ ﺗﻌﻠﻴﻢ ﻋﻠﻮﻡ ﺍﻳﺸﺎﻥ ﺑﭙﺮﺩﺍﺯﻧﺪ ﻭ ﺑﻪ ﻣـﺎ ﻧﻴﺰ ﺗﻮﻓﻴﻖ ﺩﻫﺪ ﻛﻪ ﺩﺭ ﺑﺮﻗﺮﺍﺭی ﺍﺭﺗﺒﺎﻁ ﺑـﺎ ﻋﻠﻤﺎ ﺑﻪ ﻫﺪﻑ ﻳﺎﺩﮔﻴﺮی ﺳـﺨﻨﺎﻥ ﺍﻫﻞﺑﻴـﺖ، ﺩﺭک ﻣﻌﺎﻧـﯽ ﺁﻥ ﻛﻠﻤـﺎﺕ ﻭ ﺭﻧـﮓ ﮔﺮﻓﺘـﻦ ﺍﺯ ﺁﻥ ﺑﻴﺎﻧﺎﺕ ﻛﻮﺷﺎﺗﺮ ﺑﺎﺷﻴﻢ.

ﺍﺯ ﺧﺪﺍﻭﻧﺪ ﻣﯽ ﺧﻮﺍﻫﻴﻢ ﻛﻪ ﺑﻪ ﻣﺎ ﺗﻮﻓﻴﻖ ﺩﻫﺪ ﺗﺎ ﺩﺭ ﺷـﻤﺎﺭ ﻛﺴـﺎﻧﯽ ﺩﺭﺁﻳﻴﻢ ﻛﻪ ﺑﻪ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﺭﻭﻧﻖ ﻣﯽ ﺩﻫﻨﺪ ﻭ ﺑﺎ ﭘﺬﻳﺮﺵ ﻭ ﮔﺴـﺘﺮﺵ ﺁﻥ ، ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﻳﺎﺭی ﻣﯽ ﻛﻨﻨﺪ.

ﺩﺭ ﺍﻳﻨﺠـﺎ ﺳـﻪ ﮔﻔﺘـﺎﺭ ﺍﺯ ﻛﻠﻤـﺎﺕ ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﺭﺍ ﺩﺭ ﺍﻫﻤﻴﺖ ﺗﺸـﻜﻴﻞ ﻧﺸﺴﺖﻫﺎی ﻋﻠﻤﯽ ﻭ ﻧﻘﻞ ﺭﻭﺍﻳﺎﺕ ﻭ ﻋﻠﻮﻡ ﺍﻫﻞ ﺑﻴﺖ ﻣﯽ ﺁﻭﺭﻳـﻢ ﻭ ﺍﺯ ﻧﻮﺭ ﺍﻳﻦ ﻛﻠﻤـﺎﺕ ﺩﺭ ﺭﺍﻩ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻭ ﺍﺣﻴﺎی ﺍﻣﺮ ﺍﻳﺸﺎﻥ ﭘﺮﺗﻮ ﻣﯽ ﮔﻴﺮﻳﻢ.

ﺭﻭﺍﻳﺖ ﺍﻭﻝ:

ﺍﻣﺎﻡ ﺭﺿﺎ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩﻧﺪ: «ﻫﺮﻛﺲ ﺩﺭ ﻣﺠﻠﺴـﯽ ﺑﻨﺸﻴﻨﺪ ﻛﻪ ﺍﻣﺮ ﻣﺎ (ﻳﻌﻨﯽ ﻣﻜﺘﺐ ﻣﺎ) ﺩﺭ ﺁﻥ ﺯﻧﺪﻩ ﻣﯽ ﮔﺮﺩﺩ، ﺩﺭ ﺁﻥ ﺭﻭﺯ ﻛﻪ ﺩﻝﻫﺎ ﻫﻤﻪ ﻣﯽ ﻣﻴﺮﻧﺪ ﺩﻟﻤﺮﺩﻩ ﻧﺨﻮﺍﻫﺪ ﺷﺪ.»(25)

ﺭﻭﺍﻳﺖ ﺩﻭﻡ:

ﺍﻣـﺎﻡ ﺻـﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﻳﻜـﯽ ﺍﺯ ﻳﺎﺭﺍﻥ ﺧﻮﺩ ﺑـﻪ ﻧﺎﻡ ﻓﻀﻴﻞ ﻓﺮﻣﻮﺩﻧﺪ: «ﺁﻳﺎ ﺷـﻤﺎ ﺟﻠﺴـﻪ ﺗﺸـﻜﻴﻞ ﻣﯽ ﺩﻫﻴﺪ ﻭ ﮔﻔﺘﻪ ﻫﺎی ﻣﺎ ﺭﺍ ﺑﺮﺍی ﻳﻜﺪﻳﮕـﺮ ﻧﻘﻞ ﻣﯽ ﻛﻨﻴﺪ؟» ﻓﻀﻴﻞ ﻋﺮﺿﻪ ﺩﺍﺷـﺖ: ﺁﺭی ﻓﺪﺍﻳﺖ ﺷـﻮﻡ. ﺍﻣـﺎﻡ ﺻـﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣـﻮﺩ: «ﻫﻤﺎﻧﺎ ﻣـﻦ ﺍﻳﻦ ﭼﻨﻴﻦ ﺟﻠﺴـﻪ ﺍی ﺭﺍ ﺩﻭﺳـﺖ ﻣـﯽ ﺩﺍﺭﻡ. ﭘﺲ (ﺑﺎ ﺑﺮﻗﺮﺍﺭی ﺍﻳﻦ ﺟﻠﺴـﺎﺕ) ﺃﻣـﺮ ﻣﺎ (ﻳﻌﻨﯽ ﻣﻜﺘﺐ ﻣﺎ) ﺭﺍ ﺯﻧﺪﻩ ﻛﻨﻴﺪ! ﺍی ﻓﻀﻴﻞ ﺧﺪﺍﻭﻧﺪ ﺭﺣﻤﺖ ﻛﻨﺪ ﻛﺴﯽ ﺭﺍ ﻛﻪ ﺍﻣﺮ ﻣﺎ ﺭﺍ ﺯﻧﺪﻩ ﻣﯽ ﺳﺎﺯﺩ.»(26)

ﺭﻭﺍﻳﺖ ﺳﻮﻡ:

ﻋﺒﺎﺩ ﺑﻦ ﻛﺜﻴﺮ ﮔﻮﻳﺪ ﺑﻪ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻋﺮﺿﻪ ﺩﺍﺷـﺘﻢ: «ﻣـﻦ ﺍﺯ ﻛﻨﺎﺭ ﻗﺼﻪ ﮔﻮﻳﯽ(27) ﮔﺬﺷـﺘﻢ ﻛﻪ ﺩﺭ ﺣـﺎﻝ ﻗﺼﻪ ﮔﻔﺘﻦ ﭼﻨﻴﻦ ﻣﯽ ﮔﻔـﺖ: ﺍﻳﻦ ﻣﺠﻠﺴـﯽ ﺍﺳـﺖ ﻛـﻪ ﻫﺮﻛـﺲ ﺩﺭ ﺁﻥ ﺑﻨﺸـﻴﻨﺪ ﺑﺪﺑﺨﺖ ﻧﻤﯽ ﺷـﻮﺩ.» ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩ: «ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ! ﺑﻪ ﻫﻴﭻ ﻭﺟـﻪ ! ﻣﺠﺎﻟـﺲ ﺁﻥ ﻫﺎ ﺩﺭ ﻣﺴـﻴﺮ ﺧﻄـﺎ ﺍﺳـﺖ. ﺧﺪﺍﻭﻧﺪ ﺑﻪ ﻏﻴـﺮ ﺍﺯ ﺁﻥ ﻓﺮﺷﺘﮕﺎﻥ ﮔﺮﺍﻣﯽ ﻛﻪ ﺍﻋﻤﺎﻝ ﺑﻨﺪﮔﺎﻥ ﺭﺍ ﻣﯽ ﻧﻮﻳﺴﻨﺪ، ﻓﺮﺷﺘﮕﺎﻧﯽ ﺩﺍﺭﺩ ﻛﻪ ﮔﺸﺖ ﻣﯽ ﺯﻧﻨﺪ، ﭘﺲ ﻫﺮﮔﺎﻩ ﺑﺮ ﺟﻤﻌﯽ ﺑﮕﺬﺭﻧﺪ ﻛﻪ ﻣﺤﻤﺪ ﻭ ﺁﻝ ﻣﺤﻤﺪ (ﺹ) ﺭﺍ ﻳﺎﺩ ﻣﯽ ﻛﻨﻨﺪ، ﺑﻪ ﻳﻜﺪﻳﮕﺮ ﮔﻮﻳﻨﺪ: ﺗﻮﻗﻒ ﻛﻨﻴﺪ! ﭘﺲ ﻣﯽ ﻧﺸﻴﻨﻨﺪ ﻭ ﺧﻮﺩ ﺍﺯ ﺁﻥ ﻣﺠﻠﺲ ﺑﻬﺮﻩ ﻣﻨﺪ ﻣﯽ ﺷـﻮﻧﺪ، ﻭ ﭼﻮﻥ ﺍﺯ ﺁﻧﺎﻥ ﺟﺪﺍ ﮔﺮﺩﻧﺪ ﺑﻴﻤﺎﺭﺍﻥ ﺁﻥ ﻣﺠﻠﺲ ﺭﺍ ﻋﻴﺎﺩﺕ ﻛﻨﻨﺪ ﻭ ﺩﺭ ﻣﺮﺍﺳـﻢ ﻣﺮﺩﮔﺎﻧﺸﺎﻥ ﺷﺮﻛﺖ ﻧﻤﺎﻳﻨﺪ ﻭ ﺍﺯ ﻏﺎﺋﺒﻴﻦ ﺁﻧﺎﻥ ﺗﻔﻘﺪ ﻛﻨﻨﺪ. ﭼﻨﻴﻦ ﺟﻠﺴـﻪ ﺍی، ﻣﺠﻠﺴﯽ ﻫﺴﺖ ﻛﻪ ﻫﺮﻛﺲ ﺩﺭ ﺁﻥ ﺑﻨﺸﻴﻨﺪ، ﺗﻴﺮﻩ ﺑﺨﺖ ﻧﺨﻮﺍﻫﺪ ﮔﺮﺩﻳﺪ.»(28)

ﺧﺪﺍﻳﺎ! ﻣﺎ ﺭﺍ ﺍﺯ ﻛﺴﺎﻧﯽ ﻗﺮﺍﺭ ﺩﻩ ﻛﻪ ﺑﺎ ﺗﻼﺵ ﺩﺭ ﺟﻬﺖ ﺁﻣﻮﺧﺘﻦ ﻭ ﺁﻣﻮﺯﺵ ﻣﻜﺘﺐ ﺍﻫﻞﺑﻴﺖ ﺍﻣﺎﻡ ﺧﻮﻳﺶ ﺭﺍ ﻳﺎﺭی ﻣﯽ ﻛﻨﻨﺪ.
