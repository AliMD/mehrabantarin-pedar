بازگشت

ﺑﺸﺎﺭﺕ ﺑﻪ ﻳﺎﻭﺭﺍﻥ ﺧﺪﺍ

ﺧﺪﺍﻭﻧـﺪ ﻣـﺎ ﻛـﻪ ﻋﺰﻳـﺰ ﻭ ﮔﺮﺍﻣﯽ ﺍﺳـﺖ، ﺑﺎﺭﻫـﺎ ﺩﺭ ﻛﻼﻡ ﺧـﻮﺩ ﻣﺎ ﺭﺍ ﻓﺮﺍﺧﻮﺍﻧـﺪﻩ ﺗـﺎ ﺍﻭ ﺭﺍ ﻳﺎﺭی ﻛﻨﻴﻢ ﻭ ﺑﻪ ﻣﺎ ﺑﺸـﺎﺭﺕ ﺩﺍﺩﻩ ﻛـﻪ ﺍﮔﺮ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﺑﻴﺎﻳﻴـﻢ، ﺍﻭ ﺑﻪ ﻳﺎﺭی ﻣﺎ ﺧﻮﺍﻫﺪ ﺁﻣﺪ. ﺍﻭ ﻓﺮﻣﻮﺩﻩ ﺍﺳـﺖ: «ﺧﺪﺍﻭﻧﺪ ﺑﻪ ﻳﻘﻴﻦ، ﻛﺴـﯽ ﺭﺍ ﻛﻪ ﻳﺎﺭی ﺍﺵ ﺑﻜﻨﺪ ﻳﺎﺭی ﻣﯽ ﻛﻨﺪ. ﺭﺍﺳـﺘﯽ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺗﻮﺍﻧﺎ ﻭ ﭘﻴﺮﻭﺯ ﺍﺳـﺖ.»(70) ﻭ ﻓﺮﻣﻮﺩﻩ ﺍﺳـﺖ: «ﺍﮔـﺮ ﺧﺪﺍﻭﻧﺪ ﺭﺍ ﻳﺎﺭی ﻛﻨﻴـﺪ ﺍﻭ ﺷـﻤﺎ ﺭﺍ ﻳـﺎﺭی ﺧﻮﺍﻫﺪ ﻛـﺮﺩ ﻭ ﮔﺎﻡﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺛﺎﺑﺖ ﻭ ﺍﺳـﺘﻮﺍﺭ ﻣﯽ ﺳﺎﺯﺩ.»(71)

ﻫﻤـﻪ ﻣﯽ ﺩﺍﻧﻴﻢ ﻛﻪ ﺧﺪﺍ ﺑﯽ ﻧﻴﺎﺯ ﺍﺳـﺖ ﻭ ﺑﻪ ﺧـﻮﺩی ﺧﻮﺩ ﺣﺎﺟﺘﯽ ﻧﺪﺍﺭﺩ ﺗﺎ ﻣﺎ ﺑﺮﺍی ﺭﻓﻊ ﺣﺎﺟﺖ ﺍﻭ ﺑﻪ ﺍﻭ ﻳﺎﺭی ﺭﺳـﺎﻧﻴـﻢ. ﺑﻪ ﻋﻼﻭﻩ ﻫﻤﻪ ﻧﻴﺮﻭﻫـﺎی ﻣﺎ ﺭﺍ ﺍﻭ ﺑﻪ ﻣﺎ ﺩﺍﺩﻩ ﺍﺳـﺖ ﻭ ﻣﯽ ﺩﻫﺪ، ﭘـﺲ ﭼﻪ ﻣﻌﻨﺎ ﺩﺍﺭﺩ ﻛﻪ ﺑﺎ ﻧﻴﺮﻭی ﺧﻮﺩ ﺍﻭ ﺑﻪ ﺍﻭ ﻳﺎﺭی ﺑﺮﺳـﺎﻧﻴﻢ؟ ﺍﻭ ﺧﻮﺩ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﺷﻤﺎﻳﻴﺪ ﻛـﻪ ﻫﻤﻪ ﺑـﻪ ﺧﺪﺍﻭﻧﺪ ﻣﺤﺘﺎﺝ ﻭ ﻓﻘﻴﺮﻳﺪ ﻭ ﺧﺪﺍﻭﻧﺪ ﺍﺳـﺖ ﻛـﻪ ﺑﯽ ﻧﻴﺎﺯ ﻭ ﺳﺘﻮﺩﻩ ﺍﺳﺖ.»(72)

ﺣﻘﻴﻘﺖ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺍﺯ ﻣﻴﺎﻥ ﺑﻨﺪﮔﺎﻥ ﺧﻮﺩ ﻛﺴﺎﻧﯽ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﻤﺎﻳﻨﺪﻩ ﺧﻮﻳﺶ ﺑﺮﮔﺰﻳﺪﻩ ﺍﺳﺖ. ﺍﻳﻦ ﺑﺮﮔﺰﻳﺪﮔﺎﻥ، ﺧﺎﺿﻊﺗﺮﻳﻦ ﺑﻨﺪﮔﺎﻥﺍﻧﺪ ﻭ ﺑﻪ ﺣﻖ ﺳـﺰﺍﻭﺍﺭ ﺗﺠﻠﻴﻞ ﻭ ﺗﻌﻈﻴﻢ ﻣﯽ ﺑﺎﺷـﻨﺪ. ﺧﺪﺍﻭﻧﺪ ﺁﻧﻬﺎ ﺭﺍ ﺧﻠﻴﻔـﻪ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺍﺳـﺖ. ﺁﻧﻬﺎ ﻣﻌﺪﻥ ﻋﻠـﻢ ﻭ ﺍﺭﺍﺩﻩ ﺧﺪﺍﻭﻧﺪﻧﺪ. ﺧﺪﺍﻭﻧﺪ ﺭﺿﺎﻳﺖ ﺁﻧﻬﺎ ﺭﺍ ﺭﺿﺎﻳﺖ ﺧﻮﺩ ﺩﺍﻧﺴـﺘﻪ ﻭ ﺧﺸـﻢ ﺁﻧﻬﺎ ﺭﺍ ﺧﺸﻢ ﺧﻮﺩ. ﻧﺰﺩﻳﮏ ﺷﺪﻥ ﺑﻪ ﺁﻧﻬﺎ ﺭﺍ ﻧﺰﺩﻳﮏ ﺷﺪﻥ ﺑﻪ ﺧﻮﺩ ﻭ ﺩﻭﺭ ﺷﺪﻥ ﺍﺯ ﺁﻧﻬﺎ ﺭﺍ ﺩﻭﺭ ﺷـﺪﻥ ﺍﺯ ﺧﻮﺩ ﺩﺍﻧﺴﺘﻪ ﺍﺳـﺖ. ﺧﺪﺍﻭﻧﺪ ﻋﺰﻭﺟﻞ ﺩﺷﻤﻨﯽ ﺑﺎ ﺁﻧﻬﺎ ﺭﺍ ﺩﺷـﻤﻨﯽ ﺑﺎ ﺧﻮﺩﺵ ﻭ ﺩﻭﺳـﺘﯽ ﺑﺎ ﺁﻧﻬﺎ ﺭﺍ ﺩﻭﺳﺘﯽ ﺑﺎ ﺧﻮﺩﺵ ﺧﻮﺍﻧﺪﻩ ﺍﺳﺖ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺗﺮﺗﻴﺐ ﻳﺎﺭی ﺁﻧﻬﺎ ﺭﺍ، ﻳﺎﺭی ﺑﻪ ﺧﻮﺩ ﺑﺮﺷﻤﺮﺩﻩ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳـﻦ ﺍﺻﻮﻝ ﻫﺮﻛـﺲ ﺑﺮﮔﺰﻳﺪﮔﺎﻥ ﺧﺪﺍ ﺭﺍ ﻳﺎﺭی ﻛﻨﺪ ﮔﻮﻳﯽ ﺧﺪﺍ ﺭﺍ ﻳﺎﺭی ﻛﺮﺩﻩ ﺍﺳﺖ ﻭ ﺧﺪﺍ ﻧﻴﺰ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﻣﯽ ﺁﻳﺪ.

ﺍﻳﻦ ﺑﻨﺪﮔﺎﻥ ﺑﺮﮔﺰﻳﺪﻩ، ﭘﻴﺎﻣﺒﺮﺍﻥ ﻭ ﺟﺎﻧﺸـﻴﻨﺎﻥ ﺁﻧﻬﺎ ﻫﺴﺘﻨﺪ ﻭ ﻳﺎﺭی ﺁﻧـﺎﻥ ﻳـﺎﺭی ﺧﺪﺍﺳـﺖ. ﺍﻣـﺮﻭﺯ ﺁﻥ ﻛﺴـﯽ ﻛـﻪ ﺍﺯ ﻣﻴـﺎﻥ ﺍﻳﻦ ﺑـﺰﺭﮔﺎﻥ، ﺑﺮﮔﺰﻳـﺪﻩ ﺍﻟﻬـﯽ ﺩﺭ ﻣﻴﺎﻥ ﻣﺎﺳـﺖ، ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻣﺎ ﻣﯽ ﺑﺎﺷـﺪ ﻭ ﺧﺪﺍﻭﻧﺪ ﻳﺎﺭی ﺍﻭ ﺭﺍ ﺩﺭ ﺣﻜﻢ ﻳﺎﺭی ﺧﻮﺩ ﺩﺍﻧﺴـﺘﻪ ﺍﺳـﺖ. ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻦ ﻣﺒﺎﻧﯽ ﺭﻭﺷـﻦ ﻣﯽ ﺷـﻮﺩ ﻛﻪ ﺍﮔﺮ ﺧﺪﺍ ﻣﺎ ﺭﺍ ﺩﻋﻮﺕ ﻛﺮﺩﻩ ﺗﺎ ﺑـﻪ ﻳﺎﺭی ﺍﻭ ﺑﻴﺎﻳﻴﻢ ﺩﺭ ﺣﻘﻴﻘﺖ ﻣﺎ ﺭﺍ ﺑﻪ ﻳﺎﺭی ﻧﻤﺎﻳﻨﺪﻩ ﻭ ﺧﻠﻴﻔﻪ ﺧﻮﺩ ﻳﻌﻨﯽ ﺍﻣﺎﻡ ﺯﻧﺪﻩ ﺯﻣﺎﻥ ﻓﺮﺍﺧﻮﺍﻧﺪﻩ ﺍﺳﺖ. ﺷﺎﻫﺪ ﺑﺮ ﺩﺭﺳﺘﯽ ﺍﻳﻦ ﺍﻣﺮ ﺭﻭﺍﻳﺎﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﺍﻳﻦﺑﺎﺭﻩ ﻧﻘﻞ ﺷـﺪﻩ ﺍﺳـﺖ. ﺑﻪ ﻋﻼﻭﻩ ﺩﺭ ﺁﻳﺎﺕ ﻗﺮﺁﻥ ﻧﻴﺰ ﺑﻪ ﺍﻳﻦ ﻧﻜﺘﻪ ﻣﻬﻢ ﺍﺷﺎﺭﻩ ﮔﺮﺩﻳﺪﻩ ﺍﺳﺖ.

ﺑﻴﺎﻳﻴﺪ ﺩﺭ ﺁﻳﺎﺕ ﭘﺎﻳﺎﻧﯽ ﺳﻮﺭﻩ ﺻﻒ ﺗﺪﺑﺮ ﻛﻨﻴﻢ ﻭ ﻧﻜﺎﺕ ﺍﺭﺯﺷﻤﻨﺪی ﺭﺍ ﻛﻪ ﺩﺭﺑﺎﺭﻩ ﻳﺎﺭی ﺧﺪﺍ ﻭ ﻓﺪﺍﻛﺎﺭی ﺩﺭ ﺭﺍﻩ ﺍﻭ ﺩﺭ ﺍﻳﻦ ﺁﻳﺎﺕ ﺁﻣﺪﻩ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﺩﻫﻴﻢ. ﻣﻀﻤﻮﻥ ﺍﻳﻦ ﺁﻳﺎﺕ ﭼﻨﻴﻦ ﺍﺳﺖ:

«ﺍی ﻛﺴـﺎﻧﯽ ﻛﻪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻩ ﺍﻳﺪ! ﺁﻳﺎ ﺷـﻤﺎ ﺭﺍ ﺑـﻪ ﺗﺠﺎﺭﺗﯽ ﻛﻪ ﺍﺯ ﻋﺬﺍﺏ ﺩﺭﺩﺁﻭﺭ ﺭﻫﺎﻳﺘﺎﻥ ﺳﺎﺯﺩ ﺭﺍﻩ ﺑﻨﻤﺎﻳﻢ؟

(ﺁﻥ ﺗﺠﺎﺭﺕ ﺍﻳﻦ ﺍﺳﺖ:) ﺑﻪ ﺧﺪﺍ ﻭ ﭘﻴﺎﻣﺒﺮﺵ ﺍﻳﻤﺎﻥ ﺑﻴﺎﻭﺭﻳﺪ، ﻭ ﺩﺭ ﺭﺍﻩ ﺧـﺪﺍ ﺑـﺎ ﺟـﺎﻥ ﻭ ﻣﺎﻝ ﺧﻮﻳـﺶ ﺟﻬﺎﺩ ﻛﻨﻴـﺪ. ﻭ ﺍﻳﻦ ﺑﺮﺍی ﺷـﻤﺎ ﺑﻬﺘﺮ ﺍﺳﺖ، ﺍﮔﺮ ﺩﺍﻧﺎ ﺑﺎﺷﻴﺪ.

(ﺩﺭ ﻣﻘﺎﺑﻞ، ﺧﺪﺍﻭﻧﺪ ﻧﻴﺰ) ﮔﻨﺎﻫﺎﻥ ﺷﻤﺎ ﺭﺍ ﻣﯽ ﺁﻣﺮﺯﺩ ﻭ ﺑﻪ ﺑﺎﻍﻫﺎﻳﯽ ﻛـﻪ ﺩﺭﻭﻧﺶ ﻧﻬﺮﻫﺎ ﺟﺎﺭی ﺍﺳـﺖ ﻭ ﻧﻴﺰ ﺧﺎﻧﻪ ﻫﺎﻳـﯽ ﺧﻮﺵ ﻭ ﭘﺎﻛﻴﺰﻩ ﺩﺭ ﺑﻬﺸﺖﻫﺎی ﺟﺎﻭﻳﺪ ﺩﺍﺧﻠﺘﺎﻥ ﻣﯽ ﻛﻨﺪ، ﻭ ﺍﻳﻦ ﺍﺳﺖ ﻛﺎﻣﻴﺎﺑﯽ ﺑﺰﺭگ.

ﻭ (ﺑﻪ ﭘﺎﺱ ﺗﻼﺵ ﻭ ﻓﺪﺍﻛﺎﺭﻳﺘﺎﻥ) ﻧﻌﻤﺘﯽ ﺩﻳﮕﺮ ﻛﻪ ﺩﻭﺳﺘﺶ ﺩﺍﺭﻳﺪ (ﻧﺼﻴﺒﺘﺎﻥ ﻣﯽ ﺳـﺎﺯﺩ ﻭ ﺁﻥ،) ﻧﺼﺮﺗﯽ ﺍﺳـﺖ ﺍﺯ ﺟﺎﻧﺐ ﺧﺪﺍﻭﻧﺪ ﻭ ﻓﺘﺤﯽ ﻧﺰﺩﻳﮏ، ﻭ (ﺍی ﭘﻴﺎﻣﺒﺮ) ﻣﺆﻣﻨﺎﻥ ﺭﺍ ﺑﺸﺎﺭﺕ ﺩﻩ!

ﺍی ﻛﺴـﺎﻧﯽ ﻛﻪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻩ ﺍﻳﺪ! ﻳـﺎﻭﺭﺍﻥ ﺧﺪﺍ ﺑﺎﺷـﻴﺪ، ﻫﻤﭽﻨﺎﻥ ﻛـﻪ ﻋﻴﺴـﯽ ﺑـﻦ ﻣﺮﻳﻢ ﺑﻪ ﺣﻮﺍﺭﻳـﺎﻥ ﮔﻔﺖ: ﭼﻪ ﻛﺴـﺎﻧﻨﺪ ﻳـﺎﻭﺭﺍﻥ ﻣﻦ ﺑﻪ ﺳـﻮی ﺧـﺪﺍ؟ ﺣﻮﺍﺭﻳـﺎﻥ ﮔﻔﺘﻨﺪ: ﻣﺎﺋﻴﻢ ﻳـﺎﻭﺭﺍﻥ ﺧﺪﺍ. ﭘـﺲ ﮔﺮﻭﻫﯽ ﺍﺯ ﺑﻨﯽ ﺍﺳـﺮﺍﺋﻴﻞ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻧـﺪ ﻭ ﮔﺮﻭﻫﯽ ﻛﺎﻓﺮ ﺷـﺪﻧﺪ. ﻭ ﻣﺎ ﻛﺴـﺎﻧﯽ ﺭﺍ ﻛـﻪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻩ ﺑﻮﺩﻧـﺪ ﺩﺭ ﺑﺮﺍﺑﺮ ﺩﺷﻤﻨﺎﻧﺸـﺎﻥ ﻣـﺪﺩ ﻛﺮﺩﻳﻢ ﺗـﺎ ﭘﻴﺮﻭﺯ ﺷﺪﻧﺪ.»(73)

ﺑﺮﺧﯽ ﺍﺯ ﻧﻜﺎﺗﯽ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﺁﻳﺎﺕ ﺑﻪ ﭼﺸـﻢ ﻣﯽ ﺧﻮﺭﺩ ﺍﺯ ﺍﻳﻦ ﻗﺮﺍﺭ ﺍﺳﺖ:

١) ﺩﺭ ﺍﻳﻦ ﺁﻳﺎﺕ ﻣﺎ ﻣﺄﻣﻮﺭ ﺷﺪﻩ ﺍﻳﻢ ﻛﻪ ﺑﻪ ﻳﺎﺭی ﺧﺪﺍ ﺍﻗﺪﺍﻡ ﺑﻜﻨﻴﻢ. ﺩﺭ ﺍﻳـﻦ ﺁﻳﺎﺕ ﻳـﺎﺭی ﺧﺪﺍﻭﻧﺪ ﻳﺎ ﺭﻫﺎ ﻛﺮﺩﻥ ﺍﻭ ﺑـﻪ ﺍﻧﺘﺨﺎﺏ ﻭ ﺩﻟﺨﻮﺍﻩ ﻣﺎ ﻭﺍﮔﺬﺍﺭ ﻧﺸـﺪﻩ ﺍﺳـﺖ ﻛﻪ ﺍﮔﺮ ﺧﻮﺩﻣﺎﻥ ﺧﻮﺍﺳـﺘﻴﻢ ﺑﻪ ﻳـﺎﺭی ﭘﺮﻭﺭﺩﮔﺎﺭ ﻣﺒﺎﺩﺭﺕ ﻭﺭﺯﻳﻢ ﻭ ﺍﮔﺮ ﻧﺨﻮﺍﺳﺘﻴﻢ، ﻣﺄﻣﻮﺭ ﺑﻪ ﺁﻥ ﻧﺒﺎﺷﻴﻢ. ﺩﺭ ﺁﻳﻪ ی ﺁﺧﺮ، ﺧﺪﺍﻭﻧـﺪ ﺑﻪ ﺻﺮﺍﺣﺖ ﺍﻣﺮ ﻣﯽ ﻛﻨﺪ: «ﺍی ﻛﺴـﺎﻧﯽ ﻛـﻪ ﺍﻳﻤﺎﻥ ﺁﻭﺭﺩﻩ ﺍﻳﺪ! ﻳﺎﻭﺭﺍﻥ ﺧﺪﺍ ﺑﺎﺷﻴﺪ.»

٢) ﺷـﻴﻮﻩ ﻳﺎﺭی ﺧﺪﺍ ﺩﺭ ﺍﻳﻦ ﺁﻳﺎﺕ ﺑﺮﺍی ﻣﺎ ﺭﻭﺷـﻦ ﺷـﺪﻩ ﺍﺳﺖ. ﺧﺪﺍﻭﻧـﺪ ﺩﺭ ﺍﻳﻦ ﺁﻳﺎﺕ ﺑـﻪ ﻣﺎ ﺍﻟﮕﻮ ﻣﯽ ﺩﻫﺪ ﻭ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﻳﺎﻭﺭﺍﻥ ﺧﺪﺍ ﺑﺎﺷـﻴﺪ، ﻫﻤﭽﻨﺎﻥ ﻛﻪ ﻋﻴﺴـﯽ ﺑﻦ ﻣﺮﻳﻢ ﺑﻪ ﺣﻮﺍﺭﻳﺎﻥ ﮔﻔﺖ: ﭼﻪ ﻛﺴﺎﻧﻨﺪ ﻳﺎﻭﺭﺍﻥ ﻣﻦ ﺑﻪ ﺳﻮی ﺧﺪﺍ؟»

ﺍﮔـﺮ ﺩﺭ ﺍﻳـﻦ ﻋﺒـﺎﺭﺕ ﺩﻗـﺖ ﻛﻨﻴـﺪ ﺩﺭﻣﯽ ﻳﺎﺑﻴﺪ ﻛـﻪ ﺑﻪ ﻣﺎ ﺩﺳـﺘﻮﺭ ﺩﺍﺩﻩ ﺷـﺪﻩ ﻣﺎﻧﻨﺪ ﺣﻮﺍﺭﻳﺎﻥ ﺣﻀﺮﺕ ﻋﻴﺴـﯽ ﺑﻪ ﻳﺎﺭی ﺧﺪﺍ ﺍﻗﺪﺍﻡ ﻛﻨﻴﻢ. ﺣﻀـﺮﺕ ﻋﻴﺴـﯽ ﻭﻗﺘﯽ ﺁﻧﻬﺎ ﺭﺍ ﺑـﻪ ﻳﺎﺭی ﺧﺪﺍ ﺩﻋﻮﺕ ﻣﯽ ﻛﻨـﺪ، ﺁﻧﻬﺎ ﺭﺍ ﺑـﻪ ﻳﺎﺭی ﺧﻮﺩﺵ ﻛﻪ ﺣﺠﺖ ﺧﺪﺍﺳـﺖ ﻓـﺮﺍ ﻣﯽ ﺧﻮﺍﻧـﺪ. ﺁﻥ ﺣﻀﺮﺕ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﭼﻪ ﻛﺴـﺎﻧﻨﺪ ﻳﺎﻭﺭﺍﻥ ﻣﻦ، ﺑﻪ ﺳﻮی ﺧﺪﺍ؟» ﻭ ﺣﻮﺍﺭﻳﺎﻥ ﻛﻪ ﻗﺼـﺪ ﻳﺎﺭی ﺧـﺪﺍ ﻛﺮﺩﻧﺪ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻳﺎﺭی ﻣﻮﻻی ﺯﻣﺎﻥ ﺧﻮﺩ ﺍﻧﺠـﺎﻡ ﻣﯽ ﺩﻫﻨﺪ. ﻣـﺎ ﻧﻴﺰ ﺑﻨﺎ ﺑﺮ ﺍﻳﻦ ﺁﻳﺎﺕ ﻣﻮﻇﻔﻴﻢ ﻛـﻪ ﺧﺪﺍ ﺭﺍ ﺍﺯ ﻫﻤﻴﻦ ﻃﺮﻳـﻖ ﻳﺎﺭی ﻛﻨﻴﻢ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻛﻪ ﻣـﺎ ﻧﻴﺰ ﺑﺎﻳﺪ ﻳﺎﺭی ﺧﺪﺍ ﺭﺍ ﺩﺭ ﻳﺎﺭی ﺑﻨﺪﻩ ﺑﺮﮔﺰﻳﺪﻩ ﺧﺪﺍ ﺩﺭ ﺯﻣﺎﻥ ﺧﻮﺩ، ﻃﻠﺐ ﻛﻨﻴﻢ.

ﺑﺎﺯ ﺍﺯ ﺷـﻤﺎ ﻣﯽ ﺧﻮﺍﻫﻢ ﻛﻪ ﺩﺭ ﺁﻳﻪ ﺷـﺮﻳﻔﻪ ﺩﻗﺖ ﻛﻨﻴـﺪ ﻭ ﺑﺒﻴﻨﻴﺪ ﻛﻪ ﭼﮕﻮﻧﻪ ﺧﺪﺍﻭﻧﺪ ﺍﺯ ﻣﺎ ﻣﯽ ﺧﻮﺍﻫﺪ ﺗﺎ ﺑﺮﺍﺳﺎﺱ ﺍﻟﮕﻮی ﺣﻀﺮﺕ ﻋﻴﺴﯽ ﻭ ﺣﻮﺍﺭﻳﺎﻥ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﺑﺸﺘﺎﺑﻴﻢ.

٣) ﺧﺪﺍﻭﻧـﺪ ﺩﺭ ﺍﻳـﻦ ﺁﻳـﺎﺕ ﺍﺷـﺎﺭﻩ ﻓﺮﻣﻮﺩﻩ ﻛـﻪ ﺍﮔﺮ ﻣﺎ ﺑـﻪ ﻳﺎﺭی ﺍﻭ ﺍﻗـﺪﺍﻡ ﻛﻨﻴـﻢ، ﺍﻭ ﻧﻴﺰ ﺑـﻪ ﻳﺎﺭی ﻣﺎ ﻣﯽ ﺁﻳـﺪ. ﻓﺮﺍﺭﺳـﻴﺪﻥ ﻧﺼﺮﺕ ﺧﺪﺍ ﻭ ﺩﺳـﺘﻴﺎﺑﯽ ﺍﻫـﻞ ﺍﻳﻤـﺎﻥ ﺑـﻪ ﭘﻴـﺮﻭﺯی ﻭ ﭼﻴﺮﮔـﯽ، ﺁﺭﺯﻭی ﻗﻠﺒـﯽ ﻫـﺮ ﻣﺆﻣﻨﯽ ﺍﺳـﺖ. ﺧﺪﺍﻭﻧـﺪ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﻣﺎ ﻛﺴـﺎﻧﯽ ﺭﺍ ﻛﻪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻩ ﺑﻮﺩﻧـﺪ (ﻳﻌﻨﯽ ﺑـﻪ ﺩﻋﻮﺕ ﺣﻀﺮﺕ ﻋﻴﺴـﯽ ﻟﺒﻴﮏ ﮔﻔﺘﻪ ﺑـﻮﺩ) ﺩﺭ ﺑﺮﺍﺑﺮ ﺩﺷﻤﻨﺎﻧﺸﺎﻥ ﻣﺪﺩ ﻛﺮﺩﻳﻢ ﺗﺎ ﭘﻴﺮﻭﺯ ﺷﺪﻧﺪ.»

ﺍﻳﻦ ﺁﻳﻪ ﺍﺷﺎﺭﻩ ﺍی ﺑﺮﺍی ﻳﺎﻭﺭﺍﻥ ﺧﺪﺍ ﺩﺭ ﺍﻳﻦ ﺍﻣﺖ ﺍﺳﺖ ﻛﻪ ﺍﮔﺮ ﺁﻧﻬﺎ ﻧﻴﺰ ﺑﻪ ﻳﺎﺭی ﺧﺪﺍ ﺑﻴﺎﻳﻨﺪ، ﺧﺪﺍ ﺑﻪ ﻳﺎﺭی ﺁﻧﻬﺎ ﺧﻮﺍﻫﺪ ﺁﻣﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﭘﻴﺮﻭﺯ ﺧﻮﺍﻫﺪ ﺳﺎﺧﺖ.

ﻫﻤﻴﻦ ﺑﺸـﺎﺭﺕ ﺩﺭ ﺁﻳﻪ ﺳـﻴﺰﺩﻫﻢ ﻧﻴﺰ ﺑﻪ ﻛﺴـﺎﻧﯽ ﻛﻪ ﺩﺭ ﺭﺍﻩ ﺧﺪﺍ ﻭ ﺭﺳـﻮﻝ ﺑﻪ ﺗﻼﺵ ﻭ ﻓـﺪﺍﻛﺎﺭی ﻣﺎﻟـﯽ ﻭ ﺟﺎﻧﯽ ﻣﯽ ﭘﺮﺩﺍﺯﻧﺪ، ﺩﺍﺩﻩ ﺷـﺪﻩ ﺍﺳـﺖ، ﺁﻧﺠﺎ ﻛﻪ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﻭ (ﺑﻪ ﭘﺎﺱ ﺗﻼﺵ ﻭ ﻓﺪﺍﻛﺎﺭﻳﺘﺎﻥ) ﻧﻌﻤﺘﯽ ﺩﻳﮕـﺮ ﻛـﻪ ﺩﻭﺳـﺘﺶ ﺩﺍﺭﻳـﺪ (ﻧﺼﻴﺒﺘﺎﻥ ﺳـﺎﺯﺩ ﻭ ﺁﻥ) ﻧﺼﺮﺗﯽ ﺍﺳـﺖ ﺍﺯ ﺟﺎﻧـﺐ ﺧﺪﺍﻭﻧـﺪ ﻭ ﻓﺘﺤﯽ ﻧﺰﺩﻳﮏ. ﻭ (ﺍی ﭘﻴﺎﻣﺒﺮ) ﻣﺆﻣﻨﺎﻥ ﺭﺍ ﺑﺸـﺎﺭﺕ ﺩﻩ!»

۵) ﭘـﺎﺩﺍﺵ ﺩﻳﮕـﺮی ﻛﻪ ﻧﺼﻴﺐ ﻳـﺎﻭﺭﺍﻥ ﺧﺪﺍ ﻣﯽ ﺷـﻮﺩ، ﻣﻐﻔﺮﺕ ﻭ ﺁﻣـﺮﺯﺵ ﺍﻟﻬﯽ ﺍﺳـﺖ. ﻫﻤﻪ ﻣـﺎ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﻛـﻪ ﺟﺎﻥﻫﺎﻳﻤﺎﻥ ﺩﺭ ﮔﺮﻭ ﺧﻄﺎﻫـﺎ، ﻏﻔﻠﺖﻫـﺎ ﻭ ﻟﻐﺰﺵﻫﺎﻳﻤـﺎﻥ ﺍﺳـﺖ. ﻭﻇﺎﻳﻔـﯽ ﻛﻪ ﺑـﺮ ﻋﻬﺪﻩ ﻣﺎﺳﺖ، ﺑﺴﻴﺎﺭ ﺑﺰﺭﮔﺘﺮ ﺍﺯ ﺁﻥ ﻛﺎﺭﻫﺎی ﻧﻴﻜﯽ ﺍﺳﺖ ﻛﻪ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺍﻳﻢ ﻭ ﻣﯽ ﺩﻫﻴﻢ. ﻫﻤﻪ ﻣﺎ ﺑﺮ ﻟﺒﻪ ﭘﺮﺗﮕﺎﻩ ﻏﻀﺐ ﺍﻟﻬﯽ ﻫﺴـﺘﻴﻢ ﻭ ﺍﮔﺮ ﺍﻳﻦ ﺧﻄﺮ ﺭﺍ ﺑﺮﺍی ﺧﻮﺩ ﺍﺣﺴـﺎﺱ ﻧﻤﯽ ﻛﻨﻴﻢ ﺑﺪﻳﻦ ﺟﻬﺖ ﺍﺳـﺖ ﻛﻪ ﺷﻴﻄﺎﻥ ﻣﺎ ﺭﺍ ﻧﺴـﺒﺖ ﺑـﻪ ﻭﻇﺎﻳﻔﻤﺎﻥ ﻏﺎﻓﻞ ﺳـﺎﺧﺘﻪ ﻭ ﺍﺯ ﻋﻈﻤﺖ ﻣﻌﺎﺻـﯽ ﻭ ﺑﺰﺭﮔﯽ ﮔﻨﺎﻫـﺎﻥ ﻭ ﺍﺯ ﺟﻤﻠﻪ ﮔﻨـﺎﻩ ﻏﻔﻠﺖ ﺍﺯ ﺧﺪﺍ ﺑﯽ ﺧﺒﺮﻳـﻢ. ﺩﺭ ﻗﺮﺁﻥ ﻭ ﺩﻳﮕﺮ ﺗﻌﺎﻟﻴـﻢ ﺩﻳﻨـﯽ ﺑﻪ ﻣﺎ ﺩﺭﺑﺎﺭﻩ ﮔﻨﺎﻫﺎﻧﻤﺎﻥ ﻫﺸـﺪﺍﺭ ﺩﺍﺩﻩ ﺷـﺪﻩ ﺍﺳـﺖ ﻭ ﺑﺮ ﻣﺎﺳـﺖ ﻛﻪ ﺍﻳﻦ ﻫﺸﺪﺍﺭﻫﺎ ﺭﺍ ﺟﺪی ﺑﮕﻴﺮﻳﻢ. ﺧﺪﺍﻭﻧﺪ ﺑﺰﺭگ ﻭ ﻣﻬﺮﺑﺎﻥ ﺩﺭ ﻛﻠﻤـﺎﺕ ﺧﻮﻳـﺶ ﻧﻪ ﺗﻨﻬﺎ ﺑﻪ ﻣـﺎ ﺩﺭﺑﺎﺭﻩ ﻧﺘﻴﺠﻪ ﺧﻄﺎﻫﺎﻳﻤﺎﻥ ﻫﺸـﺪﺍﺭ ﺩﺍﺩﻩ، ﺑﻠﻜـﻪ ﺭﺍﻩ ﻧﺠـﺎﺕ ﺍﺯ ﺯﻧﺠﻴـﺮ ﮔﻨﺎﻫﺎﻥ ﺭﺍ ﻧﻴﺰ ﭘﻴﺶ ﭘﺎی ﻣﺎ ﮔﺸـﻮﺩﻩ ﺍﺳﺖ.

ﻳﻜـﯽ ﺍﺯ ﻣﻬﻢ ﺗﺮﻳﻦ ﺭﺍﻩ ﻫﺎی ﺁﻣﺮﺯﻳﺪﻩ ﺷـﺪﻥ ﻭ ﻧﺠـﺎﺕ ﺍﺯ ﮔﻨﺎﻫﺎﻥ، ﻓـﺪﺍﻛﺎﺭی ﻣﺎﻟـﯽ ﻭ ﺟﺎﻧـﯽ ﺩﺭ ﺭﺍﻩ ﺧـﺪﺍ ﻳﻌﻨـﯽ ﺩﺭ ﺭﺍﻩ ﻭﻟـﯽ ﺑﺮﮔﺰﻳـﺪﻩ ﺍﻭﺳﺖ. ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﺁﻳﺎﺕ ﺁﺧﺮ ﺳﻮﺭﻩ ی ﺻﻒ ﻣﯽ ﻓﺮﻣﺎﻳﺪ:

«ﺍی ﻛﺴـﺎﻧﯽ ﻛﻪ ﺍﻳﻤـﺎﻥ ﺁﻭﺭﺩﻩ ﺍﻳﺪ! ﺁﻳﺎ ﺷـﻤﺎ ﺭﺍ ﺑـﻪ ﺗﺠﺎﺭﺗﯽ ﻛﻪ ﺍﺯ ﻋﺬﺍﺏ ﺩﺭﺩﺁﻭﺭ ﺭﻫﺎﻳﺘﺎﻥ ﺳﺎﺯﺩ ﺭﺍﻩ ﺑﻨﻤﺎﻳﻢ؟

(ﺁﻥ ﺗﺠﺎﺭﺕ ﺍﻳﻦ ﺍﺳـﺖ:) ﺑﻪ ﺧﺪﺍ ﻭ ﭘﻴﺎﻣﺒـﺮﺵ ﺍﻳﻤﺎﻥ ﺑﻴﺎﻭﺭﻳﺪ، ﻭ ﺩﺭ ﺭﺍﻩ ﺧﺪﺍ ﺑﺎ ﺟﺎﻥ ﻭ ﻣﺎﻝ ﺧﻮﻳﺶ ﺟﻬﺎﺩ ﻛﻨﻴﺪ. ﻭ ﺍﻳﻦ ﺑﺮﺍی ﺷـﻤﺎ ﺑﻬﺘﺮ ﺍﺳﺖ، ﺍﮔﺮ ﺩﺍﻧﺎ ﺑﺎﺷﻴﺪ.

(ﺩﺭ ﻣﻘﺎﺑﻞ، ﺧﺪﺍﻭﻧﺪ ﻧﻴﺰ) ﮔﻨﺎﻫﺎﻥ ﺷﻤﺎ ﺭﺍ ﻣﯽ ﺁﻣﺮﺯﺩ ﻭ ﺑﻪ ﺑﺎﻍﻫﺎﻳﯽ ﻛﻪ ﺩﺭﻭﻧﺶ ﻧﻬﺮﻫﺎ ﺟﺎﺭی ﺍﺳـﺖ ﻭ ﻧﻴﺰ ﺧﺎﻧﻪ ﻫﺎﻳـﯽ ﺧﻮﺵ ﻭ ﭘﺎﻛﻴﺰﻩ ﺩﺭ ﺑﻬﺸﺖﻫﺎی ﺟﺎﻭﻳﺪ ﺩﺍﺧﻠﺘﺎﻥ ﻣﯽ ﻛﻨﺪ، ﻭ ﺍﻳﻦ ﺍﺳﺖ ﻛﺎﻣﻴﺎﺑﯽ ﺑﺰﺭگ.» ﺑـﺮﺍی ﺍﻳﻨﻜﻪ ﻣﻔﻬﻮﻡ ﻳـﺎﺭی ﺧﺪﺍ ﺭﺍ ﺍﺯ ﺧﺎﻃﺮ ﻧﺒﺮﻳـﻢ، ﺑﺎﺯ ﻳﺎﺩﺁﻭﺭی ﻣﯽ ﺷـﻮﺩ ﻛـﻪ ﻳﺎﺭی ﺧـﺪﺍ ﺩﺭ ﻫﺮ ﺯﻣـﺎﻥ ﺍﺯ ﻃﺮﻳﻖ ﻳـﺎﺭی ﺍﻣـﺎﻡ ﺁﻥ ﺯﻣﺎﻥ ﺍﺳـﺖ. ﻛﺴـﯽ ﻛﻪ ﻧﺪﺍی ﻏﺮﺑﺖ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﺸﻨﻮﺩ ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﺑﺸﺘﺎﺑﺪ ﻭ ﺍﮔﺮ ﭼﻨﻴﻦ ﻛﻨﺪ ﻣﺸﻤﻮﻝ ﺷﻔﺎﻋﺖ ﭘﻴﺎﻣﺒﺮ ﻭ ﺷﺎﻳﺴﺘﻪ ﺭﺣﻤﺖ ﺧﺪﺍﻭﻧﺪ ﻣﯽ ﮔﺮﺩﺩ.

ﻧﻘـﻞ ﺍﺳـﺖ، ﻫﻨﮕﺎﻣـﯽ ﻛـﻪ ﺍﻣﺎﻡ ﺣﺴـﻴﻦ ﻋﻠﻴـﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﺣﺎﻝ ﺣﺮﻛﺖ ﺑﻪ ﺳﻤﺖ ﻛﻮﻓﻪ ﺑﻮﺩﻧﺪ ﺑﺎ ﻛﺴﯽ ﺑﻪ ﻧﺎﻡ ﻋﺒﻴﺪﺍﻟﻠﻪ ﺑﻦ ﺣﺮ ﺑﺮﺧﻮﺭﺩ ﻛﺮﺩﻧﺪ. ﺍﻣﺎﻡ ﺑﻪ ﺍﻭ ﻓﺮﻣﻮﺩﻧﺪ: «ﺍی ﻣﺮﺩ! ﺗﻮ ﺍﻧﺴﺎﻥ ﮔﻨﻬﻜﺎﺭ ﺧﻄﺎﭘﻴﺸﻪ ﺍی ﻫﺴـﺘﯽ. ﺍﮔـﺮ ﺩﺭ ﺍﻳﻦ ﻟﺤﻈﻪ ﺍﺯ ﮔﻨﺎﻫﺎﻥ ﺧﻮﺩ ﺗﻮﺑـﻪ ﻧﻜﻨﯽ ﻭ ﺑﻪ ﻳﺎﺭی ﻣﻦ ﻧﻴﺎﻳـﯽ ﻭ ﺟﺪﻡ ﺭﺳـﻮﻝ ﺧﺪﺍ ﺷـﻔﺎﻋﺘﺖ ﻧﻜﻨـﺪ، ﺧﺪﺍﻭﻧﺪ ﺗـﻮ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﮔﻨﺎﻫﺎﻧـﺖ ﻣﺆﺍﺧﺬﻩ ﺧﻮﺍﻫﺪ ﻛﺮﺩ.» ﻋﺒﻴﺪﺍﻟﻠﻪ ﮔﻔﺖ: ﺍی ﻓﺮﺯﻧﺪ ﺭﺳـﻮﻝ ﺧـﺪﺍ! ﺍﮔـﺮ ﺑﻪ ﻳـﺎﺭی ﺍﺕ ﺑﻴﺎﻳﻢ ﻛﺸـﺘﻪ ﺧﻮﺍﻫﻢ ﺷـﺪ. ﻣـﻦ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﻛﺎﺭ ﻣﻌـﺎﻑ ﺩﺍﺭﻳﺪ ﺍﻣﺎ ﺍﻳﻦ ﺍﺳـﺐ ﻣﻦ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺷﻤﺎﺳـﺖ. ﺍﺳـﺒﯽ ﺗﻴﺰﺗﮏ ﺍﺳـﺖ ﻭ ﺗـﺎ ﻛﻨـﻮﻥ ﭼﻴـﺰی ﺭﺍ ﺑـﺎ ﺁﻥ ﺗﻌﻘﻴﺐ ﻧﻜـﺮﺩﻩ ﺍﻡ، ﻣﮕـﺮ ﺍﻳﻨﻜﻪ ﺑﻪ ﺁﻥ ﺭﺳـﻴﺪﻩ ﺍﻡ ﻭ ﻫﻴـﭻ ﻛـﺲ ﻣـﺮﺍ ﺩﻧﺒﺎﻝ ﻧﻜـﺮﺩﻩ ﻣﮕـﺮ ﺍﻳﻨﻜﻪ ﺑـﺎ ﺁﻥ ﺍﺯ ﺍﻭ ﮔﺮﻳﺨﺘـﻪ ﺍﻡ. ﺍﻣـﺎﻡ ﺍﺯ ﻋﺒﻴﺪﺍﻟﻠـﻪ ﺭﻭی ﺑﺮﮔﺮﺩﺍﻧﺪ ﻭ ﮔﻔـﺖ: «ﻣﻦ ﻧﻪ ﺑﻪ ﺗﻮ ﻭ ﻧـﻪ ﺑﻪ ﺍﺳـﺒﺖ ﺍﺣﺘﻴﺎﺟﯽ ﻧـﺪﺍﺭﻡ ﻭ ﮔﻤﺮﺍﻫﺎﻥ ﺭﺍ ﺑـﺎﺯﻭی ﺧﻮﻳﺶ ﻗﺮﺍﺭ ﻧﻤﯽ ﺩﻫﻢ. ﺍﻳﻨﮏ ﻛﻪ ﭼﻨﻴﻦ ﺍﺳـﺖ ﺍﺯ ﺍﻳﻨﺠﺎ ﺩﻭﺭ ﺷـﻮ. ﻧﻪ ﺑﺎ ﻣﺎ ﺑﺎﺵ ﻭ ﻧﻪ ﺩﺭ ﺑﺮﺍﺑﺮ ﻣﺎ! ﭼﺮﺍ ﻛﻪ ﺍﮔﺮ ﻛﺴﯽ ﺻﺪﺍی ﻏﺮﺑﺖ ﻣﺎ ﺭﺍ ﺑﺸﻨﻮﺩ ﻭ ﺟﻮﺍﺑﻤﺎﻥ ﻧﺪﻫﺪ، ﺧﺪﺍﻭﻧﺪ ﺍﻭ ﺭﺍ ﺑﺎ ﺻﻮﺭﺕ ﺩﺭ ﺁﺗﺶ ﺧﻮﺍﻫﺪ ﺍﻓﻜﻨﺪ.»(74)

ﺑﻌﺪﻫﺎ ﻋﺒﻴﺪﺍﻟﻠﻪ ﺍﺯ ﻛﺮﺩﻩ ﺧﻮﺩ ﭘﺸـﻴﻤﺎﻥ ﮔﺸـﺖ ﻭ ﻣﻮﺭﺩ ﺳﻮﺀﻇﻦ ﺍﺑﻦﺯﻳـﺎﺩ ﻗـﺮﺍﺭ ﮔﺮﻓـﺖ ﻭ ﺑﻪ ﺩﺳـﺖ ﺟﻤﻌـﯽ ﺍﺯ ﺍﻭﺑـﺎﺵ ﻛﻮﻓﻪ ﺑـﻪ ﻗﺘﻞ ﺭﺳﻴﺪ.

ﻣﺒـﺎﺩ ﻛـﻪ ﻣﺎ، ﻫﻤﭽـﻮﻥ ﻋﺒﻴﺪﺍﻟﻠﻪ ﺑﻦ ﺣﺮ ﺍﺯ ﻳﺎﺭی ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺧﻮﺩ ﺩﺭﻳﻎ ﻭﺭﺯﻳﻢ.
