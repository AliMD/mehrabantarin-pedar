بازگشت

ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﺤﻀﺮ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﺒﻴﻨﻴﻢ!

ﻛﺴـﯽ ﺑﻪ ﻧﺎﻡ ﻣﻮﺳـﯽ ﺑﻦ ﺳـﻴﺎﺭ ﻣﯽ ﮔﻮﻳـﺪ: ﺑﻪ ﻫﻤـﺮﺍﻩ ﺍﻣﺎﻡ ﺭﺿـﺎ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺩﺭﻭﺍﺯﻩ ﺷﻬﺮ ﻃﻮﺱ ﺭﺳﻴﺪﻳﻢ. ﺻﺪﺍی ﮔﺮﻳﻪ ﺍی ﺷﻨﻴﺪﻡ ﻭ ﺩﺭ ﺟﺴﺘﺠﻮی ﺁﻥ ﺑﺮﺁﻣﺪﻡ. ﺩﺭ ﺁﻧﺠﺎ ﺟﻨﺎﺯﻩ ﺍی ﺑﻮﺩ. ﻧﺎﮔﻬﺎﻥ ﺍﻣﺎﻡ ﺭﺿﺎ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺭﺍ ﺩﻳﺪﻡ ﻛﻪ ﺑﺎﺷـﺘﺎﺏ ﺍﺯ ﺍﺳـﺐ ﭘﻴﺎﺩﻩ ﺷـﺪﻧﺪ ﻭ ﺑﻪ ﺳﻮی ﺟﻨﺎﺯﻩ ﺭﻓﺘـﻪ، ﺁﻥ ﺭﺍ ﺑـﺮ ﺩﻭﺵ ﮔﺮﻓﺘﻨـﺪ. ﺳـﭙﺲ ﺑﻪ ﺳـﻮی ﻣـﺎ ﺁﻣﺪﻧـﺪ ﻭ ﺑﻪ ﻣﻦ ﮔﻔﺘﻨﺪ:

«ﺍی ﻣﻮﺳـﯽ ﺑـﻦ ﺳـﻴﺎﺭ! ﻫﺮﻛـﺲ ﺟﻨﺎﺯﻩ ﻳﻜـﯽ ﺍﺯ ﺩﻭﺳـﺘﺎﻥ ﻣﺎ ﺭﺍ ﺗﺸـﻴﻴﻊ ﻛﻨـﺪ، ﺍﺯ ﮔﻨﺎﻫﺎﻥ ﺧﻮﺩ ﺑﻴﺮﻭﻥ ﻣـﯽ ﺭﻭﺩ ﻭ ﻫﻤﭽﻮﻥ ﺭﻭﺯی ﻛﻪ ﺍﺯ ﻣﺎﺩﺭ ﺯﺍﺩﻩ ﺷﺪﻩ، ﺍﺯ ﮔﻨﺎﻫﺎﻥ ﭘﺎک ﺧﻮﺍﻫﺪ ﺷﺪ.»

ﻣـﺮﺩﻡ، ﺟﻨـﺎﺯﻩ ﺭﺍ ﺩﺭ ﻛﻨـﺎﺭ ﻗﺒـﺮ ﺑﺮ ﺯﻣﻴـﻦ ﻧﻬﺎﺩﻧﺪ. ﻣﻮﻻﻳـﻢ ﻣﺮﺩﻡ ﺭﺍ ﻛﻨـﺎﺭ ﺯﺩﻧـﺪ ﻭ ﭘﻴـﺶ ﺟﻨﺎﺯﻩ ﻧﺸﺴـﺘﻨﺪ ﻭ ﺩﺳـﺖ ﺑـﺮ ﺳـﻴﻨﻪ ﺍﻭ ﻧﻬﺎﺩﻩ ﻭ ﻓﺮﻣﻮﺩﻧﺪ:

«ﺍی ﻓﻼﻧﯽ ﻓﺮﺯﻧﺪ ﻓﻼﻧﯽ! ﻣﮋﺩﻩ ﺑﺎﺩ ﺗﻮ ﺭﺍ ﺑﻪ ﺑﻬﺸـﺖ ﻭ ﭘﺲ ﺍﺯ ﺍﻳﻦ ﻟﺤﻈﻪ ﺑﺮ ﺗﻮ ﺗﺮﺳﯽ ﻣﺒﺎﺩ!»

ﺑﻪ ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻋﺮﺽ ﻛﺮﺩﻡ: ﻓﺪﺍﻳﺖ ﮔﺮﺩﻡ، ﺁﻳﺎ ﺍﻳﻦ ﻣﺮﺩ ﺭﺍ ﻣﯽ ﺷﻨﺎﺳـﻴﺪ؟ (ﻭ ﭼﮕﻮﻧﻪ ﻣﻤﻜﻦ ﺍﺳـﺖ ﺍﻭ ﺭﺍ ﺑﺸﻨﺎﺳـﻴﺪ ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ) ﺍﻳﻨﺠﺎ ﺳﺮﺯﻣﻴﻨﯽ ﺍﺳﺖ ﻛﻪ ﺗﺎ ﻛﻨﻮﻥ ﺑﺮ ﺁﻥ ﻗﺪﻡ ﻧﻨﻬﺎﺩﻩ ﺍﻳﺪ! ﺣﻀﺮﺕ ﺭﺿﺎ ﻓﺮﻣﻮﺩﻧﺪ:

«ﺍی ﻣﻮﺳـﯽ ﺑﻦ ﺳـﻴﺎﺭ! ﻣﮕﺮ ﻧﻤﯽ ﺩﺍﻧﯽ ﻛﻪ ﺍﻋﻤﺎﻝ ﺷﻴﻌﻴﺎﻥ ﻣﺎ ﺑﺮ ﻣﺎ ﺍﻣﺎﻣﺎﻥ، ﻫﺮ ﺻﺒﺢ ﻭ ﺷـﺎﻡ ﻋﺮﺿﻪ ﻣﯽ ﺷـﻮﺩ. ﭘﺲ ﻣﺎ ﺑﺮﺍی ﻫﺮ ﺗﻘﺼﻴﺮی ﻛـﻪ ﺩﺭ ﺍﻋﻤـﺎﻝ ﺁﻧﻬﺎﺳـﺖ، ﺍﺯ ﺧﺪﺍﻭﻧﺪ ﻣﺘﻌﺎﻝ ﻣﯽ ﺧﻮﺍﻫﻴﻢ ﻛﻪ ﮔﺬﺷـﺖ ﻛﻨﺪ ﻭ ﺑﺮﺍی ﻫﺮ ﻛﺎﺭ ﺧﻮﺏ ﻭ ﭘﻴﺸﺮﻓﺖ ﺁﻥﻫﺎ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﻫﺎی ﺩﻳﻨﯽ ﺑﺮﺍی ﺁﻧﺎﻥ ﺍﺯ ﺧﺪﺍ ﭘﺎﺩﺍﺵ ﺗﻤﻨﺎ ﻣﯽ ﻛﻨﻴﻢ.»

ﺑﺮﺭﺳـﯽ ﮔﺰﺍﺭﺵ ﺍﻋﻤﺎﻝ ﻣﺎ ﺗﻮﺳـﻂ ﭘﻴﺎﻣﺒﺮ ﻭ ﺍﺋﻤﻪ ﻋﻠﻴﻬﻢ ﺍﻟﺴﻼﻡ ﺍﻣﺮ ﻏﺮﻳﺒﯽ ﻧﻴﺴﺖ، ﺑﻠﻜﻪ ﻣﻄﺎﺑﻖ ﺻﺮﻳﺢ ﺁﻳﺎﺕ ﻗﺮﺁﻥ ﺍﺳﺖ. ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﻗﺮﺁﻥ ﻛﺮﻳﻢ ﻣﯽ ﻓﺮﻣﺎﻳﺪ:

«ﺍی ﭘﻴﺎﻣﺒـﺮ! ﺑﮕـﻮ ﻫـﺮ ﻛﺎﺭی ﻛﻪ ﺑﺨﻮﺍﻫﻴـﺪ، ﺑﻜﻨﻴﺪ؛ ﺍﻣـﺎ (ﺑﺪﺍﻧﻴﺪ ﻛـﻪ) ﺑﻪ ﺯﻭﺩی ﺧﺪﺍﻭﻧﺪ ﻭ ﺭﺳـﻮﻝ ﺍﻭ ﻭ ﻣﺆﻣﻨﺎﻥ، ﺍﻋﻤﺎﻟﺘـﺎﻥ ﺭﺍ ﺧﻮﺍﻫﻨﺪ ﺩﻳﺪ.»(44)

ﺭﻭﺍﻳـﺎﺕ ﻓﺮﺍﻭﺍﻧـﯽ ﺩﺭ ﺗﻮﺿﻴـﺢ ﺍﻳـﻦ ﻛﻼﻡ ﺍﻟﻬـﯽ ﺍﺯ ﭘﻴﺎﻣﺒـﺮ ﺧﺪﺍ ﻭ ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﺍﺳـﺖ. ﺁﻧﻬـﺎ ﻓﺮﻣﻮﺩﻩ ﺍﻧﺪ ﻣﻨﻈﻮﺭ ﺍﺯ ﻣﺆﻣﻨﺎﻧـﯽ ﻛـﻪ ﺍﻋﻤﺎﻝ ﻣﺮﺩﻡ ﺑـﻪ ﺁﻧﻬﺎ ﻋﺮﺿﻪ ﻣﯽ ﺷـﻮﺩ، ﺍﻣﺎﻣﺎﻥ ﻋﻠﻴﻬﻢ ﺍﻟﺴﻼﻡ ﻫﺴﺘﻨﺪ. ﺁﻧﻬﺎ ﺷﺎﻫﺪ ﻭ ﮔﻮﺍﻩ ﺍﺯ ﻃﺮﻑ ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﻣﻴﺎﻥ ﻣﺮﺩﻡ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ. ﺑﻨﺎ ﺑﻪ ﺍﻳﻦ ﺭﻭﺍﻳﺎﺕ، ﮔﺰﺍﺭﺵ ﺍﻋﻤﺎﻝ ﻣﺎ ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻨﻜﻪ ﺩﺭ ﺩﻳﻮﺍﻥ ﺍﻟﻬﯽ ﺛﺒﺖ ﻭ ﺿﺒﻂ ﻣﯽ ﺷﻮﺩ، ﺑﺎ ﻧﻈﻢ ﺧﺎﺻﯽ ﻣﺮﺗﺒﺎ ﺑﻪ ﻣﺤﻀﺮ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﻭ ﺍﻣﺎﻡ ﺯﻧﺪﻩ ﺯﻣﺎﻥ ﻋﺮﺿﻪ ﻣﯽ ﮔﺮﺩﺩ.

ﺍﺯ ﺍﻣـﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﻛـﻪ ﻓﺮﻣﻮﺩﻧـﺪ: «ﺍﻋﻤﺎﻝ ﺑﻨـﺪﮔﺎﻥ ﺧﺪﺍ (ﭼﻪ ﻧﻴﻜﺎﻥ ﻭ ﭼﻪ ﺯﺷـﺘﻜﺎﺭﺍﻥ) ﻫﺮ ﺭﻭﺯ ﺻﺒﺢ ﺑﻪ ﻣﺤﻀﺮ ﺭﺳـﻮﻝ ﺧﺪﺍ ﻋﺮﺿﻪ ﻣﯽ ﺷـﻮﺩ، ﭘﺲ ﻣﺮﺍﻗﺐ ﺍﻋﻤﺎﻝ ﺧﻮﺩ ﺑﺎﺷـﻴﺪ ﻭ ﺍﻳﻦ ﻫﻤﺎﻥ ﮔﻔﺘﺎﺭ ﺧﺪﺍﻭﻧﺪ ﺍﺳـﺖ ﻛﻪ ﻣﯽ ﻓﺮﻣﺎﻳـﺪ: ﻫﺮ ﻛﺎﺭی ﻛﻪ ﺑﺨﻮﺍﻫﻴﺪ، ﺑﻜﻨﻴﺪ؛ ﺍﻣﺎ (ﺑﺪﺍﻧﻴﺪ ﻛﻪ) ﺑﻪ ﺯﻭﺩی ﺧﺪﺍﻭﻧﺪ ﻭ ﺭﺳﻮﻝ ﺍﻭ ... ﺍﻋﻤﺎﻟﺘﺎﻥ ﺭﺍ ﺧﻮﺍﻫﻨﺪ ﺩﻳﺪ. »(45)

ﻋﺒﺪﺍﻟﻠـﻪ ﺑﻦ ﺍﺑـﺎﻥ ﺭﻭﻏﻦﻓـﺮﻭﺵ، ﻓﺮﺩی ﺑـﻮﺩ ﻛﻪ ﻧﺰﺩ ﺍﻣـﺎﻡ ﺭﺿﺎ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺭﺝ ﻭ ﻗﺮﺑـﯽ ﺩﺍﺷـﺖ. ﺍﻭ ﮔﻮﻳـﺪ ﺭﻭﺯی ﺑـﻪ ﺁﻥ ﺣﻀﺮﺕ ﻋـﺮﺽ ﻛـﺮﺩﻡ ﻛﻪ ﺑﺮﺍی ﻣﻦ ﻭ ﺧﺎﻧـﻮﺍﺩﻩ ﺍﻡ ﺩﻋﺎ ﻛﻨﻴﺪ. ﺍﻣـﺎﻡ ﻓﺮﻣﻮﺩ: «ﺁﻳﺎ ﮔﻤﺎﻥ ﻛﺮﺩﻩ ﺍی ﻛﻪ ﭼﻨﻴﻦ ﻧﻤﯽ ﻛﻨﻢ؟ ﻗﺴﻢ ﺑﻪ ﺧﺪﺍ ﻛﻪ ﺍﻣﻮﺭ ﺷﻤﺎ ﻫﺮ ﺷﺐ ﻭ ﻫﺮ ﺭﻭﺯ ﺑﻪ ﻣﻦ ﮔﺰﺍﺭﺵ ﻣﯽ ﺷﻮﺩ.»

ﻋﺒﺪﺍﻟﻠـﻪ ﮔﻮﻳـﺪ: ﺍﻳﻦ ﻛﻼﻡ ﺍﻣﺎﻡ ﺑﺮ ﻣﻦ ﺳـﻨﮕﻴﻦ ﺁﻣـﺪ. ﺍﻣﺎﻡ (ﺑﺮﺍی ﺍﻳﻨﻜـﻪ ﺗﻌﺠـﺐ ﻣـﻦ ﺑﺮﻃـﺮﻑ ﺷـﻮﺩ) ﻓﺮﻣـﻮﺩ: «ﻣﮕـﺮ ﻛﺘـﺎﺏ ﺧـﺪﺍی ﻋﺰﻭﺟـﻞ ﺭﺍ ﻧﺨﻮﺍﻧـﺪﻩ ﺍی ﻛﻪ ﻣﯽ ﻓﺮﻣﺎﻳـﺪ: ﺍی ﭘﻴﺎﻣﺒﺮ! ﺑﮕـﻮ ﻫﺮ ﻛﺎﺭی ﻛﻪ ﺑﺨﻮﺍﻫﻴﺪ، ﺑﻜﻨﻴﺪ؛ ﺍﻣﺎ (ﺑﺪﺍﻧﻴﺪ ﻛﻪ) ﺑﻪ ﺯﻭﺩی ﺧﺪﺍﻭﻧﺪ ﻭ ﺭﺳﻮﻝ ﺍﻭ ﻭ ﻣﺆﻣﻨﺎﻥ ﺍﻋﻤﺎﻟﺘﺎﻥ ﺭﺍ ﺧﻮﺍﻫﻨﺪ ﺩﻳﺪ. »(46)

ﺍﮔﺮ ﻣﺎ ﺑﻪ ﺍﻳﻦ ﺩﺭک ﺑﺮﺳـﻴﻢ ﻛﻪ ﺗﺤﺖ ﻧﻈﺎﺭﺕ ﺧﺪﺍ، ﭘﻴﺎﻣﺒﺮ ﻭ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺧـﻮﺩ ﻗﺮﺍﺭ ﺩﺍﺭﻳﻢ ﺩﺭ ﺭﻓﺘﺎﺭﻫﺎی ﺧـﻮﺩ ﺗﺠﺪﻳﺪ ﻧﻈﺮ ﺧﻮﺍﻫﻴﻢ ﻛﺮﺩ ﻭ ﺑﺎ ﺍﺻﻼﺡ ﺁﻧﻬﺎ ﺩﻝ ﻣﻮﻻی ﺧﻮﻳﺶ ﺭﺍ ﺷﺎﺩ ﺧﻮﺍﻫﻴﻢ ﺳﺎﺧﺖ. ﺁﻳﺎ ﺷﺎﺩ ﺳـﺎﺧﺘﻦ ﻣـﻮﻻ، ﻧﻮﻋﯽ ﻛﻤﮏ ﻭ ﻳﺎﺭی ﺑﻪ ﺍﻭ ﻧﻴﺴـﺖ؟ ﻭ ﺁﻳـﺎ ﺍﻧﺪﻭﻫﮕﻴﻦ ﺳـﺎﺧﺘﻦ ﻣﻮﻻ، ﻧﻮﻋﯽ ﺩﺍﻣﻦﺯﺩﻥ ﺑﻪ ﺩﺭﺩﻫـﺎ ﻭ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺣﻀﺮﺗﺶ ﺑﻪ ﺷـﻤﺎﺭ ﻧﻤﯽ ﺁﻳﺪ؟ ﺑﺎﻳﺪ ﺑﺎﻭﺭ ﻛﻨﻴﻢ ﻛـﻪ ﺧﻮﺑﯽ ﻳﺎ ﺑﺪی ﺭﻓﺘﺎﺭﻫﺎی ﻣﺎ ﺩﺭ ﻗﻠﺐ ﺍﻣﺎﻡ ﺯﻣﺎﻧﻤﺎﻥ ﻭ ﭘﻴﻐﻤﺒﺮﻣﺎﻥ ﺗﺄﺛﻴﺮ ﻣﺜﺒﺖ ﻳﺎ ﻣﻨﻔﯽ ﻣﯽ ﮔﺬﺍﺭﺩ.

ﻳﻜﯽ ﺍﺯ ﺷـﻴﻌﻴﺎﻥ ﺑﻪ ﻧﺎﻡ ﺳـﻤﺎﻋﻪ ﻧﻘﻞ ﻣﯽ ﻛﻨﺪ ﻛﻪ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺟﻤﻌﯽ ﻣﯽ ﻓﺮﻣﻮﺩ: «ﺷـﻤﺎ ﭼﺮﺍ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﺭﺍ ﻣﯽ ﺁﺯﺍﺭﻳﺪ؟» ﻣﺮﺩی ﺑﺎ ﺷـﻨﻴﺪﻥ ﺍﻳﻦ ﻛﻼﻡ ﺍﻣﺎﻡ ﺑﺮﺧﺎﺳﺖ ﻭ ﭘﺮﺳﻴﺪ ﻣﺎ ﭼﮕﻮﻧﻪ ﺭﺳﻮﻝ ﺍﻟﻠـﻪ ﺭﺍ ﻣﯽ ﺁﺯﺍﺭﻳﻢ؟ ﺁﻥ ﺣﻀـﺮﺕ ﻓﺮﻣﻮﺩ: «ﻣﮕﺮ ﻧﻤﯽ ﺩﺍﻧﻴـﺪ ﻛﻪ ﺍﻋﻤﺎﻝ ﺷـﻤﺎ ﺑـﻪ ﺁﻥ ﺣﻀـﺮﺕ ﮔﺰﺍﺭﺵ ﻣﯽ ﺷـﻮﺩ، ﭘـﺲ ﭼـﻮﻥ ﺩﺭ ﺁﻥ ﮔﺰﺍﺭﺵ ﻣﻌﺼﻴﺘﯽ ﻣﺸـﺎﻫﺪﻩ ﻛﻨﺪ ﺁﺯﺭﺩﻩ ﻣﯽ ﮔﺮﺩﺩ. ﺑﺮ ﺷﻤﺎ ﺑﺎﺩ ﻛﻪ ﺭﺳﻮﻝ ﺧﺪﺍ ﺭﺍ ﺁﺯﺍﺭ ﻧﺪﻫﻴﺪ، ﺑﻠﻜﻪ ﺷﺎﺩﺵ ﺳﺎﺯﻳﺪ.»(47)

ﻋﻘﻴـﺪﻩ ﺑﻪ ﺍﻳﻦﻛـﻪ ﭘﻴﺎﻣﺒﺮ ﺧﺪﺍ ﻭ ﺍﻣﺎﻡ ﻫﺮ ﺯﻣﺎﻥ ﺑﺮ ﺍﺣﻮﺍﻝ ﺷـﻴﻌﻴﺎﻥ ﺁﻥ ﻋﺼـﺮ ﺍﺣﺎﻃـﻪ ﺩﺍﺭﺩ، ﺍﺯ ﺭﻭﺍﻳـﺎﺕ ﻭ ﺍﺳـﻨﺎﺩ ﻣﺘﻌﺪﺩی ﺑﺮﮔﺮﻓﺘﻪ ﺷـﺪﻩ ﺍﺳـﺖ ﻭ ﭼﻨﺎﻧﭽﻪ ﮔﺬﺷـﺖ ﺍﻳـﻦ ﺍﺧﺒـﺎﺭ ﻭ ﺭﻭﺍﻳﺎﺕ ﻣﻮﺍﻓﻖ ﺁﻳـﺎﺕ ﻗﺮﺁﻥ ﺍﺳـﺖ.(48) ﻧﻤﻮﻧﻪ ﺍی ﺩﻳﮕﺮ ﺍﺯ ﺭﻭﺍﻳﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺧﺒﺮی ﺍﺳﺖ ﻛﻪ ﺍﺯ ﺟﻨﺎﺏ ﺍﺑﻮ ﺑﺼﻴﺮ ﻧﻘﻞ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻭ ﮔﻮﻳﺪ ﺷﻨﻴﺪﻡ ﻛﻪ ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﺑﻪ ﻣﺮﺩی ﺍﺯ ﺍﻫﻞ ﺁﻓﺮﻳﻘﺎ ﻣﯽ ﻓﺮﻣﻮﺩ: ﺣﺎﻝ ﺭﺍﺷـﺪ ﭼﻄﻮﺭ ﺍﺳﺖ؟ ﻣﺮﺩ ﮔﻔﺖ: ﺍﻭ ﺭﺍ ﺗﺮک ﮔﻔﺘﻢ ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﻫﻢ ﺍﺯ ﻟﺤﺎﻅ ﺟﺴـﻤﯽ ﻭ ﻫﻢ ﺍﺯ ﻟﺤﺎﻅ ﺩﻳﻨﯽ ﻭﺿﻌﻴﺖ ﺧﻮﺑﯽ ﺩﺍﺷـﺖ ﻭ ﺑﻪ ﺷـﻤﺎ ﺳـﻼﻡ ﻣﯽ ﺭﺳﺎﻧﺪ. ﺣﻀﺮﺕ ﺻـﺎﺩﻕ ﻓﺮﻣﻮﺩ: ﺧﺪﺍﻳﺶ ﺭﺣﻤﺖ ﻛﻨﺪ! ﻣـﺮﺩ ﮔﻔﺖ: ﻣﮕﺮ ﺍﻭ ﻣﺮﺩﻩ ﺍﺳـﺖ؟ ﺍﻣـﺎﻡ ﻓﺮﻣـﻮﺩ: ﺁﺭی. ﻣﺮﺩ ﻋـﺮﺽ ﻛﺮﺩ: ﭼﻪ ﻭﻗـﺖ؟ ﺍﻣﺎﻡ ﻓﺮﻣﻮﺩ: ﺩﻭ ﺭﻭﺯ ﭘﺲ ﺍﺯ ﺧﺎﺭﺝ ﺷـﺪﻥ ﺗﻮ. ﺁﻥ ﻣﺮﺩ ﮔﻔﺖ: ﺑﻪ ﺧﺪﺍ ﻗﺴـﻢ ﺍﻭ ﺑﻴﻤﺎﺭ ﻧﺒﻮﺩ ﻭ ﻫﻴﭻ ﻧﺎﺭﺍﺣﺘﯽ ﻫﻢ ﻧﺪﺍﺷـﺖ ﻭ ﺍﻧﺴـﺎﻥ ﺑﺮ ﺍﺛﺮ ﺑﻴﻤﺎﺭی ﻭ ﻧﺎﺭﺍﺣﺘﯽ ﺍﺳﺖ ﻛﻪ ﻣﯽ ﻣﻴﺮﺩ! ﻣﻦ (ﺍﺑﻮ ﺑﺼﻴﺮ) ﭘﺮﺳﻴﺪﻡ: ﺁﻥ ﻣﺮﺩ ﻛﻪ ﺑﻮﺩ؟ ﺍﻣﺎﻡ ﻓﺮﻣﻮﺩ: ﺍﻭ ﻳﻜﯽ ﺍﺯ ﺷـﻴﻌﻴﺎﻥ ﻣﺎ ﺑﻮﺩ ﻛﻪ ﻣﺎ ﺭﺍ ﺩﻭﺳـﺖ ﻣﯽ ﺩﺍﺷـﺖ. ﺳﭙﺲ ﻓﺮﻣﻮﺩ:

«ﺍﮔـﺮ ﭼﻨﻴـﻦ ﻓﻜـﺮ ﻛﻨﻴﺪ ﻛﻪ ﻣـﺎ ﻧﺴـﺒﺖ ﺑﻪ ﺷـﻤﺎ ﺩﻳﺪﮔﺎﻧـﯽ ﺑﻴﻨﺎ ﻭ ﮔﻮﺵﻫﺎﻳﯽ ﺷـﻨﻮﺍ ﻧﺪﺍﺭﻳﻢ، ﺑﺪ ﺗﺼﻮﺭ ﻛﺮﺩﻩ ﺍﻳﺪ. ﻗﺴـﻢ ﺑﻪ ﺧﺪﺍ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺍﻋﻤﺎﻝ ﺷﻤﺎ ﺑﺮ ﻣﺎ ﻣﺨﻔﯽ ﻧﻴﺴﺖ. ﭘﺲ ﻣﺎ ﺭﺍ ﻧﺰﺩ ﺧﻮﺩ ﺣﺎﺿﺮ ﺑﺪﺍﻧﻴﺪ ﻭ ﺧﻮﻳﺸﺘﻦ ﺭﺍ ﺑﻪ ﻛﺎﺭﻫﺎی ﺧﻮﺏ ﻋﺎﺩﺕ ﺩﻫﻴﺪ ﻭ ﺍﺯ ﺍﻫﻞ ﺧﻴﺮ ﺑﺎﺷﻴﺪ ﺗﺎ ﺑﻪ ﺍﻳﻦ ﺻﻔﺖ ﺷـﻨﺎﺧﺘﻪ ﺷـﻮﻳﺪ؛ ﻛﻪ ﻣﻦ ﻓﺮﺯﻧﺪﺍﻥ ﻭ ﺷﻴﻌﻴﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﺍﻣﺮ ﺳﻔﺎﺭﺵ ﻣﯽ ﻛﻨﻢ.»(49)

ﺭﻭﺍﻳـﺖ ﺩﻳﮕـﺮی ﻛـﻪ ﻣﺆﻳﺪ ﺍﻳﻦ ﻋﻘﻴﺪﻩ ﺍﺳـﺖ، ﺭﻭﺍﻳﺘﯽ ﺍﺳـﺖ ﻛﻪ ﺍﺯ ﺍﻣـﺎﻡ ﻣﻬﺪی ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﺍﺳـﺖ. ﺁﻥ ﺣﻀـﺮﺕ ﺗﺼﺮﻳﺢ ﻣﯽ ﻓﺮﻣﺎﻳﻨـﺪ ﻛـﻪ: «ﻣﺎ ﺑﺮ ﺍﺧﺒﺎﺭ ﻭ ﺍﺣﻮﺍﻝ ﺷـﻤﺎ ﺁﮔﺎﻫﻴـﻢ ﻭ ﻫﻴﭻ ﭼﻴﺰ ﺍﺯ ﺍﻭﺿﺎﻉ ﺷﻤﺎ ﺑﺮ ﻣﺎ ﭘﻮﺷﻴﺪﻩ ﻭ ﻣﺨﻔﯽ ﻧﻴﺴﺖ.»

ﺑﻪ ﺭﺍﺳـﺘﯽ ﺍﮔﺮ ﻣﺎ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﺤﻀﺮ ﻣﻮﻻﻳﻤﺎﻥ ﺑﺪﺍﻧﻴﻢ ﻭ ﺁﮔﺎﻫﯽ ﻭ ﻧﻈﺎﺭﺗـﺶ ﺭﺍ ﺑـﺮ ﺗﮏﺗﮏ ﺍﻋﻤﺎﻟﻤـﺎﻥ ﺣﺲ ﻛﻨﻴﻢ، ﻫﻤـﺎﻥ ﻛﺎﺭﻫﺎﻳﯽ ﺭﺍ ﻣﯽ ﻛﻨﻴﻢ ﻛﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻣﯽ ﻛﺮﺩﻳﻢ؟

ﺑﻴﺎﻳﻴـﻢ ﻫﻤـﻮﺍﺭﻩ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﺤﻀﺮ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺑﺒﻴﻨﻴﻢ ﻭ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻛﺎﺭﻫﺎی ﺧﻮﺏ ﻋﺎﺩﺕ ﺩﻫﻴﻢ. ﺑﻴﺎﻳﻴﻢ ﺑﺎ ﺷـﺎﺩ ﺳﺎﺧﺘﻦ ﺍﻣﺎﻣﻤﺎﻥ، ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺍﻭ ﺭﺍ ﻳﺎﺭی ﻛﻨﻴﻢ.
