بازگشت

ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺯ ﻃﺮﻳﻖ ﻳﺎﺭی ﺷﻴﻌﻴﺎﻥ ﺍﻳﺸﺎﻥ

ﺍﺯ ﺁﻧﺠـﺎ ﻛـﻪ ﺷـﻴﻌﻴﺎﻥ ﻫﺮ ﺯﻣـﺎﻥ ﺩﺭ ﺣﻜـﻢ ﻓﺮﺯﻧـﺪﺍﻥ ﺍﻣـﺎﻡ ﺁﻥ ﺯﻣﺎﻧﻨﺪ، ﻭ ﺍﺯ ﺁﻧﺠـﺎ ﻛـﻪ ﭘﺪﺭﺍﻥ ﺑﻪ ﺷـﺎﺩی ﻓﺮﺯﻧﺪﺍﻥ ﺷـﺎﺩ ﻣﯽ ﺷـﻮﻧﺪ ﻭ ﺑـﻪ ﺍﻧﺪﻭﻩ ﺁﻧﻬـﺎ ﺍﻧﺪﻭﻫﮕﻴـﻦ ﻣﯽ ﮔﺮﺩﻧﺪ، ﻣﯽ ﺗﻮﺍﻥ ﻳﺎﺭی ﺷـﻴﻌﻴﺎﻥ ﻭ ﺩﺳـﺘﮕﻴﺮی ﺍﺯ ﺁﻧﺎﻥ ﺭﺍ ﻫﻤﺎﻥ ﻳﺎﺭی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭ ﺩﺳـﺘﮕﻴﺮی ﺍﺯ ﺍﻳﺸــﺎﻥ ﺩﺍﻧﺴـﺖ.

ﺍﻳﻦ ﻧﻜﺘﻪ، ﻧﻜﺘﻪ ﺍی ﺑﺴـﻴﺎﺭ ﻣﻬﻢ ﺍﺳـﺖ ﻛﻪ ﺩﺭ ﺗﻌﺎﻟﻴﻢ ﺷﻴﻌﯽ ﺗﻮﺟﻪ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺪﺍﻥ ﺷـﺪﻩ ﻭ ﺩﺭ ﻣﻴﺎﻥ ﻣﺎ ﻏﻔﻠﺖ ﺯﻳﺎﺩی ﻧﺴـﺒﺖ ﺑﻪ ﺁﻥ ﺑﻪ ﭼﺸﻢ ﻣﯽ ﺧﻮﺭﺩ.

ﺧﺪﻣﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻭ ﻳﺎﺭی ﺍﻳﺸﺎﻥ ﮔﺎﻫﯽ ﻣﺴﺘﻘﻴﻤﺎ ﻣﺘﻮﺟﻪ ﺧﻮﺩ ﺁﻥ ﺣﻀـﺮﺕ ﺍﺳـﺖ، ﺍﻣـﺎ ﮔﺎﻩ ﺧﺪﻣﺖ ﺑـﻪ ﺁﻥ ﺣﻀﺮﺕ ﻭ ﻳـﺎﻭﺭی ﺍﻭ ﺑﻪ ﻃﻮﺭ ﻏﻴﺮ ﻣﺴـﺘﻘﻴﻢ ﻭ ﺩﺭ ﻗﺎﻟﺐ ﺧﺪﻣﺖ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﻭ ﺩﻭﺳﺘﺎﻥ ﺍﻭ ﻣﻴﺴﺮ ﻣﯽ ﮔﺮﺩﺩ.

ﺍﮔـﺮ ﺑﺪﺍﻧﻴـﻢ ﻛـﻪ ﺟﺎﻣﻌﻪ ﺷـﻴﻌﻪ ﺧﺎﻧـﻮﺍﺩﻩ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻣﺤﺴـﻮﺏ ﻣﯽ ﺷـﻮﺩ، ﺩﺭﺳـﺘﯽ ﺍﻳﻦ ﻣﺪﻋـﺎ ﺑﺮ ﻣﺎ ﺭﻭﺷـﻦ ﺧﻮﺍﻫﺪ ﮔﺸـﺖ ﻭ ﺍﮔﺮ ﺍﺯ ﺗﺄﻛﻴـﺪ ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢ ﺍﻟﺴـﻼﻡ ﺩﺭﺑـﺎﺭﻩ ﺧﺪﻣﺖ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﻭ ﺗﻜﺮﻳﻢ ﺁﻧﺎﻥ ﺁﮔﺎﻩ ﺑﺎﺷﻴﻢ، ﺑﻪ ﺍﻳﻦ ﻛﺎﺭ ﺑﻴﺶ ﺍﺯ ﭘﻴﺶ ﻣﺸﺘﺎﻕ ﺧﻮﺍﻫﻴﻢ ﺷﺪ.

ﭘﺲ ﺑﺎ ﺷـﻤﺎ ﻣﯽ ﮔﻮﻳﻢ: ﺍی ﻛﺴﺎﻧﯽ ﻛﻪ ﺩﻭﺳـﺖ ﺩﺍﺭﻳﺪ ﺗﺎ ﺩﺭ ﺷﻤﺎﺭ ﻳـﺎﺭﺍﻥ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻣﺤﺴـﻮﺏ ﺷـﻮﻳﺪ، ﺍﻳﻨـﮏ ﺭﺍﻩ ﺁﻥ ﭘﻴﺶ ﺭﻭی ﺷـﻤﺎ ﺑﺎﺯ ﺍﺳﺖ. ﺑﻪ ﺳﺮﺍﻍ ﺷﻴﻌﻴﺎﻥ ﺁﻥ ﺣﻀﺮﺕ ﺑﺮﻭﻳﺪ ﻭ ﺑﻪ ﺧﺎﻃﺮ ﺍﻧﺘﺴـﺎﺏ ﺁﻧﺎﻥ ﺑـﻪ ﺍﻣﺎﻡ ﻋﺼـﺮ ﺍﺯ ﺁﻧﻬﺎ ﺩﻟﺠﻮﻳـﯽ ﻛﻨﻴﺪ. ﻓﻘﺮﺍی ﺷـﻴﻌﻪ ﺭﺍ ﺩﺭ ﺍﻣـﻮﺍﻝ ﺧﻮﺩ ﺷـﺮﻳﮏ ﺳـﺎﺯﻳﺪ. ﺑﻴﻤﺎﺭﺍﻥ ﺷـﻴﻌﻪ ﺭﺍ ﻋﻴـﺎﺩﺕ ﻛﻨﻴﺪ. ﺑﻪ ﺩﺭﻣﺎﻧﺪﮔﺎﻥ ﺷـﻴﻌﻪ ﺭﺣﻢ ﺁﻭﺭﻳﺪ. ﺣﻘﻮﻕ ﺑﺮ ﺑﺎﺩ ﺭﻓﺘﻪ ﺷـﻴﻌﻴﺎﻥ ﻣﻈﻠﻮﻡ ﺭﺍ ﺍﺯ ﻇﺎﻟﻤـﺎﻥ ﻣﻄﺎﻟﺒﻪ ﻛﻨﻴﺪ. ﺍﺯ ﺷـﻴﻌﻴﺎﻧﯽ ﻛﻪ ﻣـﻮﺭﺩ ﺗﻌﺮﺽ ﻧﺎﺟﻮﺍﻧﻤﺮﺩﺍﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ، ﺟﻮﺍﻧﻤﺮﺩﺍﻧﻪ ﺩﻓﺎﻉ ﻛﻨﻴﺪ ﻭ ﺑﺪﺍﻧﻴﺪ ﻛﻪ ﻳﺎﺭی ﺷﻴﻌﻴﺎﻥ ﺍﻣﺎﻡ ﺯﻣﺎﻥ، ﻳﺎﺭی ﺧﻮﺩ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ.

ﺍی ﻛﺴـﺎﻧﯽ ﻛﻪ ﻣﯽ ﺧﻮﺍﻫﻴـﺪ ﺻـﺪﻕ ﻭ ﺻﻔـﺎی ﺧـﻮﺩ ﺭﺍ ﺩﺭ ﻳﺎﺭی ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﻋﻴـﺎﻥ ﺳـﺎﺯﻳﺪ! ﺑﺪﺍﻧﻴﺪ ﻭﻗﺘـﯽ ﺷـﻴﻌﻪ ﺍی ﮔﺮﻓﺘﺎﺭ ﺑﻪ ﺳـﻮی ﺷـﻤﺎ ﻣﯽ ﺁﻳـﺪ، ﺩﺭ ﺣﻘﻴﻘﺖ ﺑـﺎﺏ ﺗﻮﻓﻴﻘﯽ ﺍﺯ ﺟﺎﻧﺐ ﺧﺪﺍ ﺑﻪ ﺭﻭی ﺷـﻤﺎ ﮔﺸـﻮﺩﻩ ﺷـﺪﻩ ﺍﺳـﺖ. ﺍﺯ ﺗﻘﺎﺿﺎ ﻭ ﺍﺳـﺘﻤﺪﺍﺩ ﻭ ﻣﺮﺍﺟﻌﻪ ﺍی ﻛﻪ ﺑﻪ ﺷﻤﺎ ﻛﺮﺩﻩ ﻧﺎﺭﺍﺣﺖ ﻧﺸﻮﻳﺪ ﻭ ﭼﻬﺮﻩ ﺩﺭﻫﻢ ﻧﺸﻜﻨﻴﺪ، ﺑﻠﻜﻪ ﺧﻮﺷﺤﺎﻝ ﺷﻮﻳﺪ ﻭ ﺧـﺪﺍی ﺭﺍ ﺍﺯ ﺑﺎﺑـﺖ ﺍﻳـﻦ ﺗﻮﻓﻴﻖ ﺳـﭙﺎﺱ ﮔﻮﻳﻴﺪ. ﺑﺎ ﺧـﻮﺩ ﺑﮕﻮﻳﻴﺪ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻣﻦ ﻣﺮﺍ ﻻﻳﻖ ﺩﺍﻧﺴـﺘﻪ ﻛﻪ ﺑـﺮﺍی ﻳﻜﯽ ﺍﺯ ﺍﻋﻀﺎی ﺧﺎﻧﻮﺍﺩﻩ ﺍﻭ ﻛﺎﺭی ﺑﻜﻨـﻢ ﻭ ﭼـﻪ ﺗﻮﻓﻴﻘﯽ ﺍﺯ ﺍﻳﻦ ﺑﺎﻻﺗﺮ! ﺑﺪﺍﻧﻴﺪ ﻛـﻪ ﺍﮔﺮ ﺍﺯ ﻣﺎﻝ ﺧﻮﺩ ﺑـﻪ ﺍﻭ ﻗﺮﺽ ﺩﻫﻴﺪ، ﮔﻮﻳﺎ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺘﺎﻥ ﻗﺮﺽ ﺩﺍﺩﻩ ﺍﻳﺪ، ﺑﻠﻜﻪ ﮔﻮﻳﺎ ﺑﻪ ﺧﺪﺍﻭﻧﺪ ﻗﺮﺽ ﺩﺍﺩﻩ ﺍﻳﺪ.

ﺍﻳﻦ ﻧﺪﺍی ﺧﺪﺍﻭﻧﺪ ﺍﺳـﺖ ﻛﻪ ﻣﯽ ﻓﺮﻣﺎﻳﺪ: «ﻛﻴﺴﺖ ﻛﻪ ﺑﻪ ﺧﺪﺍ ﻭﺍﻡ ﻧﻴﻜﻮ ﺩﻫﺪ ﺗﺎ ﺧﺪﺍﻭﻧﺪ ﭘﺎﺳﺦ ﺍﻭ ﺭﺍ ﭼﻨﺪﻳﻦ ﺑﺮﺍﺑﺮ ﻋﻨﺎﻳﺖ ﻛﻨﺪ؟»(59)

ﺧﺪﺍﻭﻧﺪ ﺑﻪ ﺧﻮﺩی ﺧﻮﺩ ﺑﻪ ﻣﺎ ﻧﻴﺎﺯی ﻧﺪﺍﺭﺩ ﻭ ﻗﺮﺽ ﺩﺍﺩﻥ ﺑﻪ ﺧﺪﺍ ﺟﺰ ﺑﻪ ﻣﻌﻨﺎی ﺩﺳﺘﮕﻴﺮی ﺍﺯ ﻛﺴﺎﻧﯽ ﻛﻪ ﺑﻪ ﺧﺪﺍ ﻣﺘﺼﻞﺍﻧﺪ، ﻧﻴﺴﺖ. ﭘﻴﺎﻣﺒﺮ ﻭ ﺑﻪ ﺗﺒﻊ ﺍﻭ ﺍﻫﻞﺑﻴﺖ ﭘﻴﺎﻣﺒﺮ ﻭ ﺑﻪ ﺗﺒﻊ ﺍﻳﺸـﺎﻥ ﺷـﻴﻌﻴﺎﻥ ﻭ ﺩﻭﺳﺘﺎﻥ ﺁﻧﻬﺎ ﺑﻪ ﺧـﺪﺍ ﻣﺘﺼﻞﺍﻧﺪ. ﻭﺍی ﺑﺮ ﻣﺎ ﺍﮔﺮ ﺑﺎﺏ ﺧﺪﻣﺖ ﺑﻪ ﺩﻭﺳـﺘﯽ ﺍﺯ ﺩﻭﺳـﺘﺎﻥ ﺧـﺪﺍ ﺑﺮ ﻣﺎ ﮔﺸـﻮﺩﻩ ﺷـﻮﺩ ﻭ ﺧـﻮﺩ ﺍﺯ ﺁﻥ ﺭﻭی ﺑﮕﺮﺩﺍﻧﻴﻢ، ﻭ ﺧﻮﺷـﺎ ﺑﻪ ﺣـﺎﻝ ﻣـﺎ ﺍﮔﺮ ﺍﺯ ﺍﻳﻦ ﺗﻮﻓﻴﻖ ﺍﺳـﺘﻘﺒﺎﻝ ﻛﻨﻴـﻢ ﻭ ﺑﻪ ﻳﺎﺭی ﺍﻳﻨﺎﻥ ﺑﺸـﺘﺎﺑﻴﻢ. ﺑﺪﺍﻧﻴﺪ ﻛﻪ ﺧﺪﺍ ﺧﻮﺩ ﺑﺮ ﺭﻓﻊ ﻧﻴﺎﺯ ﺩﻭﺳـﺘﺎﻥ ﺧﻮﻳﺶ ﺗﻮﺍﻧﺎﺳـﺖ ﻭ ﺩﺭ ﺍﻳﻦ ﻛﺎﺭ ﻧﻴﺎﺯﻣﻨـﺪ ﻣﺎ ﻧﻴﺴـﺖ. ﺑﺪﺍﻧﻴﻢ ﺍﮔﺮ ﺍﻭ ﺯﻣﻴﻨﻪ ﺍﻳﻦﮔﻮﻧـﻪ ﻛﺎﺭﻫﺎ ﺭﺍ ﺑﺮﺍی ﻣﺎ ﻓﺮﺍﻫﻢ ﻣﯽ ﺁﻭﺭﺩ، ﺑﻨﺎ ﺩﺍﺭﺩ ﻛﻪ ﻣﺎ ﺭﺍ ﺩﺭ ﺧﺪﻣﺖ ﺑﻪ ﺍﻭﻟﻴﺎ ﻭ ﺩﻭﺳـﺘﺎﻧﺶ ﺑﻠﻜـﻪ ﺧﺪﻣﺖ ﺑﻪ ﺧـﻮﺩﺵ ﺗﻮﻓﻴﻘﯽ ﺩﻫﺪ. ﺍﮔﺮ ﺑﻪ ﺍﻳـﻦ ﻧﻜﺘﻪ ﻫﺎی ﻇﺮﻳﻒ ﺩﻗﺖ ﻛﻨﻴﻢ ﻧﻪ ﺩﺭ ﻛﺎﺭ ﺧﻴﺮ ﻭ ﺧﺪﻣﺖ ﺑﻪ ﺷﻴﻌﻴﺎﻥ ﺳﺴﺘﯽ ﺧﻮﺍﻫﻴﻢ ﻛﺮﺩ، ﻧﻪ ﺍﻳﻦ ﺧﺪﻣﺖ ﺭﺍ ﺑﺎ ﺍﻛﺮﺍﻩ ﺍﻧﺠﺎﻡ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ ﻭ ﻧﻪ ﭘﺲ ﺍﺯ ﻣﻮﻓﻖ ﺷـﺪﻥ ﺑﻪ ﺧﺪﻣﺖ، ﻣﻨ ﺘﯽ ﺑﺮ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺧﻮﺍﻫﻴﻢ ﮔﺬﺍﺷـﺖ. ﺍﮔﺮ ﻛﺴـﯽ ﺷﺎﻳﺴﺘﻪ ﻣﻨ ﺖ ﮔﺬﺍﺷـﺘﻦ ﺑﺎﺷـﺪ، ﺧﺪﺍﺳـﺖ. ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﺑﻪ ﻣﺎ ﻧﻴﺎﺯﻣﻨﺪ ﻧﺒـﻮﺩﻩ، ﺗﻮﻓﻴـﻖ ﺍﻧﺠـﺎﻡ ﺍﻳﻦ ﺧﺪﻣـﺖ ﺭﺍ ﺑﻪ ﻣﺎ ﺩﺍﺩﻩ ﺍﺳـﺖ. ﭘﺲ ﺍﻭﺳـﺖ ﺷﺎﻳﺴﺘﻪ ﻣﻨﺖ ﻭ ﺍﻭﺳﺖ ﺳﺰﺍﻭﺍﺭ ﺳﺘﺎﻳﺶ.

ﺩﺭ ﺍﻳﻨﺠـﺎ ﺑـﺮﺍی ﺁﻧﻜﻪ ﺍﺯ ﺍﻫﻤﻴﺖ ﺧﺪﻣﺖﮔﺰﺍﺭی ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﺑﻴﺸـﺘﺮ ﺁﮔﺎﻩ ﺷـﻮﻳﻢ ﻭ ﺑﺮﺍﺑﺮ ﺑﻮﺩﻥ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑـﺎ ﺧﺪﻣﺖ ﺑﻪ ﺧﻮﺩ ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭﻳﺎﺑﻴﻢ، ﺑﻪ ﮔﻨﺠﻴﻨﻪ ﺭﻭﺍﻳﺎﺕ ﻭ ﻛﻠﻤﺎﺕ ﺧﻮﺩ ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭﺍﻥ ﺳـﺮ ﻣﯽ ﻛﺸـﻴﻢ ﻭ ﺩﺭ ﺁﺋﻴﻨﻪ ﻛﻠﻤﺎﺕ ﺍﻳﺸـﺎﻥ، ﺭﺳـﻢ ﻫـﻮﺍﺩﺍﺭی، ﻳﺎﺭی ﻭ ﺧﺪﻣﺖﮔﺰﺍﺭی ﺭﺍ ﻣﯽ ﺁﻣﻮﺯﻳﻢ.

ﺭﻭﺍﻳﺖ ﺍﻭﻝ:

ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩﻩ ﺍﻧﺪ: «ﻛﺴﯽ ﺍﺯ ﺷﻤﺎ ﻧﭙﻨﺪﺍﺭﺩ ﻛﻪ ﺍﮔﺮ ﻣﺆﻣﻨﯽ ﺭﺍ ﺷـﺎﺩ ﻭ ﻣﺴـﺮﻭﺭ ﻧﻤﺎﻳﺪ، ﺗﻨﻬﺎ ﺍﻭ ﺭﺍ ﺷﺎﺩﻣﺎﻥ ﻛﺮﺩﻩ ﺍﺳﺖ، ﺑﻠﻜﻪ ﻗﺴﻢ ﺑﻪ ﺧﺪﺍ ﻣﺎ ﺭﺍ ﺧﻮﺷﺤﺎﻝ ﺳﺎﺧﺘﻪ، ﺑﻠﻜﻪ ﺳﻮﮔﻨﺪ ﺑﻪ ﺧﺪﺍ ﺭﺳﻮﻝ ﺧﺪﺍ (ﺻﻠﯽ ﺍﻟﻠﻪ ﻋﻠﻴﻪ ﻭ ﺁﻟﻪ) ﺭﺍ ﺧﻮﺷﺤﺎﻝ ﻧﻤﻮﺩﻩ ﺍﺳﺖ.»(60)

ﺭﻭﺍﻳﺖ ﺩﻭﻡ:

ﺍﻣـﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩﻧﺪ: «ﻫﺮﻛـﺲ ﻧﺘﻮﺍﻧﺪ ﻣﺎ ﺭﺍ ﺻﻠﻪ ﻧﻤﺎﻳـﺪ، ﺑﺎﻳﺪ ﺩﻭﺳـﺘﺎﻥ ﺻﺎﻟﺢ ﻣـﺎ ﺭﺍ ﺻﻠﻪ ﻛﻨﺪ ﻛﻪ ﺛـﻮﺍﺏ ﺻﻠﻪ ﻣﺎ ﺑﺮﺍی ﺍﻭ ﻧﻮﺷـﺘﻪ ﻣﯽ ﺷـﻮﺩ، ﻭ ﻫﺮﻛـﺲ ﻧﺘﻮﺍﻧﺪ ﻣـﺎ ﺭﺍ ﺯﻳﺎﺭﺕ ﻛﻨـﺪ، ﺩﺭ ﻋﻮﺽ ﺑـﻪ ﺩﻳﺪﺍﺭ ﺍﻓﺮﺍﺩ ﺻﺎﻟﺢ ﺍﺯ ﺩﻭﺳـﺘﺎﻥ ﻣﺎ ﺑﺮﻭﺩ ﻛﻪ ﺛـﻮﺍﺏ ﺩﻳﺪﺍﺭ ﻣﺎ ﺑﺮﺍﻳﺶ ﻧﻮﺷﺘﻪ ﻣﯽ ﺷﻮﺩ.»(61)

ﺗﻮﺿﻴﺢ: ﺻﻠﻪ ﻛﺮﺩﻥ ﺑﻪ ﻣﻌﻨﯽ ﺣﻔﻆ ﺍﺭﺗﺒﺎﻁ ﻣﺤﺒﺖﺁﻣﻴﺰ ﺑﺎ ﻛﺴـﯽ ﺑﻪ ﻭﺳﻴﻠﻪ ﻫﺪﻳﻪ ﺩﺍﺩﻥ ﻳﺎ ﺍﺭﺍﺋﻪ ﻛﻤﮏ ﻣﺎﻟﯽ ﺑﻪ ﺍﻭﺳﺖ.

ﺭﻭﺍﻳﺖ ﺳﻮﻡ:

ﺍﻣﺎﻡ ﺻﺎﺩﻕ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻓﺮﻣﻮﺩﻧﺪ: «ﺻﻠﻪ ﺁﻝ ﻣﺤﻤﺪ - ﺻﻠﻮﺍﺕ ﺍﻟﻠﻪ ﻋﻠﻴﻬﻢ ﺍﺟﻤﻌﻴﻦ- ﺭﺍ ﺍﺯ ﺍﻣﻮﺍﻟﺘﺎﻥ ﻭﺍﻣﮕﺬﺍﺭﻳﺪ! ﻫﺮﻛﺲ ﻏﻨﯽ ﺍﺳـﺖ ﺑـﻪ ﻣﻘـﺪﺍﺭ ﺗﻮﺍﻧﮕـﺮی ﺍﺵ، ﻭ ﻫﺮﻛﺲ ﻓﻘﻴﺮ ﺍﺳـﺖ ﺑﻪ ﻧﺴـﺒﺖ ﻓﻘﺮﺵ ﺑﻪ ﺍﻳـﻦ ﻛﺎﺭ ﺍﻗـﺪﺍﻡ ﻛﻨﺪ. ﻫﺮﺁﻧﻜﻪ ﻣﯽ ﺧﻮﺍﻫﺪ ﺧﺪﺍﻭﻧـﺪ ﻣﻬﻢ ﺗﺮﻳﻦ ﺣﺎﺟﺘﺶ ﺭﺍ ﻧﺴـﺒﺖ ﺑـﻪ ﺍﻭ ﺑـﺮﺁﻭﺭﺩ؛ ﺑﺎﻳـﺪ ﻛﻪ ﺁﻝ ﻣﺤﻤـﺪ ﺻﻠﻮﺍﺕ ﺍﻟﻠـﻪ ﻋﻠﻴﻬﻢ ﺍﺟﻤﻌﻴـﻦ ﻭ ﺷﻴﻌﻴﺎﻧﺸـﺎﻥ ﺭﺍ ﺑﻪ ﺁﻧﭽﻪ ﺑﺴـﻴﺎﺭ ﺑـﻪ ﺁﻥ ﻧﻴـﺎﺯ ﺩﺍﺭﻧﺪ، ﺻﻠﻪ ﻧﻤﺎﻳﺪ.»(62)

ﺭﻭﺍﻳﺖ ﭼﻬﺎﺭﻡ:

ﺍﻣﺎﻡ ﻣﻮﺳﯽ ﺑﻦ ﺟﻌﻔﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻓﺮﻣﻮﺩﻧﺪ:

«ﻛﺴـﯽ ﻛﻪ ﻧﻤﯽ ﺗﻮﺍﻧـﺪ ﻣﺎ ﺭﺍ ﺯﻳـﺎﺭﺕ ﻧﻤﺎﻳﺪ، ﭘﺲ ﭘﻴـﺮﻭﺍﻥ ﺻﺎﻟﺢ ﻣـﺎ ﺭﺍ ﺩﻳﺪﺍﺭ ﻛﻨﺪ، ﺗﺎ ﺛﻮﺍﺏ ﺯﻳﺎﺭﺕ ﻣﺎ ﺑﺮﺍﻳﺶ ﻧﻮﺷـﺘﻪ ﺷـﻮﺩ، ﻭ ﻫﺮﻛﺲ ﻧﻤﯽ ﺗﻮﺍﻧـﺪ ﻣﺎ ﺭﺍ ﺻﻠﻪ ﻛﻨﺪ، ﭘﺲ ﭘﻴﺮﻭﺍﻥ ﻧﻴﻜـﻮﻛﺎﺭ ﻣﺎ ﺭﺍ ﺻﻠﻪ ﻧﻤﺎﻳﺪ، ﺗﺎ ﺑﺮﺍی ﺍﻭ ﺛﻮﺍﺏ ﺻﻠﻪ ﻣﺎ ﻧﻮﺷﺘﻪ ﺷﻮﺩ.»(63)

ﺭﻭﺍﻳﺖ ﭘﻨﺠﻢ:

ﺍﺯ ﺍﻣـﺎﻡ ﻛﺎﻇﻢ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﻛـﻪ ﻓﺮﻣﻮﺩﻧـﺪ: «ﻫﺮﻛﺲ ﺣﺎﺟﺘﯽ ﺍﺯ ﺣﺎﺟﺎﺕ ﺩﻭﺳﺘﺎﻥ ﻣﺎ ﺭﺍ ﺑﺮﺁﻭﺭﺩ، ﮔﻮﻳﯽ ﺁﻥ ﺣﺎﺟﺖ ﺭﺍ ﺑﺮﺍی ﻫﻤﻪ ﻣﺎ ﺑﺮﺁﻭﺭﺩﻩ ﺍﺳﺖ.»(64)

ﺭﻭﺍﻳﺖ ﺷﺸﻢ:

ﺍﺯ ﺣﻀـﺮﺕ ﺍﺑﻮﺍﻟﺤﺴـﻦ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﻘﻞ ﺷـﺪﻩ ﻛـﻪ ﻓﺮﻣﻮﺩﻧﺪ:

«ﻫﺮﻛـﺲ ﻛـﻪ ﺑـﺮﺍﺩﺭ ﻣﺆﻣﻨﺶ ﺑـﺮﺍی ﺣﺎﺟﺘﯽ ﻧـﺰﺩ ﻭی ﺁﻳﺪ، ﺑﺎﻳـﺪ ﺍﻭ ﺭﺍ ﺭﺣﻤﺘـﯽ ﺑﺪﺍﻧﺪ ﻛـﻪ ﺧﺪﺍی ﺗﺒﺎﺭک ﻭ ﺗﻌﺎﻟﯽ ﺑﺮﺍﻳﺶ ﭘﻴﺶ ﺁﻭﺭﺩﻩ ﺍﺳـﺖ. ﭘـﺲ ﭼﻨﺎﻧﭽـﻪ ﺁﻥ ﺭﺍ ﺑﭙﺬﻳـﺮﺩ ﺧـﻮﺩ ﺭﺍ ﺑﻪ ﺭﺷـﺘﻪ ﻭﻻﻳﺖ ﻣـﺎ ﭘﻴﻮﻧﺪ ﺩﺍﺩﻩ ﺍﺳـﺖ، ﻭ ﺍﻟﺒﺘﻪ ﺭﺷـﺘﻪ ﻭﻻﻳﺖ ﻣﺎ ﺑﻪ ﺭﻳﺴـﻤﺎﻥ ﻭﻻﻳـﺖ ﺧﺪﺍﻭﻧﺪ ﻣﺘﺼﻞ ﺍﺳـﺖ. ﻭ ﺍﮔﺮ ﺍﻭ ﺭﺍ ﺍﺯ ﺑﺮﺁﻭﺭﺩﻥ ﺣﺎﺟﺘﺶ ﻣﺤﺮﻭﻡ ﻛﻨﺪ ﺩﺭ ﺻﻮﺭﺗﯽ ﻛﻪ ﻣﯽ ﺗﻮﺍﻧﺴﺘﻪ ﺣﺎﺟﺘﺶ ﺭﺍ ﺭﻭﺍ ﺳﺎﺯﺩ، ﺧﺪﺍﻭﻧﺪ ﺩﺭ ﻗﺒﺮﺵ ﻣﺎﺭی ﺍﺯ ﺟﻨﺲ ﺁﺗﺶ ﺑﺮ ﺍﻭ ﻣﺴﻠﻂ ﺧﻮﺍﻫﺪ ﻛﺮﺩ ﻛﻪ ﺗﺎ ﺭﻭﺯ ﻗﻴﺎﻣﺖ ﺍﻭ ﺭﺍ ﺑﮕﺰﺩ.»(65)

ﺭﻭﺍﻳﺖ ﻫﻔﺘﻢ:

ﺟﺎﺑـﺮ ﺟﻌﻔـﯽ ﻛﻪ ﺍﺯ ﻳـﺎﺭﺍﻥ ﺧـﺎﺹ ﺍﻣﺎﻡ ﺑﺎﻗـﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻮﺩ ﮔﻮﻳـﺪ: ﻣـﺎ ﺟﻤﻌﯽ ﺑﻮﺩﻳـﻢ ﻛﻪ ﺑﺮ ﺍﻣﺎﻡ ﺑﺎﻗﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭﺍﺭﺩ ﺷـﺪﻳﻢ. ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ ﺍﺯ ﻣﺮﺍﺳـﻢ ﺣﺞ ﻭ ﻋﺒﺎﺩﺕ ﺧﻮﺩ ﻓﺎﺭﻍ ﺷـﺪﻩ ﺑﻮﺩﻳﻢ. ﭘﺲ ﺑﻪ ﻫﻨﮕﺎﻡ ﻭﺩﺍﻉ ﺑﺎ ﺁﻥ ﺣﻀﺮﺕ ﻋﺮﺿﻪ ﺩﺍﺷـﺘﻴﻢ: ﺍی ﺯﺍﺩﻩ ﺭﺳﻮﻝ ﺧﺪﺍ ﻣﺎ ﺭﺍ ﺑﻪ ﭼﻴﺰی ﺳﻔﺎﺭﺵ ﻓﺮﻣﺎی؟ ﺁﻥ ﺣﻀﺮﺕ ﻓﺮﻣﻮﺩ:

«ﺑﺎﻳﺪ ﻛﻪ ﻗﺪﺭﺗﻤﻨﺪﺍﻥ ﺷﻤﺎ، ﺿﻌﻴﻔﺎﻥ ﺭﺍ ﻳﺎﺭی ﺩﻫﻨﺪ، ﻭ ﺛﺮﻭﺗﻤﻨﺪﺍﻥ ﺑـﺮ ﻓﻘﻴﺮﺍﻥ ﻋﻄﻮﻓـﺖ ﻧﻤﺎﻳﻨﺪ، ﻭ ﻫﺮﻛﺲ ﺧﻴﺮﺧﻮﺍﻩ ﺑﺮﺍﺩﺭ ﺩﻳﻨﻴﺶ ﺑﺎﺷـﺪ ﭼﻨﺎﻧﻜﻪ ﺑﺮﺍی ﺧﻮﺩﺵ ﺧﻴﺮﺧﻮﺍﻩ ﺍﺳﺖ، ﻭ ﺃﺳﺮﺍﺭ ﻣﺎ ﺭﺍ ﻛﺘﻤﺎﻥ ﺳﺎﺯﻳﺪ ﻭ ﻣﺮﺩﻡ ﺭﺍ ﺑﺮ ﻣﺎ ﻣﺸـﻮﺭﺍﻧﻴﺪ، ﻭ ﺩﺭ ﺃﻣﺮ ﻣﺎ ﺩﻗﺖ ﻛﻨﻴﺪ ﻭ ﺁﻧﭽﻪ ﺍﺯ ﻣﺎ ﺑﺮﺍﻳﺘﺎﻥ ﻧﻘﻞ ﺷـﺪﻩ ﺭﺍ ﺑﻨﮕﺮﻳﺪ، ﭼﻨﺎﻧﭽﻪ ﺩﻳﺪﻳﺪ ﺑـﺎ ﺁﻧﭽﻪ ﻛﻪ ﺩﺭ ﻗﺮﺁﻥ ﺁﻣﺪﻩ ﻣﻮﺍﻓﻖ ﺍﺳﺖ ﺁﻥ ﺭﺍ ﺑﭙﺬﻳﺮﻳﺪ، ﻭ ﺍﮔﺮ ﺁﻥ ﺭﺍ ﻣﺨﺎﻟﻒ ﻗﺮﺁﻥ ﻳﺎﻓﺘﻴﺪ ﺁﻥ ﺭﺍ ﺭﺩ ﻛﻨﻴﺪ، ﻭ ﭼﻨﺎﻧﭽﻪ ﺃﻣﺮ ﺑﺮ ﺷـﻤﺎ ﻣﺸﺘﺒﻪ ﺷـﺪ (ﻳﻌﻨﯽ ﻣﻔﻬﻮﻡ ﺭﻭﺍﻳﺘﯽ ﺭﺍ ﻧﺘﻮﺍﻧﺴﺘﻴﺪ ﺩﺭک ﻛﻨﻴـﺪ) ﺩﺭﺑـﺎﺭﻩ ﺍﺵ ﺗﻮﻗـﻒ ﻧﻤﺎﻳﻴﺪ، (ﻧﻪ ﺁﻥ ﺭﺍ ﺭﺩ ﻛﻨﻴـﺪ ﻭ ﻧﻪ ﺑﻪ ﺁﻥ ﻋﻤﻞ ﻧﻤﺎﻳﻴﺪ) ﺑﻠﻜﻪ (ﺗﺼﻤﻴﻢ ﮔﻴﺮی ﺩﺭﺑﺎﺭﻩ ﺩﺭﺳـﺘﯽ ﻳﺎ ﻧﺎﺩﺭﺳﺘﯽ) ﺁﻥ ﺭﺍ ﺑﻪ ﻣﺎ ﻭﺍﮔﺬﺍﺭ ﻛﻨﻴﺪ ﺗﺎ ﺁﻧﭽﻪ ﺭﺍ ﺑﺮﺍی ﻣﺎ ﺑﻴﺎﻥ ﺷـﺪﻩ ﺑﺮﺍﻳﺘﺎﻥ ﺷـﺮﺡ ﺩﻫﻴﻢ، ﭘﺲ ﺍﮔﺮ ﺍﻳﻦﭼﻨﻴﻦ ﺑﻮﺩﻳﺪ ﻛﻪ ﺷـﻤﺎ ﺭﺍ ﺳـﻔﺎﺭﺵ ﻧﻤﻮﺩﻳﻢ ﻭ ﺍﺯ ﺍﻳﻦ ﺣﺎﻝ ﺑﻴﺮﻭﻥ ﻧﺮﻓﺘﻴﺪ ﻭ ﻛﺴـﯽ ﺍﺯ ﺷـﻤﺎ ﭘﻴﺶ ﺍﺯ ﻗﻴﺎﻡ ﻗﺎﺋﻢ ﻣﺎ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﻣﺮﺩ، ﺛﻮﺍﺏ ﺷﻬﻴﺪ ﺑﺮﺍی ﺍﻭ ﻧﻮﺷﺘﻪ ﺧﻮﺍﻫﺪ ﺷﺪ.»(66)

ﻋﻼﻭﻩ ﺑﺮ ﺷـﻴﻌﻴﺎﻥ ﻣﻌﺘﻘﺪ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻛﻪ ﺧﺎﻧﻮﺍﺩﻩ ﺍﻭ ﻣﺤﺴـﻮﺏ ﻣﯽ ﺷـﻮﻧﺪ، ﺳـﺎﺩﺍﺕ (ﻳﻌﻨﯽ ﻓﺮﺯﻧﺪﺍﻥ ﭘﻴﺎﻣﺒﺮ) ﻧﻴﺰ ﺑﻪ ﻧﺤﻮ ﺩﻳﮕـﺮ ﺍﺯ ﺧﻮﻳﺸـﺎﻭﻧﺪﺍﻥ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ ﻭ ﺷـﻮﺍﻫﺪ ﺗﺎﺭﻳﺨﯽ ﻧﺸـﺎﻥ ﻣﯽ ﺩﻫﺪ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻭ ﭘﺪﺭﺍﻥ ﻣﻌﺼﻮﻣﺸـﺎﻥ ﻧﺴـﺒﺖ ﺑﻪ ﺭﺳـﻴﺪﮔﯽ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻥ ﭘﻴﺎﻣﺒﺮ ﻭﻟﻮ ﺁﻧﻜﻪ ﺷﻴﻌﻪ ﻧﺒﺎﺷـﻨﺪ، ﺗﻮﺟﻪ ﻭ ﻋﻨﺎﻳﺖ ﺩﺍﺷﺘﻪ ﺍﻧﺪ. ﺭﺳـﻴﺪﮔﯽ ﻭ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻦ ﻋﺰﻳﺰﺍﻥ ﻧﻴـﺰ ﺍﺯ ﻣﺼﺎﺩﻳـﻖ ﺭﺳـﻴﺪﮔﯽ ﺑـﻪ ﺍﻫﻞﺑﻴـﺖ ﻭ ﺧﺪﻣـﺖ ﺑـﻪ ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺍﺳﺖ.

ﺑـﺎﺯ ﺑﺎﻳﺪ ﺗﺄﻛﻴﺪ ﻛﺮﺩ ﻛﻪ ﻣﺒﺎﺩﺍ ﺧﺪﻣﺖ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ (ﻳﻌﻨﯽ ﺧﺎﻧﻮﺍﺩﻩ ﻣﻌﻨﻮی ﺍﻣﺎﻡ) ﻭ ﺳـﺎﺩﺍﺕ (ﻳﻌﻨﯽ ﺧﺎﻧﻮﺍﺩﻩ ﻧﺴﺒﯽ ﺍﻣﺎﻡ) ﺭﺍ ﺍﺯ ﺭﻭی ﺍﻛﺮﺍﻩ ﻭ ﺑـﺎ ﻣﻨـﺖ ﺍﻧﺠـﺎﻡ ﺩﻫﻴﻢ. ﺑﻠﻜـﻪ ﻫﺮ ﻛﻪ ﺗﻮﻓﻴـﻖ ﺍﻳﻦ ﺍﻣـﻮﺭ ﺭﺍ ﻳﺎﻓﺖ ﺑﺎﻳﺪ ﺑـﺎ ﺷـﻮﻕ ﻭ ﺭﻏﺒﺖ ﺑﻪ ﺍﻧﺠﺎﻡ ﺁﻧﻬﺎ ﺑﺸـﺘﺎﺑﺪ ﻭ ﺍﻳـﻦ ﻛﺎﺭ ﺭﺍ ﻭﻇﻴﻔﻪ ی ﺧﻮﺩ ﺑﺪﺍﻧﺪ ﻧﻪ ﻟﻄﻒ ﻭ ﻣﻨﺘﯽ ﺍﺯ ﺟﺎﻧﺐ ﺧﻮﻳﺶ؛ ﻛﻪ ﺍﮔﺮ ﭼﻨﻴﻦ ﺑﺎﺷﺪ، ﻗﺮﺏ ﻭ ﺍﺟﺮی ﺩﺭ ﻛﺎﺭ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.

ﺧﺪﺍﻭﻧـﺪ ﻣﯽ ﻓﺮﻣﺎﻳـﺪ: «ﻛﺴـﺎﻧﯽ ﻛـﻪ ﺍﻣـﻮﺍﻝ ﺧـﻮﺩ ﺭﺍ ﺩﺭ ﺭﺍﻩ ﺧﺪﺍ ﺍﻧﻔـﺎﻕ ﻣﯽ ﻛﻨﻨـﺪ ﻭ ﺩﺭ ﭘـﯽ ﺁﻥ ﻣﻨﺘـﯽ ﻧﻤﯽ ﻧﻬﻨـﺪ ﻭ ﺁﺯﺍﺭی ﻧﻤﯽ ﺭﺳـﺎﻧﻨﺪ، ﻣﺰﺩﺷـﺎﻥ ﻧﺰﺩ ﭘﺮﻭﺭﺩﮔﺎﺭﺷـﺎﻥ ﺍﺳـﺖ. ﺍﻳﻨﺎﻥ ﺭﺍ ﻧﻪ ﻧﮕﺮﺍﻧﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﻧـﻪ ﺍﻧﺪﻭﻩ. ﻳـﮏ ﻛﻼﻡ ﺧﻮﺵ ﻭ ﻳﮏ ﭼﺸﻢ ﭘﻮﺷـﯽ، ﺑﻬﺘﺮ ﺍﺯ ﺻﺪﻗﻪ ﺍی ﺍﺳـﺖ ﻛﻪ ﺁﺯﺍﺭی ﺑﻪ ﺩﻧﺒﺎﻝ ﺩﺍﺷﺘﻪ ﺑﺎﺷـﺪ ﻭ (ﺑﺪﺍﻧﻴﺪ ﻛﻪ) ﺧﺪﺍﻭﻧﺪ ﺑﯽ ﻧﻴﺎﺯ ﻭ ﺑﺮﺩﺑﺎﺭ ﺍﺳﺖ.»(67)
