ﺁﻥ ﻣﺮﺩ ﺩﺍﻧﺎ، ﻛﻪ ﻗﺪﺭﺵ ﺭﺍ ﻧﻤﯽ ﺩﺍﻧﻴﻢ.

ﺧﻴﻠﯽ ﻫـﺎ ﺗﺼـﻮﺭ ﺩﺭﺳـﺘﯽ ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫـﺎ ﻭ ﻏﺼﻪ ﻫـﺎی ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻧﺪﺍﺭﻧـﺪ. ﺁﻧﻬـﺎ ﺑـﺎ ﺧﻮﺩﺷـﺎﻥ ﻣﯽ ﮔﻮﻳﻨـﺪ ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﻛﻪ ﮔﺮﻓﺘﺎﺭی ﻧﺪﺍﺭﺩ ﺗﺎ ﺑﺮﺍی ﺭﻓﻊ ﻣﺸـﻜﻼﺕ ﺍﻭ ﺩﻋـﺎ ﻛﻨﻴﻢ. ﺁﻧﻬﺎ ﻣﯽ ﮔﻮﻳﻨﺪ ﺍﻳﺸـﺎﻥ ﻛﻪ ﻧﻴﺎﺯی ﺑﻪ ﻣﺮﺩﻡ ﻧﺪﺍﺭﺩ ﺗﺎ ﺑﯽ ﺗﻮﺟﻬﯽ ﻣﺮﺩﻡ ﺑﻪ ﺍﻭ ﻣﺎﻳﻪ ﻏﺮﺑﺘﺶ ﺷـﻮﺩ. ﺁﻥ ﻫـﺎ ﻣﯽ ﮔﻮﻳﻨﺪ ﺍﻣـﺎﻡ ﺯﻣـﺎﻥ ﺁﺩﺍﺏ ﺩﺭﺳـﺖ ﺯﻧﺪﮔﯽ ﻛـﺮﺩﻥ ﻭ ﺩﺭﺳـﺖ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻭ ﺩﺭﺳﺖ ﺧﻮﺍﺑﻴﺪﻥ ﺭﺍ ﻫﻢ ﻣﯽ ﺩﺍﻧﺪ، ﭘﺲ ﻣﺮﻳﺾ ﻫﻢ ﻧﻤﯽ ﺷﻮﺩ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺗﺼﻮﺭﺍﺕ، ﻧﺎﻗﺺ ﻭ ﻧﺎﺩﺭﺳﺖ ﺍﺳﺖ.

ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺭﻧﺪ ﻛﻪ ﺍﻳﺸﺎﻥ ﺭﺍ ﮔﺎﻩ ﺗﺎ ﻣﺮﺯ ﺍﺿﻄﺮﺍﺭ ﺑﻪ ﭘﻴﺶ ﻣﯽ ﺑﺮﺩ. ﺍﺿﻄﺮﺍﺭ ﺑﻪ ﻣﻌﻨﺎی ﭼﺎﺭﻩ ﻧﺪﺍﺷـﺘﻦ ﺍﺳـﺖ ﻭ ﻣﻀﻄﺮ ﺑﻪ ﻛﺴـﯽ ﻣﯽ ﮔﻮﻳﻨﺪ ﻛﻪ ﭼﺎﺭﻩ ﺍی ﺑﺮﺍی ﺍﻭ ﺑﺎﻗﯽ ﻧﻤﺎﻧﺪﻩ ﺍﺳـﺖ. ﻳﻜـﯽ ﺍﺯ ﺍﻟﻘﺎﺏ ﺍﻣﺎﻡ ﻋﺼـﺮ ﻧﻴﺰ ﻛﻪ ﺩﺭ ﻛﻼﻡ ﺧـﺪﺍ ﺫﻛﺮ ﮔﺮﺩﻳﺪﻩ ﺍﺳـﺖ، ﻟﻘﺐ «ﺍﻟﻤﻀﻄﺮ » ﺍﺳـﺖ. ﺩﺭ ﺭﻭﺍﻳﺘـﯽ ﻣﯽ ﺧﻮﺍﻧﻴﻢ ﺯﻣﺎﻧﯽ ﻏﺼﻪ ﻭ ﮔﺮﻓﺘﺎﺭی ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺣﺪی ﻣﯽ ﺭﺳﺪ ﻛﻪ ﺍﻳﺸﺎﻥ ﺑﻪ ﺣﺎﻝ ﺍﺿﻄﺮﺍﺭ ﻣﯽ ﺍﻓﺘﺪ. ﺍﻳﺸـﺎﻥ ﺩﺭ ﺁﻥ ﺣﺎﻝ ﺑﻪ ﻣﺴـﺠﺪﺍﻟﺤﺮﺍﻡ ﻣﯽ ﺭﻭﺩ ﻭ ﭘﺸـﺖ ﻣﻘﺎﻡ ﺍﺑﺮﺍﻫﻴﻢ (ﺳـﻨﮓ ﻣﻘﺪﺳﯽ ﺩﺭ ﻣﺴـﺠﺪﺍﻟﺤﺮﺍﻡ) ﺑﻪ ﻧﻤﺎﺯ ﻣﯽ ﺍﻳﺴﺘﺪ ﻭ ﺍﻳﻦ ﺁﻳـﻪ ﺭﺍ ﻣﯽ ﺧﻮﺍﻧﺪ «ﺍﻣﻦ ﻳﺠﻴﺐ ﺍﻟﻤﻀﻄﺮ ﺍﺫﺍ ﺩﻋﺎﻩ ﻭ ﻳﻜﺸـﻒ ﺍﻟﺴـﻮﺀ ﻭ ﻳﺠﻌﻠﻜـﻢ ﺧﻠﻔـﺎﺀ ﺍﻻﺭﺽ»(3) ﻳﻌﻨـﯽ : «ﻏﻴﺮ ﺍﺯ ﺧﺪﺍ ﻛﻴﺴـﺖ ﻛﻪ ﭼﻮﻥ ﺑﯽ ﭼـﺎﺭﻩ ﺍی ﺍﻭ ﺭﺍ ﺑﺨﻮﺍﻧـﺪ ﭘﺎﺳـﺦ ﺑﺪﻫـﺪ ﻭ ﮔﺮﻓﺘـﺎﺭی ﺍﺵ ﺭﺍ ﺑﺮﻃﺮﻑ ﺳﺎﺯﺩ ﻭ ﺷﻤﺎ ﺭﺍ ﺣﺎﻛﻤﺎﻥ ﺯﻣﻴﻦ ﮔﺮﺩﺍﻧﺪ؟»

ﺩﺭ ﺁﻥ ﻟﺤﻈـﻪ ﺧﺪﺍﻭﻧﺪ ﺑﻪ ﺍﺿﻄﺮﺍﺭ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺭﺣﻢ ﻣﯽ ﺁﻭﺭﺩ ﻭ ﺑﺎ ﺻﺪﻭﺭ ﺍﺫﻥ ﻗﻴﺎﻡ، ﻳﺎﺭی ﻫﺎی ﺁﺳﻤﺎﻧﯽ ﺧﻮﻳﺶ ﺭﺍ ﺑﻪ ﺳﻮی ﺍﻭ ﻧﺎﺯﻝ ﻣﯽ ﻛﻨﺪ.(4)

ﺍﺯ ﺍﻳـﻦ ﺭﻭﺍﻳﺖ ﺑﺮﻣﯽ ﺁﻳﺪ ﻛﻪ ﻧـﻪ ﺗﻨﻬﺎ ﮔﺮﻓﺘﺎﺭی ﻭ ﺩﺭﺩ ﻭ ﺭﻧﺞ ﺑﺮﺍی ﺍﻣـﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﻗﺎﺑﻞ ﺗﺼﻮﺭ ﺍﺳـﺖ، ﺑﻠﻜـﻪ ﺍﻳـﻦ ﮔﺮﻓﺘﺎﺭی ﻫﺎ ﻣﻤﻜﻦ ﺍﺳـﺖ ﺗﺎ ﺣﺪ ﺍﺿﻄﺮﺍﺭ ﻭ ﻣﺮﺯ ﺷﻜﺴﺘﻪ ﺷـﺪﻥ ﺗﻮﺍﻥ ﺁﻥ ﺣﻀﺮﺕ ﻫﻢ ﭘﻴﺶ ﺑﺮﻭﺩ. ﺍﺳﺎﺳﺎ ﺗﻌﺒﻴﺮ ﻓﺮﺝ ﻛﻪ ﺑﺮﺍی ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﻛﺎﺭ ﻣﯽ ﺭﻭﺩ ﺑﻪ ﻣﻌﻨﺎی ﺭﻓﻊ ﮔﺮﻓﺘﺎﺭی ﺍﺳﺖ. ﺗﺎ ﮔﺮﻓﺘﺎﺭی ﻧﺒﺎﺷﺪ ﻓﺮﺝ ﻣﻌﻨﺎﻳﯽ ﻧﺪﺍﺭﺩ. ﺍﻳﻨﻜـﻪ ﺑـﻪ ﻣـﺎ ﮔﻔﺘﻪ ﺷـﺪﻩ ﺍﺯ ﺧﺪﺍ ﺗﺴـﺮﻳﻊ ﻭ ﺗﻌﺠﻴـﻞ ﺩﺭ ﻓﺮﺝ ﺍﻳﺸـﺎﻥ ﺭﺍ ﺑﺨﻮﺍﻫﻴـﻢ ﺣﺎﻛﯽ ﺍﺯ ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺁﻥ ﺣﻀﺮﺕ ﮔﺮﻓﺘﺎﺭ ﺍﺳـﺖ ﻭ ﺑﺮﺍی ﺭﻓﻊ ﮔﺮﻓﺘﺎﺭی ﺍﻭ ﻧﻴﺎﺯ ﺑﻪ ﻛﻤﮏ ﺍﻟﻬﯽ ﺍﺳـﺖ. ﺩﺭ ﻫﻤﻴﻦ ﻛﺘﺎﺏ ﺑﺎ ﺑﺮﺧﯽ ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎ ﻭ ﻣﺸﻜﻼﺕ ﺍﻣﺎﻡ ﻋﺼﺮ ﺁﺷﻨﺎ ﺧﻮﺍﻫﻴﺪ ﺷﺪ.

ﺑﺰﺭﮔﺎﻥ ﮔﻔﺘﻪ ﺍﻧﺪ ﻛﻪ ﻓﺮﺝ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻫﻢ ﺩﻭ ﮔﻮﻧﻪ ﺍﺳﺖ. ﻳﻜﯽ ﻓﺮﺝ ﻛﻠﯽ ﺍﺳـﺖ ﻛﻪ ﺑﺎ ﻇﻬﻮﺭ ﺍﻳﺸـﺎﻥ ﺻﻮﺭﺕ ﻣﯽ ﭘﺬﻳﺮﺩ ﻭ ﺩﻳﮕﺮ، ﻓﺮﺟﻬﺎﻳﯽ ﺟﺰﺋﯽ ﺍﺳـﺖ ﻛﻪ ﺩﺭ ﺑﺮﻫﻪ ﻫﺎی ﻣﺨﺘﻠﻒ ﭘﻴﺶ ﺍﺯ ﻇﻬﻮﺭ ﺍﻳﺸﺎﻥ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺗﻔـﺎﻕ ﺑﻴﻔﺘـﺪ. ﻓﺮﺝﻫﺎی ﺟﺰﺋﯽ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺍﺳـﺖ ﻛﻪ ﺑﺪﻭﻥ ﻭﻗﻮﻉ ﺁﻥ ﻇﻬﻮﺭ ﻣﻈﻔﺮ، ﺑﺮﺧﯽ ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺁﻥ ﺣﻀﺮﺕ ﻭ ﺷﻴﻌﻴﺎﻥ ﺍﻳﺸﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﻣﻘﻄﻌﯽ ﺑﺮﻃﺮﻑ ﮔﺮﺩﺩ.

ﮔﺎﻫـﯽ ﺍﻭﻗـﺎﺕ ﻣﺎ ﺑﺮﺍی ﻓﺮﺝ ﺣﻀـﺮﺕ ﺩﻋﺎ ﻣﯽ ﻛﻨﻴﻢ ﺍﻣـﺎ ﺧﺪﺍﻭﻧﺪ ﻫﻨﻮﺯ ﻣﺼﻠﺤﺖ ﻧﻤﯽ ﺩﺍﻧﺪ ﻛﻪ ﻇﻬﻮﺭ ﺍﻳﺸـﺎﻥ (ﻳﺎ ﻫﻤـﺎﻥ ﻓﺮﺝ ﻛﻠﯽ ) ﻭﺍﻗﻊ ﺷـﻮﺩ ﻭ ﮔﺮﻓﺘﺎﺭی ﻫـﺎی ﺣﻀﺮﺗﺶ ﺑـﻪ ﻳﻤﻦ ﺁﻥ ﻇﻬﻮﺭ ﺑﻪ ﻃـﻮﺭ ﻛﻠﯽ ﺭﻓﻊ ﮔـﺮﺩﺩ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺧﺪﺍﻭﻧﺪ ﺑﻪ ﺑﺮﻛﺖ ﻫﻤﻴﻦ ﺩﻋﺎی ﻣﺎ ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﻓﻌﻠﯽ ﺍﻣﺎﻡ ﻋﺼﺮ ﻣﯽ ﻛﺎﻫﺪ ﻭ ﺩﺭ ﺯﻧﺪﮔﯽ ﺍﻳﺸـﺎﻥ ﻓﺮﺝ ﻭ ﮔﺸـﺎﻳﺶ ﺟﺰﺋﯽ ﺍﻳﺠﺎﺩ ﻣﯽ ﻛﻨﺪ. ﭘﺲ ﻣﺒﺎﺩﺍ ﺑﭙﻨﺪﺍﺭﻳﻢ ﻛﻪ ﺩﻋﺎﻫﺎی ﻣﺎ ﻭ ﭘﻴﺸﻴﻨﻴﺎﻥ ﻣﺎ ﺑﺮ ﻓﺮﺝ ﺁﻥ ﺣﻀﺮﺕ ﺑﯽ ﺗﺄﺛﻴﺮ ﺑﻮﺩﻩ ﻭ ﻣﺒﺎﺩﺍ ﺍﺯ ﺍﺛﺮ ﺩﻋﺎی ﺑﺮ ﻓﺮﺝ ﻧﺎﺍﻣﻴﺪ ﺷﻮﻳﻢ.

ﺑـﻪ ﻋﻼﻭﻩ ﭼﻪ ﺑﺴـﺎ ﺧﺪﺍﻭﻧـﺪ ﺑﻪ ﺟﻬﺖ ﺩﻋﺎﻫﺎی ﻣـﺎ، ﻓﺮﺝ ﻛﻠﯽ ﺭﺍ ﻧﺮﺳـﺎﻧﺪ ﻭﻟﯽ ﺁﻥ ﺭﺍ ﻧﺰﺩﻳﮏ ﺳـﺎﺯﺩ، ﻭ ﭼﻪ ﺑﺴـﺎ ﻛﻪ ﺑﻪ ﺑﺮﻛﺖ ﻳﮏ ﺩﻋﺎی ﺧﺎﺹ ﺍﺫﻥ ﻓﺮﺝ ﻛﻠﯽ ﻧﻴﺰ ﺑﻪ ﻃﻮﺭ ﻧﺎﮔﻬﺎﻧﯽ ﺻﺎﺩﺭ ﺷﻮﺩ.

ﺍﻳﻦ ﺗﺼـﻮﺭ ﻛﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﮔﺮﻓﺘـﺎﺭی ﻧﺪﺍﺭﺩ ﺑﺎ ﺗﻌﺎﻟﻴﻢ ﺩﻳﻨﯽ ﻭ ﺑﻪ ﺧﺼﻮﺹ ﺑﺎ ﻣﻔﻬﻮﻡ ﻧﻴﺎﺯ ﺍﻳﺸﺎﻥ ﺑﻪ ﻓﺮﺝ ﺍﻟﻬﯽ ﻧﺎﺳﺎﺯﮔﺎﺭ ﺍﺳﺖ. ﺑﺮﺍی ﺍﻳﻨﻜﻪ ﺗﺼﻮﺭی ﺍﺟﻤﺎﻟﯽ ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺍﺷـﺘﻪ ﺑﺎﺷـﻴﻢ ﺗﻮﺟﻪ ﺷـﻤﺎ ﺭﺍ ﺑﻪ ﻛﻢ ﺷـﻤﺎﺭ ﺑﻮﺩﻥ ﻳﺎﻭﺭﺍﻥ ﺁﻥ ﺣﻀﺮﺕ ﻭ ﻛﺜﺮﺕ ﺩﺷﻤﻨﺎﻥ ﺍﻳﺸﺎﻥ ﻭ ﻋﻠﺖ ﻏﻴﺒﺖ ﺍﻳﺸﺎﻥ ﺟﻠﺐ ﻣﯽ ﻛﻨﻢ.

ﺩﺭ ﺑﺮﺧﯽ ﺭﻭﺍﻳﺎﺕ ﺫﻛﺮ ﺷﺪﻩ ﻛﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻏﻴﺒﺖ ﺧﻮﺍﻫﺪ ﮔﺰﻳﺪ، ﺯﻳﺮﺍ ﺍﮔﺮ ﺟﺰ ﺍﻳﻦ ﺑﺎﺷـﺪ ﻛﺸﺘﻪ ﺧﻮﺍﻫﺪ ﺷـﺪ ﻭ ﺑﺎ ﻛﺸﺘﻪ ﺷﺪﻥ ﺍﻭ ﺳﻠﺴﻠﻪ ﺍﻣﺎﻣﺖ ﮔﺴﻴﺨﺘﻪ ﺷﺪﻩ ، ﻧﻈﺎﻡ ﺁﻓﺮﻳﻨﺶ ﺩﭼﺎﺭ ﺍﺧﺘﻼﻝ ﻣﯽ ﺷﻮﺩ.

ﺷـﺪﺕ ﻏﺮﺑﺖ ﺁﻥ ﺣﻀﺮﺕ ﺗﺎﺑﺪﻳﻦ ﺣﺪ ﺍﺳـﺖ ﻛﻪ ﺍﮔﺮ ﺍﻳﺸـﺎﻥ ﭘﻴﺶ ﺍﺯ ﻓﺮﺍ ﺭﺳﻴﺪﻥ ﻓﺮﺝ ﻛﻠﯽ ﺍﺯ ﺣﺎﻝ ﻏﻴﺒﺖ (ﻳﻌﻨﯽ ﺯﻧﺪﮔﯽ ﻧﺎﺷﻨﺎﺱ)ﺑﻪ ﺩﺭﺁﻳﺪ ﻭ ﺧﻮﺩ ﺭﺍ ﻣﻌﺮﻓﯽ ﻛﻨﺪ، ﺩﺷﻤﻨﺎﻥ ﺑﺮ ﺍﻭ ﺧﻮﺍﻫﻨﺪ ﺗﺎﺧﺖ، ﺍﻭ ﺭﺍ ﺧﻮﺍﻫﻨﺪ ﻛﺸﺖ ﻭ ﺑﺎ ﺍﻭ ﺷﺒﻴﻪ ﻫﻤﺎﻥ ﺭﻓﺘﺎﺭی ﺭﺍ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ ﻛﻪ ﺑﺎ ﺍﺟﺪﺍﺩﺵ ﻛﺮﺩﻧﺪ.

ﺁﻳﺎ ﺍﻳﻦ ﺯﻧﺪﮔﯽ ﺁﻣﻴﺨﺘﻪ ﺑﻪ ﻧﺎﺍﻣﻨﯽ ﻭ ﺧﻔﺎ، ﻧﻮﻋﯽ ﮔﺮﻓﺘﺎﺭی ﺑﻪ ﺷﻤﺎﺭ ﻧﻤﯽ ﺁﻳﺪ. ﺁﻳﺎ ﺍﻳﻦ ﺭﻭﺍﺳـﺖ ﻛﻪ ﻫﻤﻪ ﻣﺎ ﺑﺘﻮﺍﻧﻴﻢ ﺷﻨﺎﺳـﻨﺎﻣﻪ ﺩﺍﺷـﺘﻪ ﺑﺎﺷﻴﻢ ﻭ ﻧـﺎﻡ ﻭ ﻧﺸـﺎﻥ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺟﺎﻣﻌﻪ ﻣﻄﺮﺡ ﻛﻨﻴـﻢ ﻭ ﺍﺯ ﺣﻘﻮﻕ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮﺧـﻮﺭﺩﺍﺭ ﺑﺎﺷـﻴﻢ، ﺍﻣﺎ ﺑـﺮﺍی ﺁﻥ ﺣﻀﺮﺕ ﺍﻳﻦ ﺍﻣﻜﺎﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺷـﺘﻪ ﺑﺎﺷﺪ ﻛﻪ ﺣﺘﯽ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻧﺎﻡ ﻣﻌﺮﻓﯽ ﻧﻤﺎﻳﺪ؟

ﺍﮔﺮ ﻛﺴـﯽ ﺍﻳﻦ ﺭﺍ ﮔﺮﻓﺘـﺎﺭی ﻧﻤﯽ ﺩﺍﻧﺪ، ﭘﺲ ﮔﺮﻓﺘـﺎﺭی ﺭﺍ ﺩﺭ ﭼﻪ ﭼﻴـﺰی ﻣﯽ ﺑﻴﻨﺪ؟ ﺁﻳﺎ ﺷـﻤﺎ ﮔﻤﺎﻥ ﻣﯽ ﻛﻨﻴـﺪ ﻛﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﻭﺳﺖ ﻧﺪﺍﺭﺩ ﺩﺭ ﺟﻤﻊ ﺩﻭﺳﺘﺎﻥ ﺧﻮﺩ ﺑﻨﺸﻴﻨﺪ ﻭ ﺑﺎ ﺁﻧﻬﺎ ﺩﺭ ﺷﺮﺍﻳﻂ ﺍﻣﻦ ﻭ ﺍﻣﺎﻥ ﺳـﺨﻦ ﺑﮕﻮﻳﺪ؟ ﻳﺎ ﺍﻳﻨﻜﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﻭﺳﺖ ﻧﺪﺍﺭﺩ ﺑﺎ ﺧﻮﻳﺸﺎﻭﻧﺪﺍﻥ ﺧﻮﻳﺶ ﻳﻌﻨﯽ ﺑﺎ ﺳﺎﺩﺍﺕ ﻛﻪ ﻋﻤﻮﺯﺍﺩﻩ ﻫﺎی ﺍﻭ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﻨـﺪ، ﻣﺎﻧﻨـﺪ ﺩﻳﮕـﺮ ﻣﺮﺩﻡ ﻛـﻪ ﺑﺎ ﺧﻮﻳﺸﺎﻧﺸـﺎﻥ ﺭﻓﺖ ﻭ ﺁﻣـﺪ ﺩﺍﺭﻧﺪ، ﺍﺭﺗﺒﺎﻁ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻋﻠﻨﺎ ﺻﻠﻪ ﺭﺣﻢ ﺑﻪ ﺟﺎی ﺁﻭﺭﺩ؟

ﻧﺎﺷـﻨﺎﺱ ﺯﻧﺪﮔـﯽ ﻛـﺮﺩﻥ ﻧﻮﻋـﯽ ﻏﺮﺑـﺖ ﻭ ﮔﺮﻓﺘﺎﺭی ﺍﺳـﺖ، ﺑﻪ ﺧﺼـﻮﺹ ﺍﮔﺮ ﺍﻳﻦ ﻧﺎﺷﻨﺎﺳـﯽ ﺑـﻪ ﻭﺍﺳـﻄﻪ ﻧﺎﺍﻣﻨﯽ ﻭ ﭼﻴﺮﮔـﯽ ﻇﻠﻢ ﺑﺮ ﺍﻧﺴﺎﻥ ﺗﺤﻤﻴﻞ ﺷﺪﻩ ﺑﺎﺷﺪ.

ﺑﻌـﺪ ﺩﻳﮕـﺮی ﺍﺯ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺣﻀﺮﺕ ﻣﻬﺪی ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺩﺭ ﺍﺛﺮ ﺗﻌﺪی ﮔﺴـﺘﺮﺩﻩ ﺑﺸـﺮ ﺑﻪ ﻃﺒﻴﻌﺖ ﻭ ﺑﻪ ﭘﺎ ﺳﺎﺧﺘﻦ ﺷـﺮﺍﻳﻂ ﻧﺎﻣﺘﻌﺎﺩﻝ ﻭ ﻓﺴﺎﺩ ﺩﺭ ﺯﻣﻴﻦ ﺍﺳﺖ.

ﺧﺪﺍﻭﻧﺪﺑﺮﺍی ﺑﻨﺪﮔﺎﻥ ﺻﺎﻟﺢﺧﻮﻳﺶ،ﺩﺭﻃﺒﻴﻌﺖﺣﻘﻮﻕﻓﺮﺍﻭﺍﻧﯽ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻛﻪ ﺍﻳﻨﮏ ﺑﺴﻴﺎﺭی ﺍﺯ ﺁﻧﻬﺎ ﺑﻪ ﻳﻐﻤﺎ ﺭﻓﺘﻪ ﺍﺳﺖ. ﻃﻐﻴﺎﻥ ﮔﺮﺍﻥ ﻛﻪ ﺟﻬﺎﻥ ﺭﺍ ﺑﻪ ﺗﻨﮕﻨﺎﻳﯽ ﺟﺎﻥ ﻓﺮﺳﺎ ﺑﺮﺍی ﺑﺸﺮ ﺗﺒﺪﻳﻞ ﻣﯽ ﻛﻨﻨﺪ ﺑﻪ ﺧﺼﻮﺹ ﺑـﺎ ﺗﺼﺮﻑ ﻧﺎﺑﺠﺎ ﺩﺭ ﺯﻣﻴﻦ ﺧﺪﺍ، ﺁﺭﺍﻣـﺶ ﻃﺒﻴﻌﯽ ﺭﺍ ﺑﺮﺍی ﺣﺠﺖﻫﺎی ﺧﺪﺍﻭﻧﺪ ﻭ ﻣﺆﻣﻨﺎﻥ ﺑﺮﻫﻢ ﻣﯽ ﺯﻧﻨﺪ ﻭ ﻣﺤﻴﻂ ﺯﻳﺴـﺖ ﻭ ﺳـﻼﻣﺖ ﺁﻧﻬﺎ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﻣﺨﺎﻃﺮﻩ ﻣﯽ ﺍﻧﺪﺍﺯﻧﺪ. ﺩﺭﺳـﺖ ﺍﺳﺖ ﻛﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺍﻧﺶ ﺧﻮﺏ ﺯﻳﺴـﺘﻦ ﺭﺍ ﺩﺍﺭﺩ ﻭ ﺣﺘﯽ ﺑﻪ ﺍﺫﻥ ﺍﻟﻬﯽ ﻗﺪﺭﺕ ﺷﻔﺎ ﺩﺍﺩﻥ ﺭﺍ ﻧﻴﺰ ﺩﺍﺭﺍﺳـﺖ، ﻟﻴﻜﻦ ﺍﮔﺮ ﺑﺨﻮﺍﻫﺪ ﺑﻪ ﺭﻭﺍﻝ ﻋﺎﺩی ﺯﻧﺪﮔﯽ ﻛﻨﺪ، ﺩﺭ ﺍﺛﺮ ﻓﺴـﺎﺩی ﻛﻪ ﺩﺭ ﻧﺘﻴﺠﻪ ﺭﻓﺘﺎﺭ ﺍﻧﺴـﺎﻥ ﻫﺎ ﺩﺭ ﻃﺒﻴﻌﺖ ﺭﻭی ﺩﺍﺩﻩ ، ﭼﻪ ﺑﺴـﺎ ﺩﺳـﺘﺨﻮﺵ ﺑﻴﻤﺎﺭی ﻫﻢ ﻣﯽ ﺷـﻮﺩ ﻭ ﺷـﻔﺎی ﺑﻴﻤﺎﺭی ﺍﺵ ﻧﻴﺰ ﻣﻨﻮﻁ ﺑﻪ ﻃﻮﻝ ﺯﻣﺎﻥ ﻭ ﻣﺪﺍﻭﺍ ﻣﯽ ﮔﺮﺩﺩ. ﺑﻠﻪ ﺍﮔﺮ ﺍﻭ ﺑﺨﻮﺍﻫﺪ ﺣﺠﺎﺏ ﻃﺒﻴﻌﺖ ﺭﺍ ﺑﻪ ﻧﻴﺮﻭی ﺭﺑﺎﻧﯽ ﻭ ﺁﺳﻤﺎﻧﯽ ﺧﻮﻳﺶ ﺑﺪﺭﺩ ﻭ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺗﻮﺍﻥ ﻫﺎی ﻏﻴﺒﯽ ﻣﺪﺍﻭﺍ ﻛﻨﺪ، ﻣﯽ ﺗﻮﺍﻧﺪ، ﻟﻴﻜﻦ ﺳـﻴﺮﻩ ﻧﻴﺎﻛﺎﻥ ﺁﻥ ﺣﻀﺮﺕ ﻧﺸـﺎﻥ ﻣﯽ ﺩﻫﺪ ﻛﻪ ﺁﻧﺎﻥ ﻧﻴﺰ ﺩﺭ ﻫﻤﻪ ﻣﻮﺍﺭﺩ ﺍﺯ ﭼﻨﻴﻦ ﻗﺪﺭﺕ ﻫﺎﻳﯽ ﺍﺳـﺘﻔﺎﺩﻩ ﻧﻤﯽ ﻛﺮﺩﻧﺪ. ﺩﺭ ﺍﻳﻦ ﺷـﺮﺍﻳﻂ ﺍﺳـﺖ ﻛﻪ ﺣﻤﺎﻳـﺖ ﺍﻃﺮﺍﻓﻴﺎﻥ ﻭ ﺷـﻴﻌﻴﺎﻥ ﻭ ﺗﻮﺟﻪ ﺁﻧﺎﻥ ﺑـﻪ ﺍﻣﺎﻣـﺎﻥ ﺧﻮﺩ، ﻣﺮﻫﻤﯽ ﺑﺮ ﺯﺧﻢ ﻫﺎ ﻭ ﻣﺎﻳﻪ ﺗﺴـﻜﻴﻦ ﺩﺭﺩﻫﺎی ﺍﻳﺸـﺎﻥ ﻣﯽ ﺑﺎﺷـﺪ. ﺯﻧﺪﮔﯽ ﺍﻣﺎﻣﺎﻥ ﻣﺎ ﻧﺸـﺎﻥ ﻣﯽ ﺩﻫﺪ ﻛﻪ ﺁﻥ ﻫﺎ ﺑﻪ ﻭﺍﺳـﻄﻪ ﺟﻨﺒﻪ ﺑﺸﺮی ﺷﺎﻥ ﮔﺎﻩ ﺑﻪ ﺟﺮﻋﻪ ﺁﺑﯽ ﻧﻴﺰ ﻧﻴﺎﺯ ﺿﺮﻭﺭی ﭘﻴﺪﺍ ﻣﯽ ﻛﻨﻨﺪ ﺗﺎ ﻋﻄﺶ ﺟﺎﻧﻜﺎﻫﺸﺎﻥ ﻓﺮﻭﻧﺸﺎﻧﺪﻩ ﺷﻮﺩ.

ﻭﻗﻮﻉ ﺷـﺮﺍﻳﻂ ﺳـﺨﺖ ﺑﺮﺍی ﺑﺮﮔﺰﻳﺪﮔﺎﻥ ﺧﺪﺍ ﺯﻣﻴﻨﻪ ﺍی ﺍﺳـﺖ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺘﻮﺍﻧﻨﺪ ﻣﻴﺰﺍﻥ ﻓﺪﺍﻛﺎﺭی ﻭ ﻣﺤﺒﺖ ﺧﻮﺩ ﺭﺍ ﻧﺴـﺒﺖ ﺑﻪ ﺁﻧﺎﻥ ﺍﺑﺮﺍﺯ ﻧﻤﺎﻳﻨﺪ، ﺑﺮﺍی ﺁﻧﺎﻥ ﺩﻝ ﺑﺴﻮﺯﺍﻧﻨﺪ، ﺩﻋﺎﻳﯽ ﺑﻜﻨﻨﺪ، ﻗﺪﻣﯽ ﺑﺮﺩﺍﺭﻧﺪ ﻭ ﺣﺘﯽ ﺟﺎﻥ ﻓﺸﺎﻧﯽ ﻧﻤﺎﻳﻨﺪ.

ﺍﻳﻦ ﺭﺍ ﻫﻢ ﺑﺎﻳﺪ ﺩﺍﻧﺴﺖ ﻛﻪ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﻭﮔﻮﻧﻪ ﺍﺳـﺖ. ﮔﺮﻭﻫﯽ ﺍﺯ ﺁﻧﻬﺎ ﺩﺭ ﺍﺛﺮ ﺁﺳـﻴﺐﻫﺎﻳﯽ ﺍﺳـﺖ ﻛﻪ ﻣﺘﻮﺟﻪ ﺷـﺨﺺ ﺍﻳﺸـﺎﻥ ﺍﺳـﺖ، ﺍﻣﺎ ﺑﺨﺶ ﺑﺰﺭﮔـﯽ ﺍﺯ ﺍﻳـﻦ ﮔﺮﻓﺘﺎﺭی ﻫـﺎ، ﺩﺭ ﺣﻘﻴﻘﺖ ﮔﺮﻓﺘﺎﺭی ﻫﺎی ﻛﺴـﺎﻧﯽ ﺍﺳـﺖ ﻛﻪ ﺑﺴـﺘﮕﺎﻥ ﻭ ﻭﺍﺑﺴﺘﮕﺎﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ ﻭ ﻏﻢ ﺁﻥ ﻫﺎﺳﺖ ﻛﻪ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺳﺮﺍﻳﺖ ﻣﯽ ﻛﻨﺪ.

ﺷﻴﻌﻴﺎﻥ ﻓﺮﺯﻧﺪﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ ﻭ ﺍﻣﺎﻡ ﭘﺪﺭ ﺩﻟﺴﻮﺯ ﺁﻧﺎﻥ ﺍﺳﺖ. ﺍﻣﺎﻡ ﺭﺿﺎ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﺗﻮﺻﻴﻒ ﺷﺨﺼﻴﺖ ﺍﻣﺎﻡ ﻓﺮﻣﻮﺩﻩ ﺍﺳـﺖ: «ﺍﻣﺎﻡ، ﺍﻧﻴﺴـﯽ ﺭﻓﻴﻖ ﻭ ﺍﻫﻞ ﻣﺪﺍﺭﺍ ﺍﺳـﺖ، ﭘﺪﺭی ﺩﻟﺴﻮﺯ ﺍﺳﺖ، ﺑﺮﺍﺩﺭی ﺍﺳﺖ ﺗﻨﯽ ﺍﺯ ﻳﮏ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﻭ ﻣﺎﺩﺭی ﺍﺳﺖ ﻛﻪ ﺑﻪ ﻓﺮﺯﻧﺪ ﺧﺮﺩﺳﺎﻝ ﺧﻮﻳﺶ ﻣﺤﺒﺖ ﻣﯽ ﻛﻨﺪ.»(5)

ﮔﺮﻓﺘﺎﺭی ﻫـﺎی ﻳـﮏ ﭘـﺪﺭ ﻣﺤـﺪﻭﺩ ﺑـﻪ ﻣﺸـﻜﻼﺕ ﺷـﺨﺼﯽ ﺍﻭ ﻧﻴﺴـﺖ، ﺑﻠﻜـﻪ ﮔﺮﻓﺘـﺎﺭی ﻭ ﺑﻴﻤـﺎﺭی ﻓﺮﺯﻧـﺪﺍﻥ ﻧﻴﺰ ﺍﺯ ﻣﺸـﻜﻼﺕ ﭘﺪﺭ ﺑـﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺁﻳﺪ. ﺑﺮﺍی ﭘﺪﺭ ﻧﺰﻭﻝ ﺑـﻼ ﺑﺮ ﻓﺮﺯﻧﺪ ﺍﺯ ﻧـﺰﻭﻝ ﺑﻼ ﺑﺮ ﺧﻮﺩ ﺍﻭ ﺟﺎﻧﻜﺎﻩ ﺗـﺮ ﺍﺳـﺖ، ﺯﻳﺮﺍ ﭘـﺪﺭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﺮ ﺑﻼ ﺻﺒﺮ ﻛﻨـﺪ، ﺍﻣﺎ ﻓﺮﺯﻧﺪ، ﺻﺒﺮ ﺍﻧﺪﻛﯽ ﺩﺍﺭﺩ ﻭ ﭼﻮﻥ ﺩﺭ ﻭﺭﻃﻪ ﺑﻼ ﻣﯽ ﺍﻓﺘﺪ، ﻣﻠﺘﻤﺴـﺎﻧﻪ ﺩﺳـﺖ ﻭ ﭘﺎ ﻣﯽ ﺯﻧﺪ ﻭ ﺑﺪﻳﻦ ﺗﺮﺗﻴﺐ ﻋﺠﺰ ﻭ ﻻﺑﻪ ﻫﺎی ﺭﻗﺖﺑﺎﺭ ﺍﻭ ﻗﻠﺐ ﭘﺪﺭ ﺭﺍ ﺑﻴﺸﺘﺮ ﻭ ﺑﻴﺸﺘﺮ ﻣﯽ ﺳﻮﺯﺍﻧﺪ.

ﺑﻪ ﻋﻼﻭﻩ ، ﺍﻣﺎﻡ ﭘﺪﺭی ﺍﺳـﺖ ﺁﮔﺎﻩ ﻛﻪ ﺍﺯ ﺍﺣﻮﺍﻝ ﻳﮏﻳﮏ ﻓﺮﺯﻧﺪﺍﻥ ﺧﻮﻳﺶ ﺑﺎﺧﺒﺮ ﺍﺳﺖ، ﭘﺲ ﻗﻠﺐ ﺍﻭ ﻣﺠﻤﻊ ﺭﻧﺞﻫﺎی ﺗﻤﺎﻡ ﺷﻴﻌﻴﺎﻥ ﻳﻌﻨﯽ ﺗﻤﺎﻡ ﻓﺮﺯﻧﺪﺍﻧﺶ ﺍﺳﺖ. ﻫﺮﮔﺎﻩ ﺩﺭ ﮔﻮﺷﻪ ﺍی ﺍﺯ ﺍﻳﻦ ﺩﻧﻴﺎ ﺣﻘﯽ ﺍﺯ ﺣﻘﻮﻕ ﺷﻴﻌﻴﺎﻥ ﺍﻣﺎﻡ ﻛﻪ ﺑﻪ ﻣﻨﺰﻟﻪ ﻓﺮﺯﻧﺪﺍﻥ ﺍﻭﻳﻨﺪ، ﺿﺎﻳﻊ ﻣﯽ ﺷﻮﺩ، ﺩﻝ ﺍﻭ ﺁﺯﺭﺩﻩ ﻣﯽ ﮔـﺮﺩﺩ ﻭ ﻫﺮﮔﺎﻩ ﺧﺎﺭی ﺑﻪ ﭘﺎی ﻳﻜﯽ ﺍﺯ ﺁﻧﺎﻥ ﻣﯽ ﺧﻠﺪ، ﻗﻠﺐ ﺍﻭ ﺯﺧﻢ ﺑﺮﻣﯽ ﺩﺍﺭﺩ. ﭘﻴﻮﻧﺪ ﺭﻭﺣﯽ ﺍﻣﺎﻡ ﺑﺎ ﻓﺮﺯﻧﺪﺍﻥ ﻣﻌﻨﻮی ﺍﺵ ﺑﺴـﻴﺎﺭ ﺷـﺪﻳﺪﺗﺮ ﺍﺯ ﭘﻴﻮﻧـﺪ ﻳﮏ ﭘـﺪﺭ ﻣﻌﻤﻮﻟﯽ ﺑﺎ ﻓﺮﺯﻧﺪﺵ ﺍﺳـﺖ ﻭ ﺑﺨـﺶ ﻋﻤﺪﻩ ﺍی ﺍﺯ ﺩﻝﻣﺸﻐﻮﻟﯽ ﻫﺎ، ﺩﻭﻧﺪﮔﯽ ﻫﺎ، ﺧﺴﺘﮕﯽ ﻫﺎ ﻭ ﺩﻋﺎﻫﺎی ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑـﻪ ﺭﻓﻊ ﮔﺮﻓﺘـﺎﺭی ﻓﺮﺯﻧﺪﺍﻧﺶ ﺑﺎﺯﮔﺸـﺖ ﻣﯽ ﻛﻨـﺪ؛ ﮔﺮﻓﺘﺎﺭی ﻫﺎﻳﯽ ﻛﻪ ﮔﺎﻩ ﺑﻪ ﺟﻬﺖ ﺗﺴـﻠﻂ ﺩﺷـﻤﻨﺎﻥ ﻭ ﻇﻠﻢ ﺍﻳﺸﺎﻥ ﮔﺮﻳﺒﺎﻥ ﮔﻴﺮ ﻓﺮﺯﻧﺪﺍﻥ ﺍﻣﺎﻡ ﻣﯽ ﺷـﻮﺩ ﻭ ﺁﺗﺶ ﺧﺸـﻢ ﺍﻭ ﺭﺍ ﺷـﻌﻠﻪ ﻭﺭ ﻣﯽ ﻛﻨﺪ ﻭ ﮔﺎﻩ ﺑﻪ ﺟﻬﺖ ﻧﺎﺩﺍﻧﯽ ﻭ ﻧﺎﭘﺮﻫﻴـﺰی ﺷـﻴﻌﻴﺎﻥ ﻳﺎ ﺩﺭ ﭘﯽ ﺟﺪﺍﻝ ﺩﺍﺧﻠﯽ ﻣﻴـﺎﻥ ﻓﺮﺯﻧﺪﺍﻥ ﺩﺍﻣﻨﮕﻴﺮ ﺍﻳﺸﺎﻥ ﻣﯽ ﮔﺮﺩﺩ.

ﮔﺮﻓﺘﺎﺭی ﻫـﺎی ﺷـﻴﻌﻴﺎﻥ ﺗﻨﻬـﺎ ﻣﺤـﺪﻭﺩ ﺑـﻪ ﻣـﻮﺍﺭﺩی ﻧﻤﯽ ﺷـﻮﺩ ﻛـﻪ ﺷـﻴﻌﻴﺎﻥ ﺧـﻮﺩ ﺍﺯ ﺁﻧﻬـﺎ ﺑﺎﺧﺒﺮﻧـﺪ ﻭ ﺍﺯ ﺁﻧﻬﺎ ﺭﻧـﺞ ﻣﯽ ﻛﺸـﻨﺪ، ﺑﻠﻜﻪ ﮔﺮﻓﺘﺎﺭی ﻫﺎﻳـﯽ ﻫﻢ ﻫﺴـﺖ ﻛﻪ ﮔﺮﻳﺒﺎﻥ ﺁﻧﻬﺎ ﺭﺍ ﮔﺮﻓﺘﻪ ، ﺍﻣﺎ ﺧﻮﺩ ﺍﻳﺸـﺎﻥ ﻣﺘﻮﺟـﻪ ﺁﻥ ﻫـﺎ ﻧﻴﺴـﺘﻨﺪ. ﺩﺭ ﻧﻬﺎﻳﺖ، ﺑﺎﺭ ﺳـﻨﮕﻴﻦ ﺭﻓﻊ ﻭ ﺣـﻞ ﺗﻤﺎﻡ ﺍﻳﻦ ﻣﺸـﻜﻼﺕ ﺑـﺮ ﺩﻭﺵ ﭘـﺪﺭ ﺧﺎﻧـﻮﺍﺩﻩ ﺷـﻴﻌﻪ ﻳﻌﻨـﯽ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺍﺳـﺖ. ﻛﺴـﯽ ﭼﻪ ﻣﯽ ﺩﺍﻧﺪ ﻛﻪ ﺁﻟﻮﺩﮔﯽ ﺷـﻴﻌﻪ ﺑـﻪ ﮔﻨﺎﻩ ، ﻓﺴـﺎﺩ، ﺩﻭﺭی ﺍﺯ ﺧﺪﺍ ﻭ ﺑﻪ ﺑﻄﺎﻟﺖ ﮔﺬﺭﺍﻧﺪﻥ ﻋﻤﺮ ﺑﺎ ﺩﺭﻳﺎی ﺩﻝ ﺍﻣﺎﻡ ﭼﻪ ﻣﯽ ﻛﻨﺪ ﻭ ﻛﺴـﯽ ﭼﻪ ﻣﯽ ﺩﺍﻧﺪ ﻛﻪ ﻇﻠﻢ ﻭ ﺳـﺘﻢ ﺯﻭﺭﻣﺪﺍﺭﺍﻥ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﺿﻌﻴﻒ ﻭ ﮔﺮﻓﺘﺎﺭ ﭼﻪ ﺷﺮﺍﺭی ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺩﻝ ﺁﻥ ﺣﻀﺮﺕ ﻣﯽ ﺍﻓﻜﻨﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﻏﺼﻪ ﻫﺎ ﻭ ﻗﺼﻪ ﻫﺎ ﭼﻪ ﻓﺮﺍﻭﺍﻥ ﻭ ﻧﺎﺷﻤﺮﺩﻧﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﻭ ﭼﻪ ﺑﺴﺎ ﻛﻪ ﺭﻭﺡ ﺁﺯﺭﺩﻩ ﻭ ﻗﻠﺐ ﻣﺠﺮﻭﺡ ﺣﻀﺮﺗﺶ، ﺁﺯﺭﺩﮔﯽ ﻫﺎ ﺭﺍ ﺑـﻪ ﺟﺴـﻢ ﺷـﺮﻳﻒ ﺍﻭ ﺍﻧﺘﻘﺎﻝ ﺩﻫﺪ ﻭ ﺟﺴـﻢ ﺍﻭ ﻧﻴﺰ ﺩﺭ ﺗـﺐ ﻭ ﺗﺎﺏ ﻭ ﺑﻴﻤﺎﺭی ﺍﻓﺘﺪ ﻭ ﺑﺎ ﺷﻴﻌﻴﺎﻥ ﺭﻧﺠﻮﺭﺵ ﻫﻤﺪﺭﺩ ﮔﺮﺩﺩ.

ﻫﻴﭻ ﺑﻌﻴﺪ ﻫﻢ ﻧﻴﺴـﺖ ﻭﻗﺘﯽ ﺑﺮﺍی ﻓﺮﺝ ﺍﻭ ﺩﻋﺎ ﻣﯽ ﻛﻨﻴﻢ، ﺧﺪﺍﻭﻧﺪ ﮔﺮﻓﺘﺎﺭی ﺷـﻴﻌﻴﺎﻧﺶ ﺭﺍ ﺑﺮﻃﺮﻑ ﺳـﺎﺯﺩ ﻭ ﺍﻳﻦ ﺍﻣﺮ ﺑﻪ ﻃﻮﺭ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﮔﺸﺎﻳﺶ ﻭ ﺁﺭﺍﻣﺸﯽ ﺑﺮﺍی ﺍﻭ ﭘﺪﻳﺪ ﺁﻭﺭﺩ.

ﺷـﺨﺼﯽ ﺑﻪ ﻧـﺎﻡ ﺭﻣﻴﻠﻪ ﻣﯽ ﮔﻮﻳـﺪ: ﺩﺭ ﺯﻣـﺎﻥ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻋﻠﯽ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺳﺨﺖ ﺑﻴﻤﺎﺭ ﺷﺪﻡ ﺗﺎ ﺁﻧﻜﻪ ﺩﺭ ﺭﻭﺯ ﺟﻤﻌﻪ ﻗﺪﺭی ﺍﺣﺴﺎﺱ ﺳﺒﻜﯽ ﻛﺮﺩﻡ. ﺑﺎ ﺧﻮﺩ ﮔﻔﺘﻢ: ﺑﻬﺘﺮ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺁﺑﯽ ﺑﺮ ﺭﻭی ﺧﻮﺩ ﺑﺮﻳﺰﻡ ﻭ ﺑﺮﻭﻡ ﭘﺸـﺖ ﺳﺮ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﻧﻤﺎﺯ ﺑﺨﻮﺍﻧﻢ. ﻫﻤﻴﻦ ﻛﺎﺭ ﺭﺍ ﻛﺮﺩﻡ، ﺁﻧﮕﺎﻩ ﺑﻪ ﻣﺴـﺠﺪ ﺭﻓﺘـﻢ. ﻭﻗﺘﯽ ﺁﻥ ﺣﻀﺮﺕ ﺑﺎﻻی ﻣﻨﺒﺮ ﺭﻓﺘﻨﺪ ﺣﺎﻝ ﻣﻦ ﺩﻭﺑﺎﺭﻩ ﺳﺨﺖ ﺷﺪ. ﻭﻗﺘﯽ ﺍﻣﻴﺮﺍﻟﻤﺆﻣﻨﻴﻦ ﺍﺯ ﻣﺴﺠﺪ ﺑﺮﮔﺸﺘﻨﺪ ﻭ ﺩﺍﺧﻞ ﺩﺍﺭﺍﻟﺤﻜﻮﻣﻪ ﺷـﺪﻧﺪ، ﻣﻦ ﻫﻢ ﻫﻤﺮﺍﻩ ﺍﻳﺸﺎﻥ ﺩﺍﺧﻞ ﺷﺪﻡ. ﺍﻳﺸﺎﻥ ﺑـﻪ ﻣـﻦ ﻓﺮﻣﻮﺩﻧـﺪ: ﺍی ﺭﻣﻴﻠﻪ ، ﺩﻳﺪﻡ ﺑـﻪ ﺧﻮﺩﺕ ﻣﯽ ﭘﻴﭽﻴـﺪی ! ﻋﺮﺽ ﻛـﺮﺩﻡ: ﺑﻠـﯽ . ﺟﺮﻳﺎﻥ ﺑﻴﻤـﺎﺭی ﺧﻮﺩ ﺭﺍ ﺑﺮﺍی ﺍﻳﺸـﺎﻥ ﺗﻌﺮﻳـﻒ ﻛﺮﺩﻡ ﻭ ﺍﻧﮕﻴﺰﻩ ﺣﻀﻮﺭ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻧﻤﺎﺯ ﺑﻴﺎﻥ ﺩﺍﺷﺘﻢ. ﺍﻳﺸﺎﻥ ﻓﺮﻣﻮﺩﻧﺪ:

«ﺍی ﺭﻣﻴﻠـﻪ ، ﻫﻴﭻ ﻣﺆﻣﻨﯽ ﺑﻴﻤﺎﺭ ﻧﻤﯽ ﺷـﻮﺩ، ﻣﮕـﺮ ﺍﻳﻨﻜﻪ ﻣﺎ ﻫﻢ ﺑﻪ ﺧﺎﻃﺮ ﺑﻴﻤﺎﺭی ﺍﻭ ﺑﻴﻤﺎﺭ ﻣﯽ ﺷـﻮﻳﻢ، ﻭ ﻏﻤﮕﻴﻦ ﻧﻤﯽ ﺷﻮﺩ ﻣﮕﺮ ﺍﻳﻨﻜﻪ ﻣﺎ ﻫـﻢ ﺩﺭ ﻏﻢ ﺍﻭ ﻏﻤﮕﻴﻦ ﻣﯽ ﺷـﻮﻳﻢ، ﻭ ﻫﻴﭻ ﺩﻋﺎﻳﯽ ﻧﻤﯽ ﻛﻨـﺪ ﻣﮕﺮ ﺁﻧﻜﻪ ﺑﻪ ﺩﻋﺎی ﺍﻭ ﺁﻣﻴﻦ ﻣﯽ ﮔﻮﻳﻴﻢ، ﻭ ﺳـﻜﻮﺕ ﻧﻤﯽ ﻛﻨﺪ ﻣﮕﺮ ﺁﻧﻜﻪ ﻣﺎ ﺑﺮﺍﻳﺶ ﺩﻋﺎ ﻣﯽ ﻛﻨﻴﻢ.»

ﻋـﺮﺽ ﻛـﺮﺩﻡ: ﺍی ﺍﻣﻴﺮ ﻣﺆﻣﻨـﺎﻥ، ﻓﺪﺍﻳﺖ ﮔﺮﺩﻡ؛ ﺍﻳﻦ ﮔﻔﺘﻪ ﺷـﻤﺎ ﺣﺘﻤـﺎ ﻣﺮﺑﻮﻁ ﺑﻪ ﻛﺴـﺎﻧﯽ ﺍﺳـﺖ ﻛﻪ ﺑﺎ ﺷـﻤﺎ ﺩﺭ ﺍﻳﻨﺠـﺎ (ﺩﺍﺭﺍﻟﺤﻜﻮﻣﻪ ) ﻫﺴـﺘﻨﺪ. (ﻳﻌﻨﯽ ﻛﺴﺎﻧﻴﻜﻪ ﺷﻤﺎ ﺁﻧﻬﺎ ﺭﺍ ﻣﯽ ﺑﻴﻨﻴﺪ ﻭ ﺍﺯ ﺣﺎﻟﺸﺎﻥ ﺑﺎﺧﺒﺮﻳﺪ) ﭘﺲ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﻛﺴـﺎﻧﯽ ﻛﻪ ﺩﺭ ﺩﻳﮕﺮ ﺟﺎﻫﺎی ﺯﻣﻴﻦ ﻫﺴـﺘﻨﺪ ﭼﻄﻮﺭ؟ ﺍﻣﺎﻡ ﺩﺭ ﺟﻮﺍﺏ ﻓﺮﻣﻮﺩﻧﺪ:

«ﺍی ﺭﻣﻴﻠـﻪ ، ﻫﻴـﭻ ﻣﺆﻣﻨـﯽ ﺩﺭ ﺷـﺮﻕ ﻭ ﻏﺮﺏ ﺯﻣﻴـﻦ ﺍﺯ ﻣﺎ ﭘﻨﻬﺎﻥ ﻧﻴﺴﺖ.»(6)

ﺁﺭی ، ﺍﮔـﺮ ﭘﻴﻮﻧﺪ ﻣﻴﺎﻥ ﺍﻣﺎﻡ ﻭ ﺷـﻴﻌﻴﺎﻧﺶ ﺭﺍ ﺑـﻪ ﺧﻮﺑﯽ ﺩﺭک ﻛﻨﻴﻢ ﺑﻴـﺶ ﺍﺯ ﭘﻴـﺶ ﺍﻭ ﺭﺍ ﻏﺮﻳﺐ ﻭ ﺩﻝﺷﻜﺴـﺘﻪ ﺧﻮﺍﻫﻴﻢ ﻳﺎﻓـﺖ ﻭ ﺑﺮﺍی ﺭﻓﻊ ﻏﺮﺑﺖ ﻭ ﺩﻝﺷﻜﺴﺘﮕﯽ ﺍﺵ ﺑﻴﺸﺘﺮ ﺗﻼﺵ ﻭ ﺩﻋﺎ ﺧﻮﺍﻫﻴﻢ ﻛﺮﺩ.
