بازگشت

ﭘﻨﺞ ﺭﺍﻩ ﺑﺮﺍی ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ (ﻋﺠﻞ ﺍﻟﻠﻪ ﺗﻌﺎﻟﯽ ﻓﺮﺟﻪ ﺍﻟﺸﺮﻳﻒ)

ﺩﺭ ﻛﻠﻤﻪ ﺍی ﻗﺪﺳﯽ ﻧﻘﻞ ﺍﺳﺖ ﻛﻪ ﺧﺪﺍﻭﻧﺪ ﺗﻌﺎﻟﯽ ﻓﺮﻣﻮﺩ: «ﻣﺤﺒﻮﺏﺗﺮﻳـﻦ ﺑﻨـﺪﮔﺎﻥ ﻧـﺰﺩ ﻣـﻦ، ﻛﺴـﺎﻧﯽ ﻫﺴـﺘﻨﺪ ﻛﻪ ﺣﻖ ﻣـﺮﺍ ﺑﺮﭘﺎ ﺩﺍﺷـﺘﻪ ﺍﻧﺪ، ﻭ ﺑﺮﺗﺮﻳـﻦ ﻭ ﮔﺮﺍﻣﯽ ﺗﺮﻳـﻦ ﺍﻳـﻦ ﺍﻓـﺮﺍﺩ، ﻣﺤﻤـﺪ ﺍﺳـﺖ ﻛـﻪ ﺳـﻴﺪ ﻭ ﺁﻗـﺎی ﺧﻼﺋﻖ ﺍﺳـﺖ ﻭ ﺑﻌـﺪ ﺍﺯ ﺍﻭ ﻋﻠﯽ ﻣﺮﺗﻀﯽ ﺑـﺮﺍﺩﺭ ﻣﺤﻤﺪ ﻣﺼﻄﻔـﯽ ﺍﺳـﺖ ﻭ ﭘـﺲ ﺍﺯ ﻋﻠﯽ ﺩﻳﮕﺮ ﺍﻣﺎﻣﺎﻥ ﺭﺍﺳـﺘﻴﻦ، ﻛـﻪ ﺣﻖ ﺭﺍ ﺑﻪ ﭘﺎ ﻣﯽ ﺩﺍﺭﻧﺪ. ﺩﺭ ﺭﺗﺒﻪ ﺑﻌﺪ ﺍﺯ ﺍﻳﺸـﺎﻥ ﻛﺴـﺎﻧﯽ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ ﻛﻪ ﺁﻧﺎﻥ (ﻳﻌﻨﯽ ﭘﻴﺎﻣﺒﺮ ﻭ ﺟﺎﻧﺸـﻴﻨﺎﻥ ﺍﻭ) ﺭﺍ ﺑﺮﺍی ﺩﺳـﺘﻴﺎﺑﯽ ﺑﻪ ﺣﻘﺸـﺎﻥ ﻳـﺎﺭی ﻣﯽ ﻛﻨﻨﺪ ﻭ ﺩﺭ ﺩﺭﺟﻪ ﺑﻌﺪ ﻛﺴـﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﺁﻧﺎﻥ (ﻳﻌﻨﯽ ﭘﻴﺎﻣﺒﺮ ﻭ ﺍﻫﻞﺑﻴﺘﺶ) ﺭﺍ ﺩﻭﺳـﺖ ﺩﺍﺭﻧﺪ ﻭ ﺍﺯ ﺩﺷﻤﻨﺎﻧﺸـﺎﻥ ﺑﻴﺰﺍﺭﻧﺪ، ﺍﮔﺮﭼﻪ ﺍﻫﻞﺑﻴﺖ ﺭﺍ ﻳﺎﺭی ﻭ ﻧﺼﺮﺕ ﻧﻜﺮﺩﻩ ﺑﺎﺷﻨﺪ.»(18)

ﺍﺯ ﺍﻳـﻦ ﮔﻔﺘﺎﺭ ﺧﺪﺍی ﺑﻠﻨﺪﻣﺮﺗﺒﻪ ﺩﺭﻣﯽ ﻳﺎﺑﻴﻢ ﻛﻪ ﺑﺎﻻﺗﺮﻳﻦ ﺩﺭﺟﻪ ﺍی ﻛـﻪ ﻳـﮏ ﺷـﻴﻌﻪ ﻣﯽ ﺗﻮﺍﻧـﺪ ﺑـﺪﺍﻥ ﺩﺳـﺖ ﻳﺎﺑـﺪ «ﺭﺗﺒـﻪ ﻳـﺎﺭی ﺍﻫﻞﺑﻴﺖ ﻋﻠﻴﻬﻢﺍﻟﺴـﻼﻡ» ﺍﺳـﺖ ﻭ ﺭﺗﺒـﻪ «ﻣﺤﺒـﺖ ﺍﻫﻞﺑﻴـﺖ ﻭ ﺑﻐﺾ ﺩﺷـﻤﻨﺎﻥ ﺍﻳﺸﺎﻥ » ﺭﺗﺒﻪ ﺍی ﭘﺎﻳﻴﻦﺗﺮ ﺍﺯ ﻣﺮﺗﺒﻪ «ﻧﺼﺮﺕ ﻭ ﻳﺎﺭی ﮔﺮی» ﺍﺳﺖ.

ﻋﻤـﻮﻡ ﺷـﻴﻌﻴﺎﻥ ﺩﺭ ﺩﻝ ﺧﻮﺩ ﻣﺤﺒﺖ ﺍﻣﺎﻡ ﺯﻣﺎﻧﺸـﺎﻥ ﺭﺍ ﺩﺍﺭﻧﺪ ﻭ ﺍﺯ ﺩﺷﻤﻨﺎﻥ ﺍﻳﺸﺎﻥ ﻫﻢ ﺑﻴﺰﺍﺭﻧﺪ، ﻟﻴﻜﻦ ﻋﻤﻮﻡ ﺁﻥ ﻫﺎ ﺗﻮﻓﻴﻖ ﻳﺎﺭی ﺁﻥ ﺣﻀﺮﺕ ﺭﺍ ﻧﺪﺍﺭﻧﺪ. ﺧﻴﻠﯽ ﺍﺯ ﻣﺎ ﺍﺻﻼ ﻧﻤﯽ ﺩﺍﻧﻴﻢ ﻛﻪ ﺑﺎﻻﺗﺮ ﺍﺯ ﻣﻘﺎﻡ ﻣﺤﺒﺖ ﻣﺮﺗﺒﻪ ﺩﻳﮕﺮی ﻫﻢ ﺑﻪ ﻧﺎﻡ «ﻣﻘﺎﻡ ﻧﺼﺮﺕ» ﻳﻌﻨﯽ «ﻣﻘﺎﻡ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ» ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﭼﻪ ﺑﺴـﺎ ﺍﮔﺮ ﺑﺪﺍﻧﻴﻢ ﻛﻪ ﻋﻼﻭﻩ ﺑـﺮ ﻣﺤﺒﺖ ﻭﺭﺯﻳﺪﻥ ﺑﻪ ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﺎﻳـﺪ ﺑﻪ ﻳﺎﺭی ﺍﻭ ﻧﻴﺰ ﺍﻗﺪﺍﻡ ﻛﻨﻴﻢ، ﺑـﻪ ﺍﻳﻦ ﻛﺎﺭ ﺗﻮﻓﻴﻖ ﻳﺎﺑﻴﻢ. ﭘﺲ ﻳﻜﯽ ﺍﺯ ﺭﺳﺎﻟﺖﻫﺎی ﺍﻓﺮﺍﺩ ﺁﮔﺎﻩ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺷﻴﻌﻴﺎﻥ ﻭ ﺩﻭﺳﺘﺪﺍﺭﺍﻥ ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﺑﻪ ﻳﺎﺭی ﻭ ﻫﻮﺍﺩﺍﺭی ﺍﺯ ﺍﻳﺸﺎﻥ ﺩﻋﻮﺕ ﻧﻤﺎﻳﻨﺪ ﻭ ﺑﻪ ﺁﻧﻬﺎ ﮔﻮﺷﺰﺩ ﻛﻨﻨﺪ ﻛﻪ : «ﻣﻘﺎﻡ ﻳﺎﻭﺭﺍﻥ ﺍﻣﺎﻡ ﺑﺎﻻﺗﺮ ﺍﺯ ﻣﻘﺎﻡ ﺩﻭﺳﺘﺪﺍﺭﺍﻥ ﺍﻣﺎﻡ ﺍﺳﺖ ﻭ ﺛﻮﺍﺏ ﻭ ﺍﺭﺯﺵ ﺁﻧﻬﺎ ﻧﺰﺩ ﺧﺪﺍ ﺑﺎﻻﺗﺮ ﻭ ﻣﺤﺒﺖ ﺧﺪﺍ ﻧﺴـﺒﺖ ﺑﻪ ﺁﻧﻬﺎ ﺑﻴﺸﺘﺮ ﺍﺳﺖ. ﭘﺲ ﺳﺰﺍﺳـﺖ ﻛـﻪ ﻣﺎ ﺑﻪ ﻣﺤﺒﺖ ﻗﻠﺒﯽ ﺑﻪ ﺣﻀﺮﺗـﺶ ﺍﻛﺘﻔﺎ ﻧﻜﻨﻴﻢ، ﺑﻠﻜﻪ ﺩﺭ ﻋﻤﻞ ﺑﻪ ﺟﻤﻊ ﻳﺎﻭﺭﺍﻥ ﺣﻀﺮﺗﺶ ﺑﭙﻴﻮﻧﺪﻳﻢ.»

ﭘﺲ ﺍﺯ ﺍﻳﻨﻜﻪ ﺍﻫﻤﻴﺖ ﻭ ﺍﺭﺯﺵ «ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ» ﺭﺍ ﺩﺍﻧﺴـﺘﻴﻢ ﻭ ﺧﻮﺍﺳﺘﻴﻢ ﻛﻪ ﺑﻪ ﺍﻳﻦ ﻛﺎﺭ ﺍﻗﺪﺍﻡ ﻛﻨﻴﻢ ﺍﻳﻦ ﺳﺆﺍﻝ ﭘﻴﺶ ﻣﯽ ﺁﻳﺪ ﻛﻪ ﭼﮕﻮﻧﻪ ﻣﯽ ﺗﻮﺍﻥ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺭﺍ ﻳﺎﺭی ﻛﺮﺩ، ﺑﺎﺭی ﺍﺯ ﺩﻭﺵ ﺍﻳﺸـﺎﻥ ﺑﺮﺩﺍﺷﺖ ﻭ ﻏﻤﯽ ﺍﺯ ﻏﻢﻫﺎی ﺍﻳﺸﺎﻥ ﺯﺍﺋﻞ ﺳﺎﺧﺖ.

ﺍﻳﻦ ﺳﺆﺍﻟﯽ ﺍﺳﺖ ﻛﻪ ﺧﻴﻠﯽ ﻫﺎ ﺑﺎ ﺁﻥ ﻣﻮﺍﺟﻪ ﺍﻧﺪ، ﺍﻣﺎ ﭼﻮﻥ ﭘﺎﺳﺨﺶ ﺭﺍ ﻧﻤﯽ ﺩﺍﻧﻨﺪ ﻧﻤﯽ ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺍﻗﺪﺍﻣﯽ ﻋﻤﻠﯽ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺩﺳـﺖ ﺑﺰﻧﻨﺪ. ﺩﺭ ﻣﺴـﻴﺮ ﺩﺳـﺘﻴﺎﺑﯽ ﺑﻪ ﻣﻘﺎﻡ ﻧﺼﺮﺕ ﮔﺮﺩﻧﻪ ﻫﺎ ﻭ ﮔﻠﻮﮔﺎﻩ ﻫﺎﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻛـﻪ ﺑﺴـﻴﺎﺭی، ﺍﺯ ﮔﺬﺭﻛﺮﺩﻥ ﺍﺯ ﺁﻥ ﻫﺎ ﺑﺎﺯ ﻣﯽ ﻣﺎﻧﻨـﺪ ﻭ ﺩﺭ ﻋﻤﻞ ﻣﻮﻓﻖ ﺑﻪ ﺍﻳﻦ ﻛﺎﺭ ﻧﻤﯽ ﺷـﻮﻧﺪ. ﺍﻭﻟﻴـﻦ ﮔﻠﻮﮔﺎﻩ، ﺟﻬﻞ ﻭ ﺑﯽ ﺧﺒﺮی ﺍﺯ ﺍﻳﻦ ﺍﺳـﺖ ﻛـﻪ ﻧﺼﺮﺕ ﻭ ﻳـﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭ ﺩﻳﻦ ﺍﺳـﻼﻡ ﺟﺎﻳﮕﺎﻩ ﺭﺳـﻤﯽ ﺩﺍﺭﺩ ﻭ ﺩﻭﻣﻴﻦ ﮔﻠﻮﮔﺎﻩ، ﺍﻳﻦ ﺍﺳـﺖ ﻛﻪ ﺷﺨﺺ ﺑﺪﺍﻧﺪ ﭼﮕﻮﻧﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻣﺎﻡ ﺭﺍ ﻳﺎﺭی ﻛﻨﺪ.

ﺩﺭ ﺍﻳﻦ ﮔﻔﺘﺎﺭ ﻗﺼﺪ ﺑﺮ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺑﻪ ﭘﻨﺞ ﺭﺍﻩ ﻓﺮﺍﮔﻴﺮ ﻭ ﻋﻤﻮﻣﯽ ﺑﺮﺍی ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨﺎ ﻓﺪﺍﻩ ﺍﺷـﺎﺭﻩ ﻛﻨﻴﻢ. ﺧﺼﻮﺻﻴﺖ ﺍﻳﻦ ﺷـﻴﻮﻩ ﻫﺎ ﻗﺎﺑﻠﻴﺖ ﻓﺮﺍﮔﻴﺮ ﺷﺪﻥ ﺁﻧﻬﺎﺳﺖ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻛﻪ ﻫﻤﻪ ﺷﻴﻌﻴﺎﻥ ﺍﻋـﻢ ﺍﺯ ﻣﺘﺨﺼﺼﺎﻥ ﻭ ﻏﻴﺮﻣﺘﺨﺼﺼـﺎﻥ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺁﻧﻬﺎ ﻋﻤﻞ ﻛﻨﻨﺪ ﻭ ﺑﻪ ﻋﻨﺎﻳﺖ ﺍﻟﻬﯽ ﻧﺸﺎﻥ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﻧﺼﻴﺐ ﺧﻮﺩ ﺳﺎﺯﻧﺪ.

ﺍﻳﻦ ﭘﻨﺞ ﺷﻴﻮﻩ ﺩﺭ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺍﺯ ﺍﻳﻦ ﻗﺮﺍﺭﻧﺪ:

١. ﺍﻭﻟﻴـﻦ ﺷـﻴﻮﻩ ﻳـﺎﺭی ﺍﻣـﺎﻡ ﻋﺼﺮ: ﺩﻋـﻮﺕ ﺍﺯ ﺩﻳﮕـﺮﺍﻥ ﺟﻬﺖ ﺑﻬﺮﻩ ﻣﻨﺪ ﺷﺪﻥ ﺍﺯ ﺑﺮﻛﺎﺕ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ.

ﻫﺮ ﺷﻴﻌﻪ ﺍی ﻛﻪ ﻧﻌﻤﺖ ﻭﺟﻮﺩ ﺍﻣﺎﻡ ﻋﺼﺮ ﺭﺍ ﺩﺭک ﻛﺮﺩﻩ ﻭ ﺣﻼﻭﺕ ﻋﻨﺎﻳﺎﺕ ﺍﻭ ﺭﺍ ﭼﺸﻴﺪﻩ ﺑﺎﺷﺪ، ﻣﯽ ﺗﻮﺍﻧﺪ ﻭﺟﻮﺩ ﺍﻳﻦ ﻧﻌﻤﺖ ﺭﺍ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺧﺒﺮ ﺩﻫﺪ ﻭ ﺍﺯ ﺁﻧﻬﺎ ﺩﻋﻮﺕ ﻛﻨﺪ ﻛﻪ ﺑﺎ ﺗﻮﺳﻞ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺍﺯ ﻋﻨﺎﻳـﺎﺕ ﺍﻭ ﺑﻬﺮﻩ ﻣﻨـﺪ ﮔﺮﺩﻧـﺪ. ﺍﺭﺟﺎﻉ ﺩﻳﮕـﺮﺍﻥ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﺮﺍی ﺭﻓﻊ ﻧﻴﺎﺯﻫﺎی ﻣﺎﺩی ﻭ ﻣﻌﻨﻮﻳﺸـﺎﻥ ﻛﺎﺭی ﺑﺴـﻴﺎﺭ ﺳـﺎﺩﻩ ﺍﺳـﺖ. ﺍﻳﻦ ﻛﺎﺭ ﺯﻣﻴﻨﻪ ﻗﺪﺭﺷﻨﺎﺳﯽ ﺍﻣﺖ ﺍﺯ ﺍﻣﺎﻡ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ ﺁﻭﺭﺩ ﻭ ﺩﺭ ﻏﺮﺑﺖﺯﺩﺍﻳﯽ ﺍﺯ ﺍﻳﻦ ﻣﻮﻻی ﻛﺮﻳﻢ ﺑﺴﻴﺎﺭ ﻣﺆﺛﺮ ﺍﺳﺖ، ﺍﺯ ﺍﻳﻦ ﺭﻭ ﺑﻪ ﻧﻮﺑﻪ ﺧﻮﺩ ﻧﻮﻋﯽ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ.

٢. ﺩﻭﻣﻴـﻦ ﺷـﻴﻮﻩ ﻳﺎﺭی ﺍﻣـﺎﻡ ﻋﺼﺮ: ﺗﺠﻠﻴـﻞ ﺍﺯ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺩﺭ ﺟﺎﻣﻌﻪ

ﻫـﺮ ﺷـﻴﻌﻪ ﺍی ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﺗﺠﻠﻴﻞ ﻭ ﺑﺰﺭﮔﺪﺍﺷـﺖ ﺍﺯ ﺍﻣـﺎﻡ ﺯﻣﺎﻥ ﺧﻮﺩ ﺩﺭ ﺟﺎﻣﻌﻪ ﺑﻜﻮﺷـﺪ. ﺍﻳﻦ ﻛﺎﺭ (ﻳﻌﻨﯽ ﮔﺮﺍﻣﻴﺪﺍﺷـﺖ ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭ ﺟﺎﻣﻌـﻪ )، ﺍﺯ ﻏﺮﺑﺖ ﺁﻥ ﻣـﻮﻻ ﻣﯽ ﻛﺎﻫﺪ ﻭ ﺗﻮﺟﻪ ﻫﻤﮕﺎﻧﯽ ﺭﺍ ﺑﻪ ﺍﻳﺸـﺎﻥ ﺑﻴﺸﺘﺮ ﻣﯽ ﻛﻨﺪ.

ﻫﻤـﻪ ﻣـﺎ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺗﺎﺑﻠﻮﻳﯽ ﺍﺯ ﻧﺎﻡ ﺍﻣـﺎﻡ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺟﻬـﺖ ﺍﺑﺮﺍﺯ ﺍﺭﺍﺩﺕ ﺑـﻪ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻭ ﺗﺠﻠﻴﻞ ﺍﺯ ﺍﻳﺸـﺎﻥ ﺩﺭ ﺧﺎﻧﻪ ﻳﺎ ﻣﺤﻞ ﻛﺎﺭ ﻧﺼﺐ ﻛﻨﻴﻢ.

ﻫﻤـﻪ ﻣـﺎ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺑﻪ ﻫﻨـﮕﺎﻡ ﺫﻛﺮ ﻧﺎﻡ ﺍﻳﺸـﺎﻥ ﺑـﺎ ﺍﺣﺘـﺮﺍﻡ ﺍﺯ ﺁﻥ ﺣﻀﺮﺕ ﻳﺎﺩ ﻧﻤﺎﻳﻴﻢ ﻭ ﺑﺮ ﺍﻭ ﺩﺭﻭﺩ ﻭ ﺻﻠﻮﺍﺕ ﺑﻔﺮﺳﺘﻴﻢ.

ﻫﻤـﻪ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺑـﻪ ﻫﻨﮕﺎﻡ ﺑـﺮﺩﻥ ﻧﺎﻡ ﺧﺎﺹ ﺍﻳﺸـﺎﻥ ﻳﻌﻨـﯽ ﻟﻘﺐ «ﻗﺎﺋﻢ» ﺑﺮﺧﻴﺰﻳﻢ، ﺩﺳـﺖ ﺑﺮ ﺳﺮ ﻧﻬﻴﻢ، ﺍﺩﺍی ﺍﺣﺘﺮﺍﻡ ﻛﻨﻴﻢ ﻭ ﺑﺮﺍی ﻓﺮﺝ ﺍﻳﺸﺎﻥ ﺩﻋﺎ ﻛﻨﻴﻢ.(19)

ﻫﻤـﻪ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺩﺭ ﺭﻭﺯﻫـﺎی ﻣﻨﺴـﻮﺏ ﺑـﻪ ﺍﻳﺸـﺎﻥ ﻫﻤﭽـﻮﻥ ﻧﻴﻤﻪ ﺷـﻌﺒﺎﻥ ، ﺭﻭﺯﻫﺎی ﺟﻤﻌـﻪ ﻭ ﻋﻴﺪﻏﺪﻳﺮ ﺑﺎ ﺑﺮﮔﺰﺍﺭی ﻣﺮﺍﺳـﻢ ﻭﻳﮋﻩ ﺑﺮﺍی ﺁﻥ ﺣﻀﺮﺕ، ﻭﻻﻳﺖ ﻭ ﺍﻣﺎﻣﺖ ﺍﻭ ﺭﺍ ﺑﻪ ﻳﻜﺪﻳﮕﺮ ﻳﺎﺩﺁﻭﺭ ﺷـﻮﻳﻢ. ﺣﺘﯽ ﺍﮔﺮ ﻧﺘﻮﺍﻧﻴﻢ ﺑﺮﮔﺰﺍﺭﻛﻨﻨﺪﻩ ﭼﻨﻴﻦ ﻣﺮﺍﺳﻤﯽ ﺑﺎﺷﻴﻢ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺩﺭ ﺁﻥ ﺧﺪﻣﺖ ﻛﻨﻴﻢ ﻳﺎ ﻻﺍﻗﻞ ﺑﺎ ﺣﻀﻮﺭ ﺧﺎﻟﺼﺎﻧﻪ ﺧﻮﺩ ﺑﻪ ﺁﻥ ﺭﻭﻧﻖ ﺑﺨﺸﻴﻢ.

ﻫﻤﻪ ﺍﻳﻦ ﻛﺎﺭﻫﺎ، ﺗﺠﻠﻴﻞ ﻭ ﺑﺰﺭﮔﺪﺍﺷﺖ ﺍﺯ ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ ﻭ ﺷﻴﻮﻩ ﺍی ﺍﺯ ﺷﻴﻮﻩ ﻫﺎی ﻳﺎﺭی ﺁﻥ ﺣﻀﺮﺕ ﺍﺳﺖ.

٣. ﺳـﻮﻣﻴﻦ ﺷﻴﻮﻩ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ: ﻳـﺎﺭی ﺷـﻴﻌﻴـﺎﻥ ﺍﻣﺎﻡ ﺯﻣـﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ

ﺑﺨﺶ ﻣﻬﻤﯽ ﺍﺯ ﺩﻏﺪﻏﻪ ﻫﺎی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ ﺑﻪ ﮔﺮﻓﺘﺎﺭی ﺷـﻴﻌﻴﺎﻥ ﺍﻳﺸـﺎﻥ ﻣﺮﺑﻮﻁ ﻣﯽ ﺷـﻮﺩ. ﺷـﻴﻌﻴﺎﻥ ، ﺧﺎﻧﻮﺍﺩﻩ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﻨﺪ. ﺁﻥ ﺣﻀﺮﺕ ﺍﻳﺸـﺎﻥ ﺭﺍ ﻣﺎﻧﻨﺪ ﻓﺮﺯﻧﺪﺍﻥ ﺧﻮﺩ ﺩﻭﺳـﺖ ﻣﯽ ﺩﺍﺭﺩ، ﺑﻪ ﻭﺍﺳـﻄﻪ ﮔﺮﻓﺘﺎﺭی ﺁﻧﻬﺎ ﺍﻧﺪﻭﻫﮕﻴﻦ ﻣﯽ ﺷـﻮﺩ ﻭ ﺍﺯ ﺑﺎﺑﺖ ﭘﻴﺮﻭﺯی ﻭ ﻧﺠﺎﺕ ﺁﻧﺎﻥ ﺷﺎﺩ ﻣﯽ ﮔﺮﺩﺩ.

ﺩﺭ ﺗﻌﺎﻟﻴﻢ ﺍﻫﻞﺑﻴﺖ ﺑﻴﺎﻥ ﺷـﺪﻩ ﻛﻪ ﺧﻮﺷـﺤﺎﻝ ﻛﺮﺩﻥ ﺷﻴﻌﻴﺎﻥ ﺍﺋﻤﻪ ﺩﺭ ﺣﻜﻢ ﺧﻮﺷﺤﺎﻝ ﻛﺮﺩﻥ ﺧﻮﺩ ﺁﻧﻬﺎﺳﺖ، ﺩﻳﺪﺍﺭ ﻭ ﻋﻴﺎﺩﺕ ﺍﺯ ﺷﻴﻌﻴﺎﻥ ﺩﺭ ﺣﻜـﻢ ﺩﻳـﺪﺍﺭ ﻭ ﻋﻴﺎﺩﺕ ﺍﺯ ﺧﻮﺩ ﺍﻫﻞﺑﻴﺖ ﺍﺳـﺖ ﻭ ﻛﻤـﮏ ﻣﺎﻟﯽ ﺑﻪ ﺷـﻴﻌﻴﺎﻥ ﮔﺮﻓﺘﺎﺭ ﺑﻪ ﻣﻨﺰﻟﻪ ﻛﻤﮏ ﻣﺎﻟﯽ ﺑﻪ ﺧﻮﺩ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ ﺍﺳﺖ.

ﺑـﺮ ﺍﻳـﻦ ﻣﺒﻨـﺎ ﺍﮔـﺮ ﻛﺴـﯽ ﻧﺴـﺒﺖ ﺑـﻪ ﺷـﻴﻌﻴﺎﻥ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻛـﻪ ﺟﺎﻥ ﻫﺎﻳﻤـﺎﻥ ﻓﺪﺍﻳـﺶ ﺑـﺎﺩ ﺍﺣﺴـﺎﺱ ﻏﻴـﺮﺕ ﻭ ﻭﻇﻴﻔﻪ ﻛﻨـﺪ، ﮔﻮﻳﯽ ﻧﺴـﺒﺖ ﺑـﻪ ﭘﺎﺳﺪﺍﺷـﺖ ﺣﺮﻳﻢ ﺧـﻮﺩ ﺍﻣﺎﻡ ﻋﺼـﺮ ﻏﻴﺮﺕ ﻭﺭﺯﻳـﺪﻩ ﻭ ﺑﻪ ﺷﺨﺺ ﺍﻳﺸﺎﻥ ﻳﺎﺭی ﺭﺳﺎﻧﺪﻩ ﺍﺳﺖ.

ﺑﻴﺎﻳﻴـﺪ ﺑﻪ ﻧﻴـﺖ ﻳﺎﺭی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ، ﺑﻪ ﻳﺎﺭی ﺷـﻴﻌﻴﺎﻥ ﮔﺮﻓﺘﺎﺭ ﺁﻥ ﺣﻀﺮﺕ ﺑﺸـﺘﺎﺑﻴﻢ ﻭ ﺑﻪ ﻧﻴﺖ ﺧﺪﻣﺖ ﺑﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﺩﺭ ﻋﺰﺕ ﺑﺨﺸﻴﺪﻥ ﺑﻪ ﺟﺎﻣﻌﻪ ی ﺷﻴﻌﻪ ﻛﻮﺷﺎﺗﺮ ﺍﺯ ﭘﻴﺶ ﮔﺎﻡ ﻧﻬﻴﻢ.

۴. ﭼﻬﺎﺭﻣﻴـﻦ ﺷـﻴﻮﻩ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻋﻠﻴﻪ ﺍﻟﺴـﻼﻡ: ﺯﻳﻨﺖ ﺑﻮﺩﻥ ﺑﺮﺍی ﺻﺎﺣﺐ ﺍﻟﺰﻣﺎﻥ ﺑﻪ ﻭﺳـﻴﻠﻪ ی ﺟﺪﻳﺖ ﺩﺭ ﺩﺭﺳـﺘﻜﺎﺭی ﻭ ﭘﺮﻫﻴﺰ ﺍﺯ ﺭﻓﺘﺎﺭﻫﺎی ﺯﺷﺖ.

ﮔﻨﺎﻫﺎﻥ ﺑﻪ ﺧﺼﻮﺹ ﮔﻨﺎﻫﺎﻥ ﺑﺰﺭگ ﻭ ﺯﺷـﺖﻛﺎﺭی ﻫﺎی ﻓﺎﺣﺶ ﺍﮔـﺮ ﺍﺯ ﻧﺎﺣﻴﻪ ﻛﺴـﺎﻧﯽ ﺳـﺮ ﺑﺰﻧﺪ ﻛﻪ ﻣﻨﺘﺴـﺐ ﺑـﻪ ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻫﺴـﺘﻨﺪ ﻭ ﺷـﻴﻌﻪ ﺁﻥ ﺣﻀﺮﺕ ﺑﻪ ﺷـﻤﺎﺭ ﻣﯽ ﺭﻭﻧﺪ، ﺁﺑـﺮﻭی ﺁﻥ ﺑﺰﺭﮔﻮﺍﺭ ﺭﺍ ﺩﺭ ﻧﺰﺩ ﺑﻴﮕﺎﻧﮕﺎﻥ ﺑﻪ ﻣﺨﺎﻃﺮﻩ ﻣﯽ ﺍﻓﻜﻨﺪ ﻭ ﺩﻝ ﺁﻥ ﻋﺰﻳﺰ ﺭﺍ ﺁﺯﺭﺩﻩ ﻣﯽ ﺳـﺎﺯﺩ. ﺩﺭ ﻣﻘﺎﺑﻞ، ﺩﺭﺳـﺘﻜﺎﺭی، ﺧﻮﻳﺸـﺘﻨﺪﺍﺭی ﻭ ﭘﺮﻫﻴﺰﻛﺎﺭی ﺷﻴﻌﻴﺎﻥ ﺑﻪ ﻋﺰﺕ ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭ ﺟﺎﻣﻌﻪ ﻣﯽ ﺍﻓﺰﺍﻳﺪ ﻭ ﻗﻠﺐ ﺁﻥ ﻣﻮﻻ ﺭﺍ ﺷﺎﺩ ﻣﯽ ﺳﺎﺯﺩ. ﻫﻤﻪ ﻣـﺎ ﻣﯽ ﺗﻮﺍﻧﻴـﻢ ﺑﺮﺍی ﺣﻔﻆ ﺁﺑـﺮﻭی ﺍﻣﺎﻡ ﻋﺼﺮ ﻭ ﻣﻜﺘﺐ ﺗﺮﺑﻴﺘﯽ ﺍﻳﺸـﺎﻥ ﻧﺴـﺒﺖ ﺑﻪ ﺩﺭﺳـﺘﻜﺎﺭی ﻭ ﭘﺮﻫﻴﺰ ﺍﺯ ﻛﺎﺭﻫﺎی ﺯﺷﺖ، ﺩﻗﺖ ﻭ ﺗﻼﺵ ﺩﻭ ﭼﻨﺪﺍﻥ ﺩﺍﺷـﺘﻪ ﺑﺎﺷـﻴﻢ ﻭ ﺑﺪﻳـﻦ ﮔﻮﻧﻪ ، ﺍﻣﺎﻡ ﻭﻗﺖ ﺧـﻮﺩ ﺭﺍ ﻳﺎﺭی ﻛﻨﻴﻢ، ﭼﻨﺎﻧﻜـﻪ ﺍﻣﻴﺮﻣﺆﻣﻨﺎﻥ ﺑﻪ ﻳـﺎﺭﺍﻥ ﺧﻮﻳﺶ ﻓﺮﻣﻮﺩ: «ﻣﺮﺍ ﺑﺎ ﺧﻮﻳﺸـﺘﻨﺪﺍﺭی (ﻧﺴﺒﺖ ﺑﻪ ﺯﺷﺘﯽ ﻫﺎ) ﻭ ﺗﻼﺵ (ﺩﺭ ﺭﺍﻩ ﺧﻮﺑﯽ ﻫﺎ) ﻳﺎﺭی ﻛﻨﻴﺪ.»(20)

۵. ﭘﻨﺠﻤﻴﻦ ﺷـﻴﻮﻩ ﻳـﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ: ﺟﻠﺐ ﻳﺎﺭی ﻫﺎی ﺁﺳـﻤﺎﻧﯽ ﺑﺮﺍی ﺍﻣﺎﻡ ﺯﻣﺎﻥ ﻋﻠﻴﻪ ﺍﻟﺴﻼﻡ

ﻫﻤـﻪ ﻣـﺎ ﻣﯽ ﺩﺍﻧﻴﻢ ﻛـﻪ ﻛﻤﮏ ﺑﻪ ﻓﻘـﺮﺍ (ﻳﺎ ﻫﻤﺎﻥ ﺻﺪﻗـﻪ ) ﺭﻓﻊ ﺑﻼ ﻣﯽ ﻛﻨﺪ، ﭘﺲ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺑﺎ ﺩﺍﺩﻥ ﺻﺪﻗﻪ ﺍﺯ ﻃﺮﻑ ﺍﻣﺎﻡ ﻋﺼﺮ ﺩﺭ ﺭﻓﻊ ﺑﻼ ﺍﺯ ﻭﺟﻮﺩ ﺷﺮﻳﻔﺶ ﮔﺎﻡ ﺑﺮﺩﺍﺭﻳﻢ.

ﻧﻴـﺰ ﻫﻤـﻪ ی ﻣـﺎ ﻣﯽ ﺩﺍﻧﻴـﻢ ﻛﻪ ﺩﻋـﺎ، ﺩﺭ ﺗﻌﺠﻴـﻞ ﻓﺮﺝ ﺍﻣـﺎﻡ ﻋﺼﺮ ﻭ ﺍﻳﺠﺎﺩ ﮔﺸـﺎﻳﺶ ﺩﺭ ﻛﺎﺭ ﺍﻳﺸـﺎﻥ ﻣﺆﺛﺮ ﺍﺳﺖ، ﭘﺲ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﺩﻋﺎ ﻭ ﻧﻴﺎﻳـﺶ ﺩﺭ ﺩﺭﮔﺎﻩ ﺍﻟﻬـﯽ ﺭﻭی ﺁﻭﺭﻳﻢ ﻭ ﻣﺮﺗﺒﺎ ﺍﺯ ﺧﺪﺍ ﺑﺨﻮﺍﻫﻴﻢ ﻛﻪ ﻓﺮﺝ ﺍﻳﺸﺎﻥ ﺭﺍ ﺑﻪ ﭘﻴﺶ ﺍﻧﺪﺍﺯﺩ ﻭ ﻫﻢ ﻭ ﻏﻢ ﻭ ﮔﺮﻓﺘﺎﺭی ﺭﺍ ﺍﺯ ﻭﺟﻮﺩ ﺷﺮﻳﻔﺶ ﺑﻪ ﻛﻠﯽ ﺯﺍﻳﻞ ﺳﺎﺯﺩ.

ﺟـﺬﺏ ﻳـﺎﺭﺍﻥ ﻭ ﺟﻠﺐ ﻳﺎﺭی ﺑـﺮﺍی ﺍﻣﺎﻡ ﻋﺼﺮ، ﺧـﻮﺩ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺷﻴﻮﻩ ﻫﺎی ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺁﻳﺪ، ﺑﻪ ﺧﺼﻮﺹ ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﻢ ﺍﻳﻦ ﻳﺎﺭی ﺭﺍ ﺍﺯ ﻓﻀﻞ ﺧﺪﺍ ﺗﻤﻨﺎ ﻛﻨﻴﻢ.

ﺩﺭ ﭘﺎﻳﺎﻥ ، ﺑﺎﺭ ﺩﻳﮕﺮ ﭘﻨﺞ ﺭﺍﻩ ﻋﻤﻮﻣﯽ ﻭ ﻓﺮﺍﮔﻴﺮ ﺩﺭ ﻳﺎﺭی ﺍﻣﺎﻡ ﻋﺼﺮ ﺍﺭﻭﺍﺣﻨـﺎ ﻓﺪﺍﻩ ﺭﺍ ﻣﺮﻭﺭ ﻣﯽ ﻛﻨﻴﻢ ﻭ ﺍﺯ ﺧﺪﺍی ﺑـﺰﺭگ ﻣﯽ ﺧﻮﺍﻫﻴﻢ ﻛﻪ ﺑﻪ ﻫﻤﻪ ی ﻣﺎ ﺗﻮﻓﻴﻖ ﻋﻤﻞ ﺑﺪﺍﻥ ﻫﺎ ﺭﺍ ﺍﺭﺯﺍﻧﯽ ﻛﻨﺪ.
